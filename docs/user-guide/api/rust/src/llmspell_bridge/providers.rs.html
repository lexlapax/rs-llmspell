12/// Manages LLM providers for script access

18impl ProviderManager {
19    /// Create a new provider manager with configuration
20    ///
21    /// # Errors
22    ///
23    /// Returns an error if provider registration fails
24    pub async fn new(config: ProviderManagerConfig) -> Result<Self, LLMSpellError> {
25        let core_manager = CoreProviderManager::new();
26        let manager = Self {
27            core_manager,
28            config: config.clone(),
29        };
30
31        // Register the rig provider factory
32        manager.register_rig_provider().await?;
33
34        // Initialize configured providers
35        manager.initialize_providers().await?;
36
37        Ok(manager)
38    }

40    /// Register the rig provider factory
41    async fn register_rig_provider(&self) -> Result<(), LLMSpellError> {
42        self.core_manager
43            .register_provider("rig", llmspell_providers::create_rig_provider)
44            .await;
45        Ok(())
46    }

48    /// Initialize providers from configuration
49    async fn initialize_providers(&self) -> Result<(), LLMSpellError> {
50        // Initialize each configured provider
51        for (name, config) in &self.config.providers {
52            let provider_config = Self::create_provider_config(name, config)?;
53            self.core_manager.init_provider(provider_config).await?;
54        }
55
56        // Set default provider if specified
57        if let Some(ref default) = self.config.default_provider {
58            if !self.config.providers.contains_key(default) {
59                return Err(LLMSpellError::Validation {
60                    field: Some("default_provider".to_string()),
61                    message: format!("Default provider '{default}' not found in configuration"),
62                });
63            }
64            // The default will be set based on the provider:model format
65            let model = self.config.providers[default]
66                .default_model
67                .as_ref()
68                .ok_or_else(|| LLMSpellError::Validation {
69                    field: Some("model".to_string()),
70                    message: format!("Model not specified for provider '{default}'"),
71                })?;
72            // Get the provider config to determine the actual provider name
73            let provider_config = &self.config.providers[default];
74            let provider_name = match provider_config.provider_type.as_str() {
75                "openai" | "anthropic" | "cohere" | "groq" | "perplexity" | "together"
76                | "gemini" | "mistral" | "replicate" | "fireworks" => "rig",
77                other => other,
78            };
79
80            // Use hierarchical naming: name/provider_type/model
81            self.core_manager
82                .set_default_provider(format!(
83                    "{}/{}/{}",
84                    provider_name, provider_config.provider_type, model
85                ))
86                .await?;
87        }
88
89        Ok(())
90    }

92    /// Create a provider config from our configuration
93    fn create_provider_config(
94        name: &str,
95        config: &ProviderConfig,
96    ) -> Result<ProviderInstanceConfig, LLMSpellError> {
97        // Map provider_type to the actual provider name
98        let provider_name = match config.provider_type.as_str() {
99            "openai" | "anthropic" | "cohere" | "groq" | "perplexity" | "together" | "gemini"
100            | "mistral" | "replicate" | "fireworks" => "rig",
101            other => other,
102        };
103
104        let model = config
105            .default_model
106            .as_ref()
107            .ok_or_else(|| LLMSpellError::Validation {
108                field: Some("model".to_string()),
109                message: format!("Model not specified for provider '{name}'"),
110            })?;
111
112        // Use new_with_type to preserve provider_type information
113        let mut provider_config =
114            ProviderInstanceConfig::new_with_type(provider_name, &config.provider_type, model);
115
116        // Set API key from config (already loaded from environment by registry if needed)
117        if let Some(ref api_key) = config.api_key {
118            provider_config.api_key = Some(api_key.clone());
119        } else if let Some(ref api_key_env) = config.api_key_env {
120            // Fallback: read from environment if not in config (for backward compatibility)
121            provider_config.api_key =
122                Some(
123                    std::env::var(api_key_env).map_err(|_| LLMSpellError::Configuration {
124                        message: format!(
125                            "Environment variable '{api_key_env}' not found for provider '{name}'"
126                        ),
127                        source: None,
128                    })?,
129                );
130        }
131
132        // Set other configuration
133        if let Some(ref base_url) = config.base_url {
134            provider_config.endpoint = Some(base_url.clone());
135        }
136
137        // Add max_tokens to custom config if specified
138        if let Some(max_tokens) = config.max_tokens {
139            provider_config
140                .custom_config
141                .insert("max_tokens".to_string(), serde_json::json!(max_tokens));
142        }
143
144        // Add extra configuration
145        for (key, value) in &config.options {
146            provider_config
147                .custom_config
148                .insert(key.clone(), value.clone());
149        }
150
151        Ok(provider_config)
152    }

205    /// List all registered providers
206    pub async fn list_providers(&self) -> Vec<ProviderInfo> {
207        let providers = self.core_manager.list_providers().await;
208        let mut infos = Vec::new();
209
210        for provider_name in providers {
211            let capabilities = self
212                .core_manager
213                .query_capabilities(Some(&provider_name))
214                .await
215                .ok();
216
217            // Consider a provider enabled if we can query its capabilities
218            let enabled = capabilities.is_some();
219
220            infos.push(ProviderInfo {
221                name: provider_name,
222                enabled,
223                capabilities,
224            });
225        }
226
227        infos
228    }

230    /// Get information about a specific provider
231    pub async fn get_provider_info(&self, name: &str) -> Option<ProviderInfo> {
232        // Check if provider exists in configuration
233        let exists = self.config.providers.contains_key(name)
234            || self
235                .core_manager
236                .list_providers()
237                .await
238                .contains(&name.to_string());
239
240        if !exists {
241            return None;
242        }
243
244        let capabilities = self.core_manager.query_capabilities(Some(name)).await.ok();
245
246        let enabled = capabilities.is_some();
247
248        Some(ProviderInfo {
249            name: name.to_string(),
250            enabled,
251            capabilities,
252        })
253    }

255    /// Check if a provider supports a specific capability
256    pub async fn check_provider_capability(&self, provider_name: &str, capability: &str) -> bool {
257        self.core_manager
258            .query_capabilities(Some(provider_name))
259            .await
260            .is_ok_and(|caps| match capability {
261                "streaming" => caps.supports_streaming,
262                "multimodal" => caps.supports_multimodal,
263                _ => false,
264            })
265    }

273    /// Create an Arc to a new core provider manager with the same configuration
274    /// This is needed for components that require ownership of the core manager
275    ///
276    /// # Errors
277    ///
278    /// Returns an error if provider configuration or initialization fails
279    pub async fn create_core_manager_arc(&self) -> Result<Arc<CoreProviderManager>, LLMSpellError> {
280        // Create a new core manager
281        let core_manager = CoreProviderManager::new();
282
283        // Register the rig provider factory
284        core_manager
285            .register_provider("rig", llmspell_providers::create_rig_provider)
286            .await;
287
288        // Initialize providers from our configuration
289        for (name, config) in &self.config.providers {
290            let provider_config = Self::create_provider_config(name, config)?;
291            core_manager.init_provider(provider_config).await?;
292        }
293
294        // Set default provider if specified
295        if let Some(ref default) = self.config.default_provider {
296            if let Some(provider_config) = self.config.providers.get(default) {
297                let model = provider_config.default_model.as_ref().ok_or_else(|| {
298                    LLMSpellError::Validation {
299                        field: Some("model".to_string()),
300                        message: format!("Model not specified for provider '{default}'"),
301                    }
302                })?;
303                let provider_name = match provider_config.provider_type.as_str() {
304                    "openai" | "anthropic" | "cohere" => "rig",
305                    other => other,
306                };
307                core_manager
308                    .set_default_provider(format!(
309                        "{}/{}/{}",
310                        provider_name, provider_config.provider_type, model
311                    ))
312                    .await?;
313            }
314        }
315
316        Ok(Arc::new(core_manager))
317    }
