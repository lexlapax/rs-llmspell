44/// Initialize and register all Phase 2 tools with the bridge registry
45///
46/// # Errors
47///
48/// Returns an error if tool registration fails
49pub fn register_all_tools(
50    registry: &Arc<ComponentRegistry>,
51    tools_config: &ToolsConfig,
52) -> Result<(), Box<dyn std::error::Error>> {
53    // Create a shared file sandbox for file system tools using configured allowed paths
54    let mut security_requirements = SecurityRequirements::default();
55    for path in &tools_config.file_operations.allowed_paths {
56        security_requirements = security_requirements.with_file_access(path);
57    }
58    let sandbox_context = SandboxContext::new(
59        "bridge-tools".to_string(),
60        security_requirements,
61        ResourceLimits::default(),
62    );
63    let file_sandbox = Arc::new(FileSandbox::new(sandbox_context)?);
64
65    // Register different tool categories with their specific configurations
66    register_utility_tools(registry)?;
67    register_data_processing_tools(registry, &tools_config.http_request)?;
68    register_file_system_tools(
69        registry,
70        file_sandbox.clone(),
71        &tools_config.file_operations,
72    )?;
73    register_system_tools(registry, &file_sandbox)?;
74    register_media_tools(registry, &file_sandbox)?;
75    register_search_tools(registry, &tools_config.web_search)?;
76    register_web_tools(registry)?;
77    register_communication_tools(registry)?;
78
79    Ok(())
80}

82/// Register a single tool with the bridge registry
83fn register_tool<T, F>(
84    registry: &Arc<ComponentRegistry>,
85    name: &str,
86    tool_factory: F,
87) -> Result<(), Box<dyn std::error::Error>>
88where
89    T: Tool + Send + Sync + 'static,
90    F: FnOnce() -> T,
91{
92    let tool = tool_factory();
93    registry
94        .register_tool(name.to_string(), Arc::new(tool))
95        .map_err(|e| Box::new(e) as Box<dyn std::error::Error>)?;
96    Ok(())
97}

170/// Register data processing tools
171fn register_data_processing_tools(
172    registry: &Arc<ComponentRegistry>,
173    http_request_config: &llmspell_config::tools::HttpRequestConfig,
174) -> Result<(), Box<dyn std::error::Error>> {
175    register_tool(registry, "csv_analyzer", || {
176        CsvAnalyzerTool::new(CsvAnalyzerConfig::default())
177    })?;
178    register_tool(registry, "json_processor", || {
179        JsonProcessorTool::new(JsonProcessorConfig::default())
180    })?;
181    register_tool_result(registry, "graphql_query", || {
182        GraphQLQueryTool::new(GraphQLConfig::default())
183    })?;
184    // Use the provided configuration for HttpRequestTool
185    let http_request_config = http_request_config.clone();
186    register_tool_result(registry, "http_request", move || {
187        // Convert from llmspell_config HttpRequestConfig to llmspell_tools HttpRequestConfig
188        // Note: Some fields in llmspell_config don't exist in tool config yet
189        let tool_config = HttpRequestConfig {
190            timeout_seconds: http_request_config.timeout_seconds,
191            follow_redirects: true, // Default to following redirects
192            max_redirects: http_request_config.max_redirects as usize,
193            retry_config: RetryConfig::default(), // TODO: Add retry config to llmspell_config
194            rate_limit_per_minute: None,          // TODO: Add rate limiting to llmspell_config
195            user_agent: http_request_config
196                .default_headers
197                .get("User-Agent")
198                .cloned()
199                .unwrap_or_else(|| "llmspell-http/1.0".to_string()),
200        };
201        HttpRequestTool::new(tool_config)
202    })?;
203    // Phase 7 tools
204    register_tool(registry, "pdf-processor", PdfProcessorTool::new)?;
205    register_tool(registry, "graph-builder", GraphBuilderTool::new)?;
206    Ok(())
207}

209/// Register file system tools
210fn register_file_system_tools(
211    registry: &Arc<ComponentRegistry>,
212    file_sandbox: Arc<FileSandbox>,
213    file_ops_config: &llmspell_config::tools::FileOperationsConfig,
214) -> Result<(), Box<dyn std::error::Error>> {
215    register_tool(registry, "archive_handler", ArchiveHandlerTool::new)?;
216
217    // File converter with sandbox
218    let file_sandbox_converter = file_sandbox.clone();
219    register_tool_with_sandbox(
220        registry,
221        "file_converter",
222        file_sandbox_converter.clone(),
223        move || FileConverterTool::new(FileConverterConfig::default(), file_sandbox_converter),
224    )?;
225
226    // Use the provided configuration for FileOperationsTool with sandbox
227    let file_ops_config = file_ops_config.clone();
228    let file_sandbox_ops = file_sandbox.clone();
229    register_tool_with_sandbox(
230        registry,
231        "file_operations",
232        file_sandbox_ops.clone(),
233        move || {
234            // Convert from llmspell_config FileOperationsConfig to llmspell_tools FileOperationsConfig
235            let tool_config = FileOperationsConfig {
236                allowed_paths: file_ops_config.allowed_paths.clone(),
237                atomic_writes: file_ops_config.atomic_writes,
238                max_file_size: file_ops_config.max_file_size,
239                max_dir_entries: 1000,      // Default value
240                allow_recursive: true,      // Default value
241                default_permissions: 0o644, // Default permissions
242            };
243            FileOperationsTool::new(tool_config, file_sandbox_ops)
244        },
245    )?;
246
247    // File search with sandbox
248    let file_sandbox_search = file_sandbox.clone();
249    register_tool_with_sandbox(
250        registry,
251        "file_search",
252        file_sandbox_search.clone(),
253        move || FileSearchTool::new(FileSearchConfig::default(), file_sandbox_search),
254    )?;
255
256    // File watcher with sandbox
257    let file_sandbox_watcher = file_sandbox;
258    register_tool_with_sandbox(
259        registry,
260        "file_watcher",
261        file_sandbox_watcher.clone(),
262        move || FileWatcherTool::new(FileWatcherConfig::default(), file_sandbox_watcher),
263    )?;
264    Ok(())
265}

378/// Information about a tool type
379#[derive(Debug, Clone, Serialize, Deserialize)]
380pub struct ToolInfo {
381    /// Tool name
382    pub name: String,
383    /// Tool description
384    pub description: String,
385    /// Tool category
386    pub category: ToolCategory,
387    /// Security level
388    pub security_level: SecurityLevel,
389    /// Tool schema with parameters
390    pub schema: ToolSchema,
391    /// Security requirements
392    pub security_requirements: SecurityRequirements,
393    /// Resource limits
394    pub resource_limits: ResourceLimits,
395}

397/// Tool discovery service
398pub struct ToolDiscovery {
399    /// Component registry
400    registry: Arc<ComponentRegistry>,
401}
402
403impl ToolDiscovery {
404    /// Create a new tool discovery service
405    #[must_use]
406    pub const fn new(registry: Arc<ComponentRegistry>) -> Self {
407        Self { registry }
408    }
409
410    /// Get information about a specific tool
411    #[must_use]
412    pub fn get_tool_info(&self, tool_name: &str) -> Option<ToolInfo> {
413        let tool = self.registry.get_tool(tool_name)?;
414        let metadata = tool.metadata();
415        let schema = tool.schema();
416        let category = tool.category();
417        let security_level = tool.security_level();
418        let security_requirements = tool.security_requirements();
419        let resource_limits = tool.resource_limits();
420
421        Some(ToolInfo {
422            name: tool_name.to_string(),
423            description: metadata.description.clone(),
424            category,
425            security_level,
426            schema,
427            security_requirements,
428            resource_limits,
429        })
430    }
431
432    /// List all available tool names
433    #[must_use]
434    pub fn list_tool_names(&self) -> Vec<String> {
435        self.registry.list_tools()
436    }
437
438    /// Get tools by category
439    #[must_use]
440    pub fn get_tools_by_category(&self, category: &str) -> Vec<(String, ToolInfo)> {
441        let tool_names = self.registry.list_tools();
442        tool_names
443            .into_iter()
444            .filter_map(|name| {
445                let tool = self.registry.get_tool(&name)?;
446                let tool_category = tool.category();
447                if tool_category.to_string() == category {
448                    self.get_tool_info(&name).map(|info| (name, info))
449                } else {
450                    None
451                }
452            })
453            .collect()
454    }
455}

457/// Implementation of unified BridgeDiscovery trait for ToolDiscovery
458#[async_trait::async_trait]
459impl BridgeDiscovery<ToolInfo> for ToolDiscovery {
460    async fn discover_types(&self) -> Vec<(String, ToolInfo)> {
461        self.list_tool_names()
462            .into_iter()
463            .filter_map(|name| self.get_tool_info(&name).map(|info| (name, info)))
464            .collect()
465    }
466
467    async fn get_type_info(&self, type_name: &str) -> Option<ToolInfo> {
468        self.get_tool_info(type_name)
469    }
470
471    async fn has_type(&self, type_name: &str) -> bool {
472        self.registry.get_tool(type_name).is_some()
473    }
474
475    async fn list_types(&self) -> Vec<String> {
476        self.list_tool_names()
477    }
478
479    async fn filter_types<F>(&self, predicate: F) -> Vec<(String, ToolInfo)>
480    where
481        F: Fn(&str, &ToolInfo) -> bool + Send,
482    {
483        let tool_names = self.registry.list_tools();
484        tool_names
485            .into_iter()
486            .filter_map(|name| {
487                self.get_tool_info(&name)
488                    .filter(|info| predicate(&name, info))
489                    .map(|info| (name, info))
490            })
491            .collect()
492    }
