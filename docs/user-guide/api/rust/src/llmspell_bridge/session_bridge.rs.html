13/// Helper macro to convert `SessionError` to `LLMSpellError`
14macro_rules! convert_err {
15    ($expr:expr) => {
16        $expr.map_err(|e| LLMSpellError::Component {
17            message: format!("Session error: {}", e),
18            source: None,
19        })
20    };
21}

147    /// Replay a session
148    ///
149    /// # Errors
150    ///
151    /// Returns an error if the session cannot be replayed
152    pub async fn replay_session(
153        &self,
154        session_id: &SessionId,
155        _options: serde_json::Value,
156    ) -> Result<serde_json::Value> {
157        // For now, use default config - full implementation in Task 6.5.6
158        let config = llmspell_sessions::replay::session_adapter::SessionReplayConfig::default();
159        let result = convert_err!(
160            self.session_manager
161                .replay_session(session_id, config)
162                .await
163        )?;
164
165        Ok(serde_json::json!({
166            "session_id": result.session_id.to_string(),
167            "correlation_id": result.correlation_id.to_string(),
168            "start_time": chrono::DateTime::<chrono::Utc>::from(result.start_time).to_rfc3339(),
169            "total_duration": result.total_duration.as_secs_f64(),
170            "hooks_replayed": result.hooks_replayed,
171            "successful_replays": result.successful_replays,
172            "failed_replays": result.failed_replays,
173            "metadata": result.metadata,
174        }))
175    }

186        let timeline = convert_err!(self.session_manager.get_session_timeline(session_id).await)?;
187
188        // Convert timeline events to JSON
189        Ok(timeline
190            .into_iter()
191            .map(|event| {
192                serde_json::json!({
193                    "hook_id": event.hook_id,
194                    "execution_id": event.execution_id,
195                    "correlation_id": event.correlation_id,
196                    "timestamp": chrono::DateTime::<chrono::Utc>::from(event.timestamp).to_rfc3339(),
197                    "result": event.result,
198                })
199            })
200            .collect())
201    }

293// Thread-local storage for current session context
294thread_local! {
295    static CURRENT_SESSION: std::cell::RefCell<Option<SessionId>> = const { std::cell::RefCell::new(None) };
296}
297
298/// Session context management
299impl SessionBridge {
300    /// Get the current session ID
301    #[must_use]
302    pub fn get_current_session() -> Option<SessionId> {
303        CURRENT_SESSION.with(|current| *current.borrow())
304    }
305
306    /// Set the current session ID
307    pub fn set_current_session(session_id: Option<SessionId>) {
308        CURRENT_SESSION.with(|current| {
309            *current.borrow_mut() = session_id;
310        });
311    }
312
313    /// Clear the current session context
314    pub fn clear_current_session() {
315        Self::set_current_session(None);
316    }
317
318    /// Execute a closure with a specific session context
319    pub fn with_session_context<F, R>(session_id: SessionId, f: F) -> R
320    where
321        F: FnOnce() -> R,
322    {
323        let previous = Self::get_current_session();
324        Self::set_current_session(Some(session_id));
325        let result = f();
326        Self::set_current_session(previous);
327        result
328    }
