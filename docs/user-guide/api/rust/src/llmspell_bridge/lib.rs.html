2//! ABOUTME: Supports multiple script engines (Lua, JavaScript, Python) through `ScriptEngineBridge`
3//!
4//! # `LLMSpell` Bridge
5//!
6//! The bridge crate provides a language-agnostic runtime for executing scripts that
7//! interact with LLM agents, tools, and workflows. It implements the Bridge pattern
8//! to support multiple scripting languages through a common interface.
9//!
10//! ## Key Features
11//!
12//! - **Multi-Language Support**: Execute scripts in Lua (Phase 1), JavaScript (Phase 5),
13//!   and Python (Phase 9)
14//! - **Streaming Execution**: Support for streaming responses from LLM providers
15//! - **Provider Integration**: Access multiple LLM providers through a unified API
16//! - **Security Controls**: Fine-grained security settings for script execution
17//! - **Component Registry**: Dynamic registration of agents, tools, and workflows
18//!
19//! ## Architecture
20//!
21//! The bridge uses a three-layer architecture:
22//!
23//! 1. **`ScriptEngineBridge` Trait**: Defines the common interface for all script engines
24//! 2. **Language Implementations**: Concrete implementations for each scripting language
25//! 3. **`ScriptRuntime`**: High-level runtime that manages engines and provides the user API
26//!
27//! ## Quick Start
28//!
29//! ```rust,no_run
30//! use llmspell_bridge::ScriptRuntime;
31//! use llmspell_config::LLMSpellConfig;
32//!
33//! # async fn example() -> Result<(), Box<dyn std::error::Error>> {
34//! // Create a runtime with Lua engine
35//! let runtime = ScriptRuntime::new_with_lua(LLMSpellConfig::default()).await?;
36//!
37//! // Execute a simple script
38//! let output = runtime.execute_script(r#"
39//!     print("Hello from Lua!")
40//!     return { message = "Script completed" }
41//! "#).await?;
42//!
43//! println!("Output: {:?}", output.output);
44//! # Ok(())
45//! # }
46//! ```
47//!
48//! ## Lua Integration
49//!
50//! ### Creating Agents
51//!
52//! ```lua
53//! -- Create an agent with OpenAI provider
54//! local agent = Agent.create({
55//!     name = "assistant",
56//!     provider = "openai",
57//!     model = "gpt-4",
58//!     temperature = 0.7
59//! })
60//!
61//! -- Execute the agent
62//! local response = agent:execute("What is the capital of France?")
63//! print(response.text)
64//!
65//! -- Using streaming
66//! local stream = agent:execute_stream("Tell me a story")
67//! for chunk in stream do
68//!     io.write(chunk.content)
69//! end
70//! ```
71//!
72//! ### Using Tools
73//!
74//! ```lua
75//! -- List available tools
76//! local tools = Tool.list()
77//! for _, tool in ipairs(tools) do
78//!     print(tool.name, tool.description)
79//! end
80//!
81//! -- Execute a tool
82//! local result = Tool.execute("file_reader", {
83//!     path = "/tmp/data.txt"
84//! })
85//!
86//! -- Create custom tool wrapper
87//! local calculator = Tool.wrap("calculator")
88//! local sum = calculator({operation = "add", a = 5, b = 3})
89//! ```
90//!
91//! ### Building Workflows
92//!
93//! ```lua
94//! -- Create a sequential workflow
95//! local workflow = Workflow.sequential({
96//!     name = "data_pipeline",
97//!     steps = {
98//!         {tool = "file_reader", params = {path = "input.txt"}},
99//!         {tool = "text_processor", params = {operation = "uppercase"}},
100//!         {tool = "file_writer", params = {path = "output.txt"}}
101//!     }
102//! })
103//!
104//! -- Execute workflow
105//! local result = workflow:execute()
106//!
107//! -- Parallel workflow
108//! local parallel = Workflow.parallel({
109//!     name = "multi_search",
110//!     steps = {
111//!         {tool = "web_search", params = {query = "rust programming"}},
112//!         {tool = "arxiv_search", params = {query = "machine learning"}},
113//!         {tool = "news_search", params = {query = "technology"}}
114//!     }
115//! })
116//! ```
117//!
118//! ### Session Management
119//!
120//! ```lua
121//! -- Create or load session
122//! local session = Session.load("user-123") or Session.create({
123//!     id = "user-123",
124//!     metadata = {user = "alice", created = os.time()}
125//! })
126//!
127//! -- Store artifacts
128//! session:store_artifact("conversation", conversation_history)
129//! session:store_artifact("settings", user_preferences)
130//!
131//! -- Save session
132//! session:save()
133//!
134//! -- List all sessions
135//! local sessions = Session.list()
136//! ```
137//!
138//! ## JavaScript Support (Planned - Phase 5)
139//!
140//! ```javascript
141//! // Create agent
142//! const agent = await Agent.create({

144//!     provider: "anthropic",
145//!     model: "claude-3-opus"
146//! });
147//!
148//! // Execute with async/await
149//! const response = await agent.execute("Explain quantum computing");
150//! console.log(response.text);
151//!
152//! // Tool execution
153//! const result = await Tool.execute("web_search", {
154//!     query: "latest AI news"
155//! });
156//! ```
157//!
158//! ## Python Support (Planned - Phase 9)
159//!
160//! ```python
161//! # Create agent
162//! agent = Agent.create(
163//!     name="assistant",
164//!     provider="openai",
165//!     model="gpt-4"
166//! )
167//!
168//! # Execute
169//! response = agent.execute("What is machine learning?")
170//! print(response.text)
171//!
172//! # Use with async
173//! async def process():
174//!     async for chunk in agent.execute_stream("Tell me about Python"):
175//!         print(chunk.content, end="")
176//! ```
177//!
178//! ## Cross-Language Compatibility
179//!
180//! All scripting languages share the same underlying Rust implementations, ensuring:
181//!
182//! - **Consistent Behavior**: Same results across languages
183//! - **Shared State**: Sessions and artifacts accessible from any language
184//! - **Unified Security**: Same security policies apply to all scripts
185//! - **Performance**: Native Rust performance for all operations
186//!
187//! ## Configuration
188//!
189//! The runtime can be configured through `LLMSpellConfig`:
190//!
191//! ```rust,no_run
192//! use llmspell_bridge::ScriptRuntime;
193//! use llmspell_config::LLMSpellConfig;
194//!
195//! # async fn example() -> Result<(), Box<dyn std::error::Error>> {
196//! let mut config = LLMSpellConfig::default();
197//!
198//! // Configure security settings
199//! config.runtime.security.allow_file_access = false;
200//! config.runtime.security.allow_network_access = true;
201//! config.runtime.security.max_memory_bytes = Some(50_000_000); // 50MB
202//!
203//! // Set default engine
204//! config.default_engine = "lua".to_string();
205//!
206//! let runtime = ScriptRuntime::new_with_engine_name("lua", config).await?;
207//! # Ok(())
208//! # }
209//! ```
210//!
211//! ## Provider Access
212//!
213//! Scripts can access LLM providers configured in the runtime:
214//!
215//! ```rust,no_run
216//! # use llmspell_bridge::ScriptRuntime;
217//! # use llmspell_config::LLMSpellConfig;
218//! # async fn example() -> Result<(), Box<dyn std::error::Error>> {
219//! let runtime = ScriptRuntime::new_with_lua(LLMSpellConfig::default()).await?;
220//!
221//! let script = r#"
222//!     -- List available providers
223//!     local providers = Provider.list()
224//!     for _, p in ipairs(providers) do
225//!         print("Provider: " .. p.name)
226//!     end
227//!     
228//!     return providers
229//! "#;
230//!
231//! let output = runtime.execute_script(script).await?;
232//! # Ok(())
233//! # }
234//! ```
