34/// Initialize the standard global registry with all core globals
35///
36/// # Errors
37///
38/// Returns an error if:
39/// - Global registration fails
40/// - Registry building fails
41pub async fn create_standard_registry(context: Arc<GlobalContext>) -> Result<GlobalRegistry> {
42    let mut builder = GlobalRegistryBuilder::new();
43
44    // Register core globals in dependency order
45    builder.register(Arc::new(json_global::JsonGlobal::new()));
46    builder.register(Arc::new(core::LoggerGlobal::new()));
47    builder.register(Arc::new(core::ConfigGlobal::new(serde_json::json!({}))));
48
49    // Register Debug global for unified debugging infrastructure
50    builder.register(Arc::new(debug_global::DebugGlobal::new()));
51
52    // Create StateGlobal with migration support if configured
53    let state_global = if let Some(runtime_config) =
54        context.get_bridge::<llmspell_config::LLMSpellConfig>("runtime_config")
55    {
56        if runtime_config.runtime.state_persistence.enabled {
57            // Initialize state infrastructure
58            use crate::globals::state_infrastructure::get_or_create_state_infrastructure;
59            match get_or_create_state_infrastructure(
60                &context,
61                &runtime_config.runtime.state_persistence,
62            )
63            .await
64            {
65                Ok(infrastructure) => Arc::new(state_global::StateGlobal::with_full_support(
66                    infrastructure.state_manager,
67                    infrastructure.migration_engine,
68                    infrastructure.schema_registry,
69                    infrastructure.backup_manager,
70                )),
71                Err(e) => {
72                    tracing::warn!(
73                        "Failed to initialize state infrastructure: {}, falling back to in-memory",
74                        e
75                    );
76                    Arc::new(state_global::StateGlobal::new())
77                }
78            }
79        } else {
80            Arc::new(state_global::StateGlobal::new())
81        }
82    } else {
83        Arc::new(state_global::StateGlobal::new())
84    };
85
86    builder.register(state_global);
87    builder.register(Arc::new(core::UtilsGlobal::new()));
88    // TODO: Add Security global when implemented
89    builder.register(Arc::new(event_global::EventGlobal::new()));
90
91    // Register Session and Artifact globals if SessionManager is available
92    if let Some(session_manager) =
93        context.get_bridge::<llmspell_sessions::manager::SessionManager>("session_manager")
94    {
95        // Create bridges externally for consistency with HookBridge pattern
96        let session_bridge = Arc::new(crate::session_bridge::SessionBridge::new(
97            session_manager.clone(),
98        ));
99        let artifact_bridge =
100            Arc::new(crate::artifact_bridge::ArtifactBridge::new(session_manager));
101
102        builder.register(Arc::new(session_global::SessionGlobal::new(session_bridge)));
103        builder.register(Arc::new(artifact_global::ArtifactGlobal::new(
104            artifact_bridge,
105        )));
106    }
107
108    // Create HookBridge for hook system integration
109    let hook_bridge = Arc::new(crate::hook_bridge::HookBridge::new(context.clone())?);
110    builder.register(Arc::new(hook_global::HookGlobal::new(hook_bridge.clone())));
111
112    // Register replay global for hook debugging
113    builder.register(Arc::new(replay_global::ReplayGlobal::new()));
114
115    builder.register(Arc::new(tool_global::ToolGlobal::new(
116        context.registry.clone(),
117    )));
118
119    // Register Provider global for LLM provider information
120    builder.register(Arc::new(provider_global::ProviderGlobal::new(
121        context.providers.clone(),
122    )));
123
124    // Create agent global with state manager if available
125    let agent_global = if let Some(state_manager) =
126        context.get_bridge::<llmspell_state_persistence::StateManager>("state_manager")
127    {
128        agent_global::AgentGlobal::with_state_manager(
129            context.registry.clone(),
130            context.providers.clone(),
131            state_manager,
132        )
133        .await?
134    } else {
135        agent_global::AgentGlobal::new(context.registry.clone(), context.providers.clone()).await?
136    };
137    builder.register(Arc::new(agent_global));
138
139    // Create workflow global with state manager if available
140    let workflow_global = context
141        .get_bridge::<llmspell_state_persistence::StateManager>("state_manager")
142        .map_or_else(
143            || workflow_global::WorkflowGlobal::new(context.registry.clone()),
144            |state_manager| {
145                workflow_global::WorkflowGlobal::with_state_manager(
146                    context.registry.clone(),
147                    state_manager,
148                )
149            },
150        );
151    builder.register(Arc::new(workflow_global));
152
153    builder.register(Arc::new(streaming_global::StreamingGlobal::new()));
154
155    builder.build()
156}
