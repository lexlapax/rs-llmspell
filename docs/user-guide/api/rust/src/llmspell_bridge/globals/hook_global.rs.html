1//! ABOUTME: Hook global object providing cross-language hook system
2//! ABOUTME: Full implementation integrating with llmspell-hooks infrastructure

4#![allow(clippy::significant_drop_tightening)]
5
6use crate::globals::types::{GlobalContext, GlobalMetadata, GlobalObject};
7use crate::hook_bridge::HookBridge;
8use llmspell_core::error::LLMSpellError;
9use std::sync::Arc;
10
11/// Hook global object providing cross-language hook system
12pub struct HookGlobal {
13    hook_bridge: Arc<HookBridge>,
14}
15
16impl HookGlobal {
17    /// Create a new Hook global with the bridge
18    #[must_use]
19    pub const fn new(hook_bridge: Arc<HookBridge>) -> Self {
20        Self { hook_bridge }
21    }
22}
23
24impl GlobalObject for HookGlobal {
25    fn metadata(&self) -> GlobalMetadata {
26        GlobalMetadata {
27            name: "Hook".to_string(),
28            version: "1.0.0".to_string(),
29            description: "Cross-language hook system for lifecycle events and customization"
30                .to_string(),
31            dependencies: vec![],
32            required: false,
33        }
34    }
35
36    #[cfg(feature = "lua")]
37    fn inject_lua(&self, lua: &mlua::Lua, context: &GlobalContext) -> Result<(), LLMSpellError> {
38        // Store the hook bridge in the global context
39        context.set_bridge("hook_bridge", self.hook_bridge.clone());
40
41        // Inject the Lua-specific hook global
42        crate::lua::globals::hook::inject_hook_global(lua, context, self.hook_bridge.clone())
43    }
44
45    #[cfg(feature = "javascript")]
46    fn inject_javascript(
47        &self,
48        ctx: &mut boa_engine::Context,
49        context: &GlobalContext,
50    ) -> Result<(), LLMSpellError> {
51        // Store the hook bridge in the global context
52        context.set_bridge("hook_bridge", self.hook_bridge.clone());
53
54        // Inject the JavaScript-specific hook global
55        crate::javascript::globals::hook::inject_hook_global(ctx, context)
56    }
57}

59#[cfg(test)]
60mod tests {
61    use super::*;
62    use crate::{ComponentRegistry, ProviderManager};
63    use llmspell_config::providers::ProviderManagerConfig;
64    #[tokio::test]
65    async fn test_hook_global_metadata() {
66        let context = Arc::new(GlobalContext::new(
67            Arc::new(ComponentRegistry::new()),
68            Arc::new(
69                ProviderManager::new(ProviderManagerConfig::default())
70                    .await
71                    .unwrap(),
72            ),
73        ));
74        let hook_bridge = Arc::new(HookBridge::new(context).unwrap());
75        let global = HookGlobal::new(hook_bridge);
76        let metadata = global.metadata();
77        assert_eq!(metadata.name, "Hook");
78        assert_eq!(metadata.version, "1.0.0");
79    }
