4//! # Standardized Tool API
5//!
6//! This module defines the standard API that all script engines (Lua, JavaScript, Python)
7//! must implement for tool access. This ensures consistency across different scripting
8//! environments.
9//!
10//! ## Global Object: `Tool`
11//!
12//! The Tool global object provides access to all registered tools in the system.
13//!
14//! ### Core Methods
15//!
16//! #### `Tool.list() -> Array<ToolInfo>`
17//! Returns a list of all available tools with their metadata.
18//! ```js
19//! const tools = Tool.list();
20//! // Returns: [{ name: "calculator", description: "...", version: "..." }, ...]
21//! ```
22//!
23//! #### `Tool.get(name: string) -> ToolInstance | null`
24//! Retrieves a specific tool by name, returning a tool instance or null if not found.
25//! ```js
26//! const calc = Tool.get("calculator");
27//! if (calc) {
28//!     const result = calc.execute({ expression: "2 + 2" });
29//! }
30//! ```
31//!
32//! #### `Tool.invoke(name: string, params: object) -> object`
33//! Directly invokes a tool by name with the given parameters.
34//! ```js
35//! const result = Tool.invoke("calculator", { expression: "2 + 2" });
36//! ```
37//!
38//! #### `Tool.exists(name: string) -> boolean`
39//! Checks if a tool with the given name exists.
40//! ```js
41//! if (Tool.exists("web_search")) {
42//!     // Use web search tool
43//! }
44//! ```
45//!
46//! #### `Tool.categories() -> Array<string>`
47//! Returns all available tool categories.
48//! ```js
49//! const categories = Tool.categories();
50//! // Returns: ["utility", "data_processing", "file_system", "web", ...]
51//! ```
52//!
53//! #### `Tool.discover(filter?: { category?: string, tag?: string }) -> Array<ToolInfo>`
54//! Discovers tools matching the optional filter criteria.
55//! ```js
56//! const webTools = Tool.discover({ category: "web" });
57//! const searchTools = Tool.discover({ tag: "search" });
58//! ```
59//!
60//! ### Direct Tool Access
61//!
62//! Tools can be accessed directly via the Tool object using dot notation:
63//! ```js
64//! const result = Tool.calculator.execute({ expression: "2 + 2" });
65//! ```
66//!
67//! ### Tool Instance Methods
68//!
69//! Each tool instance returned by `Tool.get()` or accessed directly has:
70//!
71//! #### `execute(params: object) -> object`
72//! Executes the tool with the given parameters.
73//!
74//! #### `getSchema() -> ToolSchema`
75//! Returns the tool's parameter schema.
76//!
77//! ## Data Types
78//!
79//! ### `ToolInfo`
80//! ```typescript
81//! interface ToolInfo {
82//!     name: string;
83//!     description: string;
84//!     category: string;
85//!     version: string;
86//! }
87//! ```
88//!
89//! ### `ToolSchema`
90//! ```typescript
91//! interface ToolSchema {
92//!     name: string;
93//!     description: string;
94//!     parameters: ParameterDef[];
95//! }
96//!
97//! interface ParameterDef {
98//!     name: string;
99//!     type: string;
100//!     description: string;
101//!     required: boolean;
102//!     default?: any;
103//! }
104//! ```
105//!
106//! ## Security Considerations
107//!
108//! - All tool executions are subject to security constraints
109//! - File system tools respect sandbox boundaries
110//! - Network tools respect allowed domains
111//! - Resource limits are enforced per tool execution
112//!
113//! ## Error Handling
114//!
115//! Tool methods throw errors on failure:
116//! - Tool not found
117//! - Invalid parameters
118//! - Security violations
119//! - Resource limit exceeded
120//! - Tool execution failure
121//!
122//! ## Implementation Notes
123//!
124//! - All methods are synchronous from the script's perspective
125//! - Async operations are handled internally by the bridge
126//! - Tool discovery is cached for performance
127//! - Direct tool access (`Tool.tool_name`) uses lazy loading

129use llmspell_core::traits::tool::ToolSchema;
130use serde::{Deserialize, Serialize};

132/// Standard tool information exposed to scripts
133#[derive(Debug, Clone, Serialize, Deserialize)]
134pub struct StandardToolInfo {
135    /// Tool name
136    pub name: String,
137    /// Tool description
138    pub description: String,
139    /// Tool category
140    pub category: String,
141    /// Tool version
142    pub version: String,
143}
144
145/// Standard tool schema exposed to scripts
146#[derive(Debug, Clone, Serialize, Deserialize)]
147pub struct StandardToolSchema {
148    /// Schema name
149    pub name: String,
150    /// Schema description
151    pub description: String,
152    /// Parameters
153    pub parameters: Vec<StandardParameterDef>,
154}
155
156/// Standard parameter definition exposed to scripts
157#[derive(Debug, Clone, Serialize, Deserialize)]
158pub struct StandardParameterDef {
159    /// Parameter name
160    pub name: String,
161    /// Parameter type as string
162    #[serde(rename = "type")]
163    pub param_type: String,
164    /// Parameter description
165    pub description: String,
166    /// Whether required
167    pub required: bool,
168    /// Default value if any
169    pub default: Option<serde_json::Value>,
170}
171
172/// Convert from core `ToolSchema` to standard schema
173#[must_use]
174pub fn to_standard_schema(schema: ToolSchema) -> StandardToolSchema {
175    StandardToolSchema {
176        name: schema.name,
177        description: schema.description,
178        parameters: schema
179            .parameters
180            .into_iter()
181            .map(|p| StandardParameterDef {
182                name: p.name,
183                param_type: format!("{:?}", p.param_type).to_lowercase(),
184                description: p.description,
185                required: p.required,
186                default: p.default,
187            })
188            .collect(),
189    }
190}

192/// Standard tool API trait that all script engines must implement
193#[async_trait::async_trait]
194pub trait StandardToolApi {
195    /// List all available tools
196    async fn list_tools(&self) -> Vec<StandardToolInfo>;
197
198    /// Get a specific tool's information
199    async fn get_tool(&self, name: &str) -> Option<StandardToolInfo>;
200
201    /// Invoke a tool with parameters
202    async fn invoke_tool(
203        &self,
204        name: &str,
205        params: serde_json::Value,
206    ) -> Result<serde_json::Value, String>;
207
208    /// Check if a tool exists
209    async fn tool_exists(&self, name: &str) -> bool;
210
211    /// Get all tool categories
212    async fn get_categories(&self) -> Vec<String>;
213
214    /// Discover tools with optional filters
215    async fn discover_tools(
216        &self,
217        category: Option<String>,
218        tag: Option<String>,
219    ) -> Vec<StandardToolInfo>;
220
221    /// Get a tool's schema
222    async fn get_tool_schema(&self, name: &str) -> Option<StandardToolSchema>;
223}
