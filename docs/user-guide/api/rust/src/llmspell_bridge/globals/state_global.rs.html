15/// State global object providing persistent state management
16///
17/// Integrates with `StateAccess` trait for flexible state backend support.
18/// Falls back to in-memory storage when `StateAccess` is not available.
19pub struct StateGlobal {
20    /// `StateAccess` implementation for persistent storage (optional for backward compatibility)
21    pub state_access: Option<Arc<dyn StateAccess>>,
22    /// Original `StateManager` for migration/backup features (optional)
23    pub state_manager: Option<Arc<StateManager>>,
24    /// Fallback in-memory state storage (when `StateAccess` is not available)
25    pub fallback_state: Arc<RwLock<HashMap<String, serde_json::Value>>>,
26    /// Migration engine for schema transitions (optional)
27    pub migration_engine: Option<Arc<MigrationEngine>>,
28    /// Schema registry for migration planning (optional)
29    pub schema_registry: Option<Arc<SchemaRegistry>>,
30    /// Backup manager for state backup/restore operations (optional)
31    pub backup_manager: Option<Arc<llmspell_state_persistence::backup::BackupManager>>,
32}
33
34impl StateGlobal {

36    #[must_use]

38        Self {

48    /// Create a new State global with `StateAccess` integration

54            migration_engine: None,
55            schema_registry: None,
56            backup_manager: None,
57        }
58    }
59
60    /// Create a new State global with `StateManager` integration (for migration/backup support)
61    pub fn with_state_manager(state_manager: Arc<StateManager>) -> Self {
62        // Create StateManagerAdapter to provide StateAccess interface
63        let state_access: Arc<dyn StateAccess> =
64            Arc::new(crate::state_adapter::StateManagerAdapter::new(
65                state_manager.clone(),
66                StateScope::Global,
67            ));
68
69        Self {
70            state_access: Some(state_access),
71            state_manager: Some(state_manager),
72            fallback_state: Arc::new(RwLock::new(HashMap::new())),
73            migration_engine: None,
74            schema_registry: None,
75            backup_manager: None,
76        }
77    }

79    /// Create a new State global with full migration support
80    pub fn with_migration_support(
81        state_manager: Arc<StateManager>,
82        migration_engine: Arc<MigrationEngine>,
83        schema_registry: Arc<SchemaRegistry>,
84    ) -> Self {
85        // Create StateManagerAdapter to provide StateAccess interface
86        let state_access: Arc<dyn StateAccess> =
87            Arc::new(crate::state_adapter::StateManagerAdapter::new(
88                state_manager.clone(),
89                StateScope::Global,
90            ));

93            state_access: Some(state_access),
94            state_manager: Some(state_manager),
95            fallback_state: Arc::new(RwLock::new(HashMap::new())),
96            migration_engine: Some(migration_engine),
97            schema_registry: Some(schema_registry),
98            backup_manager: None,
99        }
100    }
101
102    /// Create a new State global with full infrastructure support
103    pub fn with_full_support(
104        state_manager: Arc<StateManager>,
105        migration_engine: Option<Arc<MigrationEngine>>,
106        schema_registry: Option<Arc<SchemaRegistry>>,
107        backup_manager: Option<Arc<llmspell_state_persistence::backup::BackupManager>>,
108    ) -> Self {
109        // Use StateManagerAdapter with Custom scope for StateGlobal
110        // This allows reading keys that were written by NoScopeStateAdapter
111        tracing::info!(
112            "StateGlobal: Creating StateManagerAdapter with StateManager at {:p}",
113            Arc::as_ptr(&state_manager)
114        );
115        let state_access: Arc<dyn StateAccess> =
116            Arc::new(crate::state_adapter::StateManagerAdapter::new(
117                state_manager.clone(),
118                StateScope::Custom(String::new()),
119            ));
120
121        Self {
122            state_access: Some(state_access),
123            state_manager: Some(state_manager),
124            fallback_state: Arc::new(RwLock::new(HashMap::new())),
125            migration_engine,
126            schema_registry,
127            backup_manager,
128        }
129    }
130
131    /// Helper method to parse scope string to `StateScope` enum
132    #[must_use]
133    pub fn parse_scope(scope_str: &str) -> StateScope {
134        if scope_str.starts_with("agent:") {
135            StateScope::Agent(
136                scope_str
137                    .strip_prefix("agent:")
138                    .unwrap_or("unknown")
139                    .to_string(),
140            )
141        } else if scope_str.starts_with("workflow:") {
142            StateScope::Workflow(
143                scope_str
144                    .strip_prefix("workflow:")
145                    .unwrap_or("unknown")
146                    .to_string(),
147            )
148        } else if scope_str.starts_with("session:") {
149            StateScope::Session(
150                scope_str
151                    .strip_prefix("session:")
152                    .unwrap_or("unknown")
153                    .to_string(),
154            )
155        } else if scope_str.starts_with("custom:") {
156            StateScope::Custom(
157                scope_str
158                    .strip_prefix("custom:")
159                    .unwrap_or("unknown")
160                    .to_string(),
161            )
162        } else if scope_str == "global" {
163            StateScope::Global
164        } else {
165            StateScope::Custom(scope_str.to_string())
166        }
167    }
168}

170impl GlobalObject for StateGlobal {
171    fn metadata(&self) -> GlobalMetadata {
172        let description = if self.state_access.is_some() {
173            "State management system with persistent storage".to_string()
174        } else {
175            "State management system (in-memory fallback)".to_string()
176        };
177
178        GlobalMetadata {
179            name: "State".to_string(),
180            version: "2.0.0".to_string(), // Phase 7 version with StateAccess trait
181            description,
182            dependencies: vec![], // StateAccess is optional - gracefully falls back to in-memory
183            required: false,
184        }
185    }
186

225    fn test_state_in_memory_storage() {
226        let global = StateGlobal::new();
227
228        // Test basic storage operations (fallback mode)
229        {
230            let mut state = global.fallback_state.write();
231            state.insert("test_key".to_string(), serde_json::json!("test_value"));
232        }
233
234        {
235            let state = global.fallback_state.read();
236            assert_eq!(
237                state.get("test_key"),
238                Some(&serde_json::json!("test_value"))
239            );
240        }
