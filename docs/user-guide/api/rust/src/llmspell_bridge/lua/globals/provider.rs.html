1//! ABOUTME: Lua-specific Provider global implementation
2//! ABOUTME: Provides Lua bindings for LLM provider information and capabilities
3
4use crate::globals::GlobalContext;
5use crate::ProviderManager;
6use llmspell_core::error::LLMSpellError;
7use mlua::{Lua, Result as LuaResult, Table, Value};
8use std::sync::Arc;
9
10/// Inject Provider global into Lua environment
11///
12/// # Errors
13///
14/// Returns an error if:
15/// - Lua table creation fails
16/// - Function injection fails
17/// - Global setting fails
18#[allow(clippy::too_many_lines)]
19pub fn inject_provider_global(
20    lua: &Lua,
21    _context: &GlobalContext,
22    providers: &Arc<ProviderManager>,
23) -> Result<(), LLMSpellError> {
24    // Create the Provider table
25    let provider_table = lua.create_table().map_err(|e| LLMSpellError::Component {
26        message: format!("Failed to create Provider table: {e}"),
27        source: None,
28    })?;
29
30    // Clone providers for closure capture
31    let providers_for_list = providers.clone();
32
33    // Provider.list() - List all available providers
34    let list_fn = lua
35        .create_function(move |lua, ()| -> LuaResult<Table> {
36            let result = lua.create_table()?;
37
38            // Get provider list from manager
39            let provider_list = futures::executor::block_on(providers_for_list.list_providers());
40
41            for (idx, provider_info) in provider_list.iter().enumerate() {
42                let provider_table = lua.create_table()?;
43                provider_table.set("name", provider_info.name.clone())?;
44                provider_table.set("enabled", provider_info.enabled)?;
45
46                // Add capabilities if available
47                if let Some(caps) = &provider_info.capabilities {
48                    let caps_table = lua.create_table()?;
49                    caps_table.set("supports_streaming", caps.supports_streaming)?;
50                    caps_table.set("supports_multimodal", caps.supports_multimodal)?;
51                    caps_table.set("max_context_tokens", caps.max_context_tokens)?;
52
53                    // Add available models
54                    if !caps.available_models.is_empty() {
55                        let models_table = lua.create_table()?;
56                        for (model_idx, model) in caps.available_models.iter().enumerate() {
57                            models_table.set(model_idx + 1, model.clone())?;
58                        }
59                        caps_table.set("available_models", models_table)?;
60                    }
61
62                    provider_table.set("capabilities", caps_table)?;
63                }
64
65                result.set(idx + 1, provider_table)?;
66            }
67
68            Ok(result)
69        })
70        .map_err(|e| LLMSpellError::Component {
71            message: format!("Failed to create Provider.list function: {e}"),
72            source: None,
73        })?;
74
75    provider_table
76        .set("list", list_fn)
77        .map_err(|e| LLMSpellError::Component {
78            message: format!("Failed to set Provider.list: {e}"),
79            source: None,
80        })?;
81
82    // Provider.get(name) - Get specific provider information
83    let providers_get = providers.clone();
84    let get_fn = lua
85        .create_function(move |lua, name: String| -> LuaResult<Value> {
86            let provider_info = futures::executor::block_on(providers_get.get_provider_info(&name));
87
88            match provider_info {
89                Some(info) => {
90                    let provider_table = lua.create_table()?;
91                    provider_table.set("name", info.name)?;
92                    provider_table.set("enabled", info.enabled)?;
93
94                    if let Some(caps) = info.capabilities {
95                        let caps_table = lua.create_table()?;
96                        caps_table.set("supports_streaming", caps.supports_streaming)?;
97                        caps_table.set("supports_multimodal", caps.supports_multimodal)?;
98                        caps_table.set("max_context_tokens", caps.max_context_tokens)?;
99                        provider_table.set("capabilities", caps_table)?;
100                    }
101
102                    Ok(Value::Table(provider_table))
103                }
104                None => Ok(Value::Nil),
105            }
106        })
107        .map_err(|e| LLMSpellError::Component {
108            message: format!("Failed to create Provider.get function: {e}"),
109            source: None,
110        })?;
111
112    provider_table
113        .set("get", get_fn)
114        .map_err(|e| LLMSpellError::Component {
115            message: format!("Failed to set Provider.get: {e}"),
116            source: None,
117        })?;
118
119    // Provider.getCapabilities(name) - Get provider capabilities
120    let providers_caps = providers.clone();
121    let get_caps_fn = lua
122        .create_function(move |lua, name: String| -> LuaResult<Value> {
123            let provider_info =
124                futures::executor::block_on(providers_caps.get_provider_info(&name));
125
126            match provider_info.and_then(|info| info.capabilities) {
127                Some(caps) => {
128                    let caps_table = lua.create_table()?;
129                    caps_table.set("supports_streaming", caps.supports_streaming)?;
130                    caps_table.set("supports_multimodal", caps.supports_multimodal)?;
131                    caps_table.set("max_context_tokens", caps.max_context_tokens)?;
132
133                    if !caps.available_models.is_empty() {
134                        let models_table = lua.create_table()?;
135                        for (idx, model) in caps.available_models.iter().enumerate() {
136                            models_table.set(idx + 1, model.clone())?;
137                        }
138                        caps_table.set("available_models", models_table)?;
139                    }
140
141                    Ok(Value::Table(caps_table))
142                }
143                None => Ok(Value::Nil),
144            }
145        })
146        .map_err(|e| LLMSpellError::Component {
147            message: format!("Failed to create Provider.getCapabilities function: {e}"),
148            source: None,
149        })?;
150
151    provider_table
152        .set("getCapabilities", get_caps_fn)
153        .map_err(|e| LLMSpellError::Component {
154            message: format!("Failed to set Provider.getCapabilities: {e}"),
155            source: None,
156        })?;
157
158    // Provider.isAvailable(name) - Check if provider is configured and available
159    let providers_avail = providers.clone();
160    let is_available_fn = lua
161        .create_function(move |_lua, name: String| -> LuaResult<bool> {
162            let provider_info =
163                futures::executor::block_on(providers_avail.get_provider_info(&name));
164            Ok(provider_info.is_some_and(|info| info.enabled))
165        })
166        .map_err(|e| LLMSpellError::Component {
167            message: format!("Failed to create Provider.isAvailable function: {e}"),
168            source: None,
169        })?;
170
171    provider_table
172        .set("isAvailable", is_available_fn)
173        .map_err(|e| LLMSpellError::Component {
174            message: format!("Failed to set Provider.isAvailable: {e}"),
175            source: None,
176        })?;
177
178    // Set the Provider table as a global
179    lua.globals()
180        .set("Provider", provider_table)
181        .map_err(|e| LLMSpellError::Component {
182            message: format!("Failed to set Provider global: {e}"),
183            source: None,
184        })?;
185
186    Ok(())
187}
