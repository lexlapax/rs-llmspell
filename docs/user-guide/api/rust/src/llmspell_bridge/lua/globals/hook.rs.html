15/// Lua representation of a hook handle for cleanup
16struct LuaHookHandle {
17    handle: Arc<RwLock<Option<HookHandle>>>,
18    hook_bridge: Arc<HookBridge>,
19}
20
21impl UserData for LuaHookHandle {
22    fn add_methods<'lua, M: UserDataMethods<'lua, Self>>(methods: &mut M) {
23        // Get hook ID
24        methods.add_method("id", |_, this, ()| {
25            let handle = block_on_async::<_, _, std::io::Error>(
26                "hook_handle_id",
27                async { Ok(this.handle.read().await.clone()) },
28                None,
29            )?;
30
31            match handle {
32                Some(h) => Ok(h.id),
33                None => Err(mlua::Error::RuntimeError(
34                    "Hook already unregistered".to_string(),
35                )),
36            }
37        });
38
39        // Get hook point
40        methods.add_method("hook_point", |_, this, ()| {
41            let handle = block_on_async::<_, _, std::io::Error>(
42                "hook_handle_hook_point",
43                async { Ok(this.handle.read().await.clone()) },
44                None,
45            )?;
46
47            match handle {
48                Some(h) => Ok(format!("{:?}", h.hook_point)),
49                None => Err(mlua::Error::RuntimeError(
50                    "Hook already unregistered".to_string(),
51                )),
52            }
53        });
54
55        // Unregister the hook
56        methods.add_method("unregister", |_, this, ()| {
57            block_on_async(
58                "hook_unregister",
59                async {
60                    let mut handle_lock = this.handle.write().await;
61                    if let Some(handle) = handle_lock.take() {
62                        this.hook_bridge.unregister_hook(&handle).await?;
63                        Ok::<bool, LLMSpellError>(true)
64                    } else {
65                        Ok::<bool, LLMSpellError>(false)
66                    }
67                },
68                None,
69            )
70        });
71    }
72}
73
74impl Drop for LuaHookHandle {
75    fn drop(&mut self) {
76        // Auto-unregister when the handle is dropped
77        let handle = self.handle.clone();
78        let bridge = self.hook_bridge.clone();
79        tokio::spawn(async move {
80            let mut handle_lock = handle.write().await;
81            if let Some(h) = handle_lock.take() {
82                let _ = bridge.unregister_hook(&h).await;
83            }
84        });
85    }
86}

88/// Convert `HookContext` to Lua table
89#[allow(dead_code)]
90fn hook_context_to_lua_table<'lua>(
91    lua: &'lua Lua,
92    context: &HookContext,
93) -> mlua::Result<Table<'lua>> {
94    let table = lua.create_table()?;
95
96    // Set basic fields
97    table.set("point", format!("{:?}", context.point))?;
98    table.set("correlation_id", context.correlation_id.to_string())?;
99
100    // Component ID
101    let component_table = lua.create_table()?;
102    component_table.set("type", format!("{:?}", context.component_id.component_type))?;
103    component_table.set("name", context.component_id.name.clone())?;
104    table.set("component_id", component_table)?;
105
106    // Language
107    table.set("language", context.language.to_string())?;
108
109    // Metadata
110    let metadata_table = lua.create_table()?;
111    for (key, value) in &context.metadata {
112        metadata_table.set(key.as_str(), value.as_str())?;
113    }
114    table.set("metadata", metadata_table)?;
115
116    // Data (convert JSON to Lua)
117    let data_table = lua.create_table()?;
118    for (key, value) in &context.data {
119        if let Ok(lua_value) = json_to_lua_value(lua, value) {
120            data_table.set(key.as_str(), lua_value)?;
121        }
122    }
123    table.set("data", data_table)?;
124
125    Ok(table)
126}

128/// Convert Lua value to `HookResult`
129#[allow(dead_code)]
130fn lua_value_to_hook_result(value: Value) -> mlua::Result<HookResult> {
131    match value {
132        Value::String(s) => {
133            let s = s.to_str()?;
134            match s {
135                "continue" => Ok(HookResult::Continue),
136                "skip" | "skipped" => Ok(HookResult::Skipped("Skipped by Lua hook".to_string())),
137                _ => Ok(HookResult::Cancel(s.to_string())),
138            }
139        }
140        Value::Table(table) => {
141            // Check for result type
142            if let Ok(Value::String(result_type)) = table.get::<_, Value>("type") {
143                let result_type = result_type.to_str()?;
144                match result_type {
145                    "modified" => {
146                        // Get the modified data
147                        if let Ok(data) = table.get::<_, Table>("data") {
148                            let json_data = lua_table_to_json(data)?;
149                            Ok(HookResult::Modified(json_data))
150                        } else {
151                            Ok(HookResult::Modified(serde_json::json!({})))
152                        }
153                    }
154                    "cancel" => {
155                        let reason = table
156                            .get::<_, String>("reason")
157                            .unwrap_or_else(|_| "Cancelled by Lua hook".to_string());
158                        Ok(HookResult::Cancel(reason))
159                    }
160                    "redirect" => {
161                        let target = table.get::<_, String>("target").map_err(|_| {
162                            mlua::Error::FromLuaConversionError {
163                                from: "table",
164                                to: "HookResult::Redirect",
165                                message: Some("Missing 'target' field for redirect".to_string()),
166                            }
167                        })?;
168                        Ok(HookResult::Redirect(target))
169                    }
170                    "replace" => {
171                        if let Ok(data) = table.get::<_, Table>("data") {
172                            let json_data = lua_table_to_json(data)?;
173                            Ok(HookResult::Replace(json_data))
174                        } else {
175                            Ok(HookResult::Replace(serde_json::json!({})))
176                        }
177                    }
178                    "retry" => {
179                        let delay_ms = table.get::<_, u64>("delay_ms").unwrap_or(1000);
180                        let max_attempts = table.get::<_, u32>("max_attempts").unwrap_or(3);
181                        Ok(HookResult::Retry {
182                            delay: std::time::Duration::from_millis(delay_ms),
183                            max_attempts,
184                        })
185                    }
186                    "skipped" => {
187                        let reason = table
188                            .get::<_, String>("reason")
189                            .unwrap_or_else(|_| "Skipped by Lua hook".to_string());
190                        Ok(HookResult::Skipped(reason))
191                    }
192                    _ => Ok(HookResult::Continue),
193                }
194            } else {
195                Ok(HookResult::Continue)
196            }
197        }
198        _ => Ok(HookResult::Continue),
199    }
200}

202/// Parse hook point from string
203fn parse_hook_point(s: &str) -> mlua::Result<HookPoint> {
204    match s {
205        "SystemStartup" => Ok(HookPoint::SystemStartup),
206        "SystemShutdown" => Ok(HookPoint::SystemShutdown),
207        "BeforeAgentInit" => Ok(HookPoint::BeforeAgentInit),
208        "AfterAgentInit" => Ok(HookPoint::AfterAgentInit),
209        "BeforeAgentExecution" => Ok(HookPoint::BeforeAgentExecution),
210        "AfterAgentExecution" => Ok(HookPoint::AfterAgentExecution),
211        "BeforeAgentShutdown" => Ok(HookPoint::BeforeAgentShutdown),
212        "AfterAgentShutdown" => Ok(HookPoint::AfterAgentShutdown),
213        "BeforeToolDiscovery" => Ok(HookPoint::BeforeToolDiscovery),
214        "AfterToolDiscovery" => Ok(HookPoint::AfterToolDiscovery),
215        "BeforeToolExecution" => Ok(HookPoint::BeforeToolExecution),
216        "AfterToolExecution" => Ok(HookPoint::AfterToolExecution),
217        "ToolValidation" => Ok(HookPoint::ToolValidation),
218        "ToolError" => Ok(HookPoint::ToolError),
219        "AgentError" => Ok(HookPoint::AgentError),
220        "BeforeWorkflowStart" => Ok(HookPoint::BeforeWorkflowStart),
221        "WorkflowStageTransition" => Ok(HookPoint::WorkflowStageTransition),
222        "BeforeWorkflowStage" => Ok(HookPoint::BeforeWorkflowStage),
223        "AfterWorkflowStage" => Ok(HookPoint::AfterWorkflowStage),
224        "WorkflowCheckpoint" => Ok(HookPoint::WorkflowCheckpoint),
225        "WorkflowRollback" => Ok(HookPoint::WorkflowRollback),
226        "AfterWorkflowComplete" => Ok(HookPoint::AfterWorkflowComplete),
227        "WorkflowError" => Ok(HookPoint::WorkflowError),
228        _ => Err(mlua::Error::RuntimeError(format!(
229            "Unknown hook point: {s}"
230        ))),
231    }
232}

234/// Parse priority from string or use default
235fn parse_priority(s: Option<&String>) -> Priority {
236    match s.map(String::as_str) {
237        Some("highest") => Priority::HIGHEST,
238        Some("high") => Priority::HIGH,
239        Some("low") => Priority::LOW,
240        Some("lowest") => Priority::LOWEST,
241        _ => Priority::NORMAL,
242    }
243}

245/// Inject the Hook global into Lua
246///
247/// # Errors
248///
249/// Returns an error if Lua global injection fails
250#[allow(clippy::too_many_lines)]
251pub fn inject_hook_global(
252    lua: &Lua,
253    _context: &GlobalContext,
254    hook_bridge: Arc<HookBridge>,
255) -> Result<()> {
256    // Create the Hook table
257    let hook_table = lua.create_table().map_err(|e| LLMSpellError::Component {
258        message: format!("Failed to create Hook table: {e}"),
259        source: None,
260    })?;
261
262    // Register a Lua hook adapter with the bridge
263    {
264        let adapter = Arc::new(crate::lua::hook_adapter::LuaHookAdapter::new());
265
266        block_on_async(
267            "register_lua_adapter",
268            async {
269                hook_bridge
270                    .register_adapter(
271                        Language::Lua,
272                        adapter
273                            as Arc<
274                                dyn llmspell_hooks::HookAdapter<
275                                    Context = Box<dyn std::any::Any>,
276                                    Result = Box<dyn std::any::Any>,
277                                >,
278                            >,
279                    )
280                    .await
281            },
282            None,
283        )
284        .map_err(|e| LLMSpellError::Component {
285            message: format!("Failed to register Lua adapter: {e}"),
286            source: None,
287        })?;
288    }
289
290    // Hook.register(hook_point, callback, priority) -> handle
291    let hook_bridge_clone = hook_bridge.clone();
292    let register_fn = lua
293        .create_function(
294            move |lua, (hook_point, callback, priority): (String, Function, Option<String>)| {
295                let hook_point = parse_hook_point(&hook_point)?;
296                let priority = parse_priority(priority.as_ref());
297                let hook_bridge = hook_bridge_clone.clone();
298
299                // Store the callback
300                let callback_ref = lua.create_registry_value(callback)?;
301
302                block_on_async::<_, _, LLMSpellError>(
303                    "hook_register",
304                    async move {
305                        let handle = hook_bridge
306                            .register_hook(
307                                Language::Lua,
308                                hook_point,
309                                priority,
310                                Box::new(callback_ref),
311                            )
312                            .await?;
313
314                        Ok::<LuaHookHandle, LLMSpellError>(LuaHookHandle {
315                            handle: Arc::new(RwLock::new(Some(handle))),
316                            hook_bridge,
317                        })
318                    },
319                    None,
320                )
321            },
322        )
323        .map_err(|e| LLMSpellError::Component {
324            message: format!("Failed to create Hook.register: {e}"),
325            source: None,
326        })?;
327
328    hook_table
329        .set("register", register_fn)
330        .map_err(|e| LLMSpellError::Component {
331            message: format!("Failed to set Hook.register: {e}"),
332            source: None,
333        })?;
334
335    // Hook.list(filter) -> array of hook info
336    // filter can be:
337    //   - nil: list all hooks
338    //   - string: hook point name
339    //   - table: {hook_point?, language?, priority?, tag?}
340    let hook_bridge_clone = hook_bridge;
341    let list_fn = lua
342        .create_function(move |lua, filter: Option<mlua::Value>| {
343            let hook_bridge = hook_bridge_clone.clone();
344
345            // Parse filter
346            let hook_point_filter = match &filter {
347                Some(mlua::Value::String(s)) => Some(parse_hook_point(s.to_str()?)?),
348                Some(mlua::Value::Table(table)) => {
349                    if let Ok(hook_point_str) = table.get::<_, String>("hook_point") {
350                        Some(parse_hook_point(&hook_point_str)?)
351                    } else {
352                        None
353                    }
354                }
355                _ => None,
356            };
357
358            let hooks = block_on_async::<_, _, LLMSpellError>(
359                "hook_list",
360                async move { hook_bridge.list_hooks(hook_point_filter).await },
361                None,
362            )?;
363
364            // Apply additional filters if provided
365            let filtered_hooks: Vec<_> = if let Some(mlua::Value::Table(filter_table)) = &filter {
366                hooks
367                    .into_iter()
368                    .filter(|hook| {
369                        // Language filter
370                        if let Ok(language_filter) = filter_table.get::<_, String>("language") {
371                            if hook.language.to_string().to_lowercase()
372                                != language_filter.to_lowercase()
373                            {
374                                return false;
375                            }
376                        }
377
378                        // Priority filter
379                        if let Ok(priority_filter) = filter_table.get::<_, String>("priority") {
380                            let matches_priority = match priority_filter.to_lowercase().as_str() {
381                                "highest" => hook.priority.0 == i32::MIN,
382                                "high" => hook.priority.0 == -100,
383                                "normal" => hook.priority.0 == 0,
384                                "low" => hook.priority.0 == 100,
385                                "lowest" => hook.priority.0 == i32::MAX,
386                                _ => false,
387                            };
388                            if !matches_priority {
389                                return false;
390                            }
391                        }
392
393                        // Tag filter
394                        if let Ok(tag_filter) = filter_table.get::<_, String>("tag") {
395                            if !hook.tags.iter().any(|tag| tag.contains(&tag_filter)) {
396                                return false;
397                            }
398                        }
399
400                        true
401                    })
402                    .collect()
403            } else {
404                hooks
405            };
406
407            let result = lua.create_table()?;
408            for (i, hook) in filtered_hooks.iter().enumerate() {
409                let hook_info = lua.create_table()?;
410                hook_info.set("name", hook.name.clone())?;
411                hook_info.set("priority", format!("{:?}", hook.priority))?;
412                hook_info.set("language", hook.language.to_string())?;
413                hook_info.set("version", hook.version.clone())?;
414                if let Some(desc) = &hook.description {
415                    hook_info.set("description", desc.clone())?;
416                }
417
418                let tags = lua.create_table()?;
419                for (j, tag) in hook.tags.iter().enumerate() {
420                    tags.set(j + 1, tag.clone())?;
421                }
422                hook_info.set("tags", tags)?;
423
424                result.set(i + 1, hook_info)?;
425            }
426            Ok(result)
427        })
428        .map_err(|e| LLMSpellError::Component {
429            message: format!("Failed to create Hook.list: {e}"),
430            source: None,
431        })?;
432
433    hook_table
434        .set("list", list_fn)
435        .map_err(|e| LLMSpellError::Component {
436            message: format!("Failed to set Hook.list: {e}"),
437            source: None,
438        })?;
439
440    // Hook.unregister(handle) -> bool
441    let unregister_fn = lua
442        .create_function(move |_, handle: mlua::AnyUserData| {
443            // Try to cast to LuaHookHandle and call unregister
444            #[allow(clippy::option_if_let_else)] // Complex pattern
445            if let Ok(lua_handle) = handle.borrow::<LuaHookHandle>() {
446                block_on_async::<_, _, LLMSpellError>(
447                    "hook_unregister_standalone",
448                    async move {
449                        let mut handle_lock = lua_handle.handle.write().await;
450                        if let Some(h) = handle_lock.take() {
451                            lua_handle.hook_bridge.unregister_hook(&h).await?;
452                            Ok::<bool, LLMSpellError>(true)
453                        } else {
454                            Ok::<bool, LLMSpellError>(false)
455                        }
456                    },
457                    None,
458                )
459            } else {
460                Ok(false) // Invalid handle, return false instead of error
461            }
462        })
463        .map_err(|e| LLMSpellError::Component {
464            message: format!("Failed to create Hook.unregister: {e}"),
465            source: None,
466        })?;
467
468    hook_table
469        .set("unregister", unregister_fn)
470        .map_err(|e| LLMSpellError::Component {
471            message: format!("Failed to set Hook.unregister: {e}"),
472            source: None,
473        })?;
474
475    // Set the Hook global
476    lua.globals()
477        .set("Hook", hook_table)
478        .map_err(|e| LLMSpellError::Component {
479            message: format!("Failed to set Hook global: {e}"),
480            source: None,
481        })?;
482
483    Ok(())
484}

486#[cfg(test)]
487mod tests {
488    use super::*;
489    use crate::{ComponentRegistry, ProviderManager};
490    use llmspell_config::providers::ProviderManagerConfig;
491
492    #[tokio::test(flavor = "multi_thread")]
493    async fn test_lua_hook_injection() {
494        let lua = Lua::new();
495        let context = Arc::new(GlobalContext::new(
496            Arc::new(ComponentRegistry::new()),
497            Arc::new(
498                ProviderManager::new(ProviderManagerConfig::default())
499                    .await
500                    .unwrap(),
501            ),
502        ));
503        let hook_bridge = Arc::new(HookBridge::new(context.clone()).unwrap());
504
505        inject_hook_global(&lua, &context, hook_bridge).unwrap();
506
507        // Check that Hook global exists
508        let hook: Table = lua.globals().get("Hook").unwrap();
509        assert!(hook.contains_key("register").unwrap());
510        assert!(hook.contains_key("list").unwrap());
511    }
