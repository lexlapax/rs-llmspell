1//! Stack trace collection for Lua scripts
2//!
3//! Provides comprehensive stack trace capture using Lua's debug library
4//! with local variable inspection and source location mapping.
5
6use mlua::{Function, Lua, Result as LuaResult, Table, Value};
7use serde::{Deserialize, Serialize};
8use std::collections::HashMap;
9use std::fmt::Write;

11/// Stack frame information
12#[derive(Debug, Clone, Serialize, Deserialize)]
13pub struct StackFrame {
14    /// Function name (if available)
15    pub name: Option<String>,
16    /// Source file or chunk name
17    pub source: Option<String>,
18    /// Line number in source
19    pub line: Option<i32>,
20    /// Function definition line
21    pub line_defined: Option<i32>,
22    /// Function type (Lua, C, main, tail)
23    pub what: String,
24    /// Number of upvalues
25    pub num_upvalues: u8,
26    /// Number of parameters
27    pub num_params: u8,
28    /// Local variables at this frame
29    pub locals: HashMap<String, String>,
30    /// Upvalues at this frame
31    pub upvalues: HashMap<String, String>,
32}
33
34/// Complete stack trace
35#[derive(Debug, Clone, Serialize, Deserialize)]
36pub struct StackTrace {
37    /// Stack frames from bottom to top
38    pub frames: Vec<StackFrame>,
39    /// Maximum depth captured
40    pub max_depth: usize,
41    /// Whether the trace was truncated
42    pub truncated: bool,
43    /// Error message if capture failed
44    pub error: Option<String>,
45}
46
47/// Stack trace collection options
48#[derive(Debug, Clone)]
49pub struct StackTraceOptions {
50    /// Maximum stack depth to capture
51    pub max_depth: usize,
52    /// Whether to capture local variables
53    pub capture_locals: bool,
54    /// Whether to capture upvalues
55    pub capture_upvalues: bool,
56    /// Whether to include source information
57    pub include_source: bool,
58}
59
60impl Default for StackTraceOptions {
61    fn default() -> Self {
62        Self {
63            max_depth: 50,
64            capture_locals: false, // Only at trace level
65            capture_upvalues: false,
66            include_source: true,
67        }
68    }
69}

95/// Capture a stack trace from the current Lua execution point
96#[must_use]
97pub fn capture_stack_trace(lua: &Lua, options: &StackTraceOptions) -> StackTrace {
98    match capture_stack_trace_impl(lua, options) {
99        Ok(trace) => trace,
100        Err(e) => StackTrace {
101            frames: vec![],
102            max_depth: options.max_depth,
103            truncated: false,
104            error: Some(format!("Stack trace capture failed: {e}")),
105        },
106    }
107}
108
109fn capture_stack_trace_impl(lua: &Lua, options: &StackTraceOptions) -> LuaResult<StackTrace> {
110    let debug_table: Table = lua.globals().get("debug")?;
111    let getinfo_fn: Function = debug_table.get("getinfo")?;
112
113    let mut frames = Vec::new();
114    let mut level = 1; // Start from level 1 (skip debug.getinfo itself)
115    let mut truncated = false;
116
117    while frames.len() < options.max_depth {
118        // Get basic frame info
119        let info_result: LuaResult<Table> = getinfo_fn.call((level, "nSluf"));
120
121        let Ok(info_table) = info_result else { break }; // No more frames
122
123        // Extract frame information
124        let frame = extract_frame_info(lua, &info_table, level, options)?;
125        frames.push(frame);
126
127        level += 1;
128
129        // Check if we've hit our depth limit
130        if frames.len() >= options.max_depth {
131            truncated = true;
132            break;
133        }
134    }
135
136    Ok(StackTrace {
137        frames,
138        max_depth: options.max_depth,
139        truncated,
140        error: None,
141    })
142}

144fn extract_frame_info(
145    lua: &Lua,
146    info: &Table,
147    level: i32,
148    options: &StackTraceOptions,
149) -> LuaResult<StackFrame> {
150    // Basic frame information
151    let name = info.get::<_, Option<String>>("name")?;
152    let source = if options.include_source {
153        info.get::<_, Option<String>>("source")?
154    } else {
155        None
156    };
157    let line = info.get::<_, Option<i32>>("currentline")?;
158    let line_defined = info.get::<_, Option<i32>>("linedefined")?;
159    let what = info
160        .get::<_, String>("what")
161        .unwrap_or_else(|_| "unknown".to_string());
162    let num_upvalues = info.get::<_, Option<u8>>("nups")?.unwrap_or(0);
163    let num_params = info.get::<_, Option<u8>>("nparams")?.unwrap_or(0);
164
165    // Capture local variables if requested
166    let locals = if options.capture_locals {
167        capture_locals(lua, level)?
168    } else {
169        HashMap::new()
170    };
171
172    // Capture upvalues if requested
173    let upvalues = if options.capture_upvalues && info.contains_key("func")? {
174        capture_upvalues(lua, level)?
175    } else {
176        HashMap::new()
177    };
178
179    Ok(StackFrame {
180        name,
181        source,
182        line,
183        line_defined,
184        what,
185        num_upvalues,
186        num_params,
187        locals,
188        upvalues,
189    })
190}

192fn capture_locals(lua: &Lua, level: i32) -> LuaResult<HashMap<String, String>> {
193    let debug_table: Table = lua.globals().get("debug")?;
194    let getlocal_fn: Function = debug_table.get("getlocal")?;
195
196    let mut locals = HashMap::new();
197    let mut local_index = 1;
198
199    loop {
200        let result: LuaResult<(Option<String>, Value)> = getlocal_fn.call((level, local_index));
201
202        match result {
203            Ok((Some(name), value)) => {
204                // Skip internal variables (those starting with '(')
205                if !name.starts_with('(') {
206                    locals.insert(name, value_to_debug_string(&value));
207                }
208                local_index += 1;
209            }
210            _ => break, // No more locals or error
211        }
212
213        // Safety limit to prevent infinite loops
214        if local_index > 100 {
215            break;
216        }
217    }
218
219    Ok(locals)
220}

222fn capture_upvalues(lua: &Lua, level: i32) -> LuaResult<HashMap<String, String>> {
223    let debug_table: Table = lua.globals().get("debug")?;
224    let getinfo_fn: Function = debug_table.get("getinfo")?;
225    let getupvalue_fn: Function = debug_table.get("getupvalue")?;
226
227    // Get the function at this level
228    let info: Table = getinfo_fn.call((level, "f"))?;
229    let func: Function = info.get("func")?;
230
231    let mut upvalues = HashMap::new();
232    let mut upvalue_index = 1;
233
234    loop {
235        let result: LuaResult<(Option<String>, Value)> = getupvalue_fn.call((&func, upvalue_index));
236
237        match result {
238            Ok((Some(name), value)) => {
239                upvalues.insert(name, value_to_debug_string(&value));
240                upvalue_index += 1;
241            }
242            _ => break, // No more upvalues or error
243        }
244
245        // Safety limit
246        if upvalue_index > 50 {
247            break;
248        }
249    }
250
251    Ok(upvalues)
252}

254fn value_to_debug_string(value: &Value) -> String {
255    match value {
256        Value::Nil => "nil".to_string(),
257        Value::Boolean(b) => b.to_string(),
258        Value::Integer(i) => i.to_string(),
259        Value::Number(n) => n.to_string(),
260        Value::String(s) => s.to_str().map_or_else(
261            |_| "\"<invalid string>\"".to_string(),
262            |str_val| {
263                if str_val.len() > 100 {
264                    format!("\"{}...\" ({})", &str_val[..97], str_val.len())
265                } else {
266                    format!("\"{str_val}\"")
267                }
268            },
269        ),
270        Value::Table(_) => "table".to_string(),
271        Value::Function(_) => "function".to_string(),
272        Value::Thread(_) => "thread".to_string(),
273        Value::UserData(_) | Value::LightUserData(_) => "userdata".to_string(),
274        Value::Error(e) => format!("error: {e}"),
275    }
276}

278impl StackTrace {
279    /// Format the stack trace as a readable string
280    #[must_use]
281    pub fn format(&self) -> String {
282        if let Some(error) = &self.error {
283            return format!("Stack trace error: {error}");
284        }
285
286        if self.frames.is_empty() {
287            return "No stack frames available".to_string();
288        }
289
290        let mut output = String::from("Stack trace:\n");
291
292        for (i, frame) in self.frames.iter().enumerate() {
293            let frame_header = frame.name.as_ref().map_or_else(
294                || format!("  #{i}: in <unknown>"),
295                |name| format!("  #{i}: in function '{name}'"),
296            );
297
298            output.push_str(&frame_header);
299
300            // Add source information
301            if let Some(source) = &frame.source {
302                if let Some(line) = frame.line {
303                    let _ = write!(output, " ({source}:{line})");
304                } else {
305                    let _ = write!(output, " ({source})");
306                }
307            }
308
309            output.push('\n');
310
311            // Add local variables if available
312            if !frame.locals.is_empty() {
313                output.push_str("    Locals:\n");
314                for (name, value) in &frame.locals {
315                    let _ = writeln!(output, "      {name} = {value}");
316                }
317            }
318
319            // Add upvalues if available
320            if !frame.upvalues.is_empty() {
321                output.push_str("    Upvalues:\n");
322                for (name, value) in &frame.upvalues {
323                    let _ = writeln!(output, "      {name} = {value}");
324                }
325            }
326        }
327
328        if self.truncated {
329            let _ = writeln!(output, "  ... (truncated at {} frames)", self.max_depth);
330        }
331
332        output
333    }
334
335    /// Format as JSON string
336    ///
337    /// # Errors
338    ///
339    /// Returns an error if JSON serialization fails
340    pub fn to_json(&self) -> Result<String, serde_json::Error> {
341        serde_json::to_string_pretty(self)
342    }
343
344    /// Get frame count
345    #[must_use]
346    pub fn frame_count(&self) -> usize {
347        self.frames.len()
348    }
349
350    /// Get the topmost frame (most recent call)
351    #[must_use]
352    #[allow(clippy::missing_const_for_fn)]
353    pub fn top_frame(&self) -> Option<&StackFrame> {
354        self.frames.first()
355    }
356}

358/// Create a stack trace function for Lua scripts
359///
360/// # Errors
361///
362/// Returns an error if function creation fails
363pub fn create_stacktrace_function(lua: &Lua) -> LuaResult<Function> {
364    lua.create_function(|lua, options: Option<Table>| {
365        let trace_options =
366            options.map_or_else(StackTraceOptions::default, |opts| StackTraceOptions {
367                max_depth: opts.get("max_depth").unwrap_or(50),
368                capture_locals: opts.get("capture_locals").unwrap_or(false),
369                capture_upvalues: opts.get("capture_upvalues").unwrap_or(false),
370                include_source: opts.get("include_source").unwrap_or(true),
371            });
372
373        let trace = capture_stack_trace(lua, &trace_options);
374        Ok(trace.format())
375    })
376}
