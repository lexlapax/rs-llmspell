11/// Cache for workflow type information to avoid repeated discovery
12static WORKFLOW_TYPE_CACHE: LazyLock<Arc<RwLock<HashMap<String, WorkflowTypeInfo>>>> =
13    LazyLock::new(|| Arc::new(RwLock::new(HashMap::new())));

25/// Validator function type alias
26type ValidatorFn = Box<dyn Fn(&Value) -> bool + Send + Sync>;
27
28/// Performance-optimized parameter conversion
29pub struct OptimizedConverter {
30    /// Pre-compiled parameter validators
31    validators: HashMap<String, ValidatorFn>,
32}
33
34impl Default for OptimizedConverter {
35    fn default() -> Self {
36        Self::new()
37    }
38}
39
40impl OptimizedConverter {
41    #[must_use]
42    pub fn new() -> Self {
43        let mut validators = HashMap::new();
44
45        // Pre-compile common validators
46        validators.insert(
47            "sequential".to_string(),
48            Box::new(|v: &Value| {
49                v.get("name").is_some() && v.get("steps").and_then(|s| s.as_array()).is_some()
50            }) as ValidatorFn,
51        );
52
53        validators.insert(
54            "parallel".to_string(),
55            Box::new(|v: &Value| {
56                v.get("name").is_some() && v.get("branches").and_then(|b| b.as_array()).is_some()
57            }) as ValidatorFn,
58        );
59
60        validators.insert(
61            "conditional".to_string(),
62            Box::new(|v: &Value| v.get("name").is_some() && v.get("condition").is_some())
63                as ValidatorFn,
64        );
65
66        validators.insert(
67            "loop".to_string(),
68            Box::new(|v: &Value| v.get("name").is_some() && v.get("iterator").is_some())
69                as ValidatorFn,
70        );
71
72        Self { validators }
73    }
74
75    /// Fast parameter validation without full parsing
76    #[must_use]
77    pub fn validate_params(&self, workflow_type: &str, params: &Value) -> bool {
78        self.validators
79            .get(workflow_type)
80            .is_none_or(|validator| validator(params))
81    }
82}

84/// Workflow execution cache for recently executed workflows
85pub struct ExecutionCache {
86    /// LRU cache of workflow execution results
87    cache: Arc<RwLock<lru::LruCache<String, CachedExecution>>>,
88}
89
90#[derive(Clone)]
91struct CachedExecution {
92    _workflow_id: String,
93    result: Value,
94    timestamp: std::time::Instant,
95}
96
97impl ExecutionCache {
98    /// Create a new execution cache with the specified capacity
99    ///
100    /// # Panics
101    ///
102    /// Panics if capacity is 0
103    #[must_use]
104    pub fn new(capacity: usize) -> Self {
105        Self {
106            cache: Arc::new(RwLock::new(lru::LruCache::new(
107                std::num::NonZeroUsize::new(capacity).unwrap(),
108            ))),
109        }
110    }
111
112    /// Get cached execution if available and fresh
113    #[must_use]
114    pub fn get(&self, workflow_id: &str) -> Option<Value> {
115        let mut cache = self.cache.write();
116        if let Some(cached) = cache.get(workflow_id) {
117            // Cache entries expire after 60 seconds
118            if cached.timestamp.elapsed().as_secs() < 60 {
119                return Some(cached.result.clone());
120            }
121        }
122        None
123    }
124
125    /// Store execution result in cache
126    pub fn put(&self, workflow_id: String, result: Value) {
127        let mut cache = self.cache.write();
128        cache.put(
129            workflow_id.clone(),
130            CachedExecution {
131                _workflow_id: workflow_id,
132                result,
133                timestamp: std::time::Instant::now(),
134            },
135        );
136    }
137}

139/// Performance metrics collector
140pub struct PerformanceMetrics {
141    metrics: Arc<RwLock<Metrics>>,
142}
143
144#[derive(Default)]
145struct Metrics {
146    total_operations: u64,
147    total_duration_ms: u64,
148    operation_durations: Vec<u64>,
149}
150
151impl Default for PerformanceMetrics {
152    fn default() -> Self {
153        Self::new()
154    }
155}
156
157impl PerformanceMetrics {
158    #[must_use]
159    pub fn new() -> Self {
160        Self {
161            metrics: Arc::new(RwLock::new(Metrics::default())),
162        }
163    }
164
165    /// Record operation duration
166    pub fn record_operation(&self, duration_ms: u64) {
167        let mut metrics = self.metrics.write();
168        metrics.total_operations += 1;
169        metrics.total_duration_ms += duration_ms;
170        metrics.operation_durations.push(duration_ms);
171
172        // Keep only last 1000 operations
173        if metrics.operation_durations.len() > 1000 {
174            metrics.operation_durations.remove(0);
175        }
176    }
177
178    /// Get average operation duration
179    #[must_use]
180    pub fn average_duration_ms(&self) -> f64 {
181        let metrics = self.metrics.read();
182        if metrics.total_operations == 0 {
183            0.0
184        } else {
185            #[allow(clippy::cast_precision_loss)]
186            let total_duration = metrics.total_duration_ms as f64;
187            #[allow(clippy::cast_precision_loss)]
188            let total_ops = metrics.total_operations as f64;
189            total_duration / total_ops
190        }
191    }
192
193    /// Get p99 operation duration
194    #[must_use]
195    pub fn p99_duration_ms(&self) -> u64 {
196        let metrics = self.metrics.read();
197        if metrics.operation_durations.is_empty() {
198            return 0;
199        }
200
201        let mut durations = metrics.operation_durations.clone();
202        durations.sort_unstable();
203        #[allow(clippy::cast_precision_loss)]
204        let len_f64 = durations.len() as f64;
205        #[allow(clippy::cast_possible_truncation, clippy::cast_sign_loss)]
206        let idx = (len_f64 * 0.99) as usize;
207        durations.get(idx).copied().unwrap_or(0)
208    }
209
210    /// Check if performance is within acceptable bounds
211    #[must_use]
212    pub fn is_within_bounds(&self) -> bool {
213        self.p99_duration_ms() < 10
214    }
215}

217/// Optimized workflow discovery with caching
218#[must_use]
219pub fn get_workflow_info_cached(workflow_type: &str) -> Option<WorkflowTypeInfo> {
220    // Check cache first
221    {
222        let cache = WORKFLOW_TYPE_CACHE.read();
223        if let Some(info) = cache.get(workflow_type) {
224            return Some(info.clone());
225        }
226    }
227
228    // Cache miss - would normally fetch from registry
229    // For now, return static info
230    let info = match workflow_type {
231        "sequential" => WorkflowTypeInfo {
232            workflow_type: "sequential".to_string(),
233            description: "Execute steps in sequence".to_string(),
234            features: vec!["ordered_execution".to_string()],
235            required_params: vec!["name".to_string(), "steps".to_string()],
236            optional_params: vec!["error_strategy".to_string()],
237        },
238        "parallel" => WorkflowTypeInfo {
239            workflow_type: "parallel".to_string(),
240            description: "Execute branches in parallel".to_string(),
241            features: vec!["concurrent_execution".to_string()],
242            required_params: vec!["name".to_string(), "branches".to_string()],
243            optional_params: vec!["max_concurrency".to_string()],
244        },
245        "conditional" => WorkflowTypeInfo {
246            workflow_type: "conditional".to_string(),
247            description: "Execute based on conditions".to_string(),
248            features: vec!["branching_logic".to_string()],
249            required_params: vec!["name".to_string(), "condition".to_string()],
250            optional_params: vec!["default_branch".to_string()],
251        },
252        "loop" => WorkflowTypeInfo {
253            workflow_type: "loop".to_string(),
254            description: "Execute in a loop".to_string(),
255            features: vec!["iteration".to_string()],
256            required_params: vec!["name".to_string(), "iterator".to_string()],
257            optional_params: vec!["max_iterations".to_string()],
258        },
259        _ => return None,
260    };
261
262    // Store in cache
263    {
264        let mut cache = WORKFLOW_TYPE_CACHE.write();
265        cache.insert(workflow_type.to_string(), info.clone());
266    }
267
268    Some(info)
269}

271#[cfg(test)]
272mod tests {
273    use super::*;
274    #[test]
275    fn test_optimized_converter() {
276        let converter = OptimizedConverter::new();
277
278        // Test valid sequential params
279        let params = serde_json::json!({
280            "name": "test",
281            "steps": []
282        });
283        assert!(converter.validate_params("sequential", &params));
284
285        // Test invalid sequential params
286        let params = serde_json::json!({
287            "name": "test"
288        });
289        assert!(!converter.validate_params("sequential", &params));
290    }
291    #[test]
292    fn test_execution_cache() {
293        let cache = ExecutionCache::new(10);
294
295        // Store result
296        cache.put(
297            "workflow1".to_string(),
298            serde_json::json!({"result": "success"}),
299        );
300
301        // Retrieve result
302        let cached = cache.get("workflow1");
303        assert!(cached.is_some());
304        assert_eq!(cached.unwrap()["result"], "success");
305
306        // Non-existent entry
307        assert!(cache.get("workflow2").is_none());
308    }
309    #[test]
310    fn test_performance_metrics() {
311        let metrics = PerformanceMetrics::new();
312
313        // Record some operations
314        metrics.record_operation(5);
315        metrics.record_operation(8);
316        metrics.record_operation(3);
317
318        // Check average
319        assert!((metrics.average_duration_ms() - 5.333).abs() < 0.001);
320
321        // Check p99
322        assert_eq!(metrics.p99_duration_ms(), 8);
323
324        // Check bounds
325        assert!(metrics.is_within_bounds());
326
327        // Record slow operation
328        metrics.record_operation(15);
329        assert!(!metrics.is_within_bounds());
330    }
