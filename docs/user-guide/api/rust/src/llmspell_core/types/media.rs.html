7/// Maximum file size limits for different media types (in bytes)
8pub const MAX_IMAGE_SIZE: usize = 100 * 1024 * 1024; // 100MB
9pub const MAX_AUDIO_SIZE: usize = 500 * 1024 * 1024; // 500MB
10pub const MAX_VIDEO_SIZE: usize = 5 * 1024 * 1024 * 1024; // 5GB
11pub const MAX_BINARY_SIZE: usize = 1024 * 1024 * 1024; // 1GB

13/// Multimodal content that can be processed by agents
14///
15/// # Examples
16///
17/// ```
18/// use llmspell_core::types::{MediaContent, ImageFormat, ImageMetadata, ColorSpace};
19///
20/// // Text content
21/// let text = MediaContent::Text("Hello world".to_string());
22///
23/// // Image content with metadata
24/// let image = MediaContent::Image {
25///     data: vec![0xFF, 0xD8, 0xFF], // JPEG header
26///     format: ImageFormat::Jpeg,
27///     metadata: ImageMetadata {
28///         width: 1920,
29///         height: 1080,
30///         color_space: ColorSpace::RGB,
31///         has_transparency: false,
32///         dpi: Some(72),
33///     },
34/// };
35/// ```
36#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
37#[serde(tag = "type", rename_all = "snake_case")]
38pub enum MediaContent {
39    /// Plain text content
40    Text(String),
41
42    /// Image data with format and metadata
43    Image {
44        /// Raw image bytes
45        data: Vec<u8>,
46        /// Image format
47        format: ImageFormat,
48        /// Image metadata
49        metadata: ImageMetadata,
50    },
51
52    /// Audio data with format and metadata
53    Audio {
54        /// Raw audio bytes
55        data: Vec<u8>,
56        /// Audio format
57        format: AudioFormat,
58        /// Audio metadata
59        metadata: AudioMetadata,
60    },
61
62    /// Video data with format and metadata
63    Video {
64        /// Raw video bytes
65        data: Vec<u8>,
66        /// Video format
67        format: VideoFormat,
68        /// Video metadata
69        metadata: VideoMetadata,
70    },
71
72    /// Generic binary data
73    Binary {
74        /// Raw binary bytes
75        data: Vec<u8>,
76        /// MIME type if known
77        mime_type: Option<String>,
78        /// Original filename if available
79        filename: Option<String>,
80    },
81}

190/// Media types for capability detection
191///
192/// Used to indicate which media types an agent or tool can handle.
193///
194/// # Examples
195///
196/// ```
197/// use llmspell_core::types::MediaType;
198///
199/// let supported_types = vec![
200///     MediaType::Text,
201///     MediaType::Image,
202/// ];

204/// // Check if audio is supported
205/// if supported_types.contains(&MediaType::Audio) {
206///     println!("Audio processing available");
207/// }
208/// ```
209#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, Serialize, Deserialize)]

216    /// Audio content
217    Audio,
218    /// Video content
219    Video,
220    /// Binary content
221    Binary,
222}

226impl fmt::Display for MediaContent {
227    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
228        match self {
229            MediaContent::Text(text) => {
230                write!(f, "Text({} chars)", text.len())
231            }
232            MediaContent::Image {
233                format, metadata, ..
234            } => {
235                write!(
236                    f,
237                    "Image({:?}, {}x{})",
238                    format, metadata.width, metadata.height
239                )
240            }
241            MediaContent::Audio {
242                format, metadata, ..
243            } => {
244                write!(f, "Audio({:?}, {}ms)", format, metadata.duration_ms)
245            }
246            MediaContent::Video {
247                format, metadata, ..
248            } => {
249                write!(
250                    f,
251                    "Video({:?}, {}x{}, {}ms)",
252                    format, metadata.width, metadata.height, metadata.duration_ms
253                )
254            }
255            MediaContent::Binary {
256                mime_type,
257                filename,
258                data,
259            } => {
260                write!(f, "Binary({} bytes", data.len())?;
261                if let Some(mime) = mime_type {
262                    write!(f, ", {mime}")?;
263                }
264                if let Some(name) = filename {
265                    write!(f, ", {name}")?;
266                }
267                write!(f, ")")
268            }
269        }
270    }
271}

335impl MediaContent {
336    /// Get the size of the content in bytes
337    #[must_use]
338    pub fn size_bytes(&self) -> usize {
339        match self {
340            MediaContent::Text(text) => text.len(),
341            MediaContent::Image { data, .. }
342            | MediaContent::Audio { data, .. }
343            | MediaContent::Video { data, .. }
344            | MediaContent::Binary { data, .. } => data.len(),
345        }
346    }
347
348    /// Get the media type of this content
349    #[must_use]
350    pub fn media_type(&self) -> MediaType {
351        match self {
352            MediaContent::Text(_) => MediaType::Text,
353            MediaContent::Image { .. } => MediaType::Image,
354            MediaContent::Audio { .. } => MediaType::Audio,
355            MediaContent::Video { .. } => MediaType::Video,
356            MediaContent::Binary { .. } => MediaType::Binary,
357        }
358    }
359
360    /// Validate size constraints
361    pub fn validate_size(&self) -> Result<(), String> {
362        let size = self.size_bytes();
363        let (max_size, type_name) = match self {
364            MediaContent::Text(_) => return Ok(()), // No size limit for text
365            MediaContent::Image { .. } => (MAX_IMAGE_SIZE, "image"),
366            MediaContent::Audio { .. } => (MAX_AUDIO_SIZE, "audio"),
367            MediaContent::Video { .. } => (MAX_VIDEO_SIZE, "video"),
368            MediaContent::Binary { .. } => (MAX_BINARY_SIZE, "binary"),
369        };
370
371        if size > max_size {
372            Err(format!(
373                "{type_name} size {size} bytes exceeds maximum {max_size} bytes"
374            ))
375        } else {
376            Ok(())
377        }
378    }
379}

463impl TryFrom<&str> for ImageFormat {
464    type Error = String;
465
466    fn try_from(ext: &str) -> Result<Self, Self::Error> {
467        match ext.to_lowercase().as_str() {
468            "png" => Ok(ImageFormat::Png),
469            "jpg" | "jpeg" => Ok(ImageFormat::Jpeg),
470            "webp" => Ok(ImageFormat::Webp),
471            "gif" => Ok(ImageFormat::Gif),
472            "svg" => Ok(ImageFormat::Svg),
473            "tif" | "tiff" => Ok(ImageFormat::Tiff),
474            _ => Err(format!("Unknown image format: {ext}")),
475        }
476    }
477}
