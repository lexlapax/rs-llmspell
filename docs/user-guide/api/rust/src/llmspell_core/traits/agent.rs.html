9/// Role in a conversation.
10///
11/// Defines the role of a message in a conversation history.
12/// Used for maintaining context in LLM interactions.
13///
14/// # Examples
15///
16/// ```
17/// use llmspell_core::traits::agent::MessageRole;
18///
19/// assert_eq!(MessageRole::System.to_string(), "system");
20/// assert_eq!(MessageRole::User.to_string(), "user");
21/// assert_eq!(MessageRole::Assistant.to_string(), "assistant");
22/// ```
23#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
24pub enum MessageRole {
25    System,
26    User,
27    Assistant,
28}
29
30impl std::fmt::Display for MessageRole {
31    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
32        match self {
33            MessageRole::System => write!(f, "system"),
34            MessageRole::User => write!(f, "user"),
35            MessageRole::Assistant => write!(f, "assistant"),
36        }
37    }
38}

40/// Conversation message in an agent's history.
41///
42/// Represents a single message in a conversation, including the role,
43/// content, and timestamp. Used to maintain conversation context for agents.
44///
45/// # Examples
46///
47/// ```
48/// use llmspell_core::traits::agent::{ConversationMessage, MessageRole};
49///
50/// // Create messages using convenience methods
51/// let system_msg = ConversationMessage::system("You are a helpful assistant".to_string());
52/// let user_msg = ConversationMessage::user("Hello!".to_string());
53/// let assistant_msg = ConversationMessage::assistant("Hi! How can I help?".to_string());
54///
55/// assert_eq!(system_msg.role, MessageRole::System);
56/// assert_eq!(user_msg.role, MessageRole::User);
57/// assert_eq!(assistant_msg.role, MessageRole::Assistant);
58/// ```
59#[derive(Debug, Clone, Serialize, Deserialize)]
60pub struct ConversationMessage {
61    pub role: MessageRole,
62    pub content: String,
63    pub timestamp: chrono::DateTime<chrono::Utc>,
64}
65
66impl ConversationMessage {
67    /// Create a new conversation message
68    #[must_use]
69    pub fn new(role: MessageRole, content: String) -> Self {
70        Self {
71            role,
72            content,
73            timestamp: chrono::Utc::now(),
74        }
75    }
76
77    /// Create a system message
78    #[must_use]
79    pub fn system(content: String) -> Self {
80        Self::new(MessageRole::System, content)
81    }
82
83    /// Create a user message
84    #[must_use]
85    pub fn user(content: String) -> Self {
86        Self::new(MessageRole::User, content)
87    }
88
89    /// Create an assistant message
90    #[must_use]
91    pub fn assistant(content: String) -> Self {
92        Self::new(MessageRole::Assistant, content)
93    }
94}

96/// Configuration for LLM-powered agents.
97///
98/// Controls agent behavior including conversation management, generation parameters,
99/// and system prompts. All fields are optional to allow partial configuration.
100///
101/// # Examples
102///
103/// ```
104/// use llmspell_core::traits::agent::AgentConfig;
105///
106/// let config = AgentConfig {
107///     max_conversation_length: Some(50),
108///     system_prompt: Some("You are a research assistant".to_string()),
109///     temperature: Some(0.7),
110///     max_tokens: Some(1000),
111/// };
112///
113/// // Or use default configuration
114/// let default_config = AgentConfig::default();
115/// assert_eq!(default_config.max_conversation_length, Some(100));
116/// assert_eq!(default_config.temperature, Some(0.7));
117/// ```
118#[derive(Debug, Clone, Serialize, Deserialize)]
119pub struct AgentConfig {
120    /// Maximum number of messages to retain in conversation history
121    pub max_conversation_length: Option<usize>,
122    /// System prompt for the agent
123    pub system_prompt: Option<String>,
124    /// Temperature setting for LLM generation
125    pub temperature: Option<f32>,
126    /// Maximum tokens to generate
127    pub max_tokens: Option<usize>,
128}
129
130impl Default for AgentConfig {
131    fn default() -> Self {
132        Self {
133            max_conversation_length: Some(100),
134            system_prompt: None,
135            temperature: Some(0.7),
136            max_tokens: Some(2000),
137        }
138    }
139}

141/// Agent trait for LLM-powered components.
142///
143/// Extends `BaseAgent` with conversation management and LLM-specific functionality.
144/// Agents maintain conversation history, handle message trimming, and provide
145/// configuration for LLM generation parameters.
146///
147/// # Implementation Requirements
148///
149/// - Must maintain conversation history thread-safely
150/// - Should implement conversation trimming to respect max length
151/// - Configuration should be accessible but may be immutable
152/// - All conversation operations should be atomic
153///
154/// # Examples
155///
156/// ```ignore
157/// use llmspell_core::{
158///     ComponentMetadata, Result,
159///     traits::{
160///         base_agent::{BaseAgent, AgentInput, AgentOutput, ExecutionContext},
161///         agent::{Agent, AgentConfig, ConversationMessage, MessageRole}
162///     }
163/// };
164/// use async_trait::async_trait;
165///
166/// struct MyLLMAgent {
167///     metadata: ComponentMetadata,
168///     config: AgentConfig,
169///     conversation: Vec<ConversationMessage>,
170/// }
171///
172/// #[async_trait]
173/// impl Agent for MyLLMAgent {
174///     fn config(&self) -> &AgentConfig {
175///         &self.config
176///     }
177///     
178///     async fn get_conversation(&self) -> Result<Vec<ConversationMessage>> {
179///         Ok(self.conversation.clone())
180///     }
181///     
182///     async fn add_message(&self, message: ConversationMessage) -> Result<()> {
183///         // With interior mutability pattern
184///         self.conversation.lock().await.push(message);
185///         // Trim if needed
186///         self.trim_conversation().await?;
187///         Ok(())
188///     }
189///     
190///     async fn clear_conversation(&self) -> Result<()> {
191///         // With interior mutability pattern
192///         self.conversation.lock().await.clear();
193///         Ok(())
194///     }
195/// }
196///
197/// # impl BaseAgent for MyLLMAgent {
198/// #     fn metadata(&self) -> &ComponentMetadata { &self.metadata }
199/// #     async fn execute(&self, input: AgentInput, context: ExecutionContext) -> Result<AgentOutput> {
200/// #         Ok(AgentOutput::text("Response".to_string()))
201/// #     }
202/// #     async fn validate_input(&self, input: &AgentInput) -> Result<()> { Ok(()) }
203/// #     async fn handle_error(&self, error: llmspell_core::LLMSpellError) -> Result<AgentOutput> {
204/// #         Ok(AgentOutput::text("Error".to_string()))
205/// #     }
206/// # }
207/// ```
208#[async_trait]
209pub trait Agent: BaseAgent {
210    /// Get agent configuration
211    fn config(&self) -> &AgentConfig;
212
213    /// Get conversation history
214    async fn get_conversation(&self) -> Result<Vec<ConversationMessage>>;
215
216    /// Add message to conversation
217    async fn add_message(&self, message: ConversationMessage) -> Result<()>;
218
219    /// Clear conversation history
220    async fn clear_conversation(&self) -> Result<()>;
221
222    /// Get the current conversation length
223    async fn conversation_length(&self) -> Result<usize> {
224        Ok(self.get_conversation().await?.len())
225    }
226
227    /// Trim conversation to configured max length
228    async fn trim_conversation(&self) -> Result<()> {
229        if let Some(max_len) = self.config().max_conversation_length {
230            let current_len = self.conversation_length().await?;
231            if current_len > max_len {
232                // Keep system messages and trim oldest user/assistant messages
233                let conversation = self.get_conversation().await?;
234                let system_messages: Vec<_> = conversation
235                    .iter()
236                    .filter(|msg| matches!(msg.role, MessageRole::System))
237                    .cloned()
238                    .collect();
239
240                let other_messages: Vec<_> = conversation
241                    .into_iter()
242                    .filter(|msg| !matches!(msg.role, MessageRole::System))
243                    .collect();
244
245                let skip_count = other_messages
246                    .len()
247                    .saturating_sub(max_len - system_messages.len());
248
249                self.clear_conversation().await?;
250
251                // Re-add system messages
252                for msg in system_messages {
253                    self.add_message(msg).await?;
254                }
255
256                // Add trimmed other messages
257                for msg in other_messages.into_iter().skip(skip_count) {
258                    self.add_message(msg).await?;
259                }
260            }
261        }
262        Ok(())
263    }
264}
