1//! ABOUTME: Agent builder with fluent API for easy agent configuration
2//! ABOUTME: Provides chainable methods for building agent configurations

4use crate::factory::{AgentConfig, ModelConfig, ResourceLimits};
5use anyhow::Result;
6use serde_json::Value;
7
8/// Builder for creating agent configurations with a fluent API
9#[derive(Debug, Clone)]
10pub struct AgentBuilder {
11    config: AgentConfig,
12}
13
14impl AgentBuilder {
15    /// Create a new agent builder with required fields
16    pub fn new(name: impl Into<String>, agent_type: impl Into<String>) -> Self {
17        Self {
18            config: AgentConfig {
19                name: name.into(),
20                description: String::new(),
21                agent_type: agent_type.into(),
22                model: None,
23                allowed_tools: vec![],
24                custom_config: serde_json::Map::new(),
25                resource_limits: ResourceLimits::default(),
26            },
27        }
28    }
29
30    /// Set the agent description
31    #[must_use]
32    pub fn description(mut self, description: impl Into<String>) -> Self {
33        self.config.description = description.into();
34        self
35    }
36
37    /// Configure the model for LLM-based agents
38    #[must_use]
39    pub fn with_model(mut self, provider: impl Into<String>, model_id: impl Into<String>) -> Self {
40        self.config.model = Some(ModelConfig {
41            provider: provider.into(),
42            model_id: model_id.into(),
43            temperature: None,
44            max_tokens: None,
45            settings: serde_json::Map::new(),
46        });
47        self
48    }
49
50    /// Set model temperature
51    ///
52    /// # Panics
53    ///
54    /// Will not panic. This function only modifies the temperature if a model config exists.
55    #[must_use]
56    pub const fn temperature(mut self, temperature: f32) -> Self {
57        if let Some(model) = &mut self.config.model {
58            model.temperature = Some(temperature);
59        }
60        self
61    }
62
63    /// Set maximum tokens for model
64    ///
65    /// # Panics
66    ///
67    /// Will not panic. This function only modifies the max tokens if a model config exists.
68    #[must_use]
69    pub const fn max_tokens(mut self, max_tokens: u32) -> Self {
70        if let Some(model) = &mut self.config.model {
71            model.max_tokens = Some(max_tokens);
72        }
73        self
74    }
75
76    /// Add a model setting
77    ///
78    /// # Panics
79    ///
80    /// Will not panic. This function only adds settings if a model config exists.
81    #[must_use]
82    pub fn model_setting(mut self, key: impl Into<String>, value: Value) -> Self {
83        if let Some(model) = &mut self.config.model {
84            model.settings.insert(key.into(), value);
85        }
86        self
87    }
88
89    /// Allow access to specific tools
90    #[must_use]
91    pub fn allow_tool(mut self, tool_id: impl Into<String>) -> Self {
92        self.config.allowed_tools.push(tool_id.into());
93        self
94    }
95
96    /// Allow access to multiple tools
97    #[must_use]
98    pub fn allow_tools(mut self, tool_ids: Vec<String>) -> Self {
99        self.config.allowed_tools.extend(tool_ids);
100        self
101    }
102
103    /// Allow access to all tools
104    #[must_use]
105    pub fn allow_all_tools(mut self) -> Self {
106        self.config.allowed_tools = vec!["*".to_string()];
107        self
108    }
109
110    /// Add a custom configuration parameter
111    #[must_use]
112    pub fn custom(mut self, key: impl Into<String>, value: Value) -> Self {
113        self.config.custom_config.insert(key.into(), value);
114        self
115    }
116
117    /// Set maximum execution time
118    #[must_use]
119    pub const fn max_execution_time_secs(mut self, secs: u64) -> Self {
120        self.config.resource_limits.max_execution_time_secs = secs;
121        self
122    }
123
124    /// Set maximum memory usage
125    #[must_use]
126    pub const fn max_memory_mb(mut self, mb: u64) -> Self {
127        self.config.resource_limits.max_memory_mb = mb;
128        self
129    }
130
131    /// Set maximum number of tool calls
132    #[must_use]
133    pub const fn max_tool_calls(mut self, calls: u32) -> Self {
134        self.config.resource_limits.max_tool_calls = calls;
135        self
136    }
137
138    /// Set maximum recursion depth
139    #[must_use]
140    pub const fn max_recursion_depth(mut self, depth: u8) -> Self {
141        self.config.resource_limits.max_recursion_depth = depth;
142        self
143    }
144
145    /// Set all resource limits at once
146    #[must_use]
147    pub const fn resource_limits(mut self, limits: ResourceLimits) -> Self {
148        self.config.resource_limits = limits;
149        self
150    }
151
152    /// Build the agent configuration
153    ///
154    /// # Errors
155    ///
156    /// Returns an error if:
157    /// - Agent name is empty
158    /// - Agent type is empty
159    /// - Invalid tool IDs are specified
160    /// - Configuration validation fails
161    pub fn build(self) -> Result<AgentConfig> {
162        // Validate the configuration
163        if self.config.name.is_empty() {
164            anyhow::bail!("Agent name cannot be empty");
165        }
166
167        if self.config.agent_type.is_empty() {
168            anyhow::bail!("Agent type cannot be empty");
169        }
170
171        Ok(self.config)
172    }
173}
174
175/// Convenience builders for common agent types
176impl AgentBuilder {
177    /// Create a builder for a basic agent
178    pub fn basic(name: impl Into<String>) -> Self {
179        Self::new(name, "basic").description("Basic agent with minimal capabilities")
180    }
181
182    /// Create a builder for a tool orchestrator agent
183    pub fn tool_orchestrator(name: impl Into<String>) -> Self {
184        Self::new(name, "tool-orchestrator")
185            .description("Agent that orchestrates tool execution")
186            .allow_all_tools()
187    }
188
189    /// Create a builder for an LLM agent
190    pub fn llm(
191        name: impl Into<String>,
192        provider: impl Into<String>,
193        model: impl Into<String>,
194    ) -> Self {
195        Self::new(name.into(), "llm")
196            .description("LLM-powered agent")
197            .with_model(provider, model)
198    }
199
200    /// Create a builder for a workflow agent
201    pub fn workflow(name: impl Into<String>) -> Self {
202        Self::new(name, "workflow").description("Agent that executes workflows")
203    }
204}

206#[cfg(test)]
207mod tests {
208    use super::*;
209    use serde_json::json;
210    #[test]
211    fn test_basic_builder() {
212        let config = AgentBuilder::new("test-agent", "basic")
213            .description("A test agent")
214            .build()
215            .unwrap();
216
217        assert_eq!(config.name, "test-agent");
218        assert_eq!(config.agent_type, "basic");
219        assert_eq!(config.description, "A test agent");
220    }
221    #[test]
222    fn test_fluent_api() {
223        let config = AgentBuilder::new("complex-agent", "llm")
224            .description("Complex LLM agent")
225            .with_model("openai", "gpt-4")
226            .temperature(0.7)
227            .max_tokens(2000)
228            .model_setting("top_p", json!(0.9))
229            .allow_tools(vec!["calculator".to_string(), "file_tool".to_string()])
230            .custom("retry_count", json!(3))
231            .max_execution_time_secs(600)
232            .max_memory_mb(1024)
233            .build()
234            .unwrap();
235
236        assert_eq!(config.name, "complex-agent");
237        assert!(config.model.is_some());
238
239        let model = config.model.unwrap();
240        assert_eq!(model.provider, "openai");
241        assert_eq!(model.model_id, "gpt-4");
242        assert_eq!(model.temperature, Some(0.7));
243        assert_eq!(model.max_tokens, Some(2000));
244        assert_eq!(model.settings.get("top_p"), Some(&json!(0.9)));
245
246        assert_eq!(config.allowed_tools.len(), 2);
247        assert_eq!(config.custom_config.get("retry_count"), Some(&json!(3)));
248        assert_eq!(config.resource_limits.max_execution_time_secs, 600);
249        assert_eq!(config.resource_limits.max_memory_mb, 1024);
250    }
251    #[test]
252    fn test_convenience_builders() {
253        // Test basic builder
254        let basic = AgentBuilder::basic("my-basic").build().unwrap();
255        assert_eq!(basic.agent_type, "basic");
256        assert!(basic.description.contains("Basic"));
257
258        // Test tool orchestrator builder
259        let orchestrator = AgentBuilder::tool_orchestrator("my-orchestrator")
260            .build()
261            .unwrap();
262        assert_eq!(orchestrator.agent_type, "tool-orchestrator");
263        assert_eq!(orchestrator.allowed_tools, vec!["*"]);
264
265        // Test LLM builder
266        let llm = AgentBuilder::llm("my-llm", "anthropic", "claude-3")
267            .build()
268            .unwrap();
269        assert_eq!(llm.agent_type, "llm");
270        assert!(llm.model.is_some());
271        assert_eq!(llm.model.as_ref().unwrap().provider, "anthropic");
272
273        // Test workflow builder
274        let workflow = AgentBuilder::workflow("my-workflow").build().unwrap();
275        assert_eq!(workflow.agent_type, "workflow");
276    }
277    #[test]
278    fn test_validation() {

280        let result = AgentBuilder::new("", "basic").build();
281        assert!(result.is_err());
