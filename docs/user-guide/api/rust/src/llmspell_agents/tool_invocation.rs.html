17/// Tool invocation wrapper that provides validation, error handling,
18/// and execution tracking for tool calls.
19///
20/// This wrapper sits between `ToolCapable` components and the actual tools,
21/// providing additional safety, monitoring, and debugging capabilities.
22///
23/// # Examples
24///
25/// ```
26/// use llmspell_agents::tool_invocation::{ToolInvoker, InvocationConfig};
27/// use llmspell_core::ExecutionContext;
28/// use serde_json::json;
29///
30/// # async fn example() -> llmspell_core::Result<()> {
31/// let config = InvocationConfig::default();
32/// let invoker = ToolInvoker::new(config);
33///
34/// let context = ExecutionContext::new();
35/// let params = json!({"pattern": "*.txt"});
36///
37/// // This would normally use an actual tool
38/// // let result = invoker.invoke(tool, params, context).await?;
39/// # Ok(())
40/// # }
41/// ```
42pub struct ToolInvoker {
43    config: InvocationConfig,
44}
45

234    /// Invoke a tool with full validation and error handling
235    ///
236    /// # Errors
237    ///
238    /// Returns an error if:
239    /// - Parameter validation fails
240    /// - Security checks fail
241    /// - Tool execution fails
242    /// - Resource limits are exceeded
243    pub async fn invoke(
244        &self,
245        tool: Arc<dyn Tool>,
246        parameters: JsonValue,
247        context: ExecutionContext,
248    ) -> Result<InvocationResult> {
249        let start_time = Instant::now();
250        let mut metrics = InvocationMetrics::default();
251        let mut warnings = Vec::new();
252
253        // Get tool security level
254        metrics.security_level = match tool.security_level() {
255            llmspell_core::traits::tool::SecurityLevel::Safe => "safe",
256            llmspell_core::traits::tool::SecurityLevel::Restricted => "restricted",
257            llmspell_core::traits::tool::SecurityLevel::Privileged => "privileged",
258        }
259        .to_string();
260
261        // Validate parameters if enabled
262        if self.config.feature_flags.validate_parameters {
263            let validation_start = Instant::now();
264            match Self::validate_tool_parameters(tool.as_ref(), &parameters) {
265                Ok(validation_warnings) => {
266                    warnings.extend(validation_warnings);
267                }
268                Err(e) => {
269                    metrics.validation_errors += 1;
270                    metrics.execution_time = start_time.elapsed();
271                    return Ok(InvocationResult {
272                        output: AgentOutput::text(format!("Validation failed: {e}")),
273                        metrics,
274                        warnings,
275                        success: false,
276                    });
277                }
278            }
279            metrics.validation_time = Some(validation_start.elapsed());
280        }
281
282        // Prepare input
283        let input = AgentInput::text("Tool invocation".to_string())
284            .with_parameter("parameters".to_string(), parameters);
285
286        // Execute tool with timeout
287        let execution_result =
288            timeout(self.config.max_execution_time, tool.execute(input, context)).await;
289
290        let output = match execution_result {
291            Ok(Ok(output)) => output,
292            Ok(Err(e)) => {
293                metrics.execution_time = start_time.elapsed();
294                return Ok(InvocationResult {
295                    output: AgentOutput::text(format!("Tool execution failed: {e}")),
296                    metrics,
297                    warnings,
298                    success: false,
299                });
300            }
301            Err(_) => {
302                metrics.timed_out = true;
303                metrics.execution_time = self.config.max_execution_time;
304                return Ok(InvocationResult {
305                    output: AgentOutput::text("Tool execution timed out".to_string()),
306                    metrics,
307                    warnings,
308                    success: false,
309                });
310            }
311        };
312
313        // Update final metrics
314        metrics.execution_time = start_time.elapsed();
315
316        // Log execution if debug logging is enabled
317        if self.config.feature_flags.debug_logging {
318            tracing::debug!(
319                "Tool {} executed in {:?}",
320                tool.metadata().name,
321                metrics.execution_time
322            );
323        }
324
325        Ok(InvocationResult {
326            output,
327            metrics,
328            warnings,
329            success: true,
330        })
331    }

333    /// Invoke a tool with basic error handling (convenience method)
334    ///
335    /// # Errors
336    ///
337    /// Returns an error if:
338    /// - Tool invocation fails
339    /// - Tool returns an error result
340    pub async fn invoke_simple(
341        &self,
342        tool: Arc<dyn Tool>,
343        parameters: JsonValue,
344        context: ExecutionContext,
345    ) -> Result<AgentOutput> {
346        let result = self.invoke(tool, parameters, context).await?;
347        if result.success {
348            Ok(result.output)
349        } else {
350            Err(LLMSpellError::Component {
351                message: result.output.text,
352                source: None,
353            })
354        }
355    }

357    /// Validate tool parameters against the tool's schema
358    fn validate_tool_parameters(tool: &dyn Tool, parameters: &JsonValue) -> Result<Vec<String>> {
359        let mut warnings = Vec::new();
360
361        // Get tool schema
362        let schema = tool.schema();
363
364        // Check that parameters is an object
365        let params_map = parameters
366            .as_object()
367            .ok_or_else(|| LLMSpellError::Validation {
368                message: "Parameters must be an object".to_string(),
369                field: Some("parameters".to_string()),
370            })?;
371
372        // Validate required parameters
373        for required in schema.required_parameters() {
374            if !params_map.contains_key(&required) {
375                return Err(LLMSpellError::Validation {
376                    message: format!("Missing required parameter: {required}"),
377                    field: Some(required),
378                });
379            }
380        }
381
382        // Validate parameter types and constraints
383        for param_def in &schema.parameters {
384            if let Some(value) = params_map.get(&param_def.name) {
385                // Type validation
386                let valid_type = match param_def.param_type {
387                    llmspell_core::traits::tool::ParameterType::String => value.is_string(),
388                    llmspell_core::traits::tool::ParameterType::Number => value.is_number(),
389                    llmspell_core::traits::tool::ParameterType::Boolean => value.is_boolean(),
390                    llmspell_core::traits::tool::ParameterType::Array => value.is_array(),
391                    llmspell_core::traits::tool::ParameterType::Object => value.is_object(),
392                    llmspell_core::traits::tool::ParameterType::Null => value.is_null(),
393                };
394
395                if !valid_type {
396                    return Err(LLMSpellError::Validation {
397                        message: format!(
398                            "Invalid type for parameter '{}': expected {:?}",
399                            param_def.name, param_def.param_type
400                        ),
401                        field: Some(param_def.name.clone()),
402                    });
403                }
404
405                // Additional validation for strings
406                if param_def.param_type == llmspell_core::traits::tool::ParameterType::String {
407                    if let Some(string_val) = value.as_str() {
408                        if string_val.is_empty() && param_def.required {
409                            warnings
410                                .push(format!("Required parameter '{}' is empty", param_def.name));
411                        }
412                        if string_val.len() > 10000 {
413                            warnings.push(format!(
414                                "Parameter '{}' is very long ({} chars), this may impact performance",
415                                param_def.name, string_val.len()
416                            ));
417                        }
418                    }
419                }
420
421                // Additional validation for arrays
422                if param_def.param_type == llmspell_core::traits::tool::ParameterType::Array {
423                    if let Some(array_val) = value.as_array() {
424                        if array_val.len() > 1000 {
425                            warnings.push(format!(
426                                "Parameter '{}' has many elements ({}), this may impact performance",
427                                param_def.name, array_val.len()
428                            ));
429                        }
430                    }
431                }
432            }
433        }
434
435        // Check for unexpected parameters
436        for key in params_map.keys() {
437            if !schema.parameters.iter().any(|p| &p.name == key) {
438                warnings.push(format!("Unexpected parameter '{key}' will be ignored"));
439            }
440        }
441
442        Ok(warnings)
443    }
