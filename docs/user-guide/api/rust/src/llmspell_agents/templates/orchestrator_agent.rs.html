21/// Orchestration strategy types
22#[derive(Debug, Clone, Serialize, Deserialize)]
23pub enum OrchestrationStrategy {
24    /// Execute tasks sequentially
25    Sequential,
26    /// Execute tasks in parallel where possible
27    Parallel,
28    /// Use conditional logic for task execution
29    Conditional,
30    /// Pipeline-based execution
31    Pipeline,
32    /// Event-driven orchestration
33    EventDriven,
34    /// Custom orchestration strategy
35    Custom(String),
36}
37
38impl OrchestrationStrategy {
39    #[must_use]
40    pub fn name(&self) -> String {
41        match self {
42            Self::Sequential => "sequential".to_string(),
43            Self::Parallel => "parallel".to_string(),
44            Self::Conditional => "conditional".to_string(),
45            Self::Pipeline => "pipeline".to_string(),
46            Self::EventDriven => "event_driven".to_string(),
47            Self::Custom(name) => name.clone(),
48        }
49    }
50}

476    /// Apply parameters to config
477    fn apply_parameters_to_config(
478        config: &mut OrchestratorAgentConfig,
479        params: &HashMap<String, serde_json::Value>,
480    ) {
481        if let Some(max_agents) = params.get("max_managed_agents") {
482            if let Some(value) = max_agents.as_u64() {
483                #[allow(clippy::cast_possible_truncation)]
484                let max_agents_usize = value as usize;
485                config.max_managed_agents = max_agents_usize;
486            }
487        }
488
489        if let Some(strategy) = params.get("orchestration_strategy") {
490            if let Some(value) = strategy.as_str() {
491                config.default_strategy = match value {
492                    "sequential" => OrchestrationStrategy::Sequential,
493                    "parallel" => OrchestrationStrategy::Parallel,
494                    "conditional" => OrchestrationStrategy::Conditional,
495                    "pipeline" => OrchestrationStrategy::Pipeline,
496                    "event_driven" => OrchestrationStrategy::EventDriven,
497                    _ => OrchestrationStrategy::Custom(value.to_string()),
498                };
499            }
500        }
501
502        if let Some(max_time) = params.get("max_workflow_time") {
503            if let Some(value) = max_time.as_u64() {
504                config.max_workflow_time = value;
505            }
506        }
507
508        if let Some(enable_monitoring) = params.get("enable_health_monitoring") {
509            if let Some(value) = enable_monitoring.as_bool() {
510                config.enable_health_monitoring = value;
511            }
512        }
513
514        if let Some(interval) = params.get("health_check_interval") {
515            if let Some(value) = interval.as_u64() {
516                config.health_check_interval = value;
517            }
518        }
519
520        if let Some(max_concurrent) = params.get("max_concurrent_workflows") {
521            if let Some(value) = max_concurrent.as_u64() {
522                #[allow(clippy::cast_possible_truncation)]
523                let max_concurrent_usize = value as usize;
524                config.max_concurrent_workflows = max_concurrent_usize;
525            }
526        }
527
528        if let Some(enable_rollback) = params.get("enable_rollback") {
529            if let Some(value) = enable_rollback.as_bool() {
530                config.enable_rollback = value;
531            }
532        }
533
534        if let Some(max_retries) = params.get("max_retries") {
535            if let Some(value) = max_retries.as_u64() {
536                #[allow(clippy::cast_possible_truncation)]
537                let max_retries_u32 = value as u32;
538                config.max_retries = max_retries_u32;
539            }
540        }
541
542        if let Some(templates_dir) = params.get("workflow_templates_dir") {
543            if let Some(value) = templates_dir.as_str() {
544                config.workflow_templates_dir = Some(value.to_string());
545            }
546        }
547    }

557impl AgentTemplate for OrchestratorAgentTemplate {
558    fn schema(&self) -> &TemplateSchema {
559        &self.schema
560    }

562    async fn instantiate(
563        &self,
564        mut params: TemplateInstantiationParams,
565    ) -> Result<TemplateInstantiationResult> {
566        // Apply defaults
567        self.apply_defaults(&mut params).await?;
568
569        // Validate parameters
570        self.validate_parameters(&params).await?;
571
572        // Create agent-specific configuration
573        let mut agent_config = self.config.clone();
574        Self::apply_parameters_to_config(&mut agent_config, &params.parameters);
575
576        // Build final configuration
577        let mut final_config = HashMap::new();
578        final_config.insert("agent_type".to_string(), "orchestrator_agent".into());
579        #[allow(clippy::cast_sign_loss)]
580        let max_managed_agents_u64 = agent_config.max_managed_agents as u64;
581        final_config.insert(
582            "max_managed_agents".to_string(),
583            max_managed_agents_u64.into(),
584        );
585        final_config.insert(
586            "orchestration_strategy".to_string(),
587            agent_config.default_strategy.name().into(),
588        );
589        final_config.insert(
590            "max_workflow_time".to_string(),
591            agent_config.max_workflow_time.into(),
592        );
593        final_config.insert(
594            "enable_health_monitoring".to_string(),
595            agent_config.enable_health_monitoring.into(),
596        );
597        final_config.insert(
598            "health_check_interval".to_string(),
599            agent_config.health_check_interval.into(),
600        );
601        #[allow(clippy::cast_sign_loss)]
602        let max_concurrent_workflows_u64 = agent_config.max_concurrent_workflows as u64;
603        final_config.insert(
604            "max_concurrent_workflows".to_string(),
605            max_concurrent_workflows_u64.into(),
606        );
607        final_config.insert(
608            "enable_rollback".to_string(),
609            agent_config.enable_rollback.into(),
610        );
611        final_config.insert(
612            "max_retries".to_string(),
613            u64::from(agent_config.max_retries).into(),
614        );
615
616        if let Some(templates_dir) = &agent_config.workflow_templates_dir {
617            final_config.insert(
618                "workflow_templates_dir".to_string(),
619                templates_dir.clone().into(),
620            );
621        }
622
623        // Add managed agent types if specified
624        if let Some(agent_types) = params.parameters.get("managed_agent_types") {
625            final_config.insert("managed_agent_types".to_string(), agent_types.clone());
626        }
627
628        // Apply config overrides
629        for (key, value) in params.config_overrides {
630            final_config.insert(key, value);
631        }
632
633        // Get agent name
634        let agent_name = params
635            .parameters
636            .get("agent_name")
637            .and_then(|v| v.as_str())
638            .unwrap_or(&params.agent_id)
639            .to_string();
640
641        // For now, return a mock result since we can't create actual BaseAgent instances
642        // In a real implementation, this would create the actual orchestrator agent
643        let mock_agent =
644            MockOrchestratorAgent::new(params.agent_id.clone(), agent_name, final_config.clone());
645
646        Ok(TemplateInstantiationResult {
647            agent: Box::new(mock_agent),
648            template_schema: self.schema.clone(),
649            applied_parameters: params.parameters,
650            applied_config: final_config,
651        })
652    }

655        Box::new(Self {
