10/// Capability aggregator that manages and matches capabilities

69/// Trait for scoring capabilities
70pub trait CapabilityScorer: Send + Sync {
71    /// Score a capability (0.0 to 1.0)
72    fn score(&self, capability: &Capability, stats: &CapabilityUsageStats) -> f64;
73}
74
75/// Default capability scorer based on usage statistics
76#[derive(Debug)]
77pub struct DefaultCapabilityScorer;
78
79impl CapabilityScorer for DefaultCapabilityScorer {
80    fn score(&self, _capability: &Capability, stats: &CapabilityUsageStats) -> f64 {
81        if stats.invocations == 0 {
82            return 0.5; // Neutral score for unused capabilities
83        }
84
85        #[allow(clippy::cast_precision_loss)]
86        let successes_f64 = stats.successes as f64;
87        #[allow(clippy::cast_precision_loss)]
88        let invocations_f64 = stats.invocations as f64;
89        let success_rate = successes_f64 / invocations_f64;
90        let recency_score = stats.last_invocation.map_or_else(
91            || 0.0,
92            |last| {
93                let hours_ago = (chrono::Utc::now() - last).num_hours();
94                #[allow(clippy::cast_precision_loss)]
95                let hours_ago_f64 = hours_ago as f64;
96                (1.0 / (1.0 + hours_ago_f64 / 24.0)).min(1.0)
97            },
98        );
99
100        // Weighted average of success rate and recency
101        success_rate
102            .mul_add(0.7, recency_score * 0.3)
103            .clamp(0.0, 1.0)
104    }
105}

137    /// Register a capability
138    ///
139    /// # Errors
140    ///
141    /// Returns an error if capability registration fails
142    ///
143    /// # Panics
144    ///
145    /// Panics if the `RwLock` is poisoned
146    pub fn register_capability(
147        &self,
148        capability: Capability,
149        provider_id: impl Into<String>,
150    ) -> Result<()> {
151        let provider_id = provider_id.into();
152        let capability_id = format!("{}::{}", provider_id, capability.name);
153
154        // Calculate initial score
155        let score = self
156            .scorer
157            .score(&capability, &CapabilityUsageStats::default());
158
159        let entry = CapabilityEntry {
160            capability: capability.clone(),
161            provider_id,
162            score,
163            available: true,
164            usage_stats: CapabilityUsageStats::default(),
165        };
166
167        // Add to main registry
168        let mut capabilities = self.capabilities.write().unwrap();
169        capabilities.insert(capability_id.clone(), entry);
170        drop(capabilities);
171
172        // Update category index
173        let mut index = self.category_index.write().unwrap();
174        index
175            .entry(capability.category)
176            .or_default()
177            .insert(capability_id);
178        drop(index);
179
180        Ok(())
181    }

183    /// Unregister a capability
184    ///
185    /// # Errors
186    ///
187    /// Returns an error if capability is not found
188    ///
189    /// # Panics
190    ///
191    /// Panics if the `RwLock` is poisoned
192    pub fn unregister_capability(&self, provider_id: &str, capability_name: &str) -> Result<()> {
193        let capability_id = format!("{provider_id}::{capability_name}");
194
195        let mut capabilities = self.capabilities.write().unwrap();
196        let entry_opt = capabilities.remove(&capability_id);
197        drop(capabilities);
198
199        if let Some(entry) = entry_opt {
200            // Remove from category index
201            let mut index = self.category_index.write().unwrap();
202            if let Some(set) = index.get_mut(&entry.capability.category) {
203                set.remove(&capability_id);
204            }
205            drop(index);
206        }
207
208        Ok(())
209    }

232    /// Find capabilities matching requirements
233    ///
234    /// # Panics
235    ///
236    /// Panics if the `RwLock` is poisoned
237    pub fn find_matches(&self) -> Vec<CapabilityMatch> {
238        let capabilities = self.capabilities.read().unwrap();
239        let capabilities_vec: Vec<_> = capabilities.values().cloned().collect();
240        drop(capabilities);
241
242        let requirements = self.requirements.read().unwrap();
243        let requirements_vec: Vec<_> = requirements.clone();
244        drop(requirements);
245
246        let mut matches = Vec::new();
247
248        for entry in &capabilities_vec {
249            if !entry.available {
250                continue;
251            }
252
253            let mut satisfied = Vec::new();
254            let mut total_score = 0.0;
255            let mut requirement_count = 0;
256
257            for (idx, req) in requirements_vec.iter().enumerate() {
258                if self.matches_requirement(&entry.capability, req) {
259                    satisfied.push(format!("req-{idx}"));
260                    total_score += entry.score;
261                    requirement_count += 1;
262                }
263            }
264
265            if !satisfied.is_empty() {
266                let avg_score = if requirement_count > 0 {
267                    total_score / f64::from(requirement_count)
268                } else {
269                    entry.score
270                };
271
272                matches.push(CapabilityMatch {
273                    capability: entry.capability.clone(),
274                    provider_id: entry.provider_id.clone(),
275                    score: avg_score,
276                    satisfied_requirements: satisfied,
277                });
278            }
279        }
280
281        // Sort by score descending
282        matches.sort_by(|a, b| b.score.partial_cmp(&a.score).unwrap());
283        matches
284    }

286    /// Check if a capability matches a requirement
287    ///
288    /// # Panics
289    ///
290    /// Panics if the `RwLock` is poisoned
291    fn matches_requirement(
292        &self,
293        capability: &Capability,
294        requirement: &CapabilityRequirement,
295    ) -> bool {
296        // Check name pattern
297        if !Self::matches_pattern(&capability.name, &requirement.name_pattern) {
298            return false;
299        }
300
301        // Check category
302        if requirement
303            .category
304            .as_ref()
305            .is_some_and(|req_category| &capability.category != req_category)
306        {
307            return false;
308        }
309
310        // Check version (simplified)
311        if let Some(ref min_version) = requirement.min_version {
312            // This map_or is correct: if capability has no version (None), we consider it
313            // as not meeting the requirement (true = fails check). If it has a version,
314            // we check if it's less than the minimum required version.
315            #[allow(clippy::unnecessary_map_or)]
316            if capability
317                .version
318                .as_ref()
319                .map_or(true, |cap_version| cap_version < min_version)
320            {
321                return false;
322            }
323        }
324
325        // Check metadata
326        for (key, value) in &requirement.required_metadata {
327            if capability.metadata.get(key) != Some(value) {
328                return false;
329            }
330        }
331
332        // Check score
333        if let Some(min_score) = requirement.min_score {
334            let cap_id = format!("{}::{}", "unknown", capability.name);
335            if self
336                .capabilities
337                .read()
338                .unwrap()
339                .get(&cap_id)
340                .is_some_and(|entry| entry.score < min_score)
341            {
342                return false;
343            }
344        }
345
346        true
347    }

349    /// Simple pattern matching (supports * wildcard)
350    fn matches_pattern(text: &str, pattern: &str) -> bool {
351        if pattern == "*" {
352            return true;
353        }
354
355        if pattern.contains('*') {
356            // Simple wildcard matching
357            let parts: Vec<&str> = pattern.split('*').collect();
358            if parts.is_empty() {
359                return true;
360            }
361
362            let mut text_pos = 0;
363            for (i, part) in parts.iter().enumerate() {
364                if part.is_empty() {
365                    continue;
366                }
367
368                if i == 0 && !text.starts_with(part) {
369                    return false;
370                }
371
372                if i == parts.len() - 1 && !pattern.ends_with('*') && !text.ends_with(part) {
373                    return false;
374                }
375
376                match text[text_pos..].find(part) {
377                    Some(pos) => {
378                        text_pos += pos + part.len();
379                    }
380                    None => return false,
381                }
382            }
383            true
384        } else {
385            text == pattern
386        }
387    }

389    /// Update usage statistics for a capability
390    ///
391    /// # Errors
392    ///
393    /// Returns an error if capability is not found
394    ///
395    /// # Panics
396    ///
397    /// Panics if the `RwLock` is poisoned
398    pub fn update_usage(
399        &self,
400        provider_id: &str,
401        capability_name: &str,
402        success: bool,
403        duration: std::time::Duration,
404    ) -> Result<()> {
405        let capability_id = format!("{provider_id}::{capability_name}");
406
407        let mut capabilities = self.capabilities.write().unwrap();
408        capabilities.get_mut(&capability_id).map_or_else(
409            || {
410                Err(LLMSpellError::Component {
411                    message: format!("Capability not found: {capability_id}"),
412                    source: None,
413                })
414            },
415            |entry| {
416                let stats = &mut entry.usage_stats;
417                stats.invocations += 1;
418                if success {
419                    stats.successes += 1;
420                } else {
421                    stats.failures += 1;
422                }
423
424                // Update average execution time
425                let total_time =
426                    stats.avg_execution_time.as_secs() * stats.invocations + duration.as_secs();
427                stats.avg_execution_time =
428                    std::time::Duration::from_secs(total_time / (stats.invocations + 1));
429
430                stats.last_invocation = Some(chrono::Utc::now());
431
432                // Recalculate score
433                entry.score = self.scorer.score(&entry.capability, stats);
434
435                Ok(())
436            },
437        )
438    }

504    /// Get statistics for all capabilities
505    ///
506    /// # Panics
507    ///
508    /// Panics if the `RwLock` is poisoned
509    pub fn get_statistics(&self) -> CapabilityStatistics {
510        let capabilities = self.capabilities.read().unwrap();
511        let total = capabilities.len();
512        let available = capabilities.values().filter(|e| e.available).count();
513
514        let mut by_category = HashMap::new();
515        for entry in capabilities.values() {
516            *by_category
517                .entry(entry.capability.category.clone())
518                .or_insert(0) += 1;
519        }
520
521        let avg_score = if total > 0 {
522            #[allow(clippy::cast_precision_loss)]
523            let total_f64 = total as f64;
524            capabilities.values().map(|e| e.score).sum::<f64>() / total_f64
525        } else {
526            0.0
527        };
528
529        CapabilityStatistics {
530            total_capabilities: total,
531            available_capabilities: available,
532            capabilities_by_category: by_category,
533            average_score: avg_score,
534        }
535    }
