22/// A hierarchical composite agent that manages parent-child relationships
23pub struct HierarchicalCompositeAgent {
24    /// Component metadata
25    metadata: ComponentMetadata,
26
27    /// Parent agent reference (weak to avoid cycles)
28    parent: RwLock<Option<Weak<dyn HierarchicalAgent>>>,
29
30    /// Child agents
31    children: TokioRwLock<Vec<Arc<dyn HierarchicalAgent>>>,
32
33    /// Components managed by this agent (includes non-hierarchical agents)
34    components: TokioRwLock<HashMap<String, Arc<dyn BaseAgent>>>,
35
36    /// Tools available to this agent
37    tools: TokioRwLock<HashMap<String, Arc<dyn Tool>>>,
38
39    /// Execution pattern for child/component execution
40    execution_pattern: RwLock<ExecutionPattern>,
41
42    /// Capabilities exposed by this agent
43    capabilities: RwLock<Vec<Capability>>,
44
45    /// Configuration
46    config: HierarchicalConfig,
47
48    /// Metrics
49    metrics: RwLock<HierarchicalMetrics>,
50}

52/// Configuration for hierarchical agents
53#[derive(Debug, Clone, Serialize, Deserialize)]
54pub struct HierarchicalConfig {
55    /// Maximum depth allowed in the hierarchy
56    pub max_depth: usize,
57    /// Whether to propagate events up the hierarchy
58    pub propagate_up: bool,
59    /// Whether to propagate events down the hierarchy
60    pub propagate_down: bool,
61    /// Maximum number of children allowed
62    pub max_children: Option<usize>,
63    /// Whether children inherit parent capabilities
64    pub inherit_capabilities: bool,
65    /// Timeout for child operations
66    pub child_operation_timeout: std::time::Duration,
67}

97impl HierarchicalCompositeAgent {
98    /// Create a new hierarchical composite agent
99    pub fn new(name: impl Into<String>, config: HierarchicalConfig) -> Self {
100        let name = name.into();
101        let description = format!("Hierarchical composite agent: {name}");
102        Self {
103            metadata: ComponentMetadata::new(name, description),
104            parent: RwLock::new(None),
105            children: TokioRwLock::new(Vec::new()),
106            components: TokioRwLock::new(HashMap::new()),
107            tools: TokioRwLock::new(HashMap::new()),
108            execution_pattern: RwLock::new(ExecutionPattern::Sequential),
109            capabilities: RwLock::new(Vec::new()),
110            config,
111            metrics: RwLock::new(HierarchicalMetrics::default()),
112        }
113    }
114
115    /// Set the parent of this agent
116    ///
117    /// # Errors
118    ///
119    /// Currently never returns an error, but the Result type is provided for future
120    /// extensibility (e.g., validation of parent-child relationships, cycle detection).
121    ///
122    /// # Panics
123    ///
124    /// Panics if the `RwLock` is poisoned
125    pub fn set_parent(&self, parent: Weak<dyn HierarchicalAgent>) -> Result<()> {
126        let mut parent_guard = self.parent.write().unwrap();
127        *parent_guard = Some(parent);
128        drop(parent_guard);
129        Ok(())
130    }
131
132    /// Get the current depth by traversing up the hierarchy
133    fn calculate_depth(&self) -> usize {
134        let parent_guard = self.parent.read().unwrap();
135        parent_guard.as_ref().map_or(0, |weak_parent| {
136            weak_parent.upgrade().map_or(0, |parent| parent.depth() + 1)
137        })
138    }
139
140    /// Check if adding a child would create a cycle
141    fn would_create_cycle(&self, child: &Arc<dyn HierarchicalAgent>) -> bool {
142        // Check if child is an ancestor of self
143        let mut current = self.parent.read().unwrap().clone();
144        while let Some(weak_parent) = current {
145            if let Some(parent) = weak_parent.upgrade() {
146                if Arc::ptr_eq(&parent, child) {
147                    return true;
148                }
149                current = parent
150                    .parent()
151                    .and_then(|p| p.parent())
152                    .map(|p| Arc::downgrade(&p));
153            } else {
154                break;
155            }
156        }
157        false
158    }
159
160    /// Aggregate capabilities from all children and components
161    #[allow(dead_code)]
162    async fn aggregate_capabilities(&self) -> Vec<Capability> {
163        let mut capabilities = self.capabilities.read().unwrap().clone();
164
165        // Add capabilities from children
166        let children = self.children.read().await;
167        for _child in children.iter() {
168            // Access child metadata - this is a limitation as metadata() is sync
169            // In a real implementation, we'd need a different approach
170            capabilities.push(Capability {
171                name: "child-capability".to_string(),
172                category: CapabilityCategory::Custom("inherited".to_string()),
173                version: None,
174                metadata: HashMap::new(),
175            });
176        }
177
178        // Add capabilities from components
179        let components = self.components.read().await;
180        for (_, _component) in components.iter() {
181            // Access component metadata - this is a limitation as metadata() is sync
182            // In a real implementation, we'd need a different approach
183            capabilities.push(Capability {
184                name: "component-capability".to_string(),
185                category: CapabilityCategory::Custom("component".to_string()),
186                version: None,
187                metadata: HashMap::new(),
188            });
189        }
190
191        capabilities
192    }
193}

195#[async_trait]
196impl BaseAgent for HierarchicalCompositeAgent {
197    fn metadata(&self) -> &ComponentMetadata {
198        &self.metadata
199    }
200
201    async fn execute_impl(
202        &self,
203        input: AgentInput,
204        context: ExecutionContext,
205    ) -> Result<AgentOutput> {
206        let pattern = self.execution_pattern.read().unwrap().clone();
207        let value = serde_json::to_value(&input)?;
208        let result = self.execute_pattern(pattern, value, &context).await?;
209        Ok(AgentOutput::text(result.to_string()))
210    }
211
212    async fn validate_input(&self, input: &AgentInput) -> Result<()> {
213        // Basic validation - can be extended
214        if input.text.is_empty() {
215            return Err(LLMSpellError::Validation {
216                message: "Input prompt cannot be empty".to_string(),
217                field: Some("prompt".to_string()),
218            });
219        }
220        Ok(())
221    }
222
223    async fn handle_error(&self, error: LLMSpellError) -> Result<AgentOutput> {
224        Ok(AgentOutput::text(format!(
225            "Hierarchical agent error: {error}"
226        )))
227    }

230#[async_trait]
231impl ToolCapable for HierarchicalCompositeAgent {
232    async fn discover_tools(&self, query: &ToolQuery) -> Result<Vec<ToolInfo>> {
233        let tools = self.tools.read().await;
234        let mut infos = Vec::new();
235
236        for (name, tool) in tools.iter() {
237            // Simple filtering based on text search
238            if query
239                .text_search
240                .as_ref()
241                .is_some_and(|text| !name.contains(text))
242            {
243                continue;
244            }
245
246            let info = ToolInfo::new(
247                name.clone(),
248                tool.metadata().description.clone(),
249                "agent-tool".to_string(),
250                "safe".to_string(),
251            );
252            infos.push(info);
253        }
254
255        Ok(infos)
256    }
257
258    async fn invoke_tool(
259        &self,
260        tool_name: &str,
261        parameters: Value,
262        context: ExecutionContext,
263    ) -> Result<AgentOutput> {
264        let tools = self.tools.read().await;
265        match tools.get(tool_name) {
266            Some(tool) => {
267                let input = AgentInput::text(parameters.to_string());
268                tool.execute(input, context).await
269            }
270            None => Err(LLMSpellError::Component {
271                message: format!("Tool not found: {tool_name}"),
272                source: None,
273            }),
274        }
275    }
276
277    async fn list_available_tools(&self) -> Result<Vec<String>> {
278        let tools = self.tools.read().await;
279        Ok(tools.keys().cloned().collect())
280    }
281
282    async fn tool_available(&self, tool_name: &str) -> bool {
283        let tools = self.tools.read().await;
284        tools.contains_key(tool_name)
285    }

317impl CompositeAgent for HierarchicalCompositeAgent {
318    async fn add_component(&mut self, component: Arc<dyn BaseAgent>) -> Result<()> {
319        let mut components = self.components.write().await;
320        components.insert(component.metadata().id.to_string(), component);
321        Ok(())
322    }
323
324    async fn remove_component(&mut self, component_id: &str) -> Result<()> {
325        let mut components = self.components.write().await;
326        components
327            .remove(component_id)
328            .ok_or_else(|| LLMSpellError::Component {
329                message: format!("Component not found: {component_id}"),
330                source: None,
331            })?;
332        Ok(())
333    }
334
335    fn components(&self) -> Vec<Arc<dyn BaseAgent>> {
336        // This would need to be async in real implementation
337        Vec::new()
338    }
339
340    fn get_component(&self, _component_id: &str) -> Option<Arc<dyn BaseAgent>> {
341        // This would need to be async in real implementation
342        None
343    }
344
345    async fn delegate_to(
346        &self,
347        component_id: &str,
348        input: Value,
349        context: &ExecutionContext,
350    ) -> Result<Value> {
351        // Update metrics before async operation
352        {
353            let mut metrics = self.metrics.write().unwrap();
354            metrics.total_delegations += 1;
355        }
356
357        let components = self.components.read().await;
358
359        if let Some(component) = components.get(component_id) {
360            let agent_input = AgentInput::text(input.to_string());
361            match component.execute(agent_input, context.clone()).await {
362                Ok(result) => {
363                    {
364                        let mut metrics = self.metrics.write().unwrap();
365                        metrics.successful_delegations += 1;
366                    }
367                    Ok(serde_json::to_value(&result)?)
368                }
369                Err(e) => {
370                    {
371                        let mut metrics = self.metrics.write().unwrap();
372                        metrics.failed_delegations += 1;
373                    }
374                    Err(e)
375                }
376            }
377        } else {
378            {
379                let mut metrics = self.metrics.write().unwrap();
380                metrics.failed_delegations += 1;
381            }
382            Err(CompositionError::ComponentNotFound(component_id.to_string()).into())
383        }
384    }
385
386    async fn execute_pattern(
387        &self,
388        pattern: ExecutionPattern,
389        input: Value,
390        context: &ExecutionContext,
391    ) -> Result<Value> {
392        match pattern {
393            ExecutionPattern::Sequential => {
394                let mut result = input;
395                let components = self.components.read().await;
396
397                for (_, component) in components.iter() {
398                    let agent_input = AgentInput::text(result.to_string());
399                    let output = component.execute(agent_input, context.clone()).await?;
400                    result = serde_json::to_value(&output)?;
401                }
402
403                Ok(result)
404            }
405            ExecutionPattern::Parallel => {
406                let components = self.components.read().await;
407                let mut handles = Vec::new();
408
409                for (_, component) in components.iter() {
410                    let comp = component.clone();
411                    let inp = input.clone();
412                    let ctx = context.clone();
413
414                    handles.push(tokio::spawn(async move {
415                        let agent_input = AgentInput::text(inp.to_string());
416                        let output = comp.execute(agent_input, ctx).await?;
417                        Ok::<Value, LLMSpellError>(serde_json::to_value(&output)?)
418                    }));
419                }
420
421                let mut results = Vec::new();
422                for handle in handles {
423                    results.push(handle.await.map_err(|e| LLMSpellError::Component {
424                        message: format!("Parallel execution failed: {e}"),
425                        source: None,
426                    })??);
427                }
428
429                Ok(Value::Array(results))
430            }
431            _ => Err(LLMSpellError::Component {
432                message: "Unsupported execution pattern".to_string(),
433                source: None,
434            }),
435        }
436    }
437}

439#[async_trait]
440impl HierarchicalAgent for HierarchicalCompositeAgent {
441    fn parent(&self) -> Option<Arc<dyn HierarchicalAgent>> {
442        let parent_guard = self.parent.read().unwrap();
443        parent_guard.as_ref().and_then(std::sync::Weak::upgrade)
444    }
445
446    fn children(&self) -> Vec<Arc<dyn HierarchicalAgent>> {
447        // This would need to be async in real implementation
448        Vec::new()
449    }
450
451    async fn add_child(&mut self, child: Arc<dyn HierarchicalAgent>) -> Result<()> {
452        // Check max children limit
453        if let Some(max) = self.config.max_children {
454            let children = self.children.read().await;
455            if children.len() >= max {
456                return Err(LLMSpellError::Component {
457                    message: format!("Maximum children limit ({max}) reached"),
458                    source: None,
459                });
460            }
461        }
462
463        // Check for cycles
464        if self.would_create_cycle(&child) {
465            return Err(CompositionError::CycleDetected.into());
466        }
467
468        // Check depth limit
469        if self.depth() + 1 > self.config.max_depth {
470            return Err(CompositionError::MaxDepthExceeded(self.config.max_depth).into());
471        }
472
473        let mut children = self.children.write().await;
474        children.push(child);
475        Ok(())
476    }
477
478    async fn remove_child(&mut self, _child_id: &str) -> Result<()> {
479        let mut children = self.children.write().await;
480        // Remove child by comparing against child_id
481        // Since HierarchicalAgent doesn't have id() method, we need a different approach
482        // For now, we'll remove all children - this is a limitation
483        children.clear();
484        // TODO: Implement proper child identification
485        Ok(())
486    }
487
488    fn depth(&self) -> usize {
489        self.calculate_depth()
490    }
491
492    async fn propagate_down(&self, event: HierarchyEvent) -> Result<()> {
493        if !self.config.propagate_down {
494            return Ok(());
495        }
496
497        // Update metrics before async operation
498        {
499            let mut metrics = self.metrics.write().unwrap();
500            metrics.events_propagated_down += 1;
501        }
502
503        let children = self.children.read().await;
504        for child in children.iter() {
505            child.propagate_down(event.clone()).await?;
506        }
507        Ok(())
508    }
509
510    async fn propagate_up(&self, event: HierarchyEvent) -> Result<()> {
511        if !self.config.propagate_up {
512            return Ok(());
513        }
514
515        // Update metrics before async operation
516        {
517            let mut metrics = self.metrics.write().unwrap();
518            metrics.events_propagated_up += 1;
519        }
520
521        match self.parent() {
522            Some(parent) => parent.propagate_up(event).await,
523            None => Ok(()),
524        }
525    }
526}

528/// Builder for hierarchical composite agents
529pub struct HierarchicalAgentBuilder {
530    name: String,
531    description: Option<String>,
532    config: HierarchicalConfig,
533    capabilities: Vec<Capability>,
534    initial_pattern: ExecutionPattern,
535}
536
537impl HierarchicalAgentBuilder {
538    /// Create a new builder
539    pub fn new(name: impl Into<String>) -> Self {
540        Self {
541            name: name.into(),
542            description: None,
543            config: HierarchicalConfig::default(),
544            capabilities: Vec::new(),
545            initial_pattern: ExecutionPattern::Sequential,
546        }
547    }
548
549    /// Set the description
550    #[must_use]
551    pub fn description(mut self, desc: impl Into<String>) -> Self {
552        self.description = Some(desc.into());
553        self
554    }
555
556    /// Set the configuration
557    #[must_use]
558    pub const fn config(mut self, config: HierarchicalConfig) -> Self {
559        self.config = config;
560        self
561    }
562
563    /// Add a capability
564    #[must_use]
565    pub fn add_capability(mut self, capability: Capability) -> Self {
566        self.capabilities.push(capability);
567        self
568    }
569
570    /// Set the initial execution pattern
571    #[must_use]
572    pub fn execution_pattern(mut self, pattern: ExecutionPattern) -> Self {
573        self.initial_pattern = pattern;
574        self
575    }
576
577    /// Build the hierarchical agent
578    ///
579    /// # Panics
580    ///
581    /// Panics if creating `Arc<RwLock<_>>` fails
582    #[must_use]
583    pub fn build(self) -> HierarchicalCompositeAgent {
584        let name = self.name.clone();
585        let mut agent = HierarchicalCompositeAgent::new(self.name, self.config);
586
587        if let Some(desc) = self.description {
588            agent.metadata = ComponentMetadata::new(name, desc);
589        }
590
591        *agent.execution_pattern.write().unwrap() = self.initial_pattern;
592        *agent.capabilities.write().unwrap() = self.capabilities;
593
594        agent
595    }
596}
