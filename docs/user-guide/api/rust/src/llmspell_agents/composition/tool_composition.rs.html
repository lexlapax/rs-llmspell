9/// A composition of tools that can be executed as a workflow
10///
11/// This allows multiple tools to be chained together, with output from
12/// one tool feeding into the input of subsequent tools.
13///
14/// # Examples
15///
16/// ```
17/// use llmspell_agents::composition::tool_composition::{ToolComposition, CompositionStep, DataFlow};
18/// use serde_json::json;
19///
20/// # async fn example() -> llmspell_core::Result<()> {
21/// let mut composition = ToolComposition::new("file-processing-pipeline");
22///
23/// // Add steps to the composition
24/// composition.add_step(CompositionStep::new("step1", "file_reader")
25///     .with_input_mapping("file_path", DataFlow::Parameter("input_file".to_string())));
26///
27/// composition.add_step(CompositionStep::new("step2", "text_processor")
28///     .with_input_mapping("text", DataFlow::StepOutput("step1".to_string(), "content".to_string())));
29///
30/// composition.add_step(CompositionStep::new("step3", "file_writer")
31///     .with_input_mapping("content", DataFlow::StepOutput("step2".to_string(), "processed_text".to_string()))
32///     .with_input_mapping("output_path", DataFlow::Parameter("output_file".to_string())));
33///
34/// // Execute the composition
35/// let initial_params = json!({"input_file": "/path/to/input.txt", "output_file": "/path/to/output.txt"});
36/// // let result = composition.execute(tool_registry, initial_params, context).await?;
37/// # Ok(())
38/// # }
39/// ```
40#[derive(Debug, Clone)]
41pub struct ToolComposition {
42    /// Unique identifier for this composition
43    pub id: String,
44    /// Human-readable name
45    pub name: String,
46    /// Description of what this composition does
47    pub description: String,
48    /// Steps in the composition
49    pub steps: Vec<CompositionStep>,
50    /// Global error handling strategy
51    pub error_strategy: CompositionErrorStrategy,
52    /// Maximum execution time for the entire composition
53    pub max_execution_time: Option<Duration>,
54    /// Whether to run steps in parallel where possible
55    pub parallel_execution: bool,
56    /// Shared data that persists across all steps
57    pub shared_context: HashMap<String, JsonValue>,
58}

133pub enum CompositionErrorStrategy {

317    /// Execute the composition with the given tool provider
318    ///
319    /// # Errors
320    ///
321    /// Returns an error if tool execution fails
322    ///
323    /// # Panics
324    ///
325    /// Panics if a `RwLock` is poisoned
326    #[allow(clippy::future_not_send)] // Tool provider may not be Send
327    pub async fn execute<T>(
328        &self,
329        tool_provider: &T,
330        initial_parameters: JsonValue,
331        context: ExecutionContext,
332    ) -> Result<CompositionResult>
333    where
334        T: ToolProvider,
335    {
336        let start_time = Instant::now();
337        let mut step_results = HashMap::new();
338        let mut errors = Vec::new();
339        let mut execution_context =
340            CompositionExecutionContext::new(&initial_parameters, &self.shared_context);
341
342        let mut steps_executed = 0;
343        let mut steps_failed = 0;
344        let mut total_retries = 0;
345
346        for step in &self.steps {
347            // Check if we should continue based on error strategy
348            if self.should_stop_execution(&errors) {
349                break;
350            }
351
352            // Check step conditions
353            if !self.check_step_conditions(step, &execution_context)? {
354                continue;
355            }
356
357            let step_start_time = Instant::now();
358            let mut retry_attempts = 0;
359
360            let step_result =
361                loop {
362                    steps_executed += 1;
363
364                    // Prepare step input
365                    let step_input = self.prepare_step_input(step, &execution_context)?;
366
367                    // Execute the step
368                    match self
369                        .execute_step(tool_provider, step, step_input, context.clone())
370                        .await
371                    {
372                        Ok(output) => {
373                            let result = Self::handle_step_success(
374                                &step.id,
375                                output.clone(),
376                                step_start_time.elapsed(),
377                                retry_attempts,
378                            );
379                            execution_context.set_step_output(&step.id, output);
380                            break result;
381                        }
382                        Err(e) => {
383                            steps_failed += 1;
384
385                            // Check if we should retry
386                            if step.retry_config.as_ref().is_some_and(|retry_config| {
387                                retry_attempts < retry_config.max_attempts
388                            }) {
389                                retry_attempts += 1;
390                                total_retries += 1;
391
392                                let delay = Self::calculate_retry_delay(
393                                    step.retry_config.as_ref().unwrap(),
394                                    retry_attempts,
395                                );
396                                tokio::time::sleep(delay).await;
397                                continue;
398                            }
399
400                            let result = self.handle_step_error_result(
401                                step,
402                                &e,
403                                step_start_time.elapsed(),
404                                retry_attempts,
405                                &mut errors,
406                            );
407                            break result;
408                        }
409                    }
410                };
411
412            step_results.insert(step.id.clone(), step_result);
413        }
414
415        // Determine final output
416        let output = self.steps.last().map_or_else(
417            || JsonValue::Null,
418            |last_step| {
419                step_results
420                    .get(&last_step.id)
421                    .map_or(JsonValue::Null, |r| r.output.clone())
422            },
423        );
424
425        let metrics = CompositionMetrics {
426            total_execution_time: start_time.elapsed(),
427            steps_executed,
428            steps_failed,
429            total_retries,
430        };
431
432        Ok(CompositionResult {
433            success: errors.iter().all(|e| !e.fatal),
434            step_results,
435            output,
436            errors,
437            metrics,
438        })
439    }

441    /// Check if execution should stop based on error strategy
442    fn should_stop_execution(&self, errors: &[CompositionError]) -> bool {
443        match self.error_strategy {
444            CompositionErrorStrategy::FailFast => errors.iter().any(|e| e.fatal),
445            _ => false,
446        }
447    }

449    /// Check if step conditions are met
450    fn check_step_conditions(
451        &self,
452        step: &CompositionStep,
453        context: &CompositionExecutionContext,
454    ) -> Result<bool> {
455        for condition in &step.conditions {
456            let field_value = self.resolve_data_flow(&condition.field, context)?;
457
458            let condition_met = match condition.condition_type {
459                ConditionType::Equals => field_value == condition.value,
460                ConditionType::NotEquals => field_value != condition.value,
461                ConditionType::Exists => !field_value.is_null(),
462                ConditionType::NotExists => field_value.is_null(),
463                ConditionType::Matches => {
464                    if let (JsonValue::String(field_str), JsonValue::String(pattern)) =
465                        (&field_value, &condition.value)
466                    {
467                        // Simple pattern matching - could be enhanced with regex
468                        field_str.contains(pattern)
469                    } else {
470                        false
471                    }
472                }
473                ConditionType::GreaterThan => match (&field_value, &condition.value) {
474                    (JsonValue::Number(a), JsonValue::Number(b)) => {
475                        a.as_f64().unwrap_or(0.0) > b.as_f64().unwrap_or(0.0)
476                    }
477                    _ => false,
478                },
479                ConditionType::LessThan => match (&field_value, &condition.value) {
480                    (JsonValue::Number(a), JsonValue::Number(b)) => {
481                        a.as_f64().unwrap_or(0.0) < b.as_f64().unwrap_or(0.0)
482                    }
483                    _ => false,
484                },
485            };
486
487            if !condition_met {
488                return Ok(false);
489            }
490        }
491
492        Ok(true)
493    }

495    /// Prepare input for a step
496    fn prepare_step_input(
497        &self,
498        step: &CompositionStep,
499        context: &CompositionExecutionContext,
500    ) -> Result<JsonValue> {
501        let mut input_params = Map::new();
502
503        for (param_name, data_flow) in &step.input_mappings {
504            let value = self.resolve_data_flow(data_flow, context)?;
505            input_params.insert(param_name.clone(), value);
506        }
507
508        Ok(JsonValue::Object(input_params))
509    }

511    /// Resolve a data flow to a concrete value
512    #[allow(clippy::only_used_in_recursion)]
513    fn resolve_data_flow(
514        &self,
515        data_flow: &DataFlow,
516        context: &CompositionExecutionContext,
517    ) -> Result<JsonValue> {
518        match data_flow {
519            DataFlow::Parameter(param_name) => {
520                Ok(context.get_parameter(param_name).unwrap_or(JsonValue::Null))
521            }
522            DataFlow::StepOutput(step_id, field_name) => {
523                context
524                    .get_step_output(step_id)
525                    .map_or(Ok(JsonValue::Null), |step_output| {
526                        if field_name == "*" {
527                            Ok(step_output.clone())
528                        } else if let JsonValue::Object(obj) = step_output {
529                            Ok(obj.get(field_name).cloned().unwrap_or(JsonValue::Null))
530                        } else {
531                            Ok(JsonValue::Null)
532                        }
533                    })
534            }
535            DataFlow::Constant(value) => Ok(value.clone()),
536            DataFlow::SharedContext(key) => {
537                Ok(context.get_shared_context(key).unwrap_or(JsonValue::Null))
538            }
539            DataFlow::Transform { source, transform } => {
540                let source_value = self.resolve_data_flow(source, context)?;
541                Self::apply_data_transform(&source_value, transform)
542            }
543        }
544    }

546    /// Apply a data transformation
547    fn apply_data_transform(value: &JsonValue, transform: &DataTransform) -> Result<JsonValue> {
548        match transform {
549            DataTransform::ExtractField(field_name) => {
550                if let JsonValue::Object(obj) = value {
551                    Ok(obj.get(field_name).cloned().unwrap_or(JsonValue::Null))
552                } else {
553                    Ok(JsonValue::Null)
554                }
555            }
556            DataTransform::JsonPath(_path) => {
557                // Simple JSON path implementation would go here
558                // For now, just return the original value
559                Ok(value.clone())
560            }
561            DataTransform::ToString => {
562                let string_value = match value {
563                    JsonValue::String(s) => s.clone(),
564                    JsonValue::Number(n) => n.to_string(),
565                    JsonValue::Bool(b) => b.to_string(),
566                    JsonValue::Null => "null".to_string(),
567                    _ => serde_json::to_string(value).map_err(|e| LLMSpellError::Component {
568                        message: format!("Failed to convert to string: {e}"),
569                        source: Some(Box::new(e)),
570                    })?,
571                };
572                Ok(JsonValue::String(string_value))
573            }
574            DataTransform::ToNumber => match value {
575                JsonValue::Number(n) => Ok(JsonValue::Number(n.clone())),
576                JsonValue::String(s) => Ok(JsonValue::Number(
577                    s.parse::<f64>()
578                        .ok()
579                        .and_then(serde_json::Number::from_f64)
580                        .unwrap_or_else(|| serde_json::Number::from(0)),
581                )),
582                _ => Ok(JsonValue::Number(serde_json::Number::from(0))),
583            },
584            DataTransform::Custom(function_name) => {
585                // Custom transformations would be implemented here
586                tracing::warn!("Custom transform '{}' not implemented", function_name);
587                Ok(value.clone())
588            }
589        }
590    }

631    /// Helper to handle step error result
632    fn handle_step_error_result(
633        &self,
634        step: &CompositionStep,
635        error: &LLMSpellError,
636        execution_time: Duration,
637        retry_attempts: u32,
638        errors: &mut Vec<CompositionError>,
639    ) -> StepResult {
640        let error_msg = error.to_string();
641        let fatal = self.handle_step_error(step, error);
642
643        errors.push(CompositionError {
644            step_id: step.id.clone(),
645            message: error_msg.clone(),
646            fatal,
647        });
648
649        let step_metrics = StepMetrics {
650            execution_time,
651            retry_attempts,
652            memory_used: None,
653        };
654
655        StepResult {
656            success: false,
657            output: JsonValue::Null,
658            error: Some(error_msg),
659            metrics: step_metrics,
660        }
661    }

663    /// Handle step error based on strategy
664    const fn handle_step_error(&self, step: &CompositionStep, _error: &LLMSpellError) -> bool {
665        match step.error_strategy {
666            StepErrorStrategy::Inherit => {
667                matches!(self.error_strategy, CompositionErrorStrategy::FailFast)
668            }
669            StepErrorStrategy::Stop => true,
670            StepErrorStrategy::Continue
671            | StepErrorStrategy::Skip
672            | StepErrorStrategy::Retry { .. } => false,
673        }
674    }
675
676    /// Calculate retry delay
677    fn calculate_retry_delay(retry_config: &RetryConfig, attempt: u32) -> Duration {
678        let base_delay = retry_config.base_delay;
679
680        if retry_config.exponential_backoff {
681            let multiplier = 2_u32.pow(attempt.saturating_sub(1));
682            let delay = base_delay * multiplier;
683
684            retry_config
685                .max_delay
686                .map_or(delay, |max_delay| delay.min(max_delay))
687        } else {
688            base_delay
689        }
690    }

769    fn get_parameter(&self, name: &str) -> Option<JsonValue> {
770        self.parameters.get(name).cloned()
771    }
772
773    fn get_step_output(&self, step_id: &str) -> Option<&JsonValue> {
774        self.step_outputs.get(step_id)
775    }
776
777    fn set_step_output(&mut self, step_id: &str, output: JsonValue) {
778        self.step_outputs.insert(step_id.to_string(), output);
779    }
780
781    fn get_shared_context(&self, key: &str) -> Option<JsonValue> {
782        self.shared_context.get(key).cloned()
783    }
