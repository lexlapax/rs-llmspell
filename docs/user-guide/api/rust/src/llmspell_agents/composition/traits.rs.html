10/// Trait for components that can be composed into higher-level structures
11#[async_trait]
12pub trait Composable: BaseAgent + Send + Sync {
13    /// Get the composition metadata for this component
14    fn composition_metadata(&self) -> CompositionMetadata;
15
16    /// Check if this component can be composed with another
17    fn can_compose_with(&self, other: &dyn Composable) -> bool;
18
19    /// Get the exposed capabilities of this component
20    fn exposed_capabilities(&self) -> Vec<Capability>;
21
22    /// Get the required capabilities this component needs
23    fn required_capabilities(&self) -> Vec<Capability>;
24}

26/// Trait for agents that compose other agents
27#[async_trait]
28pub trait CompositeAgent: BaseAgent + ToolCapable + Send + Sync {
29    /// Add a child agent to this composite
30    async fn add_component(&mut self, component: Arc<dyn BaseAgent>) -> Result<()>;
31
32    /// Remove a child agent by ID
33    async fn remove_component(&mut self, component_id: &str) -> Result<()>;
34
35    /// Get all child components
36    fn components(&self) -> Vec<Arc<dyn BaseAgent>>;
37
38    /// Get a specific component by ID
39    fn get_component(&self, component_id: &str) -> Option<Arc<dyn BaseAgent>>;
40
41    /// Delegate execution to a specific component
42    async fn delegate_to(
43        &self,
44        component_id: &str,
45        input: Value,
46        context: &ExecutionContext,
47    ) -> Result<Value>;
48
49    /// Execute all components in a specific pattern
50    async fn execute_pattern(
51        &self,
52        pattern: ExecutionPattern,
53        input: Value,
54        context: &ExecutionContext,
55    ) -> Result<Value>;
56}

58/// Trait for hierarchical agent composition
59#[async_trait]
60pub trait HierarchicalAgent: CompositeAgent {
61    /// Get the parent agent if this is a child
62    fn parent(&self) -> Option<Arc<dyn HierarchicalAgent>>;
63
64    /// Get all child agents
65    fn children(&self) -> Vec<Arc<dyn HierarchicalAgent>>;
66
67    /// Add a child agent
68    async fn add_child(&mut self, child: Arc<dyn HierarchicalAgent>) -> Result<()>;
69
70    /// Remove a child agent
71    async fn remove_child(&mut self, child_id: &str) -> Result<()>;
72
73    /// Get the depth of this agent in the hierarchy
74    fn depth(&self) -> usize;
75
76    /// Propagate an event down the hierarchy
77    async fn propagate_down(&self, event: HierarchyEvent) -> Result<()>;
78
79    /// Propagate an event up the hierarchy
80    async fn propagate_up(&self, event: HierarchyEvent) -> Result<()>;
81}

83/// Metadata about a composable component
84#[derive(Debug, Clone)]
85pub struct CompositionMetadata {
86    /// Type of composition this component supports
87    pub composition_type: CompositionType,
88    /// Version of the composition interface
89    pub version: String,
90    /// Maximum number of components this can compose
91    pub max_components: Option<usize>,
92    /// Whether this component can be a parent
93    pub can_be_parent: bool,
94    /// Whether this component can be a child
95    pub can_be_child: bool,
96    /// Custom metadata
97    pub custom: HashMap<String, Value>,
98}

115/// Execution patterns for composite agents
116#[derive(Debug, Clone)]
117pub enum ExecutionPattern {
118    /// Execute components sequentially
119    Sequential,
120    /// Execute components in parallel
121    Parallel,
122    /// Execute based on conditions
123    Conditional(Vec<ExecutionCondition>),
124    /// Execute in a round-robin fashion
125    RoundRobin,
126    /// Execute based on capability matching
127    CapabilityBased(Vec<Capability>),
128    /// Custom execution pattern
129    Custom(String),
130}

193/// Builder for creating composite agents
194pub struct CompositeAgentBuilder {
195    #[allow(dead_code)]
196    name: String,
197    description: String,
198    components: Vec<Arc<dyn BaseAgent>>,
199    execution_pattern: ExecutionPattern,
200    metadata: HashMap<String, Value>,
201}
202
203impl CompositeAgentBuilder {
204    /// Create a new composite agent builder
205    pub fn new(name: impl Into<String>) -> Self {
206        Self {
207            name: name.into(),
208            description: String::new(),
209            components: Vec::new(),
210            execution_pattern: ExecutionPattern::Sequential,
211            metadata: HashMap::new(),
212        }
213    }
214
215    /// Set the description
216    #[must_use]
217    pub fn description(mut self, desc: impl Into<String>) -> Self {
218        self.description = desc.into();
219        self
220    }
221
222    /// Add a component
223    #[must_use]
224    pub fn add_component(mut self, component: Arc<dyn BaseAgent>) -> Self {
225        self.components.push(component);
226        self
227    }
228
229    /// Set the execution pattern
230    #[must_use]
231    pub fn execution_pattern(mut self, pattern: ExecutionPattern) -> Self {
232        self.execution_pattern = pattern;
233        self
234    }
235
236    /// Add metadata
237    #[must_use]
238    pub fn metadata(mut self, key: impl Into<String>, value: Value) -> Self {
239        self.metadata.insert(key.into(), value);
240        self
241    }
242
243    /// Build the composite agent (to be implemented by concrete types)
244    ///
245    /// # Errors
246    ///
247    /// Returns an error if:
248    /// - The build method is called on the base builder (must be implemented by concrete types)
249    /// - Component validation fails
250    /// - Agent construction fails due to configuration issues
251    pub fn build<T: CompositeAgent>(self) -> Result<T> {
252        // This will be implemented by concrete composite agent types
253        Err(LLMSpellError::Component {
254            message: "Build method must be implemented by concrete types".to_string(),
255            source: None,
256        })
257    }
258}

260/// Error types specific to composition
261#[derive(Debug, thiserror::Error)]
262pub enum CompositionError {
263    /// Component not found
264    #[error("Component not found: {0}")]
265    ComponentNotFound(String),
266
267    /// Invalid composition
268    #[error("Invalid composition: {0}")]
269    InvalidComposition(String),
270
271    /// Capability mismatch
272    #[error("Capability mismatch: required {required:?}, provided {provided:?}")]
273    CapabilityMismatch {
274        required: Vec<Capability>,
275        provided: Vec<Capability>,
276    },
277
278    /// Cycle detected in hierarchy
279    #[error("Cycle detected in hierarchy")]
280    CycleDetected,
281
282    /// Maximum depth exceeded
283    #[error("Maximum hierarchy depth exceeded: {0}")]
284    MaxDepthExceeded(usize),
285
286    /// Delegation failed
287    #[error("Delegation failed to component {0}: {1}")]
288    DelegationFailed(String, String),
289}
290
291impl From<CompositionError> for LLMSpellError {
292    fn from(err: CompositionError) -> Self {
293        Self::Component {
294            message: err.to_string(),
295            source: Some(Box::new(err)),
296        }
297    }

320            max_components: Some(10),
321            can_be_parent: true,
322            can_be_child: false,

330        assert_eq!(metadata.max_components, Some(10));
331    }

333    fn test_execution_pattern() {
334        let condition = ExecutionCondition {
335            component_id: "comp1".to_string(),
336            field: "status".to_string(),
337            expected: Value::String("ready".to_string()),
338            then_component: "comp2".to_string(),
339            else_component: Some("comp3".to_string()),
340        };
341
342        let pattern = ExecutionPattern::Conditional(vec![condition]);
343        match pattern {
344            ExecutionPattern::Conditional(conditions) => {
345                assert_eq!(conditions.len(), 1);
346                assert_eq!(conditions[0].component_id, "comp1");
347            }
348            _ => panic!("Expected conditional pattern"),
349        }
350    }

359        assert_eq!(builder.description, "A test composite agent");
