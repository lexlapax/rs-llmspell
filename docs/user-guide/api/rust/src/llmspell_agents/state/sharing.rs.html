16/// State sharing patterns
17#[derive(Debug, Clone, Copy, PartialEq, Eq)]
18pub enum SharingPattern {
19    /// Broadcast - one agent publishes, all subscribed agents receive
20    Broadcast,
21    /// Request-Response - one agent requests, another responds
22    RequestResponse,
23    /// Collaborative - multiple agents can read/write
24    Collaborative,
25    /// Pipeline - data flows through agents in sequence
26    Pipeline,
27    /// Hierarchical - parent-child agent relationships
28    Hierarchical,
29}

56/// State sharing manager for controlled data exchange
57pub struct StateSharingManager {
58    #[allow(dead_code)]
59    state_manager: Arc<dyn std::any::Any + Send + Sync>,
60    channels: Arc<RwLock<HashMap<String, SharedStateChannel>>>,
61    subscriptions: Arc<RwLock<HashMap<String, Vec<String>>>>, // agent_id -> channel_ids
62    message_queue: Arc<RwLock<HashMap<String, Vec<StateMessage>>>>, // channel_id -> messages
63}

68            state_manager,
69            channels: Arc::new(RwLock::new(HashMap::new())),
70            subscriptions: Arc::new(RwLock::new(HashMap::new())),
71            message_queue: Arc::new(RwLock::new(HashMap::new())),

75    /// Create a new shared state channel
76    ///
77    /// # Errors
78    ///
79    /// Returns an error if:
80    /// - A channel with the given ID already exists
81    /// - Subscribing the creator agent fails
82    #[instrument(skip(self))]
83    pub fn create_channel(
84        &self,
85        channel_id: &str,
86        pattern: SharingPattern,
87        creator_agent_id: &str,
88        ttl: Option<Duration>,
89    ) -> Result<SharedStateChannel> {
90        let mut channels = self.channels.write();
91
92        if channels.contains_key(channel_id) {
93            return Err(anyhow::anyhow!("Channel {} already exists", channel_id));
94        }
95
96        let channel = SharedStateChannel {
97            channel_id: channel_id.to_string(),
98            pattern,
99            creator_agent_id: creator_agent_id.to_string(),
100            participants: vec![creator_agent_id.to_string()],
101            created_at: SystemTime::now(),
102            ttl,
103            metadata: HashMap::new(),
104        };
105
106        channels.insert(channel_id.to_string(), channel.clone());
107
108        // Initialize message queue for channel
109        let mut message_queue = self.message_queue.write();
110        message_queue.insert(channel_id.to_string(), Vec::new());
111
112        // Drop locks before calling subscribe_agent to avoid deadlock
113        drop(message_queue);
114        drop(channels);
115
116        // Subscribe creator to channel
117        self.subscribe_agent(creator_agent_id, channel_id)?;
118
119        info!(
120            "Created shared state channel {} with pattern {:?}",
121            channel_id, pattern
122        );
123        Ok(channel)
124    }

126    /// Subscribe an agent to a channel
127    ///
128    /// # Errors
129    ///
130    /// Returns an error if:
131    /// - The specified channel does not exist
132    /// - The agent ID is invalid
133    #[instrument(skip(self))]
134    pub fn subscribe_agent(&self, agent_id: &str, channel_id: &str) -> Result<()> {
135        // Verify channel exists
136        let mut channels = self.channels.write();
137        let channel = channels
138            .get_mut(channel_id)
139            .ok_or_else(|| anyhow::anyhow!("Channel {} not found", channel_id))?;
140
141        // Add to participants if not already present
142        if !channel.participants.contains(&agent_id.to_string()) {
143            channel.participants.push(agent_id.to_string());
144        }
145
146        // Update subscriptions
147        let mut subscriptions = self.subscriptions.write();
148        let agent_subs = subscriptions.entry(agent_id.to_string()).or_default();
149        if !agent_subs.contains(&channel_id.to_string()) {
150            agent_subs.push(channel_id.to_string());
151        }
152
153        debug!("Agent {} subscribed to channel {}", agent_id, channel_id);
154        Ok(())
155    }

157    /// Unsubscribe an agent from a channel
158    ///
159    /// # Errors
160    ///
161    /// Returns an error if:
162    /// - The specified channel does not exist
163    /// - The agent is not subscribed to the channel
164    #[instrument(skip(self))]
165    pub fn unsubscribe_agent(&self, agent_id: &str, channel_id: &str) -> Result<()> {
166        // Remove from channel participants
167        let mut channels = self.channels.write();
168        if let Some(channel) = channels.get_mut(channel_id) {
169            channel.participants.retain(|id| id != agent_id);
170        }
171
172        // Update subscriptions
173        let mut subscriptions = self.subscriptions.write();
174        if let Some(agent_subs) = subscriptions.get_mut(agent_id) {
175            agent_subs.retain(|id| id != channel_id);
176        }
177
178        debug!(
179            "Agent {} unsubscribed from channel {}",
180            agent_id, channel_id
181        );
182        Ok(())
183    }

185    /// Publish a message to a channel
186    ///
187    /// # Errors
188    ///
189    /// Returns an error if:
190    /// - The specified channel does not exist
191    /// - The sender agent is not a participant in the channel
192    /// - Pattern permissions validation fails
193    #[instrument(skip(self, payload))]
194    pub async fn publish_message(
195        &self,
196        channel_id: &str,
197        sender_agent_id: &str,
198        message_type: &str,
199        payload: Value,
200        correlation_id: Option<Uuid>,
201    ) -> Result<Uuid> {
202        // Verify channel exists and agent is participant
203        let channels = self.channels.read();
204        let channel = channels
205            .get(channel_id)
206            .ok_or_else(|| anyhow::anyhow!("Channel {} not found", channel_id))?;
207
208        if !channel.participants.contains(&sender_agent_id.to_string()) {
209            return Err(anyhow::anyhow!(
210                "Agent {} is not a participant in channel {}",
211                sender_agent_id,
212                channel_id
213            ));
214        }
215
216        // Validate pattern permissions
217        Self::validate_pattern_permissions(channel.pattern, sender_agent_id, channel)?;
218
219        let message = StateMessage {
220            message_id: Uuid::new_v4(),
221            channel_id: channel_id.to_string(),
222            sender_agent_id: sender_agent_id.to_string(),
223            message_type: message_type.to_string(),
224            payload: payload.clone(),
225            timestamp: SystemTime::now(),
226            correlation_id,
227            reply_to: None,
228        };
229
230        // Store message in queue
231        let mut message_queue = self.message_queue.write();
232        if let Some(queue) = message_queue.get_mut(channel_id) {
233            queue.push(message.clone());
234
235            // Trim queue if too long
236            if queue.len() > 1000 {
237                queue.drain(0..100);
238            }
239        }
240
241        // Store in persistent state for durability
242        // TODO: When StateManager is properly integrated, uncomment this:
243        // let state_key = format!("channel:{}:message:{}", channel_id, message.message_id);
244        // self.state_manager.set_scoped(
245        //     StateScope::Custom(format!("sharing:{}", channel_id)),
246        //     &state_key,
247        //     serde_json::to_value(&message)?,
248        // ).await?;
249
250        debug!(
251            "Published message {} to channel {}",
252            message.message_id, channel_id
253        );
254        Ok(message.message_id)
255    }

257    /// Reply to a message in a channel
258    ///
259    /// # Errors
260    ///
261    /// Returns an error if:
262    /// - The specified channel does not exist
263    /// - The original message to reply to is not found
264    /// - Publishing the reply message fails
265    #[instrument(skip(self, payload))]
266    pub async fn reply_to_message(
267        &self,
268        channel_id: &str,
269        sender_agent_id: &str,
270        reply_to_id: Uuid,
271        message_type: &str,
272        payload: Value,
273    ) -> Result<Uuid> {
274        // Find original message
275        let message_queue = self.message_queue.read();
276        let queue = message_queue
277            .get(channel_id)
278            .ok_or_else(|| anyhow::anyhow!("Channel {} not found", channel_id))?;
279
280        let original = queue
281            .iter()
282            .find(|m| m.message_id == reply_to_id)
283            .ok_or_else(|| anyhow::anyhow!("Message {} not found", reply_to_id))?;
284
285        let correlation_id = original.correlation_id.or(Some(original.message_id));
286        drop(message_queue);
287
288        // Create reply message
289        let message = StateMessage {
290            message_id: Uuid::new_v4(),
291            channel_id: channel_id.to_string(),
292            sender_agent_id: sender_agent_id.to_string(),
293            message_type: message_type.to_string(),
294            payload,
295            timestamp: SystemTime::now(),
296            correlation_id,
297            reply_to: Some(reply_to_id),
298        };
299
300        // Store reply
301        let mut message_queue = self.message_queue.write();
302        if let Some(queue) = message_queue.get_mut(channel_id) {
303            queue.push(message.clone());
304        }
305
306        Ok(message.message_id)
307    }

309    /// Get messages for an agent from their subscribed channels
310    ///
311    /// # Errors
312    ///
313    /// Returns an error if the agent has no subscriptions
314    #[instrument(skip(self))]
315    pub fn get_messages_for_agent(
316        &self,
317        agent_id: &str,
318        since: Option<SystemTime>,
319        limit: Option<usize>,
320    ) -> Result<Vec<StateMessage>> {
321        let subscriptions = self.subscriptions.read();
322        let agent_channels = subscriptions
323            .get(agent_id)
324            .ok_or_else(|| anyhow::anyhow!("Agent {} has no subscriptions", agent_id))?;
325
326        let message_queue = self.message_queue.read();
327        let mut all_messages = Vec::new();
328
329        for channel_id in agent_channels {
330            if let Some(queue) = message_queue.get(channel_id) {
331                let channel_messages: Vec<_> = queue
332                    .iter()
333                    .filter(|m| {
334                        // Filter by time if specified
335                        // This map_or is correct: if no since time is specified (None),
336                        // include all messages (true). Otherwise, only include messages
337                        // with timestamps after the specified time.
338                        #[allow(clippy::unnecessary_map_or)]
339                        since.map_or(true, |since_time| m.timestamp > since_time)
340                    })
341                    .filter(|m| {
342                        // Don't show agent their own messages
343                        m.sender_agent_id != agent_id
344                    })
345                    .cloned()
346                    .collect();
347
348                all_messages.extend(channel_messages);
349            }
350        }
351
352        // Sort by timestamp
353        all_messages.sort_by_key(|m| m.timestamp);
354
355        // Apply limit if specified
356        if let Some(limit) = limit {
357            all_messages.truncate(limit);
358        }
359
360        Ok(all_messages)
361    }

412    /// Create a data pipeline between agents
413    ///
414    /// # Errors
415    ///
416    /// Returns an error if:
417    /// - Creating the pipeline channel fails
418    /// - The stages list is empty
419    /// - Setting pipeline metadata fails
420    #[instrument(skip(self))]
421    pub fn create_pipeline(
422        &self,
423        pipeline_id: &str,
424        stages: &[String], // Agent IDs in order
425    ) -> Result<()> {
426        if stages.is_empty() {
427            return Err(anyhow::anyhow!("Pipeline must have at least one stage"));
428        }
429
430        // Create pipeline channel
431        self.create_channel(pipeline_id, SharingPattern::Pipeline, &stages[0], None)?;
432
433        // Subscribe all stages
434        for agent_id in stages {
435            self.subscribe_agent(agent_id, pipeline_id)?;
436        }
437
438        // Store pipeline configuration
439        let mut channels = self.channels.write();
440        if let Some(channel) = channels.get_mut(pipeline_id) {
441            channel
442                .metadata
443                .insert("pipeline_stages".to_string(), serde_json::to_value(stages)?);
444        }
445
446        info!(
447            "Created pipeline {} with {} stages",
448            pipeline_id,
449            stages.len()
450        );
451        Ok(())
452    }

454    /// Process next stage in a pipeline
455    ///
456    /// # Errors
457    ///
458    /// Returns an error if:
459    /// - The pipeline is not found
460    /// - Pipeline stages metadata is missing or invalid
461    /// - The current agent is not in the pipeline
462    /// - Publishing the stage completion message fails
463    #[instrument(skip(self, data))]
464    pub async fn process_pipeline_stage(
465        &self,
466        pipeline_id: &str,
467        current_agent_id: &str,
468        data: Value,
469    ) -> Result<Option<String>> {
470        let (stages, current_index) = {
471            let channels = self.channels.read();
472            let channel = channels
473                .get(pipeline_id)
474                .ok_or_else(|| anyhow::anyhow!("Pipeline {} not found", pipeline_id))?;
475
476            let stages: Vec<String> = serde_json::from_value(
477                channel
478                    .metadata
479                    .get("pipeline_stages")
480                    .ok_or_else(|| anyhow::anyhow!("Pipeline stages not found"))?
481                    .clone(),
482            )?;
483
484            // Find current stage index
485            let current_index = stages
486                .iter()
487                .position(|id| id == current_agent_id)
488                .ok_or_else(|| anyhow::anyhow!("Agent {} not in pipeline", current_agent_id))?;
489
490            (stages, current_index)
491        };
492
493        // Publish processed data
494        let correlation_id = Uuid::new_v4();
495
496        self.publish_message(
497            pipeline_id,
498            current_agent_id,
499            "pipeline_stage_complete",
500            data,
501            Some(correlation_id),
502        )
503        .await?;
504
505        // Return next agent in pipeline
506        if current_index + 1 < stages.len() {
507            Ok(Some(stages[current_index + 1].clone()))
508        } else {
509            Ok(None) // Pipeline complete
510        }
511    }

513    /// Clean up expired channels
514    pub fn cleanup_expired_channels(&self) {
515        let mut channels = self.channels.write();
516        let now = SystemTime::now();
517
518        let expired: Vec<_> = channels
519            .iter()
520            .filter_map(|(id, channel)| {
521                if let Some(ttl) = channel.ttl {
522                    if now.duration_since(channel.created_at).ok()? > ttl {
523                        Some(id.clone())
524                    } else {
525                        None
526                    }
527                } else {
528                    None
529                }
530            })
531            .collect();
532
533        for channel_id in expired {
534            channels.remove(&channel_id);
535
536            // Clean up message queue
537            let mut message_queue = self.message_queue.write();
538            message_queue.remove(&channel_id);
539
540            // Clean up subscriptions
541            let mut subscriptions = self.subscriptions.write();
542            for (_, agent_channels) in subscriptions.iter_mut() {
543                agent_channels.retain(|id| id != &channel_id);
544            }
545
546            info!("Cleaned up expired channel {}", channel_id);
547        }
548    }

550    // Private helper methods
551
552    fn validate_pattern_permissions(
553        pattern: SharingPattern,
554        agent_id: &str,
555        channel: &SharedStateChannel,
556    ) -> Result<()> {
557        match pattern {
558            SharingPattern::Broadcast => {
559                // Only creator can broadcast
560                if agent_id != channel.creator_agent_id {
561                    return Err(anyhow::anyhow!("Only channel creator can broadcast"));
562                }
563            }
564            // Pipeline: Agents can only publish when it's their turn (enforced by process_pipeline_stage)
565            // RequestResponse, Collaborative, Hierarchical: Allow any participant to publish
566            _ => {}
567        }
568        Ok(())
569    }
