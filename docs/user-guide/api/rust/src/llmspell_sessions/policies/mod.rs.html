52/// Policy composition pattern
53#[derive(Debug, Clone, Copy, Serialize, Deserialize, PartialEq)]
54pub enum PolicyComposition {
55    /// Execute policies sequentially (fail fast)
56    Sequential,
57    /// Execute policies in parallel (all must pass)
58    Parallel,
59    /// Execute policies with voting (majority wins)
60    Voting {
61        /// Minimum percentage of policies that must pass (0.0 to 1.0)
62        threshold: f32,
63    },
64}

66/// Session policy manager
67pub struct SessionPolicyManager {
68    /// Policy configuration
69    config: SessionPolicyConfig,
70    /// Hook registry for policy registration
71    hook_registry: Arc<HookRegistry>,
72    /// Hook executor
73    #[allow(dead_code)]
74    hook_executor: Arc<HookExecutor>,
75    /// Cached policy hooks
76    policies: Vec<Arc<dyn Hook>>,
77}
78
79impl SessionPolicyManager {
80    /// Create a new policy manager
81    pub fn new(
82        config: SessionPolicyConfig,
83        hook_registry: Arc<HookRegistry>,
84        hook_executor: Arc<HookExecutor>,
85    ) -> Self {
86        let mut policies: Vec<Arc<dyn Hook>> = Vec::new();
87
88        // Add enabled policies
89        if config.enable_timeout {
90            let timeout_policy = SessionTimeoutPolicy::new(config.timeout_config.clone());
91            policies.push(Arc::new(timeout_policy));
92        }
93
94        if config.enable_resource_limits {
95            let resource_policy = SessionResourcePolicy::new(config.resource_config.clone());
96            policies.push(Arc::new(resource_policy));
97        }
98
99        if config.enable_rate_limiting {
100            let rate_limit_policy = SessionRateLimitPolicy::new(config.rate_limit_config.clone());
101            policies.push(Arc::new(rate_limit_policy));
102        }
103
104        Self {
105            config,
106            hook_registry,
107            hook_executor,
108            policies,
109        }
110    }
111
112    /// Register session policies with the hook system
113    pub fn register_policies(&self) -> Result<()> {
114        // Register each policy with the appropriate hook points
115        for policy in &self.policies {
116            // Check which hook points this policy should execute for
117            let hook_points = vec![
118                HookPoint::SessionStart,
119                HookPoint::SessionEnd,
120                HookPoint::SessionCheckpoint,
121                HookPoint::SessionRestore,
122                HookPoint::SessionSave,
123            ];
124
125            for hook_point in hook_points {
126                // Create a temporary context to check if the policy should execute
127                let temp_context = HookContext::new(
128                    hook_point.clone(),
129                    llmspell_hooks::ComponentId::new(
130                        llmspell_hooks::ComponentType::Agent,
131                        "session".to_string(),
132                    ),
133                );
134
135                if policy.should_execute(&temp_context) {
136                    // Register the policy for this hook point
137                    self.hook_registry
138                        .register_arc(hook_point, Arc::clone(policy))
139                        .map_err(|e| anyhow::anyhow!("Failed to register policy: {:?}", e))?;
140                }
141            }
142        }
143
144        Ok(())
145    }
146
147    /// Evaluate policies for a session operation
148    pub async fn evaluate_policies(&self, context: &mut HookContext) -> Result<HookResult> {
149        // Get hooks from registry for this hook point
150        let hooks = self.hook_registry.get_hooks(&context.point);
151
152        if hooks.is_empty() {
153            return Ok(HookResult::Continue);
154        }
155
156        // Execute hooks using the executor based on composition pattern
157        match self.config.composition_pattern {
158            PolicyComposition::Sequential => {
159                // Execute sequentially using executor
160                let results = self.hook_executor.execute_hooks(&hooks, context).await?;
161
162                // Check results - executor already handles Cancel/Replace logic
163                for result in results {
164                    if !result.should_continue() {
165                        return Ok(result);
166                    }
167                }
168                Ok(HookResult::Continue)
169            }
170            PolicyComposition::Parallel => {
171                // For parallel, we still use sequential execution but don't stop on first failure
172                let mut all_results = Vec::new();
173                for hook in &hooks {
174                    let result = self
175                        .hook_executor
176                        .execute_hook(hook.as_ref(), context)
177                        .await?;
178                    all_results.push(result);
179                }
180
181                // Return first non-continue result
182                for result in all_results {
183                    if !result.should_continue() {
184                        return Ok(result);
185                    }
186                }
187                Ok(HookResult::Continue)
188            }
189            PolicyComposition::Voting { threshold } => {
190                // Execute all and count successes
191                let total = hooks.len();
192                let mut successes = 0;
193
194                for hook in &hooks {
195                    let result = self
196                        .hook_executor
197                        .execute_hook(hook.as_ref(), context)
198                        .await?;
199                    if result.should_continue() {
200                        successes += 1;
201                    }
202                }
203
204                #[allow(clippy::cast_precision_loss)]
205                let success_rate = successes as f32 / total as f32;
206                if success_rate >= threshold {
207                    Ok(HookResult::Continue)
208                } else {
209                    Ok(HookResult::Cancel("Voting threshold not met".to_string()))
210                }
211            }
212        }
213    }

215    /// Check if a specific policy is enabled
216    pub fn is_policy_enabled(&self, policy_type: PolicyType) -> bool {
217        match policy_type {
218            PolicyType::Timeout => self.config.enable_timeout,
219            PolicyType::ResourceLimit => self.config.enable_resource_limits,
220            PolicyType::RateLimit => self.config.enable_rate_limiting,
221        }
222    }
223
224    /// Update policy configuration
225    pub fn update_config(&mut self, config: SessionPolicyConfig) {
226        self.config = config;
227    }
