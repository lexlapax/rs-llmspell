18/// Register artifact collectors with the hook registry
19pub fn register_artifact_collectors(
20    registry: &Arc<HookRegistry>,
21    config: &CollectorConfig,
22) -> Result<()> {
23    // Register tool result collector
24    if config.collect_tool_results {
25        let tool_collector = ToolResultCollector::with_config(config.tool_config.clone());
26        registry
27            .register(HookPoint::AfterToolExecution, tool_collector)
28            .map_err(|e| {
29                SessionError::Configuration(format!("Failed to register tool collector: {e}"))
30            })?;
31        info!("Registered tool result collector");
32    }
33
34    // Register agent output collector
35    if config.collect_agent_outputs {
36        let agent_collector = AgentOutputCollector::with_config(config.agent_config.clone());
37        registry
38            .register(HookPoint::AfterAgentExecution, agent_collector)
39            .map_err(|e| {
40                SessionError::Configuration(format!("Failed to register agent collector: {e}"))
41            })?;
42        info!("Registered agent output collector");
43    }
44
45    Ok(())
46}

75/// Process collected artifacts from hook context
76pub async fn process_collected_artifact(
77    context: &HookContext,
78    session_id: &SessionId,
79    artifact_storage: &Arc<ArtifactStorage>,
80) -> Result<()> {
81    // Check if artifact was collected
82    if let Some(collected_data) = context.data.get("collected_artifact") {
83        let artifact_type_str = collected_data
84            .get("type")
85            .and_then(|v| v.as_str())
86            .ok_or_else(|| SessionError::Storage("Missing artifact type".to_string()))?;
87
88        let artifact_data: ArtifactData = serde_json::from_value(
89            collected_data
90                .get("data")
91                .ok_or_else(|| SessionError::Storage("Missing artifact data".to_string()))?
92                .clone(),
93        )?;
94
95        // Map string type to ArtifactType enum
96        let artifact_type = match artifact_type_str {
97            "tool_result" => ArtifactType::ToolResult,
98            "agent_output" => ArtifactType::AgentOutput,
99            _ => ArtifactType::Custom(artifact_type_str.to_string()),
100        };
101
102        // Create artifact metadata
103        let mut metadata = ArtifactMetadata::new(
104            artifact_data.name.clone(),
105            artifact_type,
106            artifact_data.content.len(),
107        );
108
109        metadata.mime_type.clone_from(&artifact_data.mime_type);
110        metadata.description = Some(format!("Auto-collected {artifact_type_str}"));
111
112        // Add artifact-specific metadata
113        if let Some(obj) = artifact_data.metadata.as_object() {
114            for (key, value) in obj {
115                metadata.custom.insert(key.clone(), value.clone());
116            }
117        }
118
119        // Add tags
120        for tag in &artifact_data.tags {
121            let _ = metadata.add_tag(tag.clone());
122        }
123
124        // Get next sequence number for this session
125        // For now, use a simple timestamp-based sequence
126        let sequence = chrono::Utc::now().timestamp_micros().unsigned_abs();
127
128        // Create the session artifact
129        let session_artifact = SessionArtifact::create_with_metadata(
130            *session_id,
131            sequence,
132            metadata.artifact_type.clone(),
133            artifact_data.name,
134            artifact_data.content,
135            metadata.created_by.clone(),
136        )?;
137
138        // Store the artifact
139        let artifact_id = artifact_storage
140            .store_artifact(&session_artifact)
141            .await
142            .map_err(|e| SessionError::Storage(format!("Failed to store artifact: {e}")))?;
143
144        debug!(
145            "Stored collected artifact {} for session {}",
146            artifact_id, session_id
147        );
148    }
149
150    Ok(())
151}

153/// Artifact collection hook processor
154///
155/// This integrates with `SessionManager` to process hooks that collect artifacts
156pub struct ArtifactCollectionProcessor {
157    artifact_storage: Arc<ArtifactStorage>,
158    config: CollectorConfig,
159}
160
161impl ArtifactCollectionProcessor {
162    /// Create a new artifact collection processor
163    pub fn new(artifact_storage: Arc<ArtifactStorage>, config: CollectorConfig) -> Self {
164        Self {
165            artifact_storage,
166            config,
167        }
168    }
169
170    /// Process hook context for artifact collection
171    ///
172    /// This is called after hooks execute to check if any artifacts were collected
173    pub async fn process_hook_context(
174        &self,
175        context: &HookContext,
176        session_id: &SessionId,
177    ) -> Result<()> {
178        if !self.config.auto_store_artifacts {
179            return Ok(());
180        }
181
182        // Check if this hook collected any artifacts
183        if context
184            .metadata
185            .get("artifact_collected")
186            .map(std::string::String::as_str)
187            == Some("true")
188        {
189            if let Err(e) =
190                process_collected_artifact(context, session_id, &self.artifact_storage).await
191            {
192                warn!("Failed to process collected artifact: {}", e);
193            }
194        }
195
196        Ok(())
197    }
198
199    /// Check if a hook point should trigger artifact collection
200    pub fn should_process_hook_point(point: &HookPoint) -> bool {
201        matches!(
202            point,
203            HookPoint::AfterToolExecution | HookPoint::AfterAgentExecution
204        )
205    }
206}

214    #[tokio::test]
215    async fn test_process_collected_artifact() {
216        let storage = Arc::new(MemoryBackend::new());
217        let artifact_storage = Arc::new(ArtifactStorage::with_backend(storage));
218        let session_id = SessionId::new();
219
220        let component_id = ComponentId::new(ComponentType::Tool, "test-tool".to_string());
221        let mut context = HookContext::new(HookPoint::AfterToolExecution, component_id);
222
223        // Add collected artifact data
224        context.insert_data(
225            "collected_artifact".to_string(),
226            json!({
227                "type": "tool_result",
228                "data": {
229                    "name": "test_result.json",
230                    "content": b"test content",
231                    "mime_type": "application/json",
232                    "metadata": {
233                        "tool_name": "test-tool",
234                        "operation_type": "test"
235                    },
236                    "tags": ["test", "tool_result"]
237                }
238            }),
239        );
240
241        // Process the collected artifact
242        let result = process_collected_artifact(&context, &session_id, &artifact_storage).await;
243        assert!(result.is_ok());
244
245        // Verify artifact was stored
246        let artifacts = artifact_storage
247            .list_session_artifacts(&session_id)
248            .await
249            .unwrap();
250        assert_eq!(artifacts.len(), 1);
251        assert_eq!(artifacts[0].name, "test_result.json");
252    }
