26/// Session-specific replay configuration
27#[derive(Debug, Clone, Serialize, Deserialize)]
28pub struct SessionReplayConfig {
29    /// Replay mode (exact, modified, simulate, debug)
30    pub mode: ReplayMode,
31    /// Target timestamp for partial replay
32    pub target_timestamp: Option<SystemTime>,
33    /// Whether to compare results with original execution
34    pub compare_results: bool,
35    /// Timeout for the entire session replay
36    pub timeout: Duration,
37    /// Whether to stop on first error
38    pub stop_on_error: bool,
39    /// Session-specific metadata
40    pub metadata: HashMap<String, serde_json::Value>,
41}
42
43impl Default for SessionReplayConfig {
44    fn default() -> Self {
45        Self {
46            mode: ReplayMode::Exact,
47            target_timestamp: None,
48            compare_results: true,
49            timeout: Duration::from_secs(300), // 5 minutes for full session
50            stop_on_error: true,
51            metadata: HashMap::new(),
52        }
53    }
54}

168    /// Load session correlation ID from storage
169    async fn load_session_correlation_id(&self, session_id: &SessionId) -> Result<Uuid> {
170        // Try to load from session_metadata first (new format)
171        let metadata_key = format!("session_metadata:{}", session_id);
172        if let Ok(Some(metadata_bytes)) = self.storage_backend.get(&metadata_key).await {
173            if let Ok(metadata) = serde_json::from_slice::<serde_json::Value>(&metadata_bytes) {
174                if let Some(correlation_id_str) =
175                    metadata.get("correlation_id").and_then(|v| v.as_str())
176                {
177                    return Uuid::parse_str(correlation_id_str).map_err(|e| {
178                        SessionError::general(format!("Invalid correlation_id: {}", e))
179                    });
180                }
181            }
182        }
183
184        // Fallback to try loading from the main session key (for backward compatibility)
185        let session_key = format!("session:{}", session_id);
186        let session_bytes = self
187            .storage_backend
188            .get(&session_key)
189            .await
190            .map_err(|e| SessionError::Storage(e.to_string()))?;
191
192        // The session data is stored in bincode format, not JSON
193        // So we can't extract correlation_id from it directly
194        // Return an error indicating the session needs to be saved in the new format
195        if session_bytes.is_some() {
196            Err(SessionError::general(
197                "Session exists but correlation_id not found. Session needs to be saved in new format.",
198            ))
199        } else {
200            Err(SessionError::general(format!(
201                "Session {} not found",
202                session_id
203            )))
204        }
205    }

207    /// Convert state-persistence executions to hooks format
208    fn convert_executions_to_hooks_format(
209        executions: Vec<SerializedHookExecution>,
210    ) -> Vec<HooksSerializedHookExecution> {
211        executions
212            .into_iter()
213            .map(|exec| HooksSerializedHookExecution {
214                hook_id: exec.hook_id,
215                execution_id: exec.execution_id,
216                correlation_id: exec.correlation_id,
217                hook_context: exec.hook_context,
218                result: exec.result,
219                timestamp: exec.timestamp,
220                duration: exec.duration,
221                metadata: exec.metadata,
222            })
223            .collect()
224    }

226    /// Check if a session can be replayed
227    pub async fn can_replay_session(&self, session_id: &SessionId) -> Result<bool> {
228        match self.load_session_correlation_id(session_id).await {
229            Ok(correlation_uuid) => {
230                // Check if we have hook executions for this correlation_id
231                let executions = self
232                    .hook_replay_manager
233                    .get_hook_executions_by_correlation(correlation_uuid)
234                    .await
235                    .map_err(|e| SessionError::replay(e.to_string()))?;
236
237                Ok(!executions.is_empty())
238            }
239            Err(e) => {
240                debug!("Cannot replay session {}: {}", session_id, e);
241                Ok(false)
242            }
243        }
244    }

246    /// Replay a session using the existing replay infrastructure
247    ///
248    /// # Panics
249    ///
250    /// Panics if the active replays mutex is poisoned
251    pub async fn replay_session(
252        &self,
253        session_id: &SessionId,
254        config: SessionReplayConfig,
255    ) -> Result<SessionReplayResult> {
256        info!("Starting replay for session {}", session_id);
257        let start_time = SystemTime::now();
258
259        // Load session correlation ID
260        let correlation_uuid = self.load_session_correlation_id(session_id).await?;
261
262        // Get hook executions for this session (in state-persistence format)
263        let mut state_executions = self
264            .hook_replay_manager
265            .get_hook_executions_by_correlation(correlation_uuid)
266            .await
267            .map_err(|e| SessionError::replay(format!("Failed to get hook executions: {}", e)))?;
268
269        if state_executions.is_empty() {
270            return Err(SessionError::replay("No hook executions found for session"));
271        }
272
273        // Sort executions by timestamp for correct replay order
274        state_executions.sort_by_key(|e| e.timestamp);
275
276        // Filter executions by target timestamp if specified
277        if let Some(target) = config.target_timestamp {
278            let before_count = state_executions.len();
279            state_executions.retain(|e| e.timestamp <= target);
280            info!(
281                "Filtered from {} to {} executions up to target timestamp",
282                before_count,
283                state_executions.len()
284            );
285        }
286
287        info!(
288            "Found {} hook executions to replay for session {}",
289            state_executions.len(),
290            session_id
291        );
292
293        // Track active replay using controls
294        self.controls
295            .update_progress(session_id, 0, None)
296            .unwrap_or_else(|e| {
297                debug!("Failed to update initial progress: {}", e);
298            });
299
300        // Also track in legacy format for backward compatibility
301        {
302            let mut active = self.active_replays.write().unwrap();
303            active.insert(
304                *session_id,
305                SessionReplayStatus {
306                    session_id: *session_id,
307                    state: ReplayState::Running,
308                    start_time: Instant::now(),
309                    hooks_processed: 0,
310                    total_hooks: state_executions.len(),
311                    current_hook: None,
312                },
313            );
314        }
315
316        // Convert state-persistence executions to hooks format
317        let hook_executions = Self::convert_executions_to_hooks_format(state_executions);
318
319        // Create batch replay request
320        let batch_request = BatchReplayRequest {
321            executions: hook_executions,
322            config: config.clone().into_replay_config(),
323            parallel: false, // Session replay should maintain order
324            max_concurrent: 1,
325        };
326
327        // Execute batch replay using the actual replay infrastructure
328        let batch_response = self
329            .replay_manager
330            .batch_replay(batch_request)
331            .await
332            .map_err(|e| SessionError::replay(format!("Batch replay failed: {}", e)))?;
333
334        // Create session-specific result
335        let result = SessionReplayResult {
336            session_id: *session_id,
337            correlation_id: correlation_uuid,
338            start_time,
339            total_duration: start_time.elapsed().unwrap_or_default(),
340            hooks_replayed: batch_response.results.len(),
341            successful_replays: batch_response.success_count,
342            failed_replays: batch_response.failure_count,
343            batch_response: batch_response.clone(),
344            metadata: config.metadata,
345        };
346
347        info!(
348            "Session replay completed: {}/{} hooks replayed successfully in {:?}",
349            result.successful_replays, result.hooks_replayed, result.total_duration
350        );
351
352        // Update replay status
353        {
354            let mut active = self.active_replays.write().unwrap();
355            if let Some(status) = active.get_mut(session_id) {
356                status.state = if result.failed_replays > 0 {
357                    ReplayState::Failed(format!("{} hooks failed", result.failed_replays))
358                } else {
359                    ReplayState::Completed
360                };
361                status.hooks_processed = result.hooks_replayed;
362            }
363        }
364
365        Ok(result)
366    }

459    /// Query hook executions for a specific session
460    pub async fn query_session_hooks(
461        &self,
462        session_id: &SessionId,
463        filter: SessionHookFilter,
464    ) -> Result<Vec<SerializedHookExecution>> {
465        // Load session correlation ID
466        let correlation_uuid = self.load_session_correlation_id(session_id).await?;
467
468        // Get all hook executions for this correlation
469        let mut executions = self
470            .hook_replay_manager
471            .get_hook_executions_by_correlation(correlation_uuid)
472            .await
473            .map_err(|e| SessionError::replay(e.to_string()))?;
474
475        // Apply filters
476        if let Some(start_time) = filter.start_time {
477            executions.retain(|e| e.timestamp >= start_time);
478        }
479
480        if let Some(end_time) = filter.end_time {
481            executions.retain(|e| e.timestamp <= end_time);
482        }
483
484        if let Some(hook_id) = filter.hook_id {
485            executions.retain(|e| e.hook_id == hook_id);
486        }
487
488        if let Some(max_results) = filter.max_results {
489            executions.truncate(max_results);
490        }
491
492        // Sort by timestamp
493        executions.sort_by_key(|e| e.timestamp);
494
495        Ok(executions)
496    }

498    /// Get session replay metadata
499    pub async fn get_session_replay_metadata(
500        &self,
501        session_id: &SessionId,
502    ) -> Result<SessionReplayMetadata> {
503        // Load correlation ID
504        let correlation_uuid = self.load_session_correlation_id(session_id).await?;
505
506        // Get hook executions count
507        let executions = self
508            .hook_replay_manager
509            .get_hook_executions_by_correlation(correlation_uuid)
510            .await
511            .map_err(|e| SessionError::replay(e.to_string()))?;
512
513        let total_hooks = executions.len();
514        let first_timestamp = executions.iter().map(|e| e.timestamp).min();
515        let last_timestamp = executions.iter().map(|e| e.timestamp).max();
516        let total_duration = match (first_timestamp, last_timestamp) {
517            (Some(start), Some(end)) => end.duration_since(start).ok(),
518            _ => None,
519        };
520
521        Ok(SessionReplayMetadata {
522            session_id: *session_id,
523            correlation_id: correlation_uuid,
524            total_hooks,
525            first_hook_timestamp: first_timestamp,
526            last_hook_timestamp: last_timestamp,
527            total_duration,
528            can_replay: total_hooks > 0,
529        })
530    }

532    /// List all sessions that can be replayed
533    pub async fn list_replayable_sessions(&self) -> Result<Vec<SessionId>> {
534        // List all session_metadata keys
535        let prefix = "session_metadata:";
536        let keys = self
537            .storage_backend
538            .list_keys(prefix)
539            .await
540            .map_err(|e| SessionError::Storage(e.to_string()))?;
541
542        let mut replayable_sessions = Vec::new();
543
544        for key in keys {
545            // Extract session ID from key
546            if let Some(session_id_str) = key.strip_prefix(prefix) {
547                if let Ok(session_id) = SessionId::from_str(session_id_str) {
548                    // Check if session has hook executions
549                    if self.can_replay_session(&session_id).await.unwrap_or(false) {
550                        replayable_sessions.push(session_id);
551                    }
552                }
553            }
554        }
555
556        Ok(replayable_sessions)
557    }

564    /// Schedule a session replay
565    pub async fn schedule_replay(
566        &self,
567        session_id: &SessionId,
568        config: SessionReplayConfig,
569        schedule: ReplaySchedule,
570    ) -> Result<llmspell_hooks::replay::ScheduledReplay> {
571        // Load correlation ID
572        let correlation_uuid = self.load_session_correlation_id(session_id).await?;
573
574        // Get hook executions
575        let executions = self
576            .hook_replay_manager
577            .get_hook_executions_by_correlation(correlation_uuid)
578            .await
579            .map_err(|e| SessionError::replay(format!("Failed to get hook executions: {}", e)))?;
580
581        if executions.is_empty() {
582            return Err(SessionError::replay("No hook executions found for session"));
583        }
584
585        // Convert to batch replay request
586        let hook_executions = Self::convert_executions_to_hooks_format(executions);
587        let batch_request = BatchReplayRequest {
588            executions: hook_executions,
589            config: config.into_replay_config(),
590            parallel: false,
591            max_concurrent: 1,
592        };
593
594        // Schedule using controls
595        self.controls
596            .schedule_replay(*session_id, batch_request, schedule)
597            .await
598    }

704    ) -> Result<super::session_debug::SessionDebugData> {
705        // Ensure we have the latest timeline data
706        self.import_debug_data(session_id).await?;
707
708        self.debugger.export_debug_data(session_id)
709    }

711    /// Get timeline navigation
712    pub fn navigate_to_timeline_point(
713        &self,
714        session_id: &SessionId,
715        entry_index: usize,
716    ) -> Result<SessionState> {
717        self.debugger
718            .navigate_to_timeline_point(session_id, entry_index)
719    }
