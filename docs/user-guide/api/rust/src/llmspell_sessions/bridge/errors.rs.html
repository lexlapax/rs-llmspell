7/// Script-friendly error representation
8#[derive(Debug, Clone)]
9pub struct ScriptError {
10    /// Error code (e.g., `SESSION_NOT_FOUND`)
11    pub code: String,
12    /// Human-readable error message
13    pub message: String,
14    /// Additional error details
15    pub details: Option<serde_json::Value>,
16}
17
18impl ScriptError {
19    /// Create a new script error
20    pub fn new(code: impl Into<String>, message: impl Into<String>) -> Self {
21        Self {
22            code: code.into(),
23            message: message.into(),
24            details: None,
25        }
26    }
27
28    /// Add details to the error
29    #[must_use]
30    pub fn with_details(mut self, details: serde_json::Value) -> Self {
31        self.details = Some(details);
32        self
33    }
34}
35
36impl fmt::Display for ScriptError {
37    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
38        write!(f, "[{}] {}", self.code, self.message)
39    }
40}

44/// Convert `SessionError` to `ScriptError`
45impl From<SessionError> for ScriptError {
46    fn from(err: SessionError) -> Self {
47        match err {
48            SessionError::SessionNotFound { id } => {
49                ScriptError::new("SESSION_NOT_FOUND", format!("Session not found: {}", id))
50            }
51
52            SessionError::SessionAlreadyExists { id } => ScriptError::new(
53                "SESSION_ALREADY_EXISTS",
54                format!("Session already exists: {}", id),
55            ),
56
57            SessionError::InvalidStateTransition { from, to } => ScriptError::new(
58                "INVALID_STATE_TRANSITION",
59                format!("Invalid state transition from {:?} to {:?}", from, to),
60            )
61            .with_details(serde_json::json!({
62                "from_state": format!("{:?}", from),
63                "to_state": format!("{:?}", to),
64            })),
65
66            SessionError::InvalidOperation { reason } => {
67                ScriptError::new("INVALID_OPERATION", reason)
68            }
69
70            SessionError::InvalidSessionState {
71                id,
72                state,
73                operation,
74            } => ScriptError::new(
75                "INVALID_SESSION_STATE",
76                format!(
77                    "Session {} is in invalid state {:?} for operation {}",
78                    id, state, operation
79                ),
80            )
81            .with_details(serde_json::json!({
82                "session_id": id,
83                "state": format!("{:?}", state),
84                "operation": operation,
85            })),
86
87            SessionError::ArtifactNotFound { id } => {
88                ScriptError::new("ARTIFACT_NOT_FOUND", format!("Artifact not found: {}", id))
89            }
90
91            SessionError::ArtifactAlreadyExists { id, session_id } => ScriptError::new(
92                "ARTIFACT_ALREADY_EXISTS",
93                format!("Artifact {} already exists in session {}", id, session_id),
94            ),
95
96            SessionError::Storage(msg) => ScriptError::new("STORAGE_ERROR", msg),
97
98            SessionError::State(state_err) => {
99                ScriptError::new("STATE_ERROR", state_err.to_string())
100            }
101
102            SessionError::Hook(msg) => ScriptError::new("HOOK_EXECUTION_ERROR", msg),
103
104            SessionError::Event(msg) => ScriptError::new("EVENT_DISPATCH_ERROR", msg),
105
106            SessionError::Serialization(msg) => ScriptError::new("SERIALIZATION_ERROR", msg),
107
108            SessionError::Deserialization(msg) => ScriptError::new("DESERIALIZATION_ERROR", msg),
109
110            SessionError::Validation(msg) => ScriptError::new("VALIDATION_ERROR", msg),
111
112            SessionError::Io(io_err) => ScriptError::new("IO_ERROR", io_err.to_string()),
113
114            SessionError::ReplayError { message, source: _ } => {
115                ScriptError::new("REPLAY_ERROR", message)
116            }
117
118            SessionError::Configuration(msg) => ScriptError::new("CONFIGURATION_ERROR", msg),
119
120            SessionError::AccessDenied { message } => ScriptError::new("ACCESS_DENIED", message),
121
122            SessionError::ResourceLimitExceeded { resource, message } => ScriptError::new(
123                "RESOURCE_LIMIT_EXCEEDED",
124                format!("Resource limit exceeded for {}: {}", resource, message),
125            )
126            .with_details(serde_json::json!({
127                "resource": resource,
128                "message": message,
129            })),
130
131            SessionError::Timeout { operation } => {
132                ScriptError::new("TIMEOUT", format!("Operation timed out: {}", operation))
133            }
134
135            SessionError::IntegrityError { message } => {
136                ScriptError::new("INTEGRITY_ERROR", message)
137            }
138
139            SessionError::General { message, source: _ } => {
140                ScriptError::new("GENERAL_ERROR", message)
141            }
142        }
143    }
144}

182/// Helper to create script errors for common cases
183pub struct ErrorBuilder;
184
185impl ErrorBuilder {
186    /// Create a not found error
187    pub fn not_found(resource: &str, id: &str) -> ScriptError {
188        ScriptError::new(
189            error_codes::SESSION_NOT_FOUND,
190            format!("{} not found: {}", resource, id),
191        )
192    }
193
194    /// Create an invalid input error
195    pub fn invalid_input(field: &str, reason: &str) -> ScriptError {
196        ScriptError::new(
197            error_codes::INVALID_INPUT,
198            format!("Invalid {}: {}", field, reason),
199        )
200        .with_details(serde_json::json!({
201            "field": field,
202            "reason": reason,
203        }))
204    }
205
206    /// Create a permission denied error
207    pub fn permission_denied(action: &str, resource: &str) -> ScriptError {
208        ScriptError::new(
209            error_codes::PERMISSION_DENIED,
210            format!("Permission denied: {} on {}", action, resource),
211        )
212        .with_details(serde_json::json!({
213            "action": action,
214            "resource": resource,
215        }))
216    }
217
218    /// Create a conversion error
219    pub fn conversion_error(from_type: &str, to_type: &str, reason: &str) -> ScriptError {
220        ScriptError::new(
221            error_codes::SCRIPT_CONVERSION_ERROR,
222            format!("Failed to convert {} to {}: {}", from_type, to_type, reason),
223        )
224        .with_details(serde_json::json!({
225            "from_type": from_type,
226            "to_type": to_type,
227            "reason": reason,
228        }))
229    }
230}
