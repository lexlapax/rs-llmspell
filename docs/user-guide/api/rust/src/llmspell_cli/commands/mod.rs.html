19/// Execute a command with the given runtime configuration
20pub async fn execute_command(
21    command: Commands,
22    engine: ScriptEngine,
23    runtime_config: LLMSpellConfig,
24    output_format: OutputFormat,
25) -> Result<()> {
26    // Use LLMSpellConfig directly now that bridge accepts it
27
28    match command {
29        Commands::Run {
30            script,
31            stream,
32            args,
33        } => {
34            run::execute_script_file(script, engine, runtime_config, stream, args, output_format)
35                .await
36        }
37        Commands::Exec { code, stream } => {
38            exec::execute_inline_script(code, engine, runtime_config, stream, output_format).await
39        }
40        Commands::Repl { history } => repl::start_repl(engine, runtime_config, history).await,
41        Commands::Providers { detailed } => {
42            providers::list_providers(runtime_config, detailed, output_format).await
43        }
44        Commands::Validate { config } => validate::validate_config(config, output_format).await,
45        Commands::Info { all } => info::show_engine_info(engine, all, output_format).await,
46        Commands::Init { output, force } => init::init_config(output, force).await,
47        Commands::Keys(keys_cmd) => keys::KeysCommand { command: keys_cmd }.execute().await,
48        Commands::Backup(backup_cmd) => {
49            backup::execute_backup(backup_cmd, &runtime_config, output_format).await
50        }
51    }
52}

54/// Create a script runtime for the specified engine
55pub async fn create_runtime(engine: ScriptEngine, config: LLMSpellConfig) -> Result<ScriptRuntime> {
56    match engine {
57        ScriptEngine::Lua => ScriptRuntime::new_with_lua(config)
58            .await
59            .map_err(|e| anyhow::anyhow!("Failed to create Lua runtime: {}", e)),
60        ScriptEngine::Javascript => {
61            anyhow::bail!("JavaScript engine not available yet (coming in Phase 5)")
62        }
63        ScriptEngine::Python => {
64            anyhow::bail!("Python engine not available yet (coming in Phase 9)")
65        }
66    }
67}
