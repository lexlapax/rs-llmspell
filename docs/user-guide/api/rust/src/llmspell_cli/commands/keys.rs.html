1//! API Key management command
2
3use anyhow::{anyhow, Result};
4use chrono::Utc;
5use clap::{Args, Subcommand};
6use llmspell_utils::api_key_manager::{ApiKeyManager, ApiKeyMetadata};
7
8/// Manage API keys for external services
9#[derive(Debug, Args)]
10pub struct KeysCommand {
11    #[command(subcommand)]
12    pub command: KeysSubcommand,
13}
14
15#[derive(Debug, Subcommand)]
16pub enum KeysSubcommand {
17    /// List all API keys
18    List {
19        /// Show detailed information
20        #[arg(short, long)]
21        detailed: bool,
22    },
23
24    /// Add a new API key
25    Add {
26        /// Service name (e.g., google_search, sendgrid)
27        service: String,
28
29        /// API key value
30        key: String,
31
32        /// Key expiration in days (optional)
33        #[arg(short, long)]
34        expires_in: Option<u64>,
35    },
36
37    /// Rotate an existing API key
38    Rotate {
39        /// Service name
40        service: String,
41
42        /// New API key value
43        new_key: String,
44    },
45
46    /// Remove an API key
47    Remove {
48        /// Service name
49        service: String,
50    },
51
52    /// Show audit log
53    Audit {
54        /// Number of entries to show
55        #[arg(short, long, default_value = "20")]
56        limit: usize,
57    },
58
59    /// Load keys from environment
60    LoadEnv {
61        /// Environment variable prefix
62        #[arg(short, long, default_value = "LLMSPELL_API_KEY_")]
63        prefix: String,
64    },
65}
66
67impl KeysCommand {
68    pub async fn execute(&self) -> Result<()> {
69        // In a real implementation, this would use persistent storage
70        // For now, we'll create a new manager each time
71        let mut manager = ApiKeyManager::new();
72
73        // Load existing keys from environment
74        let _ = manager.load_from_env();
75
76        match &self.command {
77            KeysSubcommand::List { detailed } => {
78                self.list_keys(&manager, *detailed)?;
79            }
80
81            KeysSubcommand::Add {
82                service,
83                key,
84                expires_in,
85            } => {
86                self.add_key(&manager, service, key, *expires_in)?;
87            }
88
89            KeysSubcommand::Rotate { service, new_key } => {
90                self.rotate_key(&manager, service, new_key)?;
91            }
92
93            KeysSubcommand::Remove { service } => {
94                self.remove_key(&manager, service)?;
95            }
96
97            KeysSubcommand::Audit { limit } => {
98                self.show_audit(&manager, *limit)?;
99            }
100
101            KeysSubcommand::LoadEnv { prefix } => {
102                manager.set_env_prefix(prefix);
103                let loaded = manager
104                    .load_from_env()
105                    .map_err(|e| anyhow!("Failed to load keys from environment: {}", e))?;
106                println!("Loaded {} API keys from environment", loaded);
107            }
108        }
109
110        Ok(())
111    }
112
113    fn list_keys(&self, _manager: &ApiKeyManager, detailed: bool) -> Result<()> {
114        println!("API Keys:");
115        println!("---------");
116
117        // In a real implementation, we'd iterate through stored keys
118        // For now, just show a message
119        if detailed {
120            println!("Use --detailed flag to see full key information");
121            println!("\nNote: Keys are loaded from environment variables starting with LLMSPELL_API_KEY_");
122        } else {
123            println!("No persistent key storage implemented yet.");
124            println!("Keys are currently loaded from environment variables.");
125        }
126
127        Ok(())
128    }
129
130    fn add_key(
131        &self,
132        manager: &ApiKeyManager,
133        service: &str,
134        key: &str,
135        expires_in: Option<u64>,
136    ) -> Result<()> {
137        let metadata = ApiKeyMetadata {
138            key_id: format!("cli_{}", service),
139            service: service.to_string(),
140            created_at: Utc::now(),
141            last_used: None,
142            expires_at: expires_in.map(|days| Utc::now() + chrono::Duration::days(days as i64)),
143            is_active: true,
144            usage_count: 0,
145        };
146
147        manager
148            .add_key(&metadata.key_id, key, metadata.clone())
149            .map_err(|e| anyhow!("Failed to add key: {}", e))?;
150
151        println!("Added API key for service '{}'", service);
152        if let Some(days) = expires_in {
153            println!("Key will expire in {} days", days);
154        }
155
156        Ok(())
157    }
158
159    fn rotate_key(&self, manager: &ApiKeyManager, service: &str, new_key: &str) -> Result<()> {
160        manager
161            .rotate_key(service, new_key)
162            .map_err(|e| anyhow!("Failed to rotate key: {}", e))?;
163        println!("Rotated API key for service '{}'", service);
164        Ok(())
165    }
166
167    fn remove_key(&self, _manager: &ApiKeyManager, service: &str) -> Result<()> {
168        // In a real implementation, this would deactivate the key
169        println!("Removed API key for service '{}'", service);
170        println!("Note: Key removal not yet implemented in persistent storage");
171        Ok(())
172    }
173
174    fn show_audit(&self, manager: &ApiKeyManager, limit: usize) -> Result<()> {
175        let entries = manager.get_audit_log(Some(limit));
176
177        println!("API Key Audit Log (last {} entries):", limit);
178        println!("------------------------------------");
179
180        for entry in entries {
181            println!(
182                "{} | {} | {:?} | {}",
183                entry.timestamp.format("%Y-%m-%d %H:%M:%S"),
184                entry.service,
185                entry.action,
186                entry.details.as_deref().unwrap_or("-")
187            );
188        }
189
190        Ok(())
191    }
192}
