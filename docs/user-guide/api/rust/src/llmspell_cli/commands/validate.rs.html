4use crate::cli::OutputFormat;
5use crate::config;
6use anyhow::Result;
7use serde_json::json;
8use std::path::{Path, PathBuf};

10/// Validate configuration file
11pub async fn validate_config(
12    config_path: Option<PathBuf>,
13    output_format: OutputFormat,
14) -> Result<()> {
15    let path = config_path.as_deref();
16    let mut warnings = Vec::new();
17    let mut errors = Vec::new();
18
19    // Try to load the configuration
20    let (config_result, actual_path) = match path {
21        Some(p) => {
22            let result = config::load_runtime_config(Some(p)).await;
23            (result, p.to_string_lossy().to_string())
24        }
25        None => {
26            // Try to discover config file
27            let result = config::load_runtime_config(None).await;
28            let discovered_path = discover_actual_path().await;
29            (result, discovered_path)
30        }
31    };
32
33    // Check if configuration loaded successfully
34    let valid = match config_result {
35        Ok(config) => {
36            // Validate the loaded configuration
37            match config::validate_config(&config) {
38                Ok(_) => {
39                    // Additional checks
40                    if config.providers.providers.is_empty() {
41                        warnings.push("No providers configured".to_string());
42                    }
43
44                    if !config.runtime.security.allow_network_access {
45                        warnings.push(
46                            "Network access is disabled - LLM providers won't work".to_string(),
47                        );
48                    }
49
50                    true
51                }
52                Err(e) => {
53                    errors.push(format!("Validation error: {}", e));
54                    false
55                }
56            }
57        }
58        Err(e) => {
59            errors.push(format!("Failed to load configuration: {}", e));
60            false
61        }
62    };
63
64    let validation_result = json!({
65        "valid": valid,
66        "path": actual_path,
67        "warnings": warnings,
68        "errors": errors
69    });
70
71    match output_format {
72        OutputFormat::Json => {
73            println!("{}", serde_json::to_string_pretty(&validation_result)?);
74        }
75        OutputFormat::Text | OutputFormat::Pretty => {
76            println!("Configuration validation:");
77            println!("  File: {}", actual_path);
78
79            if valid {
80                println!("  Status: ✓ Valid");
81            } else {
82                println!("  Status: ✗ Invalid");
83            }
84
85            if !warnings.is_empty() {
86                println!("\nWarnings:");
87                for warning in &warnings {
88                    println!("  ⚠ {}", warning);
89                }
90            }
91
92            if !errors.is_empty() {
93                println!("\nErrors:");
94                for error in &errors {
95                    println!("  ✗ {}", error);
96                }
97            }
98        }
99    }
100
101    if !valid {
102        anyhow::bail!("Configuration validation failed");
103    }
104
105    Ok(())
106}
107
108/// Try to discover which config file would be used
109async fn discover_actual_path() -> String {
110    // Check standard paths
111    for path in &["llmspell.toml", ".llmspell.toml", "config/llmspell.toml"] {
112        if Path::new(path).exists() {
113            return path.to_string();
114        }
115    }
116
117    // Check home directory
118    if let Ok(home) = std::env::var("HOME").or_else(|_| std::env::var("USERPROFILE")) {
119        for filename in &[".llmspell.toml", ".config/llmspell.toml"] {
120            let path = PathBuf::from(&home).join(filename);
121            if path.exists() {
122                return path.display().to_string();
123            }
124        }
125    }
126
127    "(no config file found - using defaults)".to_string()
128}
