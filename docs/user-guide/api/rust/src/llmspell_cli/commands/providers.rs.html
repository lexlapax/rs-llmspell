9/// List available providers

11    _runtime_config: LLMSpellConfig,

13    output_format: OutputFormat,

15    // TODO: Query actual providers from runtime
16    let providers = vec![json!({

19        "capabilities": {

25    match output_format {

31            println!();
32            for provider in &providers {
33                if let Some(name) = provider.get("name").and_then(|v| v.as_str()) {
34                    println!("â€¢ {}", name);
35                    if detailed {
36                        if let Some(models) = provider.get("models").and_then(|v| v.as_array()) {
37                            println!("  Models:");
38                            for model in models {
39                                if let Some(model_str) = model.as_str() {
40                                    println!("    - {}", model_str);
41                                }

43                        }
44                    }
45                }
46            }
47        }
48    }
