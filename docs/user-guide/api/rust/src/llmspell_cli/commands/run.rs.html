12/// Parse script arguments into a HashMap
13/// Supports three formats:
14/// - Positional: arg1 arg2 arg3 -> {"1": "arg1", "2": "arg2", "3": "arg3"}
15/// - Named: --key value --flag true -> {"key": "value", "flag": "true"}
16/// - Mixed: pos1 --named value pos2 -> {"1": "pos1", "named": "value", "2": "pos2"}
17fn parse_script_args(args: Vec<String>, script_path: &Path) -> HashMap<String, String> {
18    let mut parsed = HashMap::new();
19    let mut positional_index = 1;
20    let mut i = 0;
21
22    // Add script name as arg[0] for Lua compatibility
23    if let Some(script_name) = script_path.file_name() {
24        parsed.insert("0".to_string(), script_name.to_string_lossy().to_string());
25    }
26
27    while i < args.len() {
28        let arg = &args[i];
29
30        if arg.starts_with("--") {
31            // Named argument
32            let key = arg.trim_start_matches("--");
33
34            // Check if there's a value following this flag
35            if i + 1 < args.len() && !args[i + 1].starts_with("--") {
36                // --key value format
37                parsed.insert(key.to_string(), args[i + 1].clone());
38                i += 2;
39            } else {
40                // --flag format (boolean flag without value)
41                parsed.insert(key.to_string(), "true".to_string());
42                i += 1;
43            }
44        } else {
45            // Positional argument
46            parsed.insert(positional_index.to_string(), arg.clone());
47            positional_index += 1;
48            i += 1;
49        }
50    }
51
52    parsed
53}

55/// Execute a script file
56pub async fn execute_script_file(
57    script_path: PathBuf,
58    engine: ScriptEngine,
59    runtime_config: LLMSpellConfig,
60    stream: bool,
61    args: Vec<String>,
62    output_format: OutputFormat,
63) -> Result<()> {
64    // Validate script file exists
65    if !script_path.exists() {
66        anyhow::bail!("Script file not found: {}", script_path.display());
67    }
68
69    // Read script content
70    let script_content = fs::read_to_string(&script_path).await?;
71
72    // Parse script arguments
73    let parsed_args = parse_script_args(args, &script_path);
74    if !parsed_args.is_empty() {
75        tracing::debug!("Parsed script arguments: {:?}", parsed_args);
76    }
77
78    // Create runtime for the selected engine
79    let mut runtime = super::create_runtime(engine, runtime_config).await?;
80
81    // Pass script arguments to the runtime
82    runtime.set_script_args(parsed_args).await?;
83
84    // Execute script
85    if stream && runtime.supports_streaming() {
86        // Execute with streaming
87        let mut stream = runtime.execute_script_streaming(&script_content).await?;
88        print_stream(&mut stream, output_format).await?;
89    } else {
90        // Execute without streaming
91        let result = runtime.execute_script(&script_content).await?;
92        println!("{}", format_output(&result, output_format)?);
93    }
94
95    Ok(())
96}
