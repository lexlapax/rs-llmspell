14/// Format script output according to the specified format
15pub fn format_output(output: &ScriptOutput, format: OutputFormat) -> Result<String> {
16    match format {
17        OutputFormat::Json => Ok(serde_json::to_string_pretty(&output.output)?),
18        OutputFormat::Text => {
19            // Simple text representation
20            match output.output {
21                serde_json::Value::String(ref s) => Ok(s.clone()),
22                _ => Ok(output.output.to_string()),
23            }
24        }
25        OutputFormat::Pretty => {
26            // Pretty-printed output with metadata
27            let mut result = String::new();
28
29            // Add console output if any
30            if !output.console_output.is_empty() {
31                result.push_str("Console Output:\n");
32                for line in &output.console_output {
33                    result.push_str(&format!("  {}\n", line));
34                }
35                result.push('\n');
36            }
37
38            result.push_str(&format!(
39                "Output: {}\n",
40                serde_json::to_string_pretty(&output.output)?
41            ));
42
43            // Add metadata
44            result.push_str(&format!("Engine: {}\n", output.metadata.engine));
45            result.push_str(&format!(
46                "Execution time: {}ms\n",
47                output.metadata.execution_time_ms
48            ));
49
50            if !output.metadata.warnings.is_empty() {
51                result.push_str("\nWarnings:\n");
52                for warning in &output.metadata.warnings {
53                    result.push_str(&format!("  ‚ö† {}\n", warning));
54                }
55            }
56
57            Ok(result)
58        }
59    }
60}

62/// Print streaming output with the specified format
63pub async fn print_stream(stream: &mut ScriptStream, format: OutputFormat) -> Result<()> {
64    // Set up signal handler for graceful shutdown
65    let interrupted = Arc::new(Mutex::new(false));
66    let interrupted_clone = interrupted.clone();
67
68    tokio::spawn(async move {
69        signal::ctrl_c().await.expect("Failed to listen for Ctrl+C");
70        *interrupted_clone.lock().await = true;
71    });
72
73    match format {
74        OutputFormat::Json => print_stream_json(stream, interrupted).await,
75        OutputFormat::Text => print_stream_text(stream, false, interrupted).await,
76        OutputFormat::Pretty => print_stream_text(stream, true, interrupted).await,
77    }
78}

80/// Print streaming output as JSON
81async fn print_stream_json(stream: &mut ScriptStream, interrupted: Arc<Mutex<bool>>) -> Result<()> {
82    // Show progress while collecting chunks
83    let spinner = ProgressBar::new_spinner();
84    spinner.set_style(
85        ProgressStyle::default_spinner()
86            .template("{spinner:.cyan} {msg}")
87            .unwrap(),
88    );
89    spinner.set_message("Collecting stream data...");
90    spinner.enable_steady_tick(Duration::from_millis(100));
91
92    let mut chunks = Vec::new();
93    while let Some(chunk) = stream.stream.next().await {
94        // Check if interrupted
95        if *interrupted.lock().await {
96            spinner.finish_with_message("Stream interrupted by user");
97            eprintln!("\n‚ö†Ô∏è Stream interrupted by Ctrl+C");
98            break;
99        }
100
101        chunks.push(chunk?);
102        spinner.set_message(format!("Collected {} chunks", chunks.len()));
103    }
104
105    spinner.finish_and_clear();
106    println!("{}", serde_json::to_string_pretty(&chunks)?);
107    Ok(())
108}

110/// Print streaming output as text with optional progress indicators
111async fn print_stream_text(
112    stream: &mut ScriptStream,
113    show_progress: bool,
114    interrupted: Arc<Mutex<bool>>,
115) -> Result<()> {
116    use std::io::{self, Write};
117
118    let progress = if show_progress {
119        let pb = ProgressBar::new_spinner();
120        pb.set_style(
121            ProgressStyle::default_spinner()
122                .template("{spinner:.green} {msg}")
123                .unwrap(),
124        );
125        pb.set_message("Streaming output...");
126        pb.enable_steady_tick(Duration::from_millis(100));
127        Some(pb)
128    } else {
129        None
130    };
131
132    let mut chunk_count = 0;
133    let mut in_tool_call = false;
134
135    while let Some(chunk) = stream.stream.next().await {
136        // Check if interrupted
137        if *interrupted.lock().await {
138            if let Some(ref pb) = progress {
139                pb.finish_with_message("Stream interrupted by user");
140            }
141            eprintln!("\n‚ö†Ô∏è Stream interrupted by Ctrl+C");
142            break;
143        }
144
145        let chunk = chunk?;
146        chunk_count += 1;
147
148        // Update progress message
149        if let Some(ref pb) = progress {
150            pb.set_message(format!("Processing chunk {}", chunk_count));
151        }
152
153        // Print chunk content based on its type
154        match &chunk.content {
155            llmspell_core::types::ChunkContent::Text(text) => {
156                if in_tool_call && progress.is_some() {
157                    // Clear the progress bar before printing text
158                    if let Some(ref pb) = progress {
159                        pb.suspend(|| {
160                            print!("{}", text);
161                            io::stdout().flush().ok();
162                        });
163                    }
164                } else {
165                    print!("{}", text);
166                    io::stdout().flush()?;
167                }
168            }
169            llmspell_core::types::ChunkContent::ToolCallProgress { tool_name, .. } => {
170                in_tool_call = true;
171                if let Some(ref pb) = progress {
172                    pb.suspend(|| {
173                        println!("\nüîß Calling tool: {}...", tool_name);
174                    });
175                    pb.set_message(format!("Tool: {}", tool_name));
176                } else {
177                    println!("\n[Calling tool: {}...]", tool_name);
178                }
179            }
180            llmspell_core::types::ChunkContent::ToolCallComplete { tool_name, .. } => {
181                in_tool_call = false;
182                if let Some(ref pb) = progress {
183                    pb.suspend(|| {
184                        println!("‚úì Tool call complete: {}", tool_name);
185                    });
186                } else {
187                    println!("[Tool call complete: {}]", tool_name);
188                }
189            }
190            llmspell_core::types::ChunkContent::Media { caption, .. } => {
191                let media_msg = if let Some(cap) = caption {
192                    format!("üìé Media: {}", cap)
193                } else {
194                    "üìé Media content".to_string()
195                };
196
197                if let Some(ref pb) = progress {
198                    pb.suspend(|| {
199                        println!("\n{}", media_msg);
200                    });
201                } else {
202                    println!("\n[{}]", media_msg);
203                }
204            }
205            llmspell_core::types::ChunkContent::Control(msg) => {
206                // Handle control messages
207                use llmspell_core::types::ControlMessage;
208                tracing::debug!("Control message: {:?}", msg);
209                if let Some(ref pb) = progress {
210                    match msg {
211                        ControlMessage::StreamStart { .. } => {
212                            pb.set_message("Stream started");
213                        }
214                        ControlMessage::StreamEnd { .. } => {
215                            pb.set_message("Stream ending...");
216                        }
217                        ControlMessage::StreamCancelled { reason } => {
218                            pb.set_message(format!("Stream cancelled: {}", reason));
219                        }
220                        ControlMessage::Heartbeat => {
221                            // Keep spinner alive
222                        }
223                        ControlMessage::RateLimit { remaining, .. } => {
224                            pb.set_message(format!("Rate limited ({} remaining)", remaining));
225                        }
226                        ControlMessage::Custom { .. } => {
227                            // Custom control messages
228                        }
229                    }
230                }
231            }
232        }
233    }
234
235    // Clean up progress bar
236    if let Some(pb) = progress {
237        pb.finish_with_message("Stream complete");
238        pb.finish_and_clear();
239    }
240
241    println!(); // Final newline
242    Ok(())
243}
