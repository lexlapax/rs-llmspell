4use anyhow::{Context, Result};
5use llmspell_config::LLMSpellConfig;
6use std::path::Path;
7use tokio::fs;

9/// Load runtime configuration from file or use defaults
10/// Delegates to llmspell-config which handles:
11/// - Configuration file discovery
12/// - TOML parsing and validation
13/// - Environment variable overrides
14pub async fn load_runtime_config(config_path: Option<&Path>) -> Result<LLMSpellConfig> {
15    // Delegate to llmspell-config's comprehensive load_with_discovery
16    // This handles discovery, loading, environment overrides, and validation
17    let config = LLMSpellConfig::load_with_discovery(config_path)
18        .await
19        .map_err(|e| anyhow::anyhow!("Configuration error: {}", e))?;
20
21    // Validate the loaded configuration
22    config
23        .validate()
24        .map_err(|e| anyhow::anyhow!("Configuration validation failed: {}", e))?;
25
26    Ok(config)
27}

29/// Create default configuration file
30pub async fn create_default_config(path: &Path) -> Result<()> {
31    let default_config = LLMSpellConfig::default();
32    let toml_content = toml::to_string_pretty(&default_config)
33        .context("Failed to serialize default configuration")?;
34
35    // Create parent directories if needed
36    if let Some(parent) = path.parent() {
37        fs::create_dir_all(parent)
38            .await
39            .with_context(|| format!("Failed to create config directory: {}", parent.display()))?;
40    }
41
42    // Write the configuration file
43    fs::write(path, toml_content)
44        .await
45        .with_context(|| format!("Failed to write config file: {}", path.display()))?;
46
47    tracing::info!("Created default configuration at: {}", path.display());
48    Ok(())
49}

51/// Validate configuration

53pub fn validate_config(config: &LLMSpellConfig) -> Result<()> {

57        .map_err(|e| anyhow::anyhow!("Configuration validation failed: {}", e))
