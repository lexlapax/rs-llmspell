4//! Comprehensive test suite and utilities for the LLMSpell framework.
5//!
6//! This crate serves two purposes:
7//! 1. **Test Suite**: Organizes and runs all tests across the llmspell workspace
8//! 2. **Test Utilities**: Provides testing helpers for downstream users
9//!
10//! # Test Organization
11//!
12//! Tests are organized into categories:
13//! - `unit` - Unit tests for individual components
14//! - `integration` - Cross-crate integration tests
15//! - `agents` - Agent-specific tests
16//! - `scenarios` - End-to-end scenario tests
17//! - `lua` - Lua scripting tests
18//! - `performance` - Performance benchmarks
19//!
20//! # Running Tests
21//!
22//! Run specific test categories using features:
23//! ```bash
24//! cargo test -p llmspell-testing --features unit-tests
25//! cargo test -p llmspell-testing --features integration-tests
26//! cargo test -p llmspell-testing --features all-tests
27//! ```
28//!
29//! # Test Utilities
30//!
31//! The crate also provides utilities for writing tests:
32//! - Mock implementations of core traits
33//! - Property-based test generators
34//! - Benchmark helpers
35//! - Test fixtures and data generators
36//!
37//! # Examples
38//!
39//! ```rust,no_run
40//! use llmspell_testing::mocks::MockBaseAgent;
41//! use llmspell_testing::generators::component_id_strategy;
42//! use proptest::prelude::*;
43//! use llmspell_core::types::AgentOutput;
44//!
45//! // Use mock agent in tests
46//! let mut mock = MockBaseAgent::new();
47//! mock.expect_execute_impl()
48//!     .returning(|_, _| Ok(AgentOutput::text("test")));
49//!
50//! // Generate test data with proptest
51//! proptest! {
52//!     #[test]
53//!     fn test_with_random_component_id(id in component_id_strategy()) {
54//!         // Test with generated component ID
55//!     }
56//! }
57//! ```

86/// Test categories for selective execution
87#[derive(Debug, Clone, Copy, PartialEq, Eq)]
88pub enum TestCategory {
89    Unit,
90    Integration,
91    Agents,
92    Scenarios,
93    Lua,
94    Performance,
95    All,
96}
97
98impl TestCategory {
99    /// Get the feature flag name for this category
100    pub fn feature_name(&self) -> &'static str {
101        match self {
102            TestCategory::Unit => "unit-tests",
103            TestCategory::Integration => "integration-tests",
104            TestCategory::Agents => "agent-tests",
105            TestCategory::Scenarios => "scenario-tests",
106            TestCategory::Lua => "lua-tests",
107            TestCategory::Performance => "performance-tests",
108            TestCategory::All => "all-tests",
109        }
110    }
111
112    /// Get a description of this test category
113    pub fn description(&self) -> &'static str {
114        match self {
115            TestCategory::Unit => "Unit tests for individual components",
116            TestCategory::Integration => "Cross-crate integration tests",
117            TestCategory::Agents => "Agent-specific functionality tests",
118            TestCategory::Scenarios => "End-to-end scenario tests",
119            TestCategory::Lua => "Lua scripting tests",
120            TestCategory::Performance => "Performance benchmarks",
121            TestCategory::All => "All test categories",
122        }
123    }
