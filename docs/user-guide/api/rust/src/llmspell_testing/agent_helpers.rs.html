4//! Agent testing helpers.
5//!
6//! This module provides common test utilities for testing agents
7//! including provider-specific agent creation and mock agent setup.
8//!
9//! # Examples
10//!
11//! ```rust,no_run
12//! use llmspell_testing::agent_helpers::{
13//!     create_mock_provider_agent,
14//!     create_test_agent_config,
15//!     AgentTestBuilder,
16//! };
17//!
18//! # async fn test_example() {
19//! // Create a mock provider agent
20//! let agent = create_mock_provider_agent("openai", "gpt-4").await;
21//!
22//! // Use the builder for more control
23//! let agent = AgentTestBuilder::new("test-agent")
24//!     .with_provider("anthropic")
25//!     .with_model("claude-3")
26//!     .with_system_prompt("You are a test assistant")
27//!     .build()
28//!     .await
29//!     .unwrap();
30//! # }
31//! ```

41/// Test agent implementation that simulates provider behavior
42pub struct TestProviderAgent {
43    metadata: ComponentMetadata,
44    config: AgentConfig,
45    _provider: String,
46    _model: String,
47    responses: Vec<String>,
48    response_index: std::sync::Mutex<usize>,
49}
50
51impl TestProviderAgent {
52    /// Create a new test provider agent
53    pub fn new(provider: &str, model: &str) -> Self {
54        let metadata = ComponentMetadata::new(
55            format!("{}-{}-test", provider, model),
56            format!("Test {} agent with model {}", provider, model),
57        );
58
59        Self {
60            metadata,
61            config: AgentConfig::default(),
62            _provider: provider.to_string(),
63            _model: model.to_string(),
64            responses: vec!["Test response".to_string()],
65            response_index: std::sync::Mutex::new(0),
66        }
67    }
68
69    /// Add a response to the agent's response queue
70    pub fn add_response(&mut self, response: String) {
71        // Clear default response if this is the first custom response
72        if self.responses.len() == 1 && self.responses[0] == "Test response" {
73            self.responses.clear();
74        }
75        self.responses.push(response);
76    }
77
78    /// Set the agent configuration
79    pub fn with_config(mut self, config: AgentConfig) -> Self {
80        self.config = config;
81        self
82    }
83}

85#[async_trait::async_trait]
86impl llmspell_core::traits::base_agent::BaseAgent for TestProviderAgent {
87    fn metadata(&self) -> &ComponentMetadata {
88        &self.metadata
89    }
90
91    async fn execute_impl(
92        &self,
93        _input: AgentInput,
94        _context: ExecutionContext,
95    ) -> Result<AgentOutput, LLMSpellError> {
96        let mut index = self.response_index.lock().unwrap();
97        let response = if *index < self.responses.len() {
98            self.responses[*index].clone()
99        } else {
100            // Cycle through responses
101            self.responses[*index % self.responses.len()].clone()
102        };
103        *index += 1;
104
105        Ok(AgentOutput::text(response))
106    }
107
108    async fn validate_input(&self, _input: &AgentInput) -> Result<(), LLMSpellError> {
109        // Test agent accepts all inputs
110        Ok(())
111    }
112
113    async fn handle_error(&self, error: LLMSpellError) -> Result<AgentOutput, LLMSpellError> {
114        Ok(AgentOutput::text(format!("Error handled: {}", error)))
115    }
116}

140/// Builder for creating test agents
141pub struct AgentTestBuilder {
142    name: String,
143    provider: Option<String>,
144    model: Option<String>,
145    config: AgentConfig,
146    responses: Vec<String>,
147    metadata_overrides: Option<ComponentMetadata>,
148}
149
150impl AgentTestBuilder {
151    /// Create a new agent test builder
152    pub fn new(name: &str) -> Self {
153        Self {
154            name: name.to_string(),
155            provider: None,
156            model: None,
157            config: AgentConfig::default(),
158            responses: Vec::new(),
159            metadata_overrides: None,
160        }
161    }
162
163    /// Set the provider
164    pub fn with_provider(mut self, provider: &str) -> Self {
165        self.provider = Some(provider.to_string());
166        self
167    }
168
169    /// Set the model
170    pub fn with_model(mut self, model: &str) -> Self {
171        self.model = Some(model.to_string());
172        self
173    }
174
175    /// Set the system prompt
176    pub fn with_system_prompt(mut self, prompt: &str) -> Self {
177        self.config.system_prompt = Some(prompt.to_string());
178        self
179    }
180
181    /// Set the temperature
182    pub fn with_temperature(mut self, temperature: f32) -> Self {
183        self.config.temperature = Some(temperature);
184        self
185    }
186
187    /// Set max tokens
188    pub fn with_max_tokens(mut self, max_tokens: usize) -> Self {
189        self.config.max_tokens = Some(max_tokens);
190        self
191    }
192
193    /// Add a response
194    pub fn with_response(mut self, response: &str) -> Self {
195        self.responses.push(response.to_string());
196        self
197    }
198
199    /// Add multiple responses
200    pub fn with_responses(mut self, responses: Vec<String>) -> Self {
201        self.responses.extend(responses);
202        self
203    }
204
205    /// Override metadata
206    pub fn with_metadata(mut self, metadata: ComponentMetadata) -> Self {
207        self.metadata_overrides = Some(metadata);
208        self
209    }
210
211    /// Build the test agent
212    pub async fn build(self) -> Result<TestProviderAgent, LLMSpellError> {
213        let provider = self.provider.unwrap_or_else(|| "mock".to_string());
214        let model = self.model.unwrap_or_else(|| "test-model".to_string());
215
216        let mut agent = TestProviderAgent::new(&provider, &model);
217        agent.config = self.config;
218
219        if let Some(metadata) = self.metadata_overrides {
220            agent.metadata = metadata;
221        } else {
222            agent.metadata =
223                ComponentMetadata::new(self.name.clone(), format!("Test agent: {}", self.name));
224        }
225
226        if self.responses.is_empty() {
227            agent.responses = vec![format!("Response from {} agent", self.name)];
228        } else {
229            agent.responses = self.responses;
230        }
231
232        Ok(agent)
233    }
234}

304/// Test conversation builders
305pub mod conversations {
306    use super::*;
307    use llmspell_core::traits::agent::MessageRole;

309    /// Build a simple conversation
310    pub fn simple_conversation() -> Vec<ConversationMessage> {
311        vec![
312            ConversationMessage::system("You are a helpful assistant.".to_string()),
313            ConversationMessage::user("Hello!".to_string()),
314            ConversationMessage::assistant("Hello! How can I help you today?".to_string()),
315        ]
316    }
317
318    /// Build a conversation with context
319    pub fn conversation_with_context(context: &str) -> Vec<ConversationMessage> {
320        vec![
321            ConversationMessage::system(format!(
322                "You are a helpful assistant. Context: {}",
323                context
324            )),
325            ConversationMessage::user("What's the context?".to_string()),
326            ConversationMessage::assistant(format!("The context is: {}", context)),
327        ]
328    }
329
330    /// Build a multi-turn conversation
331    pub fn multi_turn_conversation(turns: usize) -> Vec<ConversationMessage> {
332        let mut messages = vec![ConversationMessage::system(
333            "You are a helpful assistant.".to_string(),
334        )];
335
336        for i in 0..turns {
337            messages.push(ConversationMessage::user(format!("Question {}", i + 1)));
338            messages.push(ConversationMessage::assistant(format!(
339                "Answer to question {}",
340                i + 1
341            )));
342        }
343
344        messages
345    }
346
347    /// Build a conversation from raw messages
348    pub fn from_raw_messages(messages: Vec<(MessageRole, String)>) -> Vec<ConversationMessage> {
349        messages
350            .into_iter()
351            .map(|(role, content)| ConversationMessage {
352                role,
353                content,
354                timestamp: chrono::Utc::now(),
355            })
356            .collect()
357    }
358}

364    /// Test error handling scenario
365    pub async fn test_agent_error_handling(
366        agent: &impl Agent,
367    ) -> Result<Vec<AgentOutput>, LLMSpellError> {
368        let test_cases = vec![
369            AgentInput::text(""),                // Empty input
370            AgentInput::text("a".repeat(10000)), // Very long input
371            create_malformed_input(),            // Malformed input
372        ];
373
374        let mut results = Vec::new();
375        for input in test_cases {
376            match agent.execute(input, ExecutionContext::default()).await {
377                Ok(output) => results.push(output),
378                Err(_) => results.push(AgentOutput::text("Error handled")),
379            }
380        }
381
382        Ok(results)
383    }
384
385    /// Test conversation memory
386    pub async fn test_conversation_memory(agent: &impl Agent) -> Result<bool, LLMSpellError> {
387        let _history = conversations::simple_conversation();
388        let input = AgentInput::text("What did we just talk about?");
389
390        // For test purposes, just check if agent can execute
391        let output = agent.execute(input, ExecutionContext::default()).await?;
392
393        // Check if we got a response
394        Ok(!output.text.is_empty())
395    }
