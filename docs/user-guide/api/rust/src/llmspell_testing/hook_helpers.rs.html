4use llmspell_hooks::{
5    context::HookContext,
6    types::{ComponentId, ComponentType, HookPoint},
7};
8use uuid::Uuid;
9
10/// Create a test hook context with default values
11pub fn create_test_hook_context() -> HookContext {
12    create_test_hook_context_with_point(HookPoint::BeforeAgentExecution)
13}
14
15/// Create a test hook context with specific hook point
16pub fn create_test_hook_context_with_point(point: HookPoint) -> HookContext {
17    let component_id = ComponentId::new(ComponentType::Agent, "test-agent".to_string());
18    HookContext::new(point, component_id)
19}
20
21/// Create a test hook context with custom component
22pub fn create_test_hook_context_with_component(
23    point: HookPoint,
24    component_type: ComponentType,
25    component_name: &str,
26) -> HookContext {
27    let component_id = ComponentId::new(component_type, component_name.to_string());
28    HookContext::new(point, component_id)
29}
30
31/// Create a test hook context with correlation ID
32pub fn create_test_hook_context_with_correlation(correlation_id: Uuid) -> HookContext {
33    let component_id = ComponentId::new(ComponentType::Agent, "test-agent".to_string());
34    let mut context = HookContext::new(HookPoint::BeforeAgentExecution, component_id);
35    context.correlation_id = correlation_id;
36    context
37}
38
39/// Create a test hook context with data
40pub fn create_test_hook_context_with_data(
41    data_key: &str,
42    data_value: serde_json::Value,
43) -> HookContext {
44    let component_id = ComponentId::new(ComponentType::Agent, "test-agent".to_string());
45    let mut context = HookContext::new(HookPoint::BeforeAgentExecution, component_id);
46    context.insert_data(data_key.to_string(), data_value);
47    context
48}
49
50/// Create a test hook context for tool execution
51pub fn create_test_tool_hook_context(tool_name: &str) -> HookContext {
52    let component_id = ComponentId::new(ComponentType::Tool, tool_name.to_string());
53    HookContext::new(HookPoint::BeforeToolExecution, component_id)
54}
55
56/// Create a test hook context for workflow execution
57pub fn create_test_workflow_hook_context(workflow_name: &str) -> HookContext {
58    let component_id = ComponentId::new(ComponentType::Workflow, workflow_name.to_string());
59    HookContext::new(HookPoint::BeforeWorkflowStart, component_id)
60}
61
62/// Create a test hook context with error
63pub fn create_test_error_hook_context(error_message: &str) -> HookContext {
64    let component_id = ComponentId::new(ComponentType::Agent, "test-agent".to_string());
65    let mut context = HookContext::new(HookPoint::AgentError, component_id);
66    context.insert_data(
67        "error".to_string(),
68        serde_json::json!({
69            "message": error_message,
70            "type": "TestError"
71        }),
72    );
73    context
74}
