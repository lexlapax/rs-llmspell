19    pub fn run_tests(
20        &self,
21        categories: Vec<String>,
22        filter: Option<String>,
23        coverage: bool,
24        format: String,
25    ) -> Result<(), Box<dyn std::error::Error>> {
26        let categories: Result<Vec<_>, _> = categories
27            .iter()
28            .map(|c| c.parse::<TestCategory>())
29            .collect();
30        let categories = categories?;
31
32        // If no categories specified, show help
33        if categories.is_empty() {
34            println!("{}", "No test categories specified!".red());
35            println!();
36            println!("Usage: llmspell-test run <CATEGORY>...");
37            println!(
38                "Available categories: unit, integration, agent, scenario, lua, performance, all"
39            );
40            return Ok(());
41        }
42
43        println!(
44            "{} {}",
45            "Running tests for:".green().bold(),
46            categories
47                .iter()
48                .map(|c| c.name())
49                .collect::<Vec<_>>()
50                .join(", ")
51        );
52        println!();
53
54        let start = Instant::now();
55        let mut total_passed = 0;
56        let mut total_failed = 0;
57        let mut failed_categories = Vec::new();
58
59        // Expand "all" category
60        let categories_to_run = if categories.contains(&TestCategory::All) {
61            TestCategory::all_categories()
62        } else {
63            categories
64        };
65
66        // Run each category
67        for category in &categories_to_run {
68            if category == &TestCategory::Performance {
69                // Skip performance tests in regular test runs
70                println!(
71                    "{}: {} (use 'llmspell-test bench' instead)",
72                    "Skipping".yellow(),
73                    category.name()
74                );
75                continue;
76            }
77
78            println!("{} {} tests...", "Running".cyan(), category.name());
79
80            match self.run_category_tests(category, &filter, &format) {
81                Ok((passed, failed)) => {
82                    total_passed += passed;
83                    total_failed += failed;
84                    if failed > 0 {
85                        failed_categories.push(category.name());
86                    }
87                }
88                Err(e) => {
89                    eprintln!(
90                        "{}: Failed to run {} tests: {}",
91                        "Error".red().bold(),
92                        category.name(),
93                        e
94                    );
95                    failed_categories.push(category.name());
96                    total_failed += 1;
97                }
98            }
99            println!();
100        }
101
102        // Summary
103        let duration = start.elapsed();
104        println!("{}", "=".repeat(60));
105        println!("{}", "Test Summary".green().bold());
106        println!("{}: {:?}", "Duration".cyan(), duration);
107        println!("{}: {}", "Passed".green(), total_passed);
108        println!("{}: {}", "Failed".red(), total_failed);
109
110        if !failed_categories.is_empty() {
111            println!();
112            println!(
113                "{}: {}",
114                "Failed categories".red().bold(),
115                failed_categories.join(", ")
116            );
117            return Err("Some tests failed".into());
118        }
119
120        println!();
121        println!("{}", "All tests passed! ðŸŽ‰".green().bold());
122
123        // Run coverage if requested
124        if coverage {
125            println!();
126            println!("{}", "Generating coverage report...".cyan());
127            self.run_coverage(&categories_to_run)?;
128        }
129
130        Ok(())
131    }

133    fn run_category_tests(
134        &self,
135        category: &TestCategory,
136        filter: &Option<String>,
137        format: &str,
138    ) -> Result<(usize, usize), Box<dyn std::error::Error>> {
139        let mut cmd = Command::new("cargo");
140        cmd.arg("test");
141        cmd.arg("-p").arg("llmspell-testing");
142        cmd.arg("--features").arg(category.feature_name());
143
144        if self.config.release {
145            cmd.arg("--release");
146        }
147
148        if self.config.jobs > 0 {
149            cmd.arg("-j").arg(self.config.jobs.to_string());
150        }
151
152        if let Some(filter) = filter {
153            cmd.arg(filter);
154        }
155
156        if self.config.nocapture {
157            cmd.arg("--").arg("--nocapture");
158        } else if !self.config.verbose {
159            cmd.arg("--quiet");
160        }
161
162        // Execute and capture output
163        let output = if self.config.verbose || format == "pretty" {
164            // Stream output in real-time
165            let mut child = cmd.stdout(Stdio::piped()).stderr(Stdio::piped()).spawn()?;
166
167            let stdout = child.stdout.take().unwrap();
168            let reader = BufReader::new(stdout);
169
170            let mut passed = 0;
171            let mut failed = 0;
172
173            for line in reader.lines() {
174                let line = line?;
175                // Parse test results
176                if line.contains("test result:") {
177                    if let Some(captures) = parse_test_results(&line) {
178                        passed = captures.0;
179                        failed = captures.1;
180                    }
181                }
182                if self.config.verbose {
183                    println!("{}", line);
184                }
185            }
186
187            let status = child.wait()?;
188            if !status.success() && failed == 0 {
189                // Compilation or other error
190                failed = 1;
191            }
192
193            Ok((passed, failed))
194        } else {
195            // Quiet mode - just get the results
196            let output = cmd.output()?;
197
198            if output.status.success() {
199                // Try to parse test count from output
200                let stdout = String::from_utf8_lossy(&output.stdout);
201                if let Some((passed, failed)) = extract_test_counts(&stdout) {
202                    Ok((passed, failed))
203                } else {
204                    // Assume all passed if we can't parse
205                    Ok((1, 0))
206                }
207            } else {
208                Ok((0, 1))
209            }
210        };
211
212        output
213    }

215    pub fn run_benchmarks(
216        &self,
217        filter: Option<String>,
218        save: Option<String>,
219        baseline: Option<String>,
220    ) -> Result<(), Box<dyn std::error::Error>> {
221        println!("{}", "Running benchmarks...".green().bold());
222        println!();
223
224        let mut cmd = Command::new("cargo");
225        cmd.arg("bench");
226        cmd.arg("-p").arg("llmspell-testing");
227
228        if self.config.release {
229            cmd.arg("--release");
230        }
231
232        if let Some(filter) = filter {
233            cmd.arg("--bench").arg(filter);
234        }
235
236        // Criterion-specific options
237        let mut bench_args = Vec::new();
238
239        if let Some(ref save_name) = save {
240            bench_args.push(format!("--save-baseline={}", save_name));
241        }
242
243        if let Some(baseline) = baseline {
244            bench_args.push(format!("--baseline={}", baseline));
245        }
246
247        if !bench_args.is_empty() {
248            cmd.arg("--");
249            for arg in bench_args {
250                cmd.arg(arg);
251            }
252        }
253
254        // Execute with streaming output
255        let mut child = cmd
256            .stdout(Stdio::inherit())
257            .stderr(Stdio::inherit())
258            .spawn()?;
259
260        let status = child.wait()?;
261
262        if !status.success() {
263            return Err("Benchmark execution failed".into());
264        }
265
266        println!();
267        println!("{}", "Benchmarks completed successfully!".green().bold());
268
269        if save.is_some() {
270            println!("Baseline saved for future comparisons");
271        }
272
273        Ok(())
274    }

276    fn run_coverage(&self, categories: &[TestCategory]) -> Result<(), Box<dyn std::error::Error>> {
277        // Delegate to the test-coverage.sh script
278        let mut cmd = Command::new("./scripts/test-coverage.sh");
279
280        // Determine coverage type based on categories
281        let coverage_type = if categories.len() == TestCategory::all_categories().len() {
282            "all"
283        } else if categories.len() == 1 {
284            match categories[0] {
285                TestCategory::Unit => "unit",
286                TestCategory::Integration => "integration",
287                _ => "all",
288            }
289        } else {
290            "all"
291        };
292
293        cmd.arg(coverage_type);
294        cmd.arg("html"); // Default to HTML output
295
296        let status = cmd.status()?;
297
298        if !status.success() {
299            return Err("Coverage generation failed".into());
300        }
301
302        Ok(())
303    }

306fn parse_test_results(line: &str) -> Option<(usize, usize)> {
307    // Parse lines like: "test result: ok. 42 passed; 1 failed; 0 ignored; 0 measured; 0 filtered out"
308    let parts: Vec<&str> = line.split(';').collect();
309
310    let mut passed = 0;
311    let mut failed = 0;
312
313    for part in parts {
314        let part = part.trim();
315        if part.contains("passed") {
316            if let Some(num) = part.split_whitespace().next() {
317                passed = num.parse().unwrap_or(0);
318            }
319        } else if part.contains("failed") {
320            if let Some(num) = part.split_whitespace().next() {
321                failed = num.parse().unwrap_or(0);
322            }
323        }
324    }
325
326    Some((passed, failed))
327}
328
329fn extract_test_counts(output: &str) -> Option<(usize, usize)> {
330    for line in output.lines() {
331        if line.contains("test result:") {
332            return parse_test_results(line);
333        }
334    }
335    None
