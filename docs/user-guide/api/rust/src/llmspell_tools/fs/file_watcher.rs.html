75            ),
76            config,
77            sandbox,
78        }

81    /// Start watching files and return events
82    #[allow(clippy::unused_async)]
83    async fn watch_files(&self, watch_config: WatchConfig) -> AnyhowResult<Vec<FileEvent>> {
84        // Validate configuration
85        watch_config.validate()?;
86
87        // Validate paths with sandbox
88        for path in &watch_config.paths {
89            if self.sandbox.validate_path(path).is_err() {
90                return Err(anyhow::anyhow!(
91                    "Path not allowed by sandbox: {}",
92                    path.display()
93                ));
94            }
95        }
96
97        if watch_config.paths.len() > self.config.max_paths {
98            return Err(anyhow::anyhow!(
99                "Too many paths to watch: {} (max: {})",
100                watch_config.paths.len(),
101                self.config.max_paths
102            ));
103        }
104
105        let (tx, rx) = mpsc::channel();
106        let mut events = Vec::new();
107
108        // Create watcher
109        let mut watcher = RecommendedWatcher::new(
110            move |res| {
111                if let Err(e) = tx.send(res) {
112                    error!("Failed to send watch event: {}", e);
113                }
114            },
115            notify::Config::default(),
116        )?;
117
118        // Start watching paths
119        for path in &watch_config.paths {
120            let mode = if watch_config.recursive {
121                RecursiveMode::Recursive
122            } else {
123                RecursiveMode::NonRecursive
124            };
125
126            if let Err(e) = watcher.watch(path, mode) {
127                warn!("Failed to watch path {}: {}", path.display(), e);
128                continue;
129            }
130
131            info!("Started watching path: {}", path.display());
132        }
133
134        // Collect events
135        let timeout = Duration::from_secs(
136            watch_config
137                .timeout_seconds
138                .unwrap_or(self.config.default_timeout),
139        );
140        let start_time = std::time::Instant::now();
141
142        while start_time.elapsed() < timeout && events.len() < self.config.max_events {
143            match rx.recv_timeout(Duration::from_millis(100)) {
144                Ok(Ok(event)) => {
145                    if let Some(file_event) = self.convert_notify_event(event, &watch_config) {
146                        events.push(file_event);
147                    }
148                }
149                Ok(Err(e)) => {
150                    warn!("Watch error: {}", e);
151                }
152                Err(mpsc::RecvTimeoutError::Timeout) => {
153                    // Check if we should continue waiting
154                    if !events.is_empty() {
155                        // We have some events, check if we should wait for more
156                        thread::sleep(Duration::from_millis(watch_config.debounce_ms));
157                        break;
158                    }
159                }
160                Err(mpsc::RecvTimeoutError::Disconnected) => {
161                    break;
162                }
163            }
164        }
165
166        // Apply debouncing
167        let debounced_events = debounce_events(events, watch_config.debounce_ms);
168
169        info!("Collected {} file events", debounced_events.len());
170        Ok(debounced_events)
171    }

173    /// Convert notify event to our `FileEvent`
174    fn convert_notify_event(
175        &self,
176        event: notify::Event,
177        config: &WatchConfig,
178    ) -> Option<FileEvent> {
179        let event_type = match event.kind {
180            notify::EventKind::Create(_) => FileEventType::Create,
181            notify::EventKind::Modify(_) => FileEventType::Modify,
182            notify::EventKind::Remove(_) => FileEventType::Delete,
183            _ => FileEventType::Other,
184        };
185
186        // Handle paths
187        let paths = event.paths;
188        if paths.is_empty() {
189            return None;
190        }
191
192        let path = paths[0].clone();
193
194        // Check if path should be watched
195        if !llmspell_utils::file_monitor::should_watch_path(&path, config) {
196            return None;
197        }
198
199        // Check sandbox permissions
200        if self.sandbox.validate_path(&path).is_err() {
201            return None;
202        }
203
204        // Handle rename events
205        if paths.len() > 1 {
206            Some(FileEvent::new_rename(paths[0].clone(), paths[1].clone()))
207        } else {
208            Some(FileEvent::new(event_type, path))
209        }
210    }
211
212    /// Validate parameters for file watching operations
213    #[allow(clippy::unused_async)]
214    async fn validate_parameters(&self, params: &Value) -> Result<()> {
215        if !params.is_object() {
216            return Err(LLMSpellError::Validation {
217                message: "Parameters must be a JSON object".to_string(),
218                field: Some("parameters".to_string()),
219            });
220        }
221
222        let operation = params.get("operation").and_then(|v| v.as_str());
223        if let Some(op) = operation {
224            if !matches!(op, "watch" | "config") {
225                return Err(LLMSpellError::Validation {
226                    message: format!("Invalid operation: {op}"),
227                    field: Some("operation".to_string()),
228                });
229            }
230        }
231
232        Ok(())
233    }

236impl Default for FileWatcherTool {
237    fn default() -> Self {
238        // Create a default sandbox context for testing
239        let security_requirements = SecurityRequirements {
240            level: SecurityLevel::Restricted,
241            file_permissions: vec!["*".to_string()],
242            network_permissions: vec![],
243            env_permissions: vec![],
244            custom_requirements: HashMap::new(),
245        };
246        let resource_limits = ResourceLimits::default();
247        let context = SandboxContext::new(
248            "default_file_watcher".to_string(),
249            security_requirements,
250            resource_limits,
251        );
252        let sandbox = Arc::new(FileSandbox::new(context).unwrap());
253        Self::new(FileWatcherConfig::default(), sandbox)
254    }

257#[async_trait]
258impl BaseAgent for FileWatcherTool {
259    fn metadata(&self) -> &ComponentMetadata {
260        &self.metadata
261    }

263    async fn execute_impl(
264        &self,
265        input: AgentInput,
266        _context: ExecutionContext,
267    ) -> Result<AgentOutput> {
268        // Get parameters using shared utility
269        let params = extract_parameters(&input)?;
270
271        // Validate parameters
272        self.validate_parameters(params).await?;
273
274        let operation = extract_required_string(params, "operation")?;
275
276        match operation {
277            "watch" => {
278                let paths_array = extract_required_array(params, "input")?;
279                let paths: Vec<PathBuf> = paths_array
280                    .iter()
281                    .map(|v| {
282                        v.as_str()
283                            .ok_or_else(|| LLMSpellError::Validation {
284                                message: "Path must be a string".to_string(),
285                                field: Some("input".to_string()),
286                            })
287                            .map(PathBuf::from)
288                    })
289                    .collect::<Result<Vec<_>>>()?;
290
291                let recursive = extract_optional_bool(params, "recursive").unwrap_or(true);
292                let pattern = extract_optional_string(params, "pattern")
293                    .map(std::string::ToString::to_string);
294                let debounce_ms = extract_optional_u64(params, "debounce_ms")
295                    .unwrap_or(self.config.default_debounce_ms);
296                // Support both timeout_ms and timeout_seconds for flexibility
297                let timeout_seconds = extract_optional_u64(params, "timeout_ms").map_or_else(
298                    || {
299                        extract_optional_u64(params, "timeout_seconds")
300                            .unwrap_or(self.config.default_timeout)
301                    },
302                    |ms| ms.div_ceil(1000), // Round up to nearest second
303                );
304                let max_events = usize::try_from(
305                    extract_optional_u64(params, "max_events")
306                        .unwrap_or(self.config.max_events as u64),
307                )
308                .unwrap_or(usize::MAX);
309
310                let mut watch_config = WatchConfig::new()
311                    .recursive(recursive)
312                    .debounce(Duration::from_millis(debounce_ms))
313                    .timeout(Duration::from_secs(timeout_seconds))
314                    .max_events(max_events);
315
316                for path in paths {
317                    watch_config = watch_config.add_path(path);
318                }
319
320                if let Some(p) = pattern {
321                    watch_config = watch_config.pattern(p);
322                }
323
324                let events =
325                    self.watch_files(watch_config)
326                        .await
327                        .map_err(|e| LLMSpellError::Tool {
328                            message: format!("File watching failed: {e}"),
329                            tool_name: Some("file_watcher".to_string()),
330                            source: None,
331                        })?;
332
333                let response = ResponseBuilder::success("watch")
334                    .with_message(format!("Captured {} file events", events.len()))
335                    .with_result(json!({
336                        "events": events,
337                        "event_count": events.len()
338                    }))
339                    .build();
340
341                Ok(AgentOutput::text(serde_json::to_string_pretty(&response)?))
342            }
343            "config" => {
344                let response = ResponseBuilder::success("config")
345                    .with_message("File watcher configuration")
346                    .with_result(json!({
347                        "max_events": self.config.max_events,
348                        "default_timeout": self.config.default_timeout,
349                        "default_debounce_ms": self.config.default_debounce_ms,
350                        "max_paths": self.config.max_paths
351                    }))
352                    .build();
353
354                Ok(AgentOutput::text(serde_json::to_string_pretty(&response)?))
355            }
356            _ => Err(LLMSpellError::Validation {
357                message: format!("Unknown operation: {operation}"),
358                field: Some("operation".to_string()),
359            }),
360        }
361    }
362
363    async fn validate_input(&self, input: &AgentInput) -> Result<()> {
364        if input.text.is_empty() {
365            return Err(LLMSpellError::Validation {
366                message: "Input prompt cannot be empty".to_string(),
367                field: Some("prompt".to_string()),
368            });
369        }
370        Ok(())
371    }
372
373    async fn handle_error(&self, error: LLMSpellError) -> Result<AgentOutput> {
374        Ok(AgentOutput::text(format!("File watcher error: {error}")))
375    }
