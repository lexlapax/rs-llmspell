24/// JSON processing operation types
25#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
26#[serde(rename_all = "snake_case")]
27pub enum JsonOperation {
28    /// Transform JSON using full jq syntax
29    Query,
30    /// Validate JSON against a schema
31    Validate,
32    /// Stream process JSON lines
33    Stream,
34}
35
36impl std::fmt::Display for JsonOperation {
37    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
38        match self {
39            Self::Query => write!(f, "query"),
40            Self::Validate => write!(f, "validate"),
41            Self::Stream => write!(f, "stream"),
42        }
43    }
44}
45
46impl std::str::FromStr for JsonOperation {
47    type Err = LLMSpellError;
48
49    fn from_str(s: &str) -> Result<Self> {
50        match s.to_lowercase().as_str() {
51            "query" | "transform" | "jq" => Ok(Self::Query),
52            "validate" => Ok(Self::Validate),
53            "stream" => Ok(Self::Stream),
54            _ => Err(LLMSpellError::Validation {
55                message: format!("Unknown JSON operation: {s}"),
56                field: Some("operation".to_string()),
57            }),
58        }
59    }
60}

101    /// Validate JQ query for security issues
102    #[allow(clippy::unused_self)]
103    fn validate_jq_query(&self, query: &str) -> Result<()> {
104        // List of dangerous JQ functions and patterns
105        let dangerous_patterns = [
106            "env",        // Access environment variables
107            "__inputs",   // Access inputs beyond the provided data
108            "input",      // Read additional inputs
109            "inputs",     // Read all inputs
110            "debug",      // Debug output that might leak info
111            "modulemeta", // Module metadata access
112            "builtins",   // List all builtins
113            "$__loc__",   // Location metadata
114            "$ENV",       // Environment access
115            "include",    // Include external files
116            "import",     // Import modules
117        ];
118
119        let query_lower = query.to_lowercase();
120
121        for pattern in &dangerous_patterns {
122            if query_lower.contains(pattern) {
123                return Err(LLMSpellError::Validation {
124                    message: format!(
125                        "Security: JQ query contains potentially dangerous function: {pattern}"
126                    ),
127                    field: Some("query".to_string()),
128                });
129            }
130        }
131
132        // Check for suspicious patterns that might indicate attempts to access system info
133        if query_lower.contains("/etc/")
134            || query_lower.contains("/proc/")
135            || query_lower.contains("passwd")
136            || query_lower.contains("shadow")
137        {
138            return Err(LLMSpellError::Validation {
139                message: "Security: JQ query contains suspicious system path references"
140                    .to_string(),
141                field: Some("query".to_string()),
142            });
143        }
144
145        Ok(())
146    }

148    /// Execute a jq query on JSON data
149    fn execute_jq_query(&self, input: &Value, query: &str) -> Result<Vec<Value>> {
150        debug!("Executing jq query: {}", query);
151
152        // Validate query for security
153        self.validate_jq_query(query)?;
154
155        // Create parse context with core and std library
156        let mut parse_ctx = jaq_interpret::ParseCtx::new(Vec::new());
157        parse_ctx.insert_natives(jaq_core::core());
158        parse_ctx.insert_defs(jaq_std::std());
159
160        // Parse the filter
161        let (filter, errs) = jaq_parse::parse(query, jaq_parse::main());
162        if !errs.is_empty() {
163            let error_msg = errs
164                .into_iter()
165                .map(|e| e.to_string())
166                .collect::<Vec<_>>()
167                .join("; ");
168            return Err(LLMSpellError::Validation {
169                message: format!("Invalid jq syntax: {error_msg}"),
170                field: Some("query".to_string()),
171            });
172        }
173
174        let filter = filter.ok_or_else(|| LLMSpellError::Validation {
175            message: "Failed to parse jq filter".to_string(),
176            field: Some("query".to_string()),
177        })?;
178
179        // Compile the filter
180        let filter = parse_ctx.compile(filter);
181
182        if !parse_ctx.errs.is_empty() {
183            // jaq_interpret errors don't implement Display or Debug, so we'll use a generic message
184            return Err(LLMSpellError::Validation {
185                message: "Failed to compile jq filter: compilation errors occurred".to_string(),
186                field: Some("query".to_string()),
187            });
188        }
189
190        // Convert serde_json::Value to jaq Val
191        let jaq_input = Val::from(input.clone());
192
193        // Create empty inputs iterator
194        let inputs = jaq_interpret::RcIter::new(core::iter::empty());
195
196        // Run the filter
197        let ctx = jaq_interpret::Ctx::new([], &inputs);
198        let outputs = filter.run((ctx, jaq_input));
199
200        // Convert results back to serde_json::Value
201        let mut results = Vec::new();
202        for output in outputs {
203            match output {
204                Ok(val) => {
205                    // Convert Val to serde_json::Value
206                    let json_val: Value = val.into();
207                    results.push(json_val);
208                }
209                Err(e) => {
210                    return Err(LLMSpellError::Tool {
211                        message: format!("jq execution error: {e}"),
212                        tool_name: Some("json_processor".to_string()),
213                        source: None,
214                    })
215                }
216            }
217        }
218
219        Ok(results)
220    }

222    /// Validate JSON against a schema
223    #[allow(clippy::unused_async)]
224    async fn validate_json(&self, input: &Value, schema: &Value) -> Result<ValidationResult> {
225        debug!("Validating JSON against schema");
226
227        let compiled = JSONSchema::options()
228            .with_draft(Draft::Draft7)
229            .compile(schema)
230            .map_err(|e| LLMSpellError::Validation {
231                message: format!("Invalid JSON schema: {e}"),
232                field: Some("schema".to_string()),
233            })?;
234
235        let validation_result = compiled.validate(input);
236
237        match validation_result {
238            Ok(()) => Ok(ValidationResult {
239                is_valid: true,
240                errors: vec![],
241            }),
242            Err(errors) => {
243                let error_list: Vec<ValidationError> = errors
244                    .map(|error| ValidationError {
245                        path: error.instance_path.to_string(),
246                        message: error.to_string(),
247                        keyword: format!("{:?}", error.kind)
248                            .split('(')
249                            .next()
250                            .unwrap_or("Unknown")
251                            .to_string(),
252                    })
253                    .collect();
254
255                Ok(ValidationResult {
256                    is_valid: false,
257                    errors: error_list,
258                })
259            }
260        }
261    }

263    /// Process JSON lines with streaming
264    ///
265    /// # Errors
266    ///
267    /// Returns an error if:
268    /// - The JQ query is invalid or contains security risks
269    /// - JSON parsing fails for any line
270    /// - I/O errors occur while reading from the stream
271    pub async fn process_json_stream<R: AsyncRead + Unpin>(
272        &self,
273        reader: R,
274        query: &str,
275    ) -> Result<Vec<Value>> {
276        // Validate query for security before processing
277        self.validate_jq_query(query)?;
278
279        let mut buffer = BufReader::new(reader);
280        let mut line = String::new();
281        let mut results = Vec::new();
282
283        while buffer.read_line(&mut line).await? > 0 {
284            let trimmed = line.trim();
285            if trimmed.is_empty() {
286                line.clear();
287                continue;
288            }
289
290            // Parse JSON line
291            let value: Value =
292                serde_json::from_str(trimmed).map_err(|e| LLMSpellError::Validation {
293                    message: format!("Invalid JSON: {e}"),
294                    field: Some("input".to_string()),
295                })?;
296
297            // Apply jq query
298            let query_results = self.execute_jq_query(&value, query)?;
299            results.extend(query_results);
300
301            line.clear();
302        }
303
304        Ok(results)
305    }

307    /// Parse parameters from input
308    ///
309    /// # Errors
310    ///
311    /// Returns an error if the operation string cannot be parsed
312    fn parse_parameters(params: &Value) -> Result<(JsonOperation, Option<Value>, Option<String>)> {
313        let operation_str = params
314            .get("operation")
315            .and_then(|v| v.as_str())
316            .unwrap_or("query");
317        let operation: JsonOperation = operation_str.parse()?;
318
319        // Handle input - could be a JSON string or already parsed
320        let input = params.get("input").map(|v| {
321            v.as_str().map_or_else(
322                || v.clone(),
323                |s| {
324                    // Try to parse string as JSON
325                    serde_json::from_str(s).unwrap_or_else(|_| v.clone())
326                },
327            )
328        });
329        let query = params
330            .get("query")
331            .and_then(|v| v.as_str())
332            .map(String::from);
333
334        Ok((operation, input, query))
335    }

363    async fn execute_impl(
364        &self,
365        input: AgentInput,
366        _context: ExecutionContext,
367    ) -> Result<AgentOutput> {
368        // Get parameters using shared utility
369        let params = extract_parameters(&input)?;
370
371        let (operation, input_json, query) = Self::parse_parameters(params)?;
372
373        info!("Executing JSON {} operation", operation);
374        eprintln!("DEBUG: input_json = {:?}", input_json);
375
376        let result = match operation {
377            JsonOperation::Query => {
378                let input_val = input_json.ok_or_else(|| LLMSpellError::Validation {
379                    message: "Query operation requires 'input' parameter".to_string(),
380                    field: Some("input".to_string()),
381                })?;
382                let query_str = query.ok_or_else(|| LLMSpellError::Validation {
383                    message: "Query operation requires 'query' parameter".to_string(),
384                    field: Some("query".to_string()),
385                })?;
386
387                let results = self.execute_jq_query(&input_val, &query_str)?;
388
389                // If single result, return it directly; otherwise return array
390                if results.len() == 1 {
391                    results.into_iter().next().unwrap()
392                } else {
393                    Value::Array(results)
394                }
395            }
396            JsonOperation::Validate => {
397                let input_val = input_json.ok_or_else(|| LLMSpellError::Validation {
398                    message: "Validate operation requires 'input' parameter".to_string(),
399                    field: Some("input".to_string()),
400                })?;
401                let schema = params
402                    .get("schema")
403                    .ok_or_else(|| LLMSpellError::Validation {
404                        message: "Validate operation requires 'schema' parameter".to_string(),
405                        field: Some("schema".to_string()),
406                    })?;
407
408                let validation = self.validate_json(&input_val, schema).await?;
409                serde_json::to_value(validation)?
410            }
411            JsonOperation::Stream => {
412                // For stream operation, input should contain the JSON lines content
413                let stream_content = if let Some(input_val) = &input_json {
414                    // If input is a string, use it directly
415                    if let Some(content_str) = input_val.as_str() {
416                        content_str.to_string()
417                    } else {
418                        // Otherwise convert to string
419                        serde_json::to_string(input_val)?
420                    }
421                } else {
422                    return Err(LLMSpellError::Validation {
423                        message:
424                            "Stream operation requires 'input' parameter with JSON lines content"
425                                .to_string(),
426                        field: Some("input".to_string()),
427                    });
428                };
429                let query_str = query.unwrap_or_else(|| ".".to_string());
430
431                // Process JSON lines
432                let reader = stream_content.as_bytes();
433                let results = self.process_json_stream(reader, &query_str).await?;
434
435                // Return as array
436                Value::Array(results)
437            }
438        };
439
440        // Use ResponseBuilder for metadata, but return actual result as text
441        let message = match operation {
442            JsonOperation::Query => "JSON query executed successfully",
443            JsonOperation::Validate => "JSON validation completed",
444            JsonOperation::Stream => "JSON stream processing completed",
445        };
446
447        let response = ResponseBuilder::success(operation.to_string())
448            .with_message(message.to_string())
449            .with_result(result.clone())
450            .build();
451
452        let mut metadata = llmspell_core::types::OutputMetadata::default();
453        metadata.extra.insert(
454            "operation".to_string(),
455            Value::String(operation.to_string()),
456        );
457        metadata.extra.insert("response".to_string(), response);
458
459        // For data processing tools, return the actual result as text
460        let output_text = serde_json::to_string_pretty(&result)?;
461        Ok(AgentOutput::text(output_text).with_metadata(metadata))
462    }
463
464    async fn validate_input(&self, input: &AgentInput) -> Result<()> {
465        if input.parameters.is_empty() {
466            return Err(LLMSpellError::Validation {
467                message: "No parameters provided".to_string(),
468                field: Some("parameters".to_string()),
469            });
470        }
471
472        // Check size limit
473        if let Some(params) = input.parameters.get("parameters") {
474            let size = serde_json::to_string(params)?.len();
475            if size > self.config.max_input_size {
476                return Err(LLMSpellError::Validation {
477                    message: format!(
478                        "Input size {} bytes exceeds maximum {} bytes",
479                        size, self.config.max_input_size
480                    ),
481                    field: Some("input".to_string()),
482                });
483            }
484        }
485
486        Ok(())
487    }
488
489    async fn handle_error(&self, error: LLMSpellError) -> Result<AgentOutput> {
490        Ok(AgentOutput::text(format!("JSON processing error: {error}")))
491    }
