19/// Trait for tools that support resource limiting
20#[async_trait]
21pub trait ResourceLimited: Tool {
22    /// Get the resource limits for this tool
23    fn resource_limits(&self) -> ResourceLimits {
24        ResourceLimits::default()
25    }
26
27    /// Create a resource tracker for this tool
28    fn create_tracker(&self) -> ResourceTracker {
29        ResourceTracker::new(<Self as ResourceLimited>::resource_limits(self))
30    }
31}
32
33/// Wrapper that adds resource limiting to any tool
34pub struct ResourceLimitedTool<T: Tool> {
35    inner: T,
36    limits: ResourceLimits,
37}
38
39impl<T: Tool> ResourceLimitedTool<T> {
40    /// Create a new resource-limited tool
41    pub const fn new(tool: T, limits: ResourceLimits) -> Self {
42        Self {
43            inner: tool,
44            limits,
45        }
46    }
47
48    /// Create with default limits
49    pub fn with_defaults(tool: T) -> Self {
50        Self::new(tool, ResourceLimits::default())
51    }
52
53    /// Create with strict limits
54    pub fn with_strict_limits(tool: T) -> Self {
55        Self::new(tool, ResourceLimits::strict())
56    }
57}

59#[async_trait]
60impl<T: Tool + Send + Sync> BaseAgent for ResourceLimitedTool<T> {
61    fn metadata(&self) -> &ComponentMetadata {
62        self.inner.metadata()
63    }
64
65    async fn execute_impl(
66        &self,
67        input: AgentInput,
68        context: ExecutionContext,
69    ) -> LLMResult<AgentOutput> {
70        let tracker = ResourceTracker::new(self.limits.clone());
71
72        // Track concurrent operation
73        let _guard = tracker.track_concurrent_start()?;
74
75        // Execute with timeout
76        let result = tracker
77            .with_timeout(self.inner.execute(input, context))
78            .await??;
79
80        // Add resource metrics to output
81        let metrics = tracker.get_metrics();
82        let metrics_json = json!({
83            "resource_usage": {
84                "memory_bytes": metrics.memory_bytes,
85                "cpu_time_ms": metrics.cpu_time_ms,
86                "operations_count": metrics.operations_count,
87            }
88        });
89
90        let output_text = format!(
91            "{}
92\nResource usage: {}",
93            result.text,
94            serde_json::to_string_pretty(&metrics_json).unwrap()
95        );
96        Ok(AgentOutput::text(output_text))
97    }
98
99    async fn validate_input(&self, input: &AgentInput) -> LLMResult<()> {
100        self.inner.validate_input(input).await
101    }
102
103    async fn handle_error(&self, error: LLMSpellError) -> LLMResult<AgentOutput> {
104        self.inner.handle_error(error).await
105    }
106}

108#[async_trait]
109impl<T: Tool + Send + Sync> Tool for ResourceLimitedTool<T> {
110    fn category(&self) -> ToolCategory {
111        self.inner.category()
112    }
113
114    fn security_level(&self) -> SecurityLevel {
115        self.inner.security_level()
116    }
117
118    fn schema(&self) -> ToolSchema {
119        self.inner.schema()
120    }
121
122    fn security_requirements(&self) -> SecurityRequirements {
123        self.inner.security_requirements()
124    }
125
126    fn resource_limits(&self) -> ToolResourceLimits {
127        // Convert our ResourceLimits to Tool's ResourceLimits
128        match self.limits.max_memory_bytes {
129            Some(mem) if mem <= 10 * 1024 * 1024 => ToolResourceLimits::strict(),
130            Some(_) => ToolResourceLimits::default(),
131            None => ToolResourceLimits::unlimited(),
132        }
133    }
134}

136/// Extension trait to easily add resource limiting to tools
137pub trait ResourceLimitExt: Tool + Sized {
138    /// Wrap this tool with resource limits
139    fn with_resource_limits(self, limits: ResourceLimits) -> ResourceLimitedTool<Self> {
140        ResourceLimitedTool::new(self, limits)
141    }
142
143    /// Wrap with default resource limits
144    fn with_default_limits(self) -> ResourceLimitedTool<Self> {
145        ResourceLimitedTool::with_defaults(self)
146    }
147
148    /// Wrap with strict resource limits
149    fn with_strict_limits(self) -> ResourceLimitedTool<Self> {
150        ResourceLimitedTool::with_strict_limits(self)
151    }
152}

157/// Helper to track file operations with size limits
158///
159/// # Errors
160///
161/// Returns an error if:
162/// - Operation tracking fails due to resource limits
163/// - File size exceeds configured limits
164#[allow(clippy::unused_async)]
165pub async fn check_file_operation(
166    tracker: &ResourceTracker,
167    path: &std::path::Path,
168    operation: &str,
169) -> LLMResult<()> {
170    use std::fs;
171
172    tracker.track_operation()?;
173
174    if operation == "read" || operation == "write" {
175        if let Ok(metadata) = fs::metadata(path) {
176            #[allow(clippy::cast_possible_truncation)]
177            let size = metadata.len() as usize;
178            tracker.check_file_size(size)?;
179        }
180    }
181
182    Ok(())
183}

185/// Helper to track memory allocation for data processing
186///
187/// # Errors
188///
189/// Returns an error if:
190/// - Memory allocation would exceed configured limits
191/// - The operation function returns an error
192pub fn track_data_processing<T, F>(
193    tracker: &ResourceTracker,
194    estimated_size: usize,
195    operation: F,
196) -> LLMResult<T>
197where
198    F: FnOnce() -> LLMResult<T>,
199{
200    use llmspell_utils::resource_limits::MemoryGuard;
201
202    // Track the memory allocation
203    let _guard = MemoryGuard::new(tracker, estimated_size)?;
204
205    // Track the operation
206    tracker.track_operation()?;
207    tracker.check_cpu_time()?;
208
209    // Execute the operation
210    operation()
211}
