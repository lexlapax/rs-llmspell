153    /// Get basic system information
154    ///
155    /// # Errors
156    ///
157    /// Returns an error if system information cannot be retrieved
158    #[allow(clippy::unused_async)]
159    async fn get_basic_system_info(&self) -> LLMResult<SystemStats> {
160        let system_info = get_system_info().map_err(|e| LLMSpellError::Tool {
161            message: format!("Failed to get system information: {e}"),
162            tool_name: Some("system_monitor".to_string()),
163            source: None,
164        })?;
165
166        // Calculate memory usage
167        let total_memory = system_info.total_memory.unwrap_or(0);
168        let available_memory = system_info.available_memory.unwrap_or(0);
169        let used_memory = total_memory.saturating_sub(available_memory);
170        let memory_usage_percent = if total_memory > 0 {
171            #[allow(clippy::cast_precision_loss)]
172            let usage_percent = (used_memory as f64 / total_memory as f64) * 100.0;
173            usage_percent
174        } else {
175            0.0
176        };
177
178        let stats = SystemStats {
179            cpu_usage_percent: 0.0, // Will be updated by CPU monitoring if enabled
180            cpu_count: system_info.cpu_cores,
181            total_memory_bytes: total_memory,
182            available_memory_bytes: available_memory,
183            used_memory_bytes: used_memory,
184            memory_usage_percent,
185            disk_usage: HashMap::new(), // Will be updated by disk monitoring if enabled
186            uptime_seconds: None,
187            load_average: None,
188            process_count: None,
189        };
190
191        debug!(
192            "Basic system info - CPU cores: {}, Memory: {} / {} bytes",
193            stats.cpu_count, stats.used_memory_bytes, stats.total_memory_bytes
194        );
195
196        Ok(stats)
197    }

199    /// Get CPU usage (simplified version without external dependencies)
200    #[allow(clippy::unused_async)]
201    async fn get_cpu_usage(&self) -> f64 {
202        if !self.config.collect.is_enabled(StatType::Cpu) {
203            return 0.0;
204        }
205
206        // For a basic implementation, we'll simulate CPU usage measurement
207        // In a real implementation, this would use platform-specific APIs
208        // or a crate like `sysinfo` to get actual CPU usage
209
210        // Simple CPU load approximation based on system load
211        #[cfg(unix)]
212        {
213            if let Ok(load_avg) = self.get_load_average(&self.sandbox) {
214                #[allow(clippy::cast_precision_loss)]
215                let cpu_count = get_cpu_count() as f64;
216                // Convert load average to approximate CPU percentage
217                let cpu_percent = (load_avg[0] / cpu_count * 100.0).min(100.0);
218                debug!("Estimated CPU usage: {:.1}%", cpu_percent);
219                return cpu_percent;
220            }
221        }
222
223        // Fallback: return a reasonable default
224        0.0
225    }

227    /// Get system load average (Unix only)
228    #[cfg(unix)]
229    #[allow(clippy::unused_self)]
230    fn get_load_average(&self, sandbox: &FileSandbox) -> Result<[f64; 3], std::io::Error> {
231        use std::fs;
232
233        let loadavg_path = sandbox
234            .validate_path("/proc/loadavg".as_ref())
235            .map_err(|e| std::io::Error::new(std::io::ErrorKind::PermissionDenied, e))?;
236        let loadavg_content = fs::read_to_string(&loadavg_path)?;
237        let parts: Vec<&str> = loadavg_content.split_whitespace().collect();
238
239        if parts.len() >= 3 {
240            let one_minute = parts[0].parse::<f64>().unwrap_or(0.0);
241            let five_minutes = parts[1].parse::<f64>().unwrap_or(0.0);
242            let fifteen_minutes = parts[2].parse::<f64>().unwrap_or(0.0);
243            Ok([one_minute, five_minutes, fifteen_minutes])
244        } else {
245            Err(std::io::Error::new(
246                std::io::ErrorKind::InvalidData,
247                "Invalid loadavg format",
248            ))
249        }
250    }

252    /// Get disk usage statistics
253    #[allow(clippy::unused_async)]
254    async fn get_disk_usage(&self) -> HashMap<String, DiskStats> {
255        if !self.config.collect.is_enabled(StatType::Disk) {
256            return HashMap::new();
257        }
258
259        let mut disk_stats = HashMap::new();
260
261        // Cross-platform disk usage implementation
262        #[cfg(unix)]
263        {
264            if let Ok(mounts) = self.get_unix_mounts(&self.sandbox) {
265                let mut count = 0;
266                for (mount_point, fs_type) in mounts {
267                    if count >= self.config.max_disk_mounts {
268                        break;
269                    }
270
271                    if let Ok(stats) = self.get_disk_stats_for_path(&mount_point) {
272                        let disk_stat = DiskStats {
273                            total_bytes: stats.total_bytes,
274                            available_bytes: stats.available_bytes,
275                            used_bytes: stats.used_bytes,
276                            usage_percent: stats.usage_percent,
277                            filesystem: if self.config.include_disk_details {
278                                Some(fs_type)
279                            } else {
280                                None
281                            },
282                        };
283                        disk_stats.insert(mount_point, disk_stat);
284                        count += 1;
285                    }
286                }
287            }
288        }
289
290        #[cfg(windows)]
291        {
292            // Windows implementation would use GetDiskFreeSpaceEx
293            // For now, we'll provide a minimal implementation
294            if let Ok(stats) = self.get_disk_stats_for_path("C:\\") {
295                let disk_stat = DiskStats {
296                    total_bytes: stats.total_bytes,
297                    available_bytes: stats.available_bytes,
298                    used_bytes: stats.used_bytes,
299                    usage_percent: stats.usage_percent,
300                    filesystem: if self.config.include_disk_details {
301                        Some("NTFS".to_string())
302                    } else {
303                        None
304                    },
305                };
306                disk_stats.insert("C:\\".to_string(), disk_stat);
307            }
308        }
309
310        debug!("Collected disk statistics for {} mounts", disk_stats.len());
311        disk_stats
312    }

314    /// Get Unix mount points
315    #[cfg(unix)]
316    #[allow(clippy::unused_self)]
317    fn get_unix_mounts(
318        &self,
319        sandbox: &FileSandbox,
320    ) -> Result<Vec<(String, String)>, std::io::Error> {
321        use std::fs;
322
323        let mounts_path = sandbox
324            .validate_path("/proc/mounts".as_ref())
325            .map_err(|e| std::io::Error::new(std::io::ErrorKind::PermissionDenied, e))?;
326        let mounts_content = fs::read_to_string(&mounts_path)?;
327        let mut mounts = Vec::new();
328
329        for line in mounts_content.lines() {
330            let parts: Vec<&str> = line.split_whitespace().collect();
331            if parts.len() >= 3 {
332                let mount_point = parts[1].to_string();
333                let fs_type = parts[2].to_string();
334
335                // Skip virtual filesystems
336                if !mount_point.starts_with("/proc")
337                    && !mount_point.starts_with("/sys")
338                    && !mount_point.starts_with("/dev/")
339                    && !fs_type.starts_with("tmpfs")
340                    && fs_type != "proc"
341                    && fs_type != "sysfs"
342                    && fs_type != "devpts"
343                {
344                    mounts.push((mount_point, fs_type));
345                }
346            }
347        }
348
349        Ok(mounts)
350    }

352    /// Get disk statistics for a specific path
353    #[allow(clippy::unused_self)]
354    fn get_disk_stats_for_path(&self, path: &str) -> Result<DiskStats, std::io::Error> {
355        #[cfg(unix)]
356        {
357            use std::ffi::CString;
358            use std::mem;
359
360            let path_c = CString::new(path)?;
361            // SAFETY: Creating a zeroed libc::statvfs struct is safe as all fields are scalar types
362            #[allow(unsafe_code)]
363            let mut statvfs: libc::statvfs = unsafe { mem::zeroed() };
364
365            // SAFETY: path_c is a valid C string and statvfs is a valid mutable reference
366            #[allow(unsafe_code)]
367            let result = unsafe { libc::statvfs(path_c.as_ptr(), &mut statvfs) };
368
369            if result == 0 {
370                let block_size = statvfs.f_bsize;
371                let total_blocks = u64::from(statvfs.f_blocks);
372                let available_blocks = u64::from(statvfs.f_bavail);
373
374                let total_bytes = total_blocks * block_size;
375                let available_bytes = available_blocks * block_size;
376                let used_bytes = total_bytes.saturating_sub(available_bytes);
377                let usage_percent = if total_bytes > 0 {
378                    #[allow(clippy::cast_precision_loss)]
379                    let percent = (used_bytes as f64 / total_bytes as f64) * 100.0;
380                    percent
381                } else {
382                    0.0
383                };
384
385                Ok(DiskStats {
386                    total_bytes,
387                    available_bytes,
388                    used_bytes,
389                    usage_percent,
390                    filesystem: None, // Will be set by caller if needed
391                })
392            } else {
393                Err(std::io::Error::last_os_error())
394            }
395        }
396
397        #[cfg(windows)]
398        {
399            use std::ffi::OsStr;
400            use std::os::windows::ffi::OsStrExt;
401            use std::ptr;
402
403            let path_wide: Vec<u16> = OsStr::new(path)
404                .encode_wide()
405                .chain(std::iter::once(0))
406                .collect();
407
408            let mut free_bytes = 0u64;
409            let mut total_bytes = 0u64;
410
411            let result = unsafe {
412                winapi::um::fileapi::GetDiskFreeSpaceExW(
413                    path_wide.as_ptr(),
414                    &mut free_bytes,
415                    &mut total_bytes,
416                    ptr::null_mut(),
417                )
418            };
419
420            if result != 0 {
421                let used_bytes = total_bytes.saturating_sub(free_bytes);
422                let usage_percent = if total_bytes > 0 {
423                    #[allow(clippy::cast_precision_loss)]
424                    let percent = (used_bytes as f64 / total_bytes as f64) * 100.0;
425                    percent
426                } else {
427                    0.0
428                };
429
430                Ok(DiskStats {
431                    total_bytes,
432                    available_bytes: free_bytes,
433                    used_bytes,
434                    usage_percent,
435                    filesystem: None,
436                })
437            } else {
438                Err(std::io::Error::last_os_error())
439            }
440        }
441
442        #[cfg(not(any(unix, windows)))]
443        {
444            // Fallback for unsupported platforms
445            Err(std::io::Error::new(
446                std::io::ErrorKind::Unsupported,
447                "Disk statistics not supported on this platform",
448            ))
449        }
450    }

452    /// Get process count (simplified implementation)
453    #[allow(clippy::unused_async)]
454    async fn get_process_count(&self, sandbox: &FileSandbox) -> Option<u32> {
455        if !self.config.collect.is_enabled(StatType::Process) {
456            return None;
457        }
458
459        #[cfg(unix)]
460        {
461            let proc_path = sandbox.validate_path("/proc".as_ref()).ok()?;
462            if let Ok(entries) = std::fs::read_dir(&proc_path) {
463                let count = entries
464                    .filter_map(std::result::Result::ok)
465                    .filter(|entry| {
466                        entry
467                            .file_name()
468                            .to_string_lossy()
469                            .chars()
470                            .all(|c| c.is_ascii_digit())
471                    })
472                    .count();
473                #[allow(clippy::cast_possible_truncation)]
474                let count = count as u32;
475                debug!("Process count: {}", count);
476                return Some(count);
477            }
478        }
479
480        #[cfg(windows)]
481        {
482            // Windows implementation would use process enumeration APIs
483            // For now, return None
484        }
485
486        None
487    }

489    /// Get system uptime (simplified implementation)
490    #[allow(clippy::unused_async)]
491    async fn get_uptime(&self, sandbox: &FileSandbox) -> Option<u64> {
492        #[cfg(unix)]
493        {
494            let uptime_path = sandbox.validate_path("/proc/uptime".as_ref()).ok()?;
495            if let Ok(uptime_content) = std::fs::read_to_string(&uptime_path) {
496                if let Some(uptime_str) = uptime_content.split_whitespace().next() {
497                    if let Ok(uptime_seconds) = uptime_str.parse::<f64>() {
498                        #[allow(clippy::cast_possible_truncation, clippy::cast_sign_loss)]
499                        let uptime_u64 = uptime_seconds as u64;
500                        debug!("System uptime: {} seconds", uptime_u64);
501                        return Some(uptime_u64);
502                    }
503                }
504            }
505        }
506
507        None
508    }

510    /// Collect comprehensive system statistics
511    ///
512    /// # Errors
513    ///
514    /// Returns an error if system information cannot be retrieved
515    async fn collect_system_stats(&self) -> LLMResult<SystemStats> {
516        let start_time = Instant::now();
517
518        // Start with basic system info
519        let mut stats = self.get_basic_system_info().await?;
520
521        // Collect CPU usage
522        if self.config.collect.is_enabled(StatType::Cpu) {
523            stats.cpu_usage_percent = self.get_cpu_usage().await;
524        }
525
526        // Collect disk usage
527        if self.config.collect.is_enabled(StatType::Disk) {
528            stats.disk_usage = self.get_disk_usage().await;
529        }
530
531        // Collect process count
532        if self.config.collect.is_enabled(StatType::Process) {
533            stats.process_count = self.get_process_count(&self.sandbox).await;
534        }
535
536        // Get uptime
537        stats.uptime_seconds = self.get_uptime(&self.sandbox).await;
538
539        // Get load average (Unix only)
540        #[cfg(unix)]
541        {
542            if let Ok(load_avg) = self.get_load_average(&self.sandbox) {
543                stats.load_average = Some(load_avg);
544            }
545        }
546
547        let collection_time = start_time.elapsed();
548        info!(
549            "System statistics collected in {}ms - CPU: {:.1}%, Memory: {:.1}%, Disks: {}",
550            {
551                #[allow(clippy::cast_possible_truncation)]
552                let millis = collection_time.as_millis() as u64;
553                millis
554            },
555            stats.cpu_usage_percent,
556            stats.memory_usage_percent,
557            stats.disk_usage.len()
558        );
559
560        Ok(stats)
561    }

563    /// Validate monitoring parameters
564    ///
565    /// # Errors
566    ///
567    /// Returns an error if:
568    /// - An invalid operation is specified
569    #[allow(clippy::unused_async)]
570    async fn validate_monitoring_parameters(&self, params: &serde_json::Value) -> LLMResult<()> {
571        // Validate operation if provided
572        if let Some(operation) = extract_optional_string(params, "operation") {
573            match operation {
574                "stats" | "cpu" | "memory" | "disk" | "all" => {}
575                _ => {
576                    return Err(LLMSpellError::Validation {
577                        message: format!(
578                            "Invalid operation: {operation}. Supported operations: stats, cpu, memory, disk, all"
579                        ),
580                        field: Some("operation".to_string()),
581                    });
582                }
583            }
584        }
585
586        Ok(())
587    }

591impl BaseAgent for SystemMonitorTool {

596    async fn execute_impl(
597        &self,
598        input: AgentInput,
599        _context: ExecutionContext,
600    ) -> LLMResult<AgentOutput> {
601        // Get parameters using shared utility
602        let params = extract_parameters(&input)?;
603
604        self.validate_monitoring_parameters(params).await?;
605
606        // Extract operation (default to "all")
607        let operation = extract_optional_string(params, "operation").unwrap_or("all");
608
609        // Collect system statistics
610        let stats = self.collect_system_stats().await?;
611
612        // Format response based on operation
613        let response = match operation {
614            "cpu" => {
615                let message = format!(
616                    "CPU usage: {:.1}% ({} cores)",
617                    stats.cpu_usage_percent, stats.cpu_count
618                );
619                ResponseBuilder::success("cpu")
620                    .with_message(message)
621                    .with_result(json!({
622                        "cpu_usage_percent": stats.cpu_usage_percent,
623                        "cpu_count": stats.cpu_count,
624                        "load_average": stats.load_average
625                    }))
626                    .build()
627            }
628            "memory" => {
629                let message = format!(
630                    "Memory usage: {:.1}% ({} / {} used)",
631                    stats.memory_usage_percent,
632                    format_bytes(stats.used_memory_bytes),
633                    format_bytes(stats.total_memory_bytes)
634                );
635                ResponseBuilder::success("memory")
636                    .with_message(message)
637                    .with_result(json!({
638                        "total_memory_bytes": stats.total_memory_bytes,
639                        "used_memory_bytes": stats.used_memory_bytes,
640                        "available_memory_bytes": stats.available_memory_bytes,
641                        "memory_usage_percent": stats.memory_usage_percent,
642                        "total_memory_formatted": format_bytes(stats.total_memory_bytes),
643                        "used_memory_formatted": format_bytes(stats.used_memory_bytes),
644                        "available_memory_formatted": format_bytes(stats.available_memory_bytes)
645                    }))
646                    .build()
647            }
648            "disk" => {
649                let message = format!(
650                    "Disk usage for {} mount points collected",
651                    stats.disk_usage.len()
652                );
653                ResponseBuilder::success("disk")
654                    .with_message(message)
655                    .with_result(json!({
656                        "disk_usage": stats.disk_usage
657                    }))
658                    .build()
659            }
660            "stats" | "all" => {
661                let message = format!(
662                    "System stats: CPU {:.1}%, Memory {:.1}% ({}/{}), {} disks, {} processes",
663                    stats.cpu_usage_percent,
664                    stats.memory_usage_percent,
665                    format_bytes(stats.used_memory_bytes),
666                    format_bytes(stats.total_memory_bytes),
667                    stats.disk_usage.len(),
668                    stats.process_count.unwrap_or(0)
669                );
670                ResponseBuilder::success("all")
671                    .with_message(message)
672                    .with_result(json!({
673                        "cpu_usage_percent": stats.cpu_usage_percent,
674                        "cpu_count": stats.cpu_count,
675                        "total_memory_bytes": stats.total_memory_bytes,
676                        "used_memory_bytes": stats.used_memory_bytes,
677                        "available_memory_bytes": stats.available_memory_bytes,
678                        "memory_usage_percent": stats.memory_usage_percent,
679                        "disk_usage": stats.disk_usage,
680                        "uptime_seconds": stats.uptime_seconds,
681                        "load_average": stats.load_average,
682                        "process_count": stats.process_count,
683                        "formatted": {
684                            "total_memory": format_bytes(stats.total_memory_bytes),
685                            "used_memory": format_bytes(stats.used_memory_bytes),
686                            "available_memory": format_bytes(stats.available_memory_bytes)
687                        }
688                    }))
689                    .build()
690            }
691            _ => unreachable!(), // Already validated above
692        };
693
694        Ok(AgentOutput::text(serde_json::to_string_pretty(&response)?))
695    }
