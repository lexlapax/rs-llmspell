27/// Service check result
28#[derive(Debug, Clone, Serialize, Deserialize)]
29pub struct ServiceCheckResult {
30    /// Target being checked (host:port or URL)
31    pub target: String,
32    /// Whether the service is available
33    pub available: bool,
34    /// Response time in milliseconds
35    pub response_time_ms: u64,
36    /// Status message
37    pub status: String,
38    /// Error message if check failed
39    pub error: Option<String>,
40    /// Additional metadata about the check
41    pub metadata: HashMap<String, String>,
42}
43
44/// Service check type
45#[derive(Debug, Clone, Serialize, Deserialize)]
46pub enum ServiceCheckType {
47    /// TCP port connectivity check
48    TcpPort,
49    /// HTTP/HTTPS health check
50    Http,
51    /// HTTPS health check
52    Https,
53    /// DNS resolution check
54    Dns,
55}

144    /// Check if a port is allowed to be checked
145    fn is_port_allowed(&self, port: u16) -> bool {
146        // Check blocked ports first (takes precedence)
147        if self.config.blocked_ports.contains(&port) {
148            debug!("Port {} is blocked", port);
149            return false;
150        }
151
152        // If arbitrary ports are allowed, allow it
153        if self.config.allow_arbitrary_ports {
154            return true;
155        }
156
157        // Check allowed ports
158        if self.config.allowed_ports.contains(&port) {
159            debug!("Port {} is allowed", port);
160            return true;
161        }
162
163        debug!("Port {} is not in allowed list", port);
164        false
165    }
166
167    /// Check if a domain is allowed to be checked
168    fn is_domain_allowed(&self, domain: &str) -> bool {
169        // If any domain is allowed, allow it
170        if self.config.allow_any_domain {
171            return true;
172        }
173
174        // Check allowed domains
175        for allowed in &self.config.allowed_domains {
176            if domain == allowed || domain.ends_with(&format!(".{allowed}")) {
177                debug!("Domain '{}' is allowed", domain);
178                return true;
179            }
180        }
181
182        debug!("Domain '{}' is not in allowed list", domain);
183        false
184    }

186    /// Parse target into host and port
187    #[allow(clippy::unused_self)]
188    fn parse_target(&self, target: &str) -> LLMResult<(String, u16)> {
189        if let Some(pos) = target.rfind(':') {
190            let host = target[..pos].to_string();
191            let port_str = &target[pos + 1..];
192            let port = port_str
193                .parse::<u16>()
194                .map_err(|_| LLMSpellError::Validation {
195                    message: format!("Invalid port number: {port_str}"),
196                    field: Some("target".to_string()),
197                })?;
198
199            Ok((host, port))
200        } else {
201            Err(LLMSpellError::Validation {
202                message: "Target must be in format 'host:port'".to_string(),
203                field: Some("target".to_string()),
204            })
205        }
206    }

208    /// Check TCP port connectivity
209    #[allow(clippy::too_many_lines)]
210    async fn check_tcp_port(
211        &self,
212        host: &str,
213        port: u16,
214        timeout_duration: Duration,
215    ) -> ServiceCheckResult {
216        let target = format!("{host}:{port}");
217        let start_time = Instant::now();
218
219        debug!("Checking TCP port: {}", target);
220
221        // Validate port is allowed
222        if !self.is_port_allowed(port) {
223            return ServiceCheckResult {
224                target: target.clone(),
225                available: false,
226                response_time_ms: 0,
227                status: "Port check not allowed".to_string(),
228                error: Some(format!("Port {port} is not allowed")),
229                metadata: HashMap::new(),
230            };
231        }
232
233        // Validate domain is allowed
234        if !self.is_domain_allowed(host) {
235            return ServiceCheckResult {
236                target: target.clone(),
237                available: false,
238                response_time_ms: 0,
239                status: "Domain check not allowed".to_string(),
240                error: Some(format!("Domain '{host}' is not allowed")),
241                metadata: HashMap::new(),
242            };
243        }
244
245        // Attempt to resolve and connect
246        let socket_addrs: Vec<SocketAddr> = match target.to_socket_addrs() {
247            Ok(addrs) => addrs.collect(),
248            Err(e) => {
249                return ServiceCheckResult {
250                    target: target.clone(),
251                    available: false,
252                    response_time_ms: u64::try_from(start_time.elapsed().as_millis())
253                        .unwrap_or(u64::MAX),
254                    status: "DNS resolution failed".to_string(),
255                    error: Some(e.to_string()),
256                    metadata: HashMap::new(),
257                }
258            }
259        };
260
261        if socket_addrs.is_empty() {
262            return ServiceCheckResult {
263                target: target.clone(),
264                available: false,
265                response_time_ms: u64::try_from(start_time.elapsed().as_millis())
266                    .unwrap_or(u64::MAX),
267                status: "No addresses resolved".to_string(),
268                error: Some("DNS resolution returned no addresses".to_string()),
269                metadata: HashMap::new(),
270            };
271        }
272
273        // Try to connect to the first resolved address
274        let socket_addr = socket_addrs[0];
275        match timeout(timeout_duration, TcpStream::connect(socket_addr)).await {
276            Ok(Ok(_stream)) => {
277                let response_time =
278                    u64::try_from(start_time.elapsed().as_millis()).unwrap_or(u64::MAX);
279                info!("TCP port {} available in {}ms", target, response_time);
280                ServiceCheckResult {
281                    target,
282                    available: true,
283                    response_time_ms: response_time,
284                    status: "Port open and reachable".to_string(),
285                    error: None,
286                    metadata: {
287                        let mut meta = HashMap::new();
288                        meta.insert("resolved_ip".to_string(), socket_addr.ip().to_string());
289                        meta.insert("resolved_port".to_string(), socket_addr.port().to_string());
290                        meta
291                    },
292                }
293            }
294            Ok(Err(e)) => {
295                let response_time =
296                    u64::try_from(start_time.elapsed().as_millis()).unwrap_or(u64::MAX);
297                warn!("TCP port {} connection failed: {}", target, e);
298                ServiceCheckResult {
299                    target,
300                    available: false,
301                    response_time_ms: response_time,
302                    status: "Connection failed".to_string(),
303                    error: Some(e.to_string()),
304                    metadata: {
305                        let mut meta = HashMap::new();
306                        meta.insert("resolved_ip".to_string(), socket_addr.ip().to_string());
307                        meta
308                    },
309                }
310            }
311            Err(_) => {
312                let response_time =
313                    u64::try_from(start_time.elapsed().as_millis()).unwrap_or(u64::MAX);
314                warn!("TCP port {} check timed out", target);
315                ServiceCheckResult {
316                    target,
317                    available: false,
318                    response_time_ms: response_time,
319                    status: "Connection timed out".to_string(),
320                    error: Some(format!("Timeout after {}ms", timeout_duration.as_millis())),
321                    metadata: {
322                        let mut meta = HashMap::new();
323                        meta.insert("resolved_ip".to_string(), socket_addr.ip().to_string());
324                        meta
325                    },
326                }
327            }
328        }
329    }

331    /// Check HTTP/HTTPS service health
332    #[allow(clippy::too_many_lines)]
333    async fn check_http_service(
334        &self,
335        url: &str,
336        timeout_duration: Duration,
337    ) -> ServiceCheckResult {
338        let start_time = Instant::now();
339
340        debug!("Checking HTTP service: {}", url);
341
342        // Basic URL validation
343        if !url.starts_with("http://") && !url.starts_with("https://") {
344            return ServiceCheckResult {
345                target: url.to_string(),
346                available: false,
347                response_time_ms: 0,
348                status: "Invalid URL format".to_string(),
349                error: Some("URL must start with http:// or https://".to_string()),
350                metadata: HashMap::new(),
351            };
352        }
353
354        // Extract domain from URL for validation
355        let domain = if let Ok(parsed_url) = url::Url::parse(url) {
356            if let Some(host) = parsed_url.host_str() {
357                host.to_string()
358            } else {
359                return ServiceCheckResult {
360                    target: url.to_string(),
361                    available: false,
362                    response_time_ms: 0,
363                    status: "Invalid host in URL".to_string(),
364                    error: Some("Could not extract host from URL".to_string()),
365                    metadata: HashMap::new(),
366                };
367            }
368        } else {
369            return ServiceCheckResult {
370                target: url.to_string(),
371                available: false,
372                response_time_ms: 0,
373                status: "URL parsing failed".to_string(),
374                error: Some("Could not parse URL".to_string()),
375                metadata: HashMap::new(),
376            };
377        };
378
379        // Validate domain is allowed
380        if !self.is_domain_allowed(&domain) {
381            return ServiceCheckResult {
382                target: url.to_string(),
383                available: false,
384                response_time_ms: 0,
385                status: "Domain check not allowed".to_string(),
386                error: Some(format!("Domain '{domain}' is not allowed")),
387                metadata: HashMap::new(),
388            };
389        }
390
391        // Create HTTP client with timeout
392        let client = reqwest::Client::builder()
393            .timeout(timeout_duration)
394            .build()
395            .map_err(|e| LLMSpellError::Tool {
396                message: format!("Failed to create HTTP client: {e}"),
397                tool_name: Some("service_checker".to_string()),
398                source: None,
399            })
400            .unwrap();
401
402        // Make HEAD request to check service availability
403        match client.head(url).send().await {
404            Ok(response) => {
405                let response_time =
406                    u64::try_from(start_time.elapsed().as_millis()).unwrap_or(u64::MAX);
407                let status_code = response.status().as_u16();
408                let is_available =
409                    response.status().is_success() || response.status().is_redirection();
410
411                info!(
412                    "HTTP service {} returned {} in {}ms",
413                    url, status_code, response_time
414                );
415
416                ServiceCheckResult {
417                    target: url.to_string(),
418                    available: is_available,
419                    response_time_ms: response_time,
420                    status: format!("HTTP {status_code}"),
421                    error: if is_available {
422                        None
423                    } else {
424                        Some(format!("HTTP error {status_code}"))
425                    },
426                    metadata: {
427                        let mut meta = HashMap::new();
428                        meta.insert("status_code".to_string(), status_code.to_string());
429                        meta.insert("method".to_string(), "HEAD".to_string());
430                        if let Some(content_type) = response.headers().get("content-type") {
431                            if let Ok(ct_str) = content_type.to_str() {
432                                meta.insert("content_type".to_string(), ct_str.to_string());
433                            }
434                        }
435                        meta
436                    },
437                }
438            }
439            Err(e) => {
440                let response_time =
441                    u64::try_from(start_time.elapsed().as_millis()).unwrap_or(u64::MAX);
442                warn!("HTTP service {} check failed: {}", url, e);
443
444                let (status, error_msg) = if e.is_timeout() {
445                    (
446                        "Request timed out".to_string(),
447                        format!("Timeout after {}ms", timeout_duration.as_millis()),
448                    )
449                } else if e.is_connect() {
450                    (
451                        "Connection failed".to_string(),
452                        "Failed to connect to service".to_string(),
453                    )
454                } else {
455                    ("Request failed".to_string(), e.to_string())
456                };
457
458                ServiceCheckResult {
459                    target: url.to_string(),
460                    available: false,
461                    response_time_ms: response_time,
462                    status,
463                    error: Some(error_msg),
464                    metadata: HashMap::new(),
465                }
466            }
467        }
468    }

471#[async_trait]
472impl BaseAgent for ServiceCheckerTool {
473    fn metadata(&self) -> &ComponentMetadata {
474        &self.metadata
475    }
476
477    #[allow(clippy::too_many_lines)]
478    async fn execute_impl(
479        &self,
480        input: AgentInput,
481        _context: ExecutionContext,
482    ) -> LLMResult<AgentOutput> {
483        // Get parameters using shared utility
484        let params = extract_parameters(&input)?;
485
486        // Extract required parameters
487        let target = extract_required_string(params, "target")?;
488        let check_type = extract_required_string(params, "check_type")?;
489
490        // Validate check_type
491        match check_type {
492            "tcp" | "http" | "https" | "dns" => {}
493            _ => {
494                return Err(LLMSpellError::Validation {
495                    message: format!(
496                        "Invalid check_type: {check_type}. Supported types: tcp, http, https, dns"
497                    ),
498                    field: Some("check_type".to_string()),
499                });
500            }
501        }
502
503        // Validate target is not empty
504        if target.trim().is_empty() {
505            return Err(LLMSpellError::Validation {
506                message: "Target cannot be empty".to_string(),
507                field: Some("target".to_string()),
508            });
509        }
510
511        // Extract optional parameters
512        let timeout_seconds = extract_optional_u64(params, "timeout_seconds")
513            .unwrap_or(self.config.default_timeout_seconds);
514
515        // Validate timeout if provided
516        if timeout_seconds > self.config.max_timeout_seconds {
517            return Err(LLMSpellError::Validation {
518                message: format!(
519                    "Timeout {} exceeds maximum allowed timeout of {} seconds",
520                    timeout_seconds, self.config.max_timeout_seconds
521                ),
522                field: Some("timeout_seconds".to_string()),
523            });
524        }
525
526        let timeout_duration = Duration::from_secs(timeout_seconds);
527
528        // Perform the check based on type
529        let result = match check_type {
530            "tcp" => {
531                let (host, port) = self.parse_target(target)?;
532                self.check_tcp_port(&host, port, timeout_duration).await
533            }
534            "http" => {
535                let url = if target.starts_with("http://") || target.starts_with("https://") {
536                    target.to_string()
537                } else {
538                    format!("http://{target}")
539                };
540                self.check_http_service(&url, timeout_duration).await
541            }
542            "https" => {
543                let url = if target.starts_with("https://") {
544                    target.to_string()
545                } else if target.starts_with("http://") {
546                    target.replace("http://", "https://")
547                } else {
548                    format!("https://{target}")
549                };
550                self.check_http_service(&url, timeout_duration).await
551            }
552            "dns" => {
553                // Simple DNS resolution check
554                let start_time = Instant::now();
555                match target.to_socket_addrs() {
556                    Ok(addrs) => {
557                        let addr_list: Vec<SocketAddr> = addrs.collect();
558                        let response_time =
559                            u64::try_from(start_time.elapsed().as_millis()).unwrap_or(u64::MAX);
560                        ServiceCheckResult {
561                            target: target.to_string(),
562                            available: !addr_list.is_empty(),
563                            response_time_ms: response_time,
564                            status: format!("Resolved {} addresses", addr_list.len()),
565                            error: None,
566                            metadata: {
567                                let mut meta = HashMap::new();
568                                meta.insert(
569                                    "address_count".to_string(),
570                                    addr_list.len().to_string(),
571                                );
572                                if let Some(first_addr) = addr_list.first() {
573                                    meta.insert(
574                                        "first_address".to_string(),
575                                        first_addr.to_string(),
576                                    );
577                                }
578                                meta
579                            },
580                        }
581                    }
582                    Err(e) => {
583                        let response_time =
584                            u64::try_from(start_time.elapsed().as_millis()).unwrap_or(u64::MAX);
585                        ServiceCheckResult {
586                            target: target.to_string(),
587                            available: false,
588                            response_time_ms: response_time,
589                            status: "DNS resolution failed".to_string(),
590                            error: Some(e.to_string()),
591                            metadata: HashMap::new(),
592                        }
593                    }
594                }
595            }
596            _ => unreachable!(), // Already validated above
597        };
598
599        // Format response
600        let message = if result.available {
601            format!(
602                "Service '{}' is available ({}ms response time)",
603                result.target, result.response_time_ms
604            )
605        } else {
606            format!(
607                "Service '{}' is not available: {}",
608                result.target,
609                result.error.as_ref().unwrap_or(&result.status)
610            )
611        };
612
613        let response = ResponseBuilder::success("check")
614            .with_message(message)
615            .with_result(json!({
616                "target": result.target,
617                "check_type": check_type,
618                "available": result.available,
619                "response_time_ms": result.response_time_ms,
620                "status": result.status,
621                "error": result.error,
622                "metadata": result.metadata,
623                "timeout_seconds": timeout_seconds
624            }))
625            .build();
626
627        Ok(AgentOutput::text(serde_json::to_string_pretty(&response)?))
628    }
629
630    async fn validate_input(&self, input: &AgentInput) -> LLMResult<()> {
631        if input.text.is_empty() {
632            return Err(LLMSpellError::Validation {
633                message: "Input prompt cannot be empty".to_string(),
634                field: Some("prompt".to_string()),
635            });
636        }
637        Ok(())
638    }
639
640    async fn handle_error(&self, error: LLMSpellError) -> LLMResult<AgentOutput> {
641        Ok(AgentOutput::text(format!("Service checker error: {error}")))
642    }
