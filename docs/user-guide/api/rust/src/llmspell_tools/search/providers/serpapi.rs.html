12/// `SerpApi` provider - supports multiple search engines
13pub struct SerpApiProvider {
14    client: Client,
15    api_key: Option<String>,
16    default_engine: String,
17}
18
19impl SerpApiProvider {
20    #[must_use]
21    pub fn new(config: ProviderConfig) -> Self {
22        let default_engine = config
23            .additional_config
24            .get("default_engine")
25            .and_then(|v| v.as_str())
26            .unwrap_or("google")
27            .to_string();
28
29        Self {
30            client: Client::new(),
31            api_key: config.api_key,
32            default_engine,
33        }
34    }
35}

37/// `SerpApi` response structure (varies by engine, this is common subset)
38#[derive(Debug, Deserialize)]
39#[allow(dead_code)]
40struct SerpApiResponse {
41    search_metadata: Option<SearchMetadata>,
42    search_parameters: Option<SearchParameters>,
43    search_information: Option<SearchInformation>,
44
45    // Different result types based on search type
46    organic_results: Option<Vec<OrganicResult>>,
47    news_results: Option<Vec<NewsResult>>,
48    images_results: Option<Vec<ImageResult>>,
49
50    // Error information
51    error: Option<String>,
52}

116#[async_trait]
117impl SearchProvider for SerpApiProvider {
118    fn name(&self) -> &'static str {
119        "serpapi"
120    }
121
122    fn is_available(&self) -> bool {
123        self.api_key.is_some()
124    }
125
126    fn rate_limit(&self) -> Option<u32> {
127        Some(100) // Base plan: 100 searches/month
128    }
129
130    #[allow(clippy::too_many_lines)]
131    async fn search(&self, query: &str, options: &SearchOptions) -> Result<Vec<SearchResult>> {
132        let api_key = self
133            .api_key
134            .as_ref()
135            .ok_or_else(|| LLMSpellError::Configuration {
136                message: "SerpApi key not configured".to_string(),
137                source: None,
138            })?;
139
140        let url = "https://serpapi.com/search";
141        let num_results = options.max_results.min(100).to_string();
142        let mut params = vec![
143            ("api_key", api_key.as_str()),
144            ("q", query),
145            ("num", num_results.as_str()),
146            ("engine", &self.default_engine),
147        ];
148
149        if options.safe_search {
150            params.push(("safe", "active"));
151        }
152
153        if let Some(lang) = &options.language {
154            params.push(("hl", lang));
155        }
156
157        // Handle different search types
158        match options.search_type {
159            SearchType::News => {
160                params.push(("tbm", "nws")); // Google-style news search
161            }
162            SearchType::Images => {
163                params.push(("tbm", "isch")); // Google-style image search
164            }
165            SearchType::Web => {} // Default
166        }
167
168        debug!(
169            "Searching via SerpApi ({}) for: {}",
170            self.default_engine, query
171        );
172
173        let response = self
174            .client
175            .get(url)
176            .query(&params)
177            .send()
178            .await
179            .map_err(|e| LLMSpellError::Network {
180                message: format!("SerpApi request failed: {e}"),
181                source: Some(Box::new(e)),
182            })?;
183
184        if !response.status().is_success() {
185            let status = response.status();
186            let error_body = response.text().await.unwrap_or_default();
187            return Err(LLMSpellError::Network {
188                message: format!("SerpApi returned status {status}: {error_body}"),
189                source: None,
190            });
191        }
192
193        let serpapi_response: SerpApiResponse =
194            response.json().await.map_err(|e| LLMSpellError::Network {
195                message: format!("Failed to parse SerpApi response: {e}"),
196                source: Some(Box::new(e)),
197            })?;
198
199        // Check for API errors
200        if let Some(error) = serpapi_response.error {
201            return Err(LLMSpellError::Network {
202                message: format!("SerpApi error: {error}"),
203                source: None,
204            });
205        }
206
207        let mut results = Vec::new();
208
209        // Process results based on search type
210        match options.search_type {
211            SearchType::Web => {
212                if let Some(organic_results) = serpapi_response.organic_results {
213                    for result in organic_results {
214                        if let (Some(title), Some(link)) = (result.title, result.link) {
215                            results.push(SearchResult {
216                                title,
217                                url: link,
218                                snippet: result
219                                    .snippet
220                                    .unwrap_or_else(|| "No description available".to_string()),
221                                provider: format!("{} ({})", self.name(), self.default_engine),
222                                rank: {
223                                    #[allow(
224                                        clippy::cast_possible_truncation,
225                                        clippy::cast_sign_loss,
226                                        clippy::cast_possible_wrap
227                                    )]
228                                    let fallback_rank = results.len() as i32 + 1;
229                                    #[allow(clippy::cast_sign_loss)]
230                                    let position =
231                                        result.position.unwrap_or(fallback_rank) as usize;
232                                    position
233                                },
234                            });
235                        }
236                    }
237                }
238            }
239            SearchType::News => {
240                if let Some(news_results) = serpapi_response.news_results {
241                    for result in news_results {
242                        if let (Some(title), Some(link)) = (result.title, result.link) {
243                            results.push(SearchResult {
244                                title,
245                                url: link,
246                                snippet: result
247                                    .snippet
248                                    .unwrap_or_else(|| "No description available".to_string()),
249                                provider: format!("{} ({})", self.name(), self.default_engine),
250                                rank: {
251                                    #[allow(
252                                        clippy::cast_possible_truncation,
253                                        clippy::cast_sign_loss,
254                                        clippy::cast_possible_wrap
255                                    )]
256                                    let fallback_rank = results.len() as i32 + 1;
257                                    #[allow(clippy::cast_sign_loss)]
258                                    let position =
259                                        result.position.unwrap_or(fallback_rank) as usize;
260                                    position
261                                },
262                            });
263                        }
264                    }
265                }
266            }
267            SearchType::Images => {
268                if let Some(images_results) = serpapi_response.images_results {
269                    for result in images_results {
270                        if let Some(original) = result.original {
271                            results.push(SearchResult {
272                                title: result.title.unwrap_or_else(|| "Image".to_string()),
273                                url: original,
274                                snippet: result
275                                    .source
276                                    .unwrap_or_else(|| "Image result".to_string()),
277                                provider: format!("{} ({})", self.name(), self.default_engine),
278                                rank: {
279                                    #[allow(
280                                        clippy::cast_possible_truncation,
281                                        clippy::cast_sign_loss,
282                                        clippy::cast_possible_wrap
283                                    )]
284                                    let fallback_rank = results.len() as i32 + 1;
285                                    #[allow(clippy::cast_sign_loss)]
286                                    let position =
287                                        result.position.unwrap_or(fallback_rank) as usize;
288                                    position
289                                },
290                            });
291                        }
292                    }
293                }
294            }
295        }
296
297        if let Some(metadata) = serpapi_response.search_metadata {
298            if let Some(time) = metadata.total_time_taken {
299                info!("SerpApi search completed in {:.2}s", time);
300            }
301        }
302
303        if results.is_empty() {
304            warn!("SerpApi returned no results for query: {}", query);
305        }
306
307        Ok(results)
308    }
309
310    fn metadata(&self) -> Value {
311        let mut meta = serde_json::json!({
312            "name": self.name(),
313            "available": self.is_available(),
314            "rate_limit": self.rate_limit(),
315            "default_engine": self.default_engine,
316        });
317
318        // List supported engines
319        meta["supported_engines"] =
320            serde_json::json!(["google", "bing", "duckduckgo", "yahoo", "yandex", "baidu"]);
321
322        meta
323    }
324}
