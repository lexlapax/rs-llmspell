32/// Web search configuration
33#[derive(Debug, Clone, Serialize, Deserialize)]
34pub struct WebSearchConfig {
35    /// Default search provider
36    pub default_provider: String,
37    /// API keys and configuration for different providers
38    pub providers: HashMap<String, ProviderConfig>,
39    /// Maximum results per search
40    pub max_results: usize,
41    /// Enable safe search
42    pub safe_search: bool,
43    /// Language preference
44    pub language: Option<String>,
45    /// Provider fallback chain
46    pub fallback_chain: Vec<String>,
47}

49impl Default for WebSearchConfig {
50    fn default() -> Self {
51        // Default fallback chain prioritizes free/high-limit providers
52        let fallback_chain = vec![
53            "duckduckgo".to_string(), // No API key required
54            "serperdev".to_string(),  // 2,500/month free
55            "brave".to_string(),      // 2,000/month free
56            "google".to_string(),     // 100/day free
57            "serpapi".to_string(),    // 100/month free
58        ];
59
60        Self {
61            default_provider: "duckduckgo".to_string(),
62            providers: HashMap::new(),
63            max_results: 10,
64            safe_search: true,
65            language: Some("en".to_string()),
66            fallback_chain,
67        }
68    }
69}

71impl WebSearchConfig {
72    /// Load configuration from environment variables
73    #[must_use]
74    pub fn from_env() -> Self {
75        let mut config = Self::default();
76        let mut providers = HashMap::new();
77
78        // Google configuration - use API key manager with fallback to env vars
79        if let Some(api_key) =
80            get_api_key("google_search").or_else(|| std::env::var("WEBSEARCH_GOOGLE_API_KEY").ok())
81        {
82            let additional_config = std::env::var("WEBSEARCH_GOOGLE_SEARCH_ENGINE_ID").map_or(
83                serde_json::Value::Null,
84                |engine_id| {
85                    serde_json::json!({
86                        "search_engine_id": engine_id
87                    })
88                },
89            );
90            let google_config = ProviderConfig {
91                api_key: Some(api_key),
92                additional_config,
93            };
94            providers.insert("google".to_string(), google_config);
95        }
96
97        // Brave configuration - use API key manager with fallback to env vars
98        if let Some(api_key) = get_api_key("brave_search")
99            .or_else(|| std::env::var("WEBSEARCH_BRAVE_API_KEY").ok())
100            .or_else(|| std::env::var("BRAVE_API_KEY").ok())
101        {
102            let brave_config = ProviderConfig {
103                api_key: Some(api_key),
104                ..Default::default()
105            };
106            providers.insert("brave".to_string(), brave_config);
107        }
108
109        // SerpApi configuration - use API key manager with fallback to env vars
110        if let Some(api_key) = get_api_key("serpapi")
111            .or_else(|| std::env::var("WEBSEARCH_SERPAPI_API_KEY").ok())
112            .or_else(|| std::env::var("SERPAPI_API_KEY").ok())
113        {
114            let serpapi_config = ProviderConfig {
115                api_key: Some(api_key),
116                ..Default::default()
117            };
118            providers.insert("serpapi".to_string(), serpapi_config);
119        }
120
121        // SerperDev configuration - use API key manager with fallback to env vars
122        if let Some(api_key) = get_api_key("serperdev")
123            .or_else(|| std::env::var("WEBSEARCH_SERPERDEV_API_KEY").ok())
124            .or_else(|| std::env::var("SERPERDEV_API_KEY").ok())
125        {
126            let serperdev_config = ProviderConfig {
127                api_key: Some(api_key),
128                ..Default::default()
129            };
130            providers.insert("serperdev".to_string(), serperdev_config);
131        }
132
133        // DuckDuckGo doesn't need configuration
134        providers.insert("duckduckgo".to_string(), ProviderConfig::default());
135
136        config.providers = providers;
137
138        // Override default provider if specified
139        if let Ok(default) = std::env::var("WEBSEARCH_DEFAULT_PROVIDER") {
140            config.default_provider = default;
141        }
142
143        config
144    }

171impl WebSearchTool {
172    /// Create a new web search tool
173    ///
174    /// # Errors
175    ///
176    /// Returns an error if provider initialization fails
177    #[allow(clippy::too_many_lines)]
178    pub fn new(config: WebSearchConfig) -> Result<Self> {
179        let mut providers = HashMap::new();
180
181        // Initialize DuckDuckGo (always available)
182        let ddg_provider = DuckDuckGoProvider::new();
183        providers.insert(
184            "duckduckgo".to_string(),
185            ProviderWrapper {
186                provider: Box::new(ddg_provider),
187                rate_limiter: None, // No official rate limit
188            },
189        );
190
191        // Initialize configured providers
192        for (name, provider_config) in &config.providers {
193            let wrapper = match name.as_str() {
194                "google" => {
195                    let provider = GoogleSearchProvider::new(provider_config.clone());
196                    if provider.is_available() {
197                        Some(ProviderWrapper {
198                            provider: Box::new(provider),
199                            rate_limiter: Some(
200                                RateLimiterBuilder::default()
201                                    .custom(100, Duration::from_secs(24 * 60 * 60)) // 100 per day
202                                    .sliding_window()
203                                    .build()
204                                    .map_err(|e| LLMSpellError::Internal {
205                                        message: format!("Failed to create rate limiter: {e}"),
206                                        source: None,
207                                    })?,
208                            ),
209                        })
210                    } else {
211                        None
212                    }
213                }
214                "brave" => {
215                    let provider = BraveSearchProvider::new(provider_config.clone());
216                    if provider.is_available() {
217                        Some(ProviderWrapper {
218                            provider: Box::new(provider),
219                            rate_limiter: Some(
220                                RateLimiterBuilder::default()
221                                    .per_minute(60) // ~2000/month spread evenly
222                                    .sliding_window()
223                                    .build()
224                                    .map_err(|e| LLMSpellError::Internal {
225                                        message: format!("Failed to create rate limiter: {e}"),
226                                        source: None,
227                                    })?,
228                            ),
229                        })
230                    } else {
231                        None
232                    }
233                }
234                "serpapi" => {
235                    let provider = SerpApiProvider::new(provider_config.clone());
236                    if provider.is_available() {
237                        Some(ProviderWrapper {
238                            provider: Box::new(provider),
239                            rate_limiter: Some(
240                                RateLimiterBuilder::default()
241                                    .per_minute(2) // ~100/month conservative
242                                    .sliding_window()
243                                    .build()
244                                    .map_err(|e| LLMSpellError::Internal {
245                                        message: format!("Failed to create rate limiter: {e}"),
246                                        source: None,
247                                    })?,
248                            ),
249                        })
250                    } else {
251                        None
252                    }
253                }
254                "serperdev" => {
255                    let provider = SerperDevProvider::new(provider_config.clone());
256                    if provider.is_available() {
257                        Some(ProviderWrapper {
258                            provider: Box::new(provider),
259                            rate_limiter: Some(
260                                RateLimiterBuilder::default()
261                                    .per_minute(80) // ~2500/month spread evenly
262                                    .sliding_window()
263                                    .build()
264                                    .map_err(|e| LLMSpellError::Internal {
265                                        message: format!("Failed to create rate limiter: {e}"),
266                                        source: None,
267                                    })?,
268                            ),
269                        })
270                    } else {
271                        None
272                    }
273                }
274                _ => None,
275            };
276
277            if let Some(wrapper) = wrapper {
278                providers.insert(name.clone(), wrapper);
279                info!("Initialized search provider: {}", name);
280            } else if name != "duckduckgo" {
281                warn!("Search provider {} not available (missing API key?)", name);
282            }
283        }
284
285        Ok(Self {
286            metadata: ComponentMetadata::new(
287                "web-search-tool".to_string(),
288                "Search the web using multiple providers with fallback support".to_string(),
289            ),
290            config,
291            providers: Arc::new(Mutex::new(providers)),
292        })
293    }

306    /// Perform search with fallback support
307    async fn search_with_fallback(
308        &self,
309        query: &str,
310        requested_provider: Option<&str>,
311        options: SearchOptions,
312    ) -> Result<Vec<SearchResult>> {
313        let providers = self.providers.lock().await;
314
315        // Determine provider order
316        let provider_chain = requested_provider.map_or_else(
317            || {
318                // Use default provider first, then fallback chain
319                let mut chain = vec![self.config.default_provider.clone()];
320                for fallback in &self.config.fallback_chain {
321                    if fallback != &self.config.default_provider {
322                        chain.push(fallback.clone());
323                    }
324                }
325                chain
326            },
327            |requested| {
328                // If specific provider requested, try it first
329                let mut chain = vec![requested.to_string()];
330                // Then add fallback chain excluding the requested one
331                for fallback in &self.config.fallback_chain {
332                    if fallback != requested {
333                        chain.push(fallback.clone());
334                    }
335                }
336                chain
337            },
338        );
339
340        // Try each provider in order
341        let mut last_error = None;
342
343        for provider_name in provider_chain {
344            if let Some(wrapper) = providers.get(&provider_name) {
345                // Check rate limit if applicable
346                if let Some(rate_limiter) = &wrapper.rate_limiter {
347                    match rate_limiter.try_acquire().await {
348                        Ok(()) => {}
349                        Err(e) => {
350                            warn!("Rate limit exceeded for {}: {}", provider_name, e);
351                            continue;
352                        }
353                    }
354                }
355
356                // Try the search
357                info!("Searching with provider: {}", provider_name);
358                match wrapper.provider.search(query, &options).await {
359                    Ok(results) => {
360                        if results.is_empty() {
361                            warn!("Provider {} returned no results", provider_name);
362                        } else {
363                            return Ok(results);
364                        }
365                    }
366                    Err(e) => {
367                        error!("Search failed with provider {}: {}", provider_name, e);
368                        last_error = Some(e);
369                    }
370                }
371            } else {
372                warn!(
373                    "Provider '{}' not found, skipping to next provider",
374                    provider_name
375                );
376            }
377        }
378
379        // All providers failed
380        Err(last_error.unwrap_or_else(|| LLMSpellError::Network {
381            message: "All search providers failed or returned no results".to_string(),
382            source: None,
383        }))
384    }

386    /// Parse search type from string
387    fn parse_search_type(type_str: Option<&str>) -> SearchType {
388        match type_str {
389            Some("news") => SearchType::News,
390            Some("images") => SearchType::Images,
391            _ => SearchType::Web,
392        }

458#[async_trait]
459impl BaseAgent for WebSearchTool {
460    fn metadata(&self) -> &ComponentMetadata {
461        &self.metadata
462    }
463
464    async fn execute_impl(
465        &self,
466        input: AgentInput,
467        _context: ExecutionContext,
468    ) -> Result<AgentOutput> {
469        // Get parameters
470        let params = extract_parameters(&input)?;
471
472        // Parse required parameters
473        let query = extract_required_string(params, "input")?.to_string();
474
475        // Parse optional parameters
476        let provider = extract_optional_string(params, "provider");
477        let max_results = params
478            .get("max_results")
479            .and_then(serde_json::Value::as_u64)
480            .map_or(self.config.max_results, |n| {
481                usize::try_from(n).unwrap_or(usize::MAX)
482            });
483        let search_type = Self::parse_search_type(extract_optional_string(params, "search_type"));
484        let safe_search = params
485            .get("safe_search")
486            .and_then(serde_json::Value::as_bool)
487            .unwrap_or(self.config.safe_search);
488        let language = extract_optional_string(params, "language")
489            .map(std::string::ToString::to_string)
490            .or_else(|| self.config.language.clone());
491
492        debug!(
493            "Executing web search: query='{}', provider={:?}, type={:?}, results={}",
494            query, provider, search_type, max_results
495        );
496
497        // Build search options
498        let options = SearchOptions {
499            max_results,
500            safe_search,
501            language,
502            search_type,
503        };
504
505        // Perform search with fallback
506        let results = self.search_with_fallback(&query, provider, options).await?;
507
508        // Create response
509        let provider_used = results
510            .first()
511            .map_or_else(|| "unknown".to_string(), |r| r.provider.clone());
512
513        let message = format!(
514            "Found {} results for '{}' using {}",
515            results.len(),
516            query,
517            provider_used
518        );
519
520        let response = ResponseBuilder::success("search")
521            .with_message(&message)
522            .with_result(serde_json::json!({
523                "query": query,
524                "provider": provider_used,
525                "count": results.len(),
526                "results": results,
527            }))
528            .build();
529
530        Ok(AgentOutput::text(serde_json::to_string_pretty(&response)?))
531    }
532
533    async fn validate_input(&self, input: &AgentInput) -> Result<()> {
534        if input.parameters.is_empty() {
535            return Err(LLMSpellError::Validation {
536                message: "No parameters provided".to_string(),
537                field: Some("parameters".to_string()),
538            });
539        }
540        Ok(())
541    }
542
543    async fn handle_error(&self, error: LLMSpellError) -> Result<AgentOutput> {
544        let response = ResponseBuilder::error("search", error.to_string()).build();
545        Ok(AgentOutput::text(serde_json::to_string_pretty(&response)?))
546    }
