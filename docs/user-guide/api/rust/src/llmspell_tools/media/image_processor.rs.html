34/// Image format types
35#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
36#[serde(rename_all = "lowercase")]
37pub enum ImageFormat {
38    Png,
39    Jpeg,
40    Gif,
41    Webp,
42    Bmp,
43    Tiff,
44    Ico,
45    Unknown,
46}
47
48impl ImageFormat {
49    /// Detect format from file extension
50    #[must_use]
51    pub fn from_extension(path: &Path) -> Self {
52        match path
53            .extension()
54            .and_then(|ext| ext.to_str())
55            .map(str::to_lowercase)
56            .as_deref()
57        {
58            Some("png") => Self::Png,
59            Some("jpg" | "jpeg") => Self::Jpeg,
60            Some("gif") => Self::Gif,
61            Some("webp") => Self::Webp,
62            Some("bmp") => Self::Bmp,
63            Some("tiff" | "tif") => Self::Tiff,
64            Some("ico") => Self::Ico,
65            _ => Self::Unknown,
66        }
67    }
68
69    /// Get MIME type for the format
70    #[must_use]
71    pub const fn mime_type(&self) -> &'static str {
72        match self {
73            Self::Png => "image/png",
74            Self::Jpeg => "image/jpeg",
75            Self::Gif => "image/gif",
76            Self::Webp => "image/webp",
77            Self::Bmp => "image/bmp",
78            Self::Tiff => "image/tiff",
79            Self::Ico => "image/x-icon",
80            Self::Unknown => "application/octet-stream",
81        }
82    }
83}

85/// Image dimensions
86#[derive(Debug, Clone, Serialize, Deserialize)]
87pub struct ImageDimensions {
88    pub width: u32,
89    pub height: u32,
90}
91
92impl ImageDimensions {
93    /// Calculate aspect ratio
94    #[must_use]
95    pub fn aspect_ratio(&self) -> f64 {
96        f64::from(self.width) / f64::from(self.height)
97    }
98
99    /// Get orientation
100    #[must_use]
101    pub fn orientation(&self) -> &'static str {
102        match self.aspect_ratio() {
103            r if r > 1.2 => "landscape",
104            r if r < 0.8 => "portrait",
105            _ => "square",
106        }
107    }
108}

369    /// Validate processing parameters
370    #[allow(clippy::unused_async)]
371    async fn validate_parameters(&self, params: &serde_json::Value) -> LLMResult<()> {
372        // Validate operation
373        if let Some(operation) = extract_optional_string(params, "operation") {
374            match operation {
375                "detect" | "metadata" | "resize" | "convert" | "crop" | "rotate" | "thumbnail" => {}
376                _ => {
377                    return Err(LLMSpellError::Validation {
378                        message: format!(
379                            "Invalid operation: {operation}. Supported operations: detect, metadata, resize, convert, crop, rotate, thumbnail"
380                        ),
381                        field: Some("operation".to_string()),
382                    });
383                }
384            }
385        }
386
387        // Validate file paths
388        if let Some(file_path) = extract_optional_string(params, "file_path") {
389            if file_path.is_empty() {
390                return Err(LLMSpellError::Validation {
391                    message: "File path cannot be empty".to_string(),
392                    field: Some("file_path".to_string()),
393                });
394            }
395        }
396
397        // Validate resize parameters
398        if extract_optional_string(params, "operation") == Some("resize")
399            && params.get("width").is_none()
400            && params.get("height").is_none()
401        {
402            return Err(LLMSpellError::Validation {
403                message: "At least one of width or height must be specified for resize".to_string(),
404                field: Some("dimensions".to_string()),
405            });
406        }
407
408        // Validate crop parameters
409        if extract_optional_string(params, "operation") == Some("crop") {
410            for field in ["x", "y", "width", "height"] {
411                if params.get(field).is_none() {
412                    return Err(LLMSpellError::Validation {
413                        message: format!("{field} is required for crop operation"),
414                        field: Some(field.to_string()),
415                    });
416                }
417            }
418        }
419
420        Ok(())
421    }

430    #[allow(clippy::too_many_lines)]
431    async fn execute_impl(
432        &self,
433        input: AgentInput,
434        _context: ExecutionContext,
435    ) -> LLMResult<AgentOutput> {
436        // Get parameters using shared utility
437        let params = extract_parameters(&input)?;
438
439        self.validate_parameters(params).await?;
440
441        let operation = extract_optional_string(params, "operation").unwrap_or("metadata");
442
443        match operation {
444            "detect" => {
445                let file_path = extract_required_string(params, "file_path")?;
446
447                let path = Path::new(file_path);
448                let format = self.detect_format(path).await?;
449
450                let response = ResponseBuilder::success("detect")
451                    .with_message(format!("Detected image format: {format:?}"))
452                    .with_result(json!({
453                        "file_path": file_path,
454                        "format": format,
455                        "mime_type": format.mime_type(),
456                        "supported": format != ImageFormat::Unknown && self.config.supported_formats.contains(&format)
457                    }))
458                    .build();
459
460                Ok(AgentOutput::text(serde_json::to_string_pretty(&response)?))
461            }
462
463            "metadata" => {
464                use std::fmt::Write;
465
466                let file_path = extract_required_string(params, "file_path")?;
467
468                let path = Path::new(file_path);
469                let metadata = self.extract_metadata(path, &self.sandbox).await?;
470
471                let mut message = format!("Image file: {:?} format", metadata.format);
472
473                if let Some(dims) = &metadata.dimensions {
474                    let _ = write!(
475                        message,
476                        ", {}x{} pixels ({})",
477                        dims.width,
478                        dims.height,
479                        dims.orientation()
480                    );
481                }
482
483                let _ = write!(message, ", Size: {} bytes", metadata.file_size);
484
485                let response = ResponseBuilder::success("metadata")
486                    .with_message(message)
487                    .with_result(json!({
488                        "file_path": file_path,
489                        "metadata": metadata
490                    }))
491                    .build();
492
493                Ok(AgentOutput::text(serde_json::to_string_pretty(&response)?))
494            }
495
496            "resize" => {
497                let input_path = extract_required_string(params, "source_path")?;
498                let output_path = extract_required_string(params, "target_path")?;
499                let width = extract_optional_u64(params, "width").map(|w| {
500                    #[allow(clippy::cast_possible_truncation)]
501                    let width_val = w.min(u64::from(u32::MAX)) as u32;
502                    width_val
503                });
504                let height = extract_optional_u64(params, "height").map(|h| {
505                    #[allow(clippy::cast_possible_truncation)]
506                    let height_val = h.min(u64::from(u32::MAX)) as u32;
507                    height_val
508                });
509                let maintain_aspect_ratio =
510                    extract_optional_bool(params, "maintain_aspect_ratio").unwrap_or(true);
511
512                self.resize_image(
513                    Path::new(input_path),
514                    Path::new(output_path),
515                    width,
516                    height,
517                    maintain_aspect_ratio,
518                )
519                .await?;
520
521                let response = ResponseBuilder::success("resize")
522                    .with_message(format!("Resized image from {input_path} to {output_path}"))
523                    .with_result(json!({
524                        "source_path": input_path,
525                        "target_path": output_path,
526                        "width": width,
527                        "height": height,
528                        "maintain_aspect_ratio": maintain_aspect_ratio
529                    }))
530                    .build();
531
532                Ok(AgentOutput::text(serde_json::to_string_pretty(&response)?))
533            }
534
535            "convert" => {
536                let input_path = extract_required_string(params, "source_path")?;
537                let output_path = extract_required_string(params, "target_path")?;
538                let target_format = extract_optional_string(params, "target_format").map_or_else(
539                    || ImageFormat::from_extension(Path::new(output_path)),
540                    |s| match s.to_lowercase().as_str() {
541                        "png" => ImageFormat::Png,
542                        "jpeg" | "jpg" => ImageFormat::Jpeg,
543                        "gif" => ImageFormat::Gif,
544                        "webp" => ImageFormat::Webp,
545                        _ => ImageFormat::Unknown,
546                    },
547                );
548
549                self.convert_format(
550                    Path::new(input_path),
551                    Path::new(output_path),
552                    target_format.clone(),
553                )
554                .await?;
555
556                let response = ResponseBuilder::success("convert")
557                    .with_message(format!(
558                        "Converted image from {input_path} to {output_path} ({target_format:?} format)"
559                    ))
560                    .with_result(json!({
561                        "source_path": input_path,
562                        "target_path": output_path,
563                        "target_format": target_format
564                    }))
565                    .build();

713    async fn validate_input(&self, input: &AgentInput) -> LLMResult<()> {
714        if input.text.is_empty() {
715            return Err(LLMSpellError::Validation {
716                message: "Input prompt cannot be empty".to_string(),
717                field: Some("prompt".to_string()),
718            });
719        }
720        Ok(())
721    }
722
723    async fn handle_error(&self, error: LLMSpellError) -> LLMResult<AgentOutput> {
724        Ok(AgentOutput::text(format!("Image processor error: {error}")))
725    }
726}
