29pub struct ApiTesterTool {

39impl ApiTesterTool {

47        }

51#[async_trait]
52impl Tool for ApiTesterTool {
53    fn schema(&self) -> ToolSchema {
54        ToolSchema::new(
55            self.metadata.name.clone(),
56            self.metadata.description.clone(),
57        )
58        .with_parameter(ParameterDef {
59            name: "input".to_string(),
60            param_type: ParameterType::String,
61            description: "API endpoint URL".to_string(),
62            required: true,
63            default: None,
64        })
65        .with_parameter(ParameterDef {
66            name: "method".to_string(),
67            param_type: ParameterType::String,
68            description: "HTTP method (GET, POST, PUT, DELETE, PATCH)".to_string(),
69            required: false,
70            default: Some(json!("GET")),
71        })
72        .with_parameter(ParameterDef {
73            name: "headers".to_string(),
74            param_type: ParameterType::Object,
75            description: "HTTP headers as key-value pairs".to_string(),
76            required: false,
77            default: None,
78        })
79        .with_parameter(ParameterDef {
80            name: "body".to_string(),
81            param_type: ParameterType::Object,
82            description: "Request body (for POST, PUT, PATCH)".to_string(),
83            required: false,
84            default: None,
85        })
86        .with_parameter(ParameterDef {
87            name: "timeout".to_string(),
88            param_type: ParameterType::Number,
89            description: "Request timeout in seconds".to_string(),
90            required: false,
91            default: Some(json!(30)),
92        })
93    }
94
95    fn category(&self) -> ToolCategory {
96        ToolCategory::Data
97    }
98
99    fn security_level(&self) -> SecurityLevel {
100        SecurityLevel::Restricted
101    }
102}

104#[async_trait]
105impl BaseAgent for ApiTesterTool {
106    fn metadata(&self) -> &ComponentMetadata {
107        &self.metadata
108    }
109
110    async fn validate_input(&self, _input: &AgentInput) -> Result<()> {
111        Ok(())
112    }
113
114    async fn handle_error(&self, error: llmspell_core::LLMSpellError) -> Result<AgentOutput> {
115        Ok(AgentOutput::text(format!("ApiTester error: {error}")))
116    }
117
118    async fn execute_impl(
119        &self,
120        input: AgentInput,
121        _context: ExecutionContext,
122    ) -> Result<AgentOutput> {
123        let params = extract_parameters(&input)?;
124        let url = extract_required_string(params, "input")?;
125        let method_str = extract_optional_string(params, "method").unwrap_or("GET");
126        let headers = extract_optional_object(params, "headers");
127        let body = extract_optional_object(params, "body");
128        let timeout = extract_optional_u64(params, "timeout").unwrap_or(30);
129
130        // Validate URL with SSRF protection
131        let ssrf_protector = SsrfProtector::new();
132        if let Err(e) = ssrf_protector.validate_url(url) {
133            return Err(validation_error(
134                format!("URL validation failed: {e}"),
135                Some("input".to_string()),
136            ));
137        }
138
139        // Parse HTTP method
140        let method = match method_str.to_uppercase().as_str() {
141            "GET" => Method::GET,
142            "POST" => Method::POST,
143            "PUT" => Method::PUT,
144            "DELETE" => Method::DELETE,
145            "PATCH" => Method::PATCH,
146            "HEAD" => Method::HEAD,
147            "OPTIONS" => Method::OPTIONS,
148            _ => {
149                return Err(validation_error(
150                    format!("Invalid HTTP method: {method_str}"),
151                    Some("method".to_string()),
152                ));
153            }
154        };
155
156        // Build client
157        let client = Client::builder()
158            .timeout(Duration::from_secs(timeout))
159            .user_agent("Mozilla/5.0 (compatible; LLMSpell-ApiTester/1.0)")
160            .build()
161            .unwrap_or_default();
162
163        // Build request
164        let mut request = client.request(method.clone(), url);
165
166        // Add headers
167        if let Some(headers_map) = headers {
168            for (key, value) in headers_map {
169                if let Some(val_str) = value.as_str() {
170                    request = request.header(key, val_str);
171                }
172            }
173        }
174
175        // Add body for methods that support it
176        if matches!(method, Method::POST | Method::PUT | Method::PATCH) {
177            if let Some(body_data) = body {
178                request = request.json(&body_data);
179            }
180        }
181
182        // Execute request and measure time
183        let start = Instant::now();
184        let response = request
185            .send()
186            .await
187            .map_err(|e| component_error(format!("Request failed: {e}")))?;
188        let duration = start.elapsed();
189
190        // Extract response data
191        let status = response.status();
192        let headers = response.headers().clone();
193
194        let mut response_headers = HashMap::new();
195        for (name, value) in &headers {
196            if let Ok(val) = value.to_str() {
197                response_headers.insert(name.to_string(), val.to_string());
198            }
199        }
200
201        // Try to parse response body
202        let body_text = response.text().await.unwrap_or_default();
203        let body_json: Option<Value> = serde_json::from_str(&body_text).ok();
204
205        let body_value = body_json.clone().unwrap_or_else(|| json!(body_text));
206        let result = json!({
207            "request": {
208                "method": method_str,
209                "url": url,
210                "timeout": timeout,
211            },
212            "response": {
213                "status_code": status.as_u16(),
214                "status_text": status.canonical_reason().unwrap_or("Unknown"),
215                "is_success": status.is_success(),
216                "headers": response_headers,
217                "body": body_value,
218                "body_is_json": body_json.is_some(),
219            },
220            "timing": {
221                "duration_ms": duration.as_millis(),
222                "duration_secs": duration.as_secs_f64(),
223            }
224        });
225
226        let response = ResponseBuilder::success("test").with_result(result).build();
227
228        Ok(AgentOutput::text(
229            serde_json::to_string_pretty(&response).unwrap(),
230        ))
231    }
