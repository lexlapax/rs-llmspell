62    /// Process Base64 operation
63    #[allow(clippy::unused_async)]
64    #[allow(clippy::too_many_lines)]
65    async fn process_operation(&self, params: &Value) -> Result<Value> {
66        // Extract parameters using utilities
67        let operation = extract_required_string(params, "operation")?;
68        validate_enum(&operation, &["encode", "decode"], "operation")?;
69
70        let variant = extract_optional_string(params, "variant").unwrap_or("standard");
71        validate_enum(&variant, &["standard", "url-safe"], "variant")?;
72
73        // Get input data
74        let input_file = extract_optional_string(params, "input_file");
75        let input_str = extract_optional_string(params, "input");
76        let binary_input = extract_optional_bool(params, "binary_input").unwrap_or(false);
77
78        let input_data = if let Some(file_path) = input_file {
79            // Read from file
80            fs::read(file_path).map_err(|e| {
81                storage_error(
82                    format!("Failed to read input file: {e}"),
83                    Some("read_file".to_string()),
84                )
85            })?
86        } else if let Some(input) = input_str {
87            if binary_input {
88                // Parse hex string as binary
89                hex::decode(input).map_err(|e| {
90                    validation_error(
91                        format!("Failed to parse hex input: {e}"),
92                        Some("input".to_string()),
93                    )
94                })?
95            } else {
96                // Use text input
97                input.as_bytes().to_vec()
98            }
99        } else {
100            return Err(validation_error(
101                "Either 'input' or 'input_file' must be provided",
102                Some("input".to_string()),
103            ));
104        };
105
106        // Perform operation
107        let result_data = match operation {
108            "encode" => {
109                let encoded = match variant {
110                    "url-safe" => base64_encode_url_safe(&input_data),
111                    _ => base64_encode(&input_data),
112                };
113                encoded.into_bytes()
114            }
115            "decode" => {
116                let input_str = if let Some(input) = input_str {
117                    input.to_string()
118                } else {
119                    // Convert file data to string for decoding
120                    String::from_utf8(input_data).map_err(|e| {
121                        validation_error(
122                            format!("Input file contains invalid UTF-8 for Base64 decoding: {e}"),
123                            Some("input_file".to_string()),
124                        )
125                    })?
126                };
127
128                let decoded = match variant {
129                    "url-safe" => base64_decode_url_safe(&input_str),
130                    _ => base64_decode(&input_str),
131                };
132
133                decoded.map_err(|e| {
134                    validation_error(
135                        format!("Base64 decode error: {e}"),
136                        Some("input".to_string()),
137                    )
138                })?
139            }
140            _ => unreachable!(), // Already validated
141        };
142
143        // Handle output
144        let output_path = extract_optional_string(params, "output_file");
145
146        if let Some(path) = output_path {
147            // Write to file
148            fs::write(path, &result_data).map_err(|e| {
149                storage_error(
150                    format!("Failed to write output file: {e}"),
151                    Some("write_file".to_string()),
152                )
153            })?;
154
155            info!(
156                "Base64 {} completed: {} -> {}",
157                operation,
158                input_file.unwrap_or("input"),
159                path
160            );
161
162            Ok(ResponseBuilder::success(operation)
163                .with_message(format!("Base64 {operation} completed successfully"))
164                .with_metadata("variant", json!(variant))
165                .with_file_info(path, Some(result_data.len() as u64))
166                .build())
167        } else {
168            // Return as string
169            let output = match operation {
170                "encode" => String::from_utf8_lossy(&result_data).to_string(),
171                "decode" => {
172                    // Try to convert to string, otherwise return hex
173                    String::from_utf8(result_data.clone())
174                        .unwrap_or_else(|_| hex::encode(&result_data))
175                }
176                _ => unreachable!(),
177            };
178
179            Ok(ResponseBuilder::success(operation)
180                .with_message(format!("Base64 {operation} completed"))
181                .with_result(json!({
182                    "output": output,
183                    "variant": variant,
184                    "binary": operation == "decode" && !output.is_ascii()
185                }))
186                .build())
187        }
188    }
189}

191#[async_trait]
192impl BaseAgent for Base64EncoderTool {
193    fn metadata(&self) -> &ComponentMetadata {
194        &self.metadata
195    }
196
197    async fn execute_impl(
198        &self,
199        input: AgentInput,
200        _context: ExecutionContext,
201    ) -> Result<AgentOutput> {
202        // Extract parameters using shared utility
203        let params = extract_parameters(&input)?;
204
205        // Process the operation
206        let result = self.process_operation(params).await?;
207
208        // Return the result as JSON formatted text
209        Ok(AgentOutput::text(serde_json::to_string_pretty(&result)?))
210    }
211
212    async fn validate_input(&self, input: &AgentInput) -> Result<()> {
213        if input.text.is_empty() {
214            return Err(validation_error(
215                "Input prompt cannot be empty",
216                Some("prompt".to_string()),
217            ));
218        }
219        Ok(())
220    }
221
222    async fn handle_error(&self, error: LLMSpellError) -> Result<AgentOutput> {
223        Ok(AgentOutput::text(format!("Base64 encoding error: {error}")))
224    }
225}

227#[async_trait]
228impl Tool for Base64EncoderTool {
229    fn category(&self) -> ToolCategory {
230        ToolCategory::Utility
231    }
232
233    fn security_level(&self) -> SecurityLevel {
234        SecurityLevel::Safe
235    }
236
237    fn schema(&self) -> ToolSchema {
238        ToolSchema::new(
239            "base64_encoder".to_string(),
240            "Base64 encoding and decoding tool with standard and URL-safe variants".to_string(),
241        )
242        .with_parameter(ParameterDef {
243            name: "operation".to_string(),
244            param_type: ParameterType::String,
245            description: "Operation to perform: 'encode' or 'decode'".to_string(),
246            required: true,
247            default: None,
248        })
249        .with_parameter(ParameterDef {
250            name: "variant".to_string(),
251            param_type: ParameterType::String,
252            description: "Base64 variant: 'standard' or 'url-safe' (default: 'standard')"
253                .to_string(),
254            required: false,
255            default: Some(json!("standard")),
256        })
257        .with_parameter(ParameterDef {
258            name: "input".to_string(),
259            param_type: ParameterType::String,
260            description: "Input data (text for encode, base64 for decode)".to_string(),
261            required: false,
262            default: None,
263        })
264        .with_parameter(ParameterDef {
265            name: "input_file".to_string(),
266            param_type: ParameterType::String,
267            description: "Input file path (alternative to input)".to_string(),
268            required: false,
269            default: None,
270        })
271        .with_parameter(ParameterDef {
272            name: "output_file".to_string(),
273            param_type: ParameterType::String,
274            description: "Output file path (optional)".to_string(),
275            required: false,
276            default: None,
277        })
278        .with_parameter(ParameterDef {
279            name: "binary_input".to_string(),
280            param_type: ParameterType::Boolean,
281            description: "Treat input as hex string for binary data".to_string(),
282            required: false,
283            default: Some(json!(false)),
284        })
285        .with_returns(ParameterType::Object)
286    }
287
288    fn security_requirements(&self) -> SecurityRequirements {
289        SecurityRequirements::safe()
290    }
291
292    fn resource_limits(&self) -> ResourceLimits {
293        ResourceLimits::strict()
294            .with_memory_limit(50 * 1024 * 1024) // 50MB
295            .with_cpu_limit(5000) // 5 seconds
296    }

304    async fn test_encode_decode_text() {
305        let tool = Base64EncoderTool::new();
306        let test_text = "Hello, Base64!";
307
308        // Test standard encoding
309        let input = AgentInput::text("encode text").with_parameter(
310            "parameters",
311            json!({
312                "operation": "encode",
313                "input": test_text
314            }),
315        );
316
317        let result = tool
318            .execute(input, ExecutionContext::default())
319            .await
320            .unwrap();
321        let output: Value = serde_json::from_str(&result.text).unwrap();
322        assert!(output["success"].as_bool().unwrap_or(false));
323        let encoded = output["result"]["output"].as_str().unwrap();
324        assert_eq!(encoded, "SGVsbG8sIEJhc2U2NCE=");
325
326        // Test standard decoding
327        let input = AgentInput::text("decode text").with_parameter(
328            "parameters",

330                "operation": "decode",
