33/// Diff output format
34#[derive(Debug, Clone)]
35enum DiffFormat {
36    /// Unified diff format (like git diff)
37    Unified,
38    /// Context diff format
39    Context,
40    /// Inline diff format
41    Inline,
42    /// Simple list of changes
43    Simple,
44}
45
46impl DiffFormat {
47    fn from_str(s: &str) -> Result<Self> {
48        match s.to_lowercase().as_str() {
49            "unified" => Ok(Self::Unified),
50            "context" => Ok(Self::Context),
51            "inline" => Ok(Self::Inline),
52            "simple" => Ok(Self::Simple),
53            _ => Err(validation_error(
54                format!("Invalid diff format: {s}"),
55                Some("format".to_string()),
56            )),
57        }
58    }
59}

86    /// Calculate text diff
87    #[allow(clippy::unused_self)]
88    #[allow(clippy::too_many_lines)]
89    fn calculate_text_diff(&self, old: &str, new: &str, format: &DiffFormat) -> String {
90        use std::fmt::Write;
91
92        let diff = TextDiff::from_lines(old, new);
93
94        match format {
95            DiffFormat::Unified => diff
96                .unified_diff()
97                .context_radius(3)
98                .header("old", "new")
99                .to_string(),
100            DiffFormat::Context => {
101                // Context format - show more surrounding lines
102                let mut output = String::new();
103                output.push_str("*** old\n--- new\n");
104                output.push_str("***************\n");
105
106                for group in diff.grouped_ops(3) {
107                    for op in group {
108                        let (tag, old_range, new_range) =
109                            (op.tag(), op.old_range(), op.new_range());
110
111                        match tag {
112                            DiffTag::Equal => {
113                                // Get the old slices and extract lines for the range
114                                let old_slices = diff.old_slices();
115                                for idx in old_range {
116                                    if let Some(line) = old_slices.get(idx) {
117                                        let _ = writeln!(output, "  {line}");
118                                    }
119                                }
120                            }
121                            DiffTag::Delete => {
122                                // Get the old slices and extract lines for the range
123                                let old_slices = diff.old_slices();
124                                for idx in old_range {
125                                    if let Some(line) = old_slices.get(idx) {
126                                        let _ = writeln!(output, "- {line}");
127                                    }
128                                }
129                            }
130                            DiffTag::Insert => {
131                                // Get the new slices and extract lines for the range
132                                let new_slices = diff.new_slices();
133                                for idx in new_range {
134                                    if let Some(line) = new_slices.get(idx) {
135                                        let _ = writeln!(output, "+ {line}");
136                                    }
137                                }
138                            }
139                            DiffTag::Replace => {
140                                // For replace, show both delete and insert
141                                let old_slices = diff.old_slices();
142                                for idx in old_range {
143                                    if let Some(line) = old_slices.get(idx) {
144                                        let _ = writeln!(output, "- {line}");
145                                    }
146                                }
147                                let new_slices = diff.new_slices();
148                                for idx in new_range {
149                                    if let Some(line) = new_slices.get(idx) {
150                                        let _ = writeln!(output, "+ {line}");
151                                    }
152                                }
153                            }
154                        }
155                    }
156                    output.push_str("***************\n");
157                }
158                output
159            }
160            DiffFormat::Inline => {
161                // For inline diff, we'll use iter_all_changes to show inline changes
162                let mut output = String::new();
163                for change in diff.iter_all_changes() {
164                    let sign = match change.tag() {
165                        ChangeTag::Delete => "-",
166                        ChangeTag::Insert => "+",
167                        ChangeTag::Equal => " ",
168                    };
169                    let _ = write!(output, "{sign}{change}");
170                }
171                output
172            }
173            DiffFormat::Simple => {
174                let mut output = String::new();
175                let mut changes = 0;
176
177                for op in diff.ops() {
178                    match op.tag() {
179                        DiffTag::Delete => {
180                            changes += 1;
181                            let _ = writeln!(
182                                output,
183                                "Removed at line {}: {} line(s)",
184                                op.old_range().start + 1,
185                                op.old_range().len()
186                            );
187                        }
188                        DiffTag::Insert => {
189                            changes += 1;
190                            let _ = writeln!(
191                                output,
192                                "Added at line {}: {} line(s)",
193                                op.new_range().start + 1,
194                                op.new_range().len()
195                            );
196                        }
197                        DiffTag::Equal => {}
198                        DiffTag::Replace => {
199                            changes += 1;
200                            let _ = writeln!(
201                                output,
202                                "Replaced at lines {}-{} with lines {}-{}",
203                                op.old_range().start + 1,
204                                op.old_range().end,
205                                op.new_range().start + 1,
206                                op.new_range().end
207                            );
208                        }
209                    }
210                }
211
212                output.insert_str(0, &format!("Total changes: {changes}\n\n"));
213                output
214            }
215        }
216    }

218    /// Calculate JSON diff
219    #[allow(clippy::unused_self)]
220    fn calculate_json_diff(&self, old_json: &Value, new_json: &Value) -> Result<Value> {
221        let mut diff = json!({
222            "added": {},
223            "removed": {},
224            "modified": {},
225            "unchanged": []
226        });
227
228        compare_json_values(old_json, new_json, "", &mut diff)?;
229
230        Ok(diff)
231    }
232}
233
234/// Recursively compare JSON values
235fn compare_json_values(old: &Value, new: &Value, path: &str, diff: &mut Value) -> Result<()> {
236    match (old, new) {
237        (Value::Object(old_map), Value::Object(new_map)) => {
238            // Check for removed keys
239            for (key, value) in old_map {
240                let current_path = if path.is_empty() {
241                    key.clone()
242                } else {
243                    format!("{path}.{key}")
244                };
245
246                if !new_map.contains_key(key) {
247                    diff["removed"][&current_path] = value.clone();
248                }
249            }
250
251            // Check for added and modified keys
252            for (key, new_value) in new_map {
253                let current_path = if path.is_empty() {
254                    key.clone()
255                } else {
256                    format!("{path}.{key}")
257                };
258
259                if let Some(old_value) = old_map.get(key) {
260                    if old_value == new_value {
261                        diff["unchanged"]
262                            .as_array_mut()
263                            .ok_or_else(|| LLMSpellError::Internal {
264                                message: "Failed to access unchanged array".to_string(),
265                                source: None,
266                            })?
267                            .push(json!(current_path));
268                    } else if old_value.is_object() && new_value.is_object() {
269                        compare_json_values(old_value, new_value, &current_path, diff)?;
270                    } else {
271                        diff["modified"][&current_path] = json!({
272                            "old": old_value,
273                            "new": new_value
274                        });
275                    }
276                } else {
277                    diff["added"][&current_path] = new_value.clone();
278                }
279            }
280        }
281        (Value::Array(old_arr), Value::Array(new_arr)) => {
282            let current_path = if path.is_empty() {
283                "[]".to_string()
284            } else {
285                format!("{path}[]")
286            };
287
288            if old_arr.len() != new_arr.len() || old_arr != new_arr {
289                diff["modified"][&current_path] = json!({
290                    "old_length": old_arr.len(),
291                    "new_length": new_arr.len(),
292                    "old": old_arr,
293                    "new": new_arr
294                });
295            } else {
296                diff["unchanged"]
297                    .as_array_mut()
298                    .ok_or_else(|| LLMSpellError::Internal {
299                        message: "Failed to access unchanged array".to_string(),
300                        source: None,
301                    })?
302                    .push(json!(current_path));
303            }
304        }
305        _ => {
306            if old == new {
307                diff["unchanged"]
308                    .as_array_mut()
309                    .ok_or_else(|| LLMSpellError::Internal {
310                        message: "Failed to access unchanged array".to_string(),
311                        source: None,
312                    })?
313                    .push(json!(path));
314            } else {
315                diff["modified"][path] = json!({
316                    "old": old,
317                    "new": new
318                });
319            }
320        }
321    }
322
323    Ok(())
324}

326impl DiffCalculatorTool {
327    /// Process diff operation
328    #[allow(clippy::unused_async)]
329    #[allow(clippy::too_many_lines)]
330    async fn process_operation(&self, params: &Value) -> Result<Value> {
331        let diff_type = extract_string_with_default(params, "type", "text");
332
333        match diff_type {
334            "text" => {
335                // Get text inputs
336                let (old_text, new_text) = if let (Some(old_file), Some(new_file)) = (
337                    extract_optional_string(params, "old_file"),
338                    extract_optional_string(params, "new_file"),
339                ) {
340                    // Read from files
341                    let old = fs::read_to_string(old_file).map_err(|e| {
342                        tool_error(
343                            format!("Failed to read old file: {e}"),
344                            Some(self.metadata.name.clone()),
345                        )
346                    })?;
347                    let new = fs::read_to_string(new_file).map_err(|e| {
348                        tool_error(
349                            format!("Failed to read new file: {e}"),
350                            Some(self.metadata.name.clone()),
351                        )
352                    })?;
353                    (old, new)
354                } else if let (Some(old), Some(new)) = (
355                    extract_optional_string(params, "old_text"),
356                    extract_optional_string(params, "new_text"),
357                ) {
358                    (old.to_string(), new.to_string())
359                } else {
360                    return Err(validation_error(
361                        "Either (old_text, new_text) or (old_file, new_file) must be provided",
362                        Some("input".to_string()),
363                    ));
364                };
365
366                // Get format
367                let format_str = extract_string_with_default(params, "format", "unified");
368                let format = DiffFormat::from_str(format_str)?;
369
370                // Calculate diff
371                let diff_output = self.calculate_text_diff(&old_text, &new_text, &format);
372
373                let response = ResponseBuilder::success("text_diff")
374                    .with_message("Text diff calculated successfully")
375                    .with_result(json!({
376                        "type": "text",
377                        "format": format_str,
378                        "diff": diff_output,
379                        "stats": {
380                            "old_lines": old_text.lines().count(),
381                            "new_lines": new_text.lines().count()
382                        }
383                    }))
384                    .build();
385                Ok(response)
386            }
387            "json" => {
388                // Get JSON inputs
389                let (old_json, new_json) = if let (Some(old_file), Some(new_file)) = (
390                    extract_optional_string(params, "old_file"),
391                    extract_optional_string(params, "new_file"),
392                ) {
393                    // Read from files
394                    let old_content = fs::read_to_string(old_file).map_err(|e| {
395                        tool_error(
396                            format!("Failed to read old file: {e}"),
397                            Some(self.metadata.name.clone()),
398                        )
399                    })?;
400                    let new_content = fs::read_to_string(new_file).map_err(|e| {
401                        tool_error(
402                            format!("Failed to read new file: {e}"),
403                            Some(self.metadata.name.clone()),
404                        )
405                    })?;
406
407                    let old_json: Value = serde_json::from_str(&old_content).map_err(|e| {
408                        tool_error(
409                            format!("Failed to parse old JSON: {e}"),
410                            Some(self.metadata.name.clone()),
411                        )
412                    })?;
413                    let new_json: Value = serde_json::from_str(&new_content).map_err(|e| {
414                        tool_error(
415                            format!("Failed to parse new JSON: {e}"),
416                            Some(self.metadata.name.clone()),
417                        )
418                    })?;
419                    (old_json, new_json)
420                } else if let (Some(old), Some(new)) =
421                    (params.get("old_json"), params.get("new_json"))
422                {
423                    (old.clone(), new.clone())
424                } else {
425                    return Err(validation_error(
426                        "Either (old_json, new_json) or (old_file, new_file) must be provided",
427                        Some("input".to_string()),
428                    ));
429                };
430
431                // Calculate JSON diff
432                let diff = self.calculate_json_diff(&old_json, &new_json)?;
433
434                let response = ResponseBuilder::success("json_diff")
435                    .with_message("JSON diff calculated successfully")
436                    .with_result(json!({
437                        "type": "json",
438                        "diff": diff,
439                        "summary": {
440                            "added": diff["added"].as_object().map_or(0, serde_json::Map::len),
441                            "removed": diff["removed"].as_object().map_or(0, serde_json::Map::len),
442                            "modified": diff["modified"].as_object().map_or(0, serde_json::Map::len),
443                            "unchanged": diff["unchanged"].as_array().map_or(0, std::vec::Vec::len)
444                        }
445                    }))
446                    .build();
447                Ok(response)
448            }
449            _ => Err(validation_error(
450                format!("Invalid diff type: {diff_type}"),
451                Some("type".to_string()),
452            )),
453        }
454    }
455}

457#[async_trait]
458impl BaseAgent for DiffCalculatorTool {
459    fn metadata(&self) -> &ComponentMetadata {
460        &self.metadata
461    }
462
463    async fn execute_impl(
464        &self,
465        input: AgentInput,
466        _context: ExecutionContext,
467    ) -> Result<AgentOutput> {
468        // Get parameters using shared utility
469        let params = extract_parameters(&input)?;
470
471        // Process the operation
472        let result = self.process_operation(params).await?;
473
474        // Return the result as JSON formatted text
475        Ok(AgentOutput::text(
476            serde_json::to_string_pretty(&result).unwrap(),
477        ))
478    }
479
480    async fn validate_input(&self, input: &AgentInput) -> Result<()> {
481        if input.text.is_empty() {
482            return Err(validation_error(
483                "Input prompt cannot be empty",
484                Some("prompt".to_string()),
485            ));
486        }
487        Ok(())
488    }
489
490    async fn handle_error(&self, error: LLMSpellError) -> Result<AgentOutput> {
491        Ok(AgentOutput::text(format!(
492            "Diff calculation error: {error}"
493        )))
494    }
495}
