1//! ABOUTME: Standard environment variable registrations for LLMSpell
2//! ABOUTME: Defines and registers all built-in environment variables using config paths
3
4use crate::env::{EnvCategory, EnvRegistry, EnvVarDefBuilder};

6/// Register all standard LLMSpell environment variables
7pub fn register_standard_vars(registry: &EnvRegistry) -> Result<(), String> {
8    // Core Runtime Variables
9    register_runtime_vars(registry)?;
10
11    // State Persistence Variables
12    register_state_vars(registry)?;
13
14    // Provider Configuration Variables
15    register_provider_vars(registry)?;
16
17    // Tool Configuration Variables
18    register_tool_vars(registry)?;
19
20    // Session/Hook Variables
21    register_session_hook_vars(registry)?;
22
23    // Path Discovery Variables
24    register_path_vars(registry)?;
25
26    // Event System Variables
27    register_event_vars(registry)?;
28
29    Ok(())
30}
