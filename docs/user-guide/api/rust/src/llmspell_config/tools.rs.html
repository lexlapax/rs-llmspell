264impl FileOperationsConfig {
265    /// Create a new builder for `FileOperationsConfig`
266    #[must_use]
267    pub fn builder() -> FileOperationsConfigBuilder {
268        FileOperationsConfigBuilder::new()
269    }
270
271    /// Check if a path is allowed
272    #[must_use]
273    pub fn is_path_allowed(&self, path: &str) -> bool {
274        if self.allowed_paths.contains(&"*".to_string()) {
275            return true;
276        }
277
278        for allowed_path in &self.allowed_paths {
279            if path == *allowed_path || path.starts_with(allowed_path) {
280                return true;
281            }
282        }
283
284        false
285    }
286
287    /// Check if a file extension is allowed
288    #[must_use]
289    pub fn is_extension_allowed(&self, extension: &str) -> bool {
290        // If extension is blocked, deny
291        if self.blocked_extensions.contains(&extension.to_lowercase()) {
292            return false;
293        }
294
295        // If allowed_extensions is empty, allow all (except blocked)
296        if self.allowed_extensions.is_empty() {
297            return true;
298        }
299
300        // Check if extension is in allowed list
301        self.allowed_extensions.contains(&extension.to_lowercase())
302    }
303}

433    /// Check if a domain is allowed
434    #[must_use]
435    pub fn is_domain_allowed(&self, domain: &str) -> bool {
436        // Check if domain is blocked
437        if self
438            .blocked_domains
439            .iter()
440            .any(|blocked| domain.contains(blocked))
441        {
442            return false;
443        }
444
445        // If allowed_domains contains "*", allow all (except blocked)
446        if self.allowed_domains.contains(&"*".to_string()) {
447            return true;
448        }
449
450        // Check if domain matches any allowed domain
451        self.allowed_domains
452            .iter()
453            .any(|allowed| domain.contains(allowed))
454    }

790pub struct DocumentToolsConfig {

794    pub extract_images: Option<bool>,
795    /// OCR enable for scanned documents
796    pub enable_ocr: Option<bool>,

854            .blocked_extensions(vec!["exe".to_string()])
855            .build();

858        assert!(config.is_extension_allowed("json"));
859        assert!(!config.is_extension_allowed("exe"));
860        assert!(!config.is_extension_allowed("dll")); // Not in allowed list
861
862        // Test default (empty allowed = all allowed except blocked)
863        let config = FileOperationsConfig::default();
864        assert!(config.is_extension_allowed("txt"));
865        assert!(!config.is_extension_allowed("exe")); // Blocked by default
