95    /// Parse TOML content with environment variable overrides and validation
96    pub fn from_toml(content: &str) -> Result<Self, ConfigError> {
97        let mut config: LLMSpellConfig = toml::from_str(content)?;
98
99        // Use registry for environment overrides
100        config.apply_env_registry()?;
101        config.validate()?;
102
103        Ok(config)
104    }

106    /// Apply environment variable overrides using the centralized registry
107    pub fn apply_env_registry(&mut self) -> Result<(), ConfigError> {
108        // Create registry and load standard variables
109        let registry = EnvRegistry::new();
110        register_standard_vars(&registry).map_err(|e| ConfigError::Environment { message: e })?;
111
112        // Load environment variables
113        registry
114            .load_from_env()
115            .map_err(|e| ConfigError::Environment { message: e })?;
116
117        // Build config from registry
118        let env_config = registry
119            .build_config()
120            .map_err(|e| ConfigError::Environment { message: e })?;
121
122        // Merge environment config into self
123        self.merge_from_json_impl(&env_config)?;
124
125        Ok(())
126    }

133    /// Internal implementation of merge_from_json
134    fn merge_from_json_impl(&mut self, json: &serde_json::Value) -> Result<(), ConfigError> {
135        // Merge top-level values
136        if let Some(engine) = json.get("default_engine").and_then(|v| v.as_str()) {
137            debug!("Overriding default engine from env: {}", engine);
138            self.default_engine = engine.to_string();
139        }
140
141        // Merge runtime values
142        if let Some(runtime) = json.get("runtime").and_then(|v| v.as_object()) {
143            if let Some(max_scripts) = runtime
144                .get("max_concurrent_scripts")
145                .and_then(|v| v.as_u64())
146            {
147                debug!(
148                    "Overriding max_concurrent_scripts from env: {}",
149                    max_scripts
150                );
151                self.runtime.max_concurrent_scripts = max_scripts as usize;
152            }
153
154            if let Some(timeout) = runtime
155                .get("script_timeout_seconds")
156                .and_then(|v| v.as_u64())
157            {
158                debug!("Overriding script_timeout_seconds from env: {}", timeout);
159                self.runtime.script_timeout_seconds = timeout;
160            }
161
162            // Merge security settings
163            if let Some(security) = runtime.get("security").and_then(|v| v.as_object()) {
164                if let Some(allow_file) =
165                    security.get("allow_file_access").and_then(|v| v.as_bool())
166                {
167                    debug!("Overriding allow_file_access from env: {}", allow_file);
168                    self.runtime.security.allow_file_access = allow_file;
169                }
170
171                if let Some(allow_network) = security
172                    .get("allow_network_access")
173                    .and_then(|v| v.as_bool())
174                {
175                    debug!(
176                        "Overriding allow_network_access from env: {}",
177                        allow_network
178                    );
179                    self.runtime.security.allow_network_access = allow_network;
180                }
181
182                if let Some(allow_spawn) = security
183                    .get("allow_process_spawn")
184                    .and_then(|v| v.as_bool())
185                {
186                    self.runtime.security.allow_process_spawn = allow_spawn;
187                }
188
189                if let Some(max_memory) = security.get("max_memory_bytes").and_then(|v| v.as_u64())
190                {
191                    self.runtime.security.max_memory_bytes = Some(max_memory as usize);
192                }
193
194                if let Some(max_time) = security
195                    .get("max_execution_time_ms")
196                    .and_then(|v| v.as_u64())
197                {
198                    self.runtime.security.max_execution_time_ms = Some(max_time);
199                }
200            }
201
202            // Merge state persistence settings
203            if let Some(state) = runtime.get("state_persistence").and_then(|v| v.as_object()) {
204                if let Some(backend) = state.get("backend_type").and_then(|v| v.as_str()) {
205                    self.runtime.state_persistence.backend_type = backend.to_string();
206                }
207
208                // Flattened structure - direct access
209                if let Some(enabled) = state.get("enabled").and_then(|v| v.as_bool()) {
210                    self.runtime.state_persistence.enabled = enabled;
211                }
212                if let Some(migration) = state.get("migration_enabled").and_then(|v| v.as_bool()) {
213                    self.runtime.state_persistence.migration_enabled = migration;
214                }
215                if let Some(backup_on_migration) =
216                    state.get("backup_on_migration").and_then(|v| v.as_bool())
217                {
218                    self.runtime.state_persistence.backup_on_migration = backup_on_migration;
219                }
220                if let Some(backup_enabled) = state.get("backup_enabled").and_then(|v| v.as_bool())
221                {
222                    self.runtime.state_persistence.backup_enabled = backup_enabled;
223                }
224            }
225
226            // Merge session settings
227            if let Some(sessions) = runtime.get("sessions").and_then(|v| v.as_object()) {
228                if let Some(enabled) = sessions.get("enabled").and_then(|v| v.as_bool()) {
229                    self.runtime.sessions.enabled = enabled;
230                }
231                if let Some(backend) = sessions.get("storage_backend").and_then(|v| v.as_str()) {
232                    self.runtime.sessions.storage_backend = backend.to_string();
233                }
234            }
235        }
236
237        // Merge provider configurations
238        if let Some(providers) = json.get("providers").and_then(|v| v.as_object()) {
239            // Handle flattened structure - provider configs are direct children
240            for (name, config) in providers {
241                // Skip default_provider field, only process provider configs
242                if name == "default_provider" {
243                    continue;
244                }
245
246                if let Some(provider_obj) = config.as_object() {
247                    // Check if provider already exists in config
248                    if let Some(existing_provider) = self.providers.providers.get_mut(name) {
249                        // Provider exists - only update fields that are present in env config
250                        if let Some(api_key) = provider_obj.get("api_key").and_then(|v| v.as_str())
251                        {
252                            existing_provider.api_key = Some(api_key.to_string());
253                        }
254                        if let Some(base_url) =
255                            provider_obj.get("base_url").and_then(|v| v.as_str())
256                        {
257                            existing_provider.base_url = Some(base_url.to_string());
258                        }
259                        if let Some(model) =
260                            provider_obj.get("default_model").and_then(|v| v.as_str())
261                        {
262                            existing_provider.default_model = Some(model.to_string());
263                        }
264                        if let Some(timeout) =
265                            provider_obj.get("timeout_seconds").and_then(|v| v.as_u64())
266                        {
267                            existing_provider.timeout_seconds = Some(timeout);
268                        }
269                        if let Some(max_retries) =
270                            provider_obj.get("max_retries").and_then(|v| v.as_u64())
271                        {
272                            existing_provider.max_retries = Some(max_retries as u32);
273                        }
274                        // Only update other fields if they're present
275                        if let Some(provider_type) =
276                            provider_obj.get("provider_type").and_then(|v| v.as_str())
277                        {
278                            existing_provider.provider_type = provider_type.to_string();
279                        }
280                        if let Some(enabled) = provider_obj.get("enabled").and_then(|v| v.as_bool())
281                        {
282                            existing_provider.enabled = enabled;
283                        }
284                        // Do NOT insert/replace - we already modified in place
285                    } else {
286                        // Provider doesn't exist in config - only create if it has minimum required fields
287                        // Don't create incomplete providers from just API keys
288                        if provider_obj.contains_key("default_model")
289                            || provider_obj.contains_key("provider_type")
290                        {
291                            let mut provider_config = ProviderConfig {
292                                name: name.clone(),
293                                provider_type: provider_obj
294                                    .get("provider_type")
295                                    .and_then(|v| v.as_str())
296                                    .unwrap_or(name)
297                                    .to_string(),
298                                enabled: true,
299                                ..Default::default()
300                            };
301
302                            if let Some(api_key) =
303                                provider_obj.get("api_key").and_then(|v| v.as_str())
304                            {
305                                provider_config.api_key = Some(api_key.to_string());
306                            }
307                            if let Some(base_url) =
308                                provider_obj.get("base_url").and_then(|v| v.as_str())
309                            {
310                                provider_config.base_url = Some(base_url.to_string());
311                            }
312                            if let Some(model) =
313                                provider_obj.get("default_model").and_then(|v| v.as_str())
314                            {
315                                provider_config.default_model = Some(model.to_string());
316                            }
317
318                            self.providers
319                                .providers
320                                .insert(name.clone(), provider_config);
321                        }
322                        // Otherwise skip - don't create incomplete provider from just api_key
323                    }
324                }
325            }
326        }
327
328        // Merge tool configurations
329        if let Some(tools) = json.get("tools").and_then(|v| v.as_object()) {
330            if let Some(file_ops) = tools.get("file_operations").and_then(|v| v.as_object()) {
331                if let Some(enabled) = file_ops.get("enabled").and_then(|v| v.as_bool()) {
332                    self.tools.file_operations.enabled = enabled;
333                }
334                if let Some(max_size) = file_ops.get("max_file_size").and_then(|v| v.as_u64()) {
335                    self.tools.file_operations.max_file_size = max_size as usize;
336                }
337                // Handle allowed_paths - can be either string (from env) or array (from JSON)
338                if let Some(paths_value) = file_ops.get("allowed_paths") {
339                    if let Some(paths_str) = paths_value.as_str() {
340                        // From environment variable - comma-separated string
341                        self.tools.file_operations.allowed_paths =
342                            paths_str.split(',').map(|s| s.trim().to_string()).collect();
343                    } else if let Some(paths_array) = paths_value.as_array() {
344                        // From JSON - array of strings
345                        self.tools.file_operations.allowed_paths = paths_array
346                            .iter()
347                            .filter_map(|v| v.as_str().map(String::from))
348                            .collect();
349                    }
350                }
351            }
352            if let Some(network) = tools.get("network").and_then(|v| v.as_object()) {
353                let net_config = self.tools.network.get_or_insert(Default::default());
354                if let Some(timeout) = network.get("timeout_seconds").and_then(|v| v.as_u64()) {
355                    net_config.timeout_seconds = timeout;
356                }
357            }
358            if let Some(rate_limit) = tools.get("rate_limit_per_minute").and_then(|v| v.as_u64()) {
359                self.tools.rate_limit_per_minute = Some(rate_limit as u32);
360            }
361        }
362
363        // Merge hook configuration
364        if let Some(hooks) = json.get("hooks").and_then(|v| v.as_object()) {
365            let hook_config = self.hooks.get_or_insert(Default::default());
366            if let Some(enabled) = hooks.get("enabled").and_then(|v| v.as_bool()) {
367                hook_config.enabled = enabled;
368            }
369            if let Some(rate_limit) = hooks.get("rate_limit_per_minute").and_then(|v| v.as_u64()) {
370                hook_config.rate_limit_per_minute = Some(rate_limit as u32);
371            }
372        }
373
374        // Merge events configuration
375        if let Some(events) = json.get("events").and_then(|v| v.as_object()) {
376            if let Some(enabled) = events.get("enabled").and_then(|v| v.as_bool()) {
377                debug!("Overriding events.enabled from env: {}", enabled);
378                self.events.enabled = enabled;
379            }
380
381            if let Some(buffer_size) = events.get("buffer_size").and_then(|v| v.as_u64()) {
382                debug!("Overriding events.buffer_size from env: {}", buffer_size);
383                self.events.buffer_size = buffer_size as usize;
384            }
385
386            if let Some(emit_timing) = events.get("emit_timing_events").and_then(|v| v.as_bool()) {
387                debug!(
388                    "Overriding events.emit_timing_events from env: {}",
389                    emit_timing
390                );
391                self.events.emit_timing_events = emit_timing;
392            }
393
394            if let Some(emit_state) = events.get("emit_state_events").and_then(|v| v.as_bool()) {
395                debug!(
396                    "Overriding events.emit_state_events from env: {}",
397                    emit_state
398                );
399                self.events.emit_state_events = emit_state;
400            }
401
402            if let Some(emit_debug) = events.get("emit_debug_events").and_then(|v| v.as_bool()) {
403                debug!(
404                    "Overriding events.emit_debug_events from env: {}",
405                    emit_debug
406                );
407                self.events.emit_debug_events = emit_debug;
408            }
409
410            if let Some(max_events) = events.get("max_events_per_second").and_then(|v| v.as_u64()) {
411                debug!(
412                    "Overriding events.max_events_per_second from env: {}",
413                    max_events
414                );
415                self.events.max_events_per_second = Some(max_events as u32);
416            }
417
418            // Merge filtering configuration
419            if let Some(filtering) = events.get("filtering").and_then(|v| v.as_object()) {
420                if let Some(include_types_value) = filtering.get("include_types") {
421                    if let Some(include_types_str) = include_types_value.as_str() {
422                        // From environment variable - comma-separated string
423                        self.events.filtering.include_types = include_types_str
424                            .split(',')
425                            .map(|s| s.trim().to_string())
426                            .collect();
427                    } else if let Some(include_types_array) = include_types_value.as_array() {
428                        // From JSON - array of strings
429                        self.events.filtering.include_types = include_types_array
430                            .iter()
431                            .filter_map(|v| v.as_str().map(String::from))
432                            .collect();
433                    }
434                }
435
436                if let Some(exclude_types_value) = filtering.get("exclude_types") {
437                    if let Some(exclude_types_str) = exclude_types_value.as_str() {
438                        // From environment variable - comma-separated string
439                        self.events.filtering.exclude_types = exclude_types_str
440                            .split(',')
441                            .map(|s| s.trim().to_string())
442                            .collect();
443                    } else if let Some(exclude_types_array) = exclude_types_value.as_array() {
444                        // From JSON - array of strings
445                        self.events.filtering.exclude_types = exclude_types_array
446                            .iter()
447                            .filter_map(|v| v.as_str().map(String::from))
448                            .collect();
449                    }
450                }
451
452                if let Some(include_components_value) = filtering.get("include_components") {
453                    if let Some(include_components_str) = include_components_value.as_str() {
454                        // From environment variable - comma-separated string
455                        self.events.filtering.include_components = include_components_str
456                            .split(',')
457                            .map(|s| s.trim().to_string())
458                            .collect();
459                    } else if let Some(include_components_array) =
460                        include_components_value.as_array()
461                    {
462                        // From JSON - array of strings
463                        self.events.filtering.include_components = include_components_array
464                            .iter()
465                            .filter_map(|v| v.as_str().map(String::from))
466                            .collect();
467                    }
468                }
469
470                if let Some(exclude_components_value) = filtering.get("exclude_components") {
471                    if let Some(exclude_components_str) = exclude_components_value.as_str() {
472                        // From environment variable - comma-separated string
473                        self.events.filtering.exclude_components = exclude_components_str
474                            .split(',')
475                            .map(|s| s.trim().to_string())
476                            .collect();
477                    } else if let Some(exclude_components_array) =
478                        exclude_components_value.as_array()
479                    {
480                        // From JSON - array of strings
481                        self.events.filtering.exclude_components = exclude_components_array
482                            .iter()
483                            .filter_map(|v| v.as_str().map(String::from))
484                            .collect();
485                    }
486                }
487            }
488
489            // Merge export configuration
490            if let Some(export) = events.get("export").and_then(|v| v.as_object()) {
491                if let Some(stdout) = export.get("stdout").and_then(|v| v.as_bool()) {
492                    debug!("Overriding events.export.stdout from env: {}", stdout);
493                    self.events.export.stdout = stdout;
494                }
495
496                if let Some(file) = export.get("file").and_then(|v| v.as_str()) {
497                    debug!("Overriding events.export.file from env: {}", file);
498                    self.events.export.file = Some(file.to_string());
499                }
500
501                if let Some(webhook) = export.get("webhook").and_then(|v| v.as_str()) {
502                    debug!("Overriding events.export.webhook from env: {}", webhook);
503                    self.events.export.webhook = Some(webhook.to_string());
504                }
505
506                if let Some(pretty_json) = export.get("pretty_json").and_then(|v| v.as_bool()) {
507                    debug!(
508                        "Overriding events.export.pretty_json from env: {}",
509                        pretty_json
510                    );
511                    self.events.export.pretty_json = pretty_json;
512                }
513            }
514        }
515
516        Ok(())
517    }

557    /// Discover configuration file in standard locations
558    pub async fn discover_config_file() -> Result<Option<PathBuf>, ConfigError> {
559        // Check current directory first
560        for path in CONFIG_SEARCH_PATHS {
561            let path = PathBuf::from(path);
562            if path.exists() {
563                return Ok(Some(path));
564            }
565        }
566
567        // Check home directory
568        if let Ok(home_dir) = std_env::var("HOME").or_else(|_| std_env::var("USERPROFILE")) {
569            let home_path = PathBuf::from(home_dir);
570
571            for filename in &[".llmspell.toml", ".config/llmspell.toml"] {
572                let path = home_path.join(filename);
573                if path.exists() {
574                    return Ok(Some(path));
575                }
576            }
577        }
578
579        // Check system config directories (Linux/macOS style)
580        if let Ok(xdg_config) = std_env::var("XDG_CONFIG_HOME") {
581            let path = PathBuf::from(xdg_config)
582                .join("llmspell")
583                .join("config.toml");
584            if path.exists() {
585                return Ok(Some(path));
586            }
587        }
588
589        Ok(None)
590    }

592    /// Load configuration with automatic discovery
593    pub async fn load_with_discovery(explicit_path: Option<&Path>) -> Result<Self, ConfigError> {
594        // If explicit path provided, use it
595        if let Some(path) = explicit_path {
596            if path.exists() {
597                return Self::load_from_file(path).await;
598            } else {
599                return Err(ConfigError::NotFound {
600                    path: path.to_string_lossy().to_string(),
601                    message: "Explicitly specified config file not found".to_string(),
602                });
603            }
604        }
605
606        // Discover config file
607        if let Some(discovered_path) = Self::discover_config_file().await? {
608            return Self::load_from_file(&discovered_path).await;
609        }
610
611        // No config file found, use defaults with environment overrides
612        let mut config = Self::default();
613        config.apply_env_registry()?;
614        config.validate()?;
615
616        Ok(config)
617    }

1036/// Configuration errors
1037#[derive(Debug, thiserror::Error)]
1038pub enum ConfigError {
1039    #[error("Configuration file not found: {path} - {message}")]
1040    NotFound { path: String, message: String },
1041
1042    #[error("Configuration validation failed in field '{field:?}': {message}")]
1043    Validation {
1044        field: Option<String>,
1045        message: String,
1046    },
1047
1048    #[error("Environment variable parsing error: {message}")]
1049    Environment { message: String },
1050
1051    #[error("IO error: {0}")]
1052    Io(#[from] std::io::Error),
1053
1054    #[error("TOML parsing error: {0}")]
1055    Toml(#[from] toml::de::Error),
1056
1057    #[error("JSON serialization error: {0}")]
1058    Json(#[from] serde_json::Error),
1059
1060    #[error("General error: {0}")]
1061    Other(#[from] anyhow::Error),
1062}
1063
1064impl From<ConfigError> for LLMSpellError {
1065    fn from(err: ConfigError) -> Self {
1066        match err {
1067            ConfigError::Validation { field, message } => {
1068                LLMSpellError::Validation { field, message }
1069            }
1070            ConfigError::NotFound { path, message } => LLMSpellError::Configuration {
1071                message: format!("Configuration file not found: {} - {}", path, message),
1072                source: None,
1073            },
1074            _ => LLMSpellError::Configuration {
1075                message: err.to_string(),
1076                source: Some(Box::new(err)),
1077            },
1078        }
1079    }
1080}
