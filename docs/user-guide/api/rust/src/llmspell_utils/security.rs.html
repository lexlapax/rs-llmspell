43/// Configuration for expression complexity analysis
44#[derive(Debug, Clone)]
45pub struct ExpressionComplexityConfig {
46    /// Maximum expression length in characters
47    pub max_length: usize,
48    /// Maximum nesting depth for parentheses and functions
49    pub max_depth: usize,
50    /// Maximum number of operations allowed
51    pub max_operations: usize,
52    /// Maximum number of function calls allowed
53    pub max_functions: usize,
54    /// Maximum evaluation time allowed
55    pub max_evaluation_time: Duration,
56}
57
58impl Default for ExpressionComplexityConfig {
59    fn default() -> Self {
60        Self {
61            max_length: 1000,
62            max_depth: 20,
63            max_operations: 100,
64            max_functions: 50,
65            max_evaluation_time: Duration::from_millis(100),
66        }
67    }
68}
69
70impl ExpressionComplexityConfig {
71    /// Create a new configuration with default values
72    #[must_use]
73    pub fn new() -> Self {
74        Self::default()
75    }
76
77    /// Create a strict configuration for untrusted input
78    #[must_use]
79    pub fn strict() -> Self {
80        Self {
81            max_length: 500,
82            max_depth: 10,
83            max_operations: 50,
84            max_functions: 20,
85            max_evaluation_time: Duration::from_millis(50),
86        }
87    }
88
89    /// Create a relaxed configuration for trusted input
90    #[must_use]
91    pub fn relaxed() -> Self {
92        Self {
93            max_length: 5000,
94            max_depth: 50,
95            max_operations: 500,
96            max_functions: 200,
97            max_evaluation_time: Duration::from_millis(500),
98        }
99    }
100}

102/// Result of expression complexity analysis
103#[derive(Debug, Clone)]
104pub struct ExpressionComplexity {
105    /// Length of the expression
106    pub length: usize,
107    /// Maximum nesting depth
108    pub max_depth: usize,
109    /// Number of operations
110    pub operation_count: usize,
111    /// Number of function calls
112    pub function_count: usize,
113    /// Whether the expression is safe to evaluate
114    pub is_safe: bool,
115    /// Reason if expression is unsafe
116    pub unsafe_reason: Option<String>,
117}
118
119/// Analyzes expression complexity to prevent `DoS` attacks
120#[derive(Debug, Clone)]
121pub struct ExpressionAnalyzer {
122    config: ExpressionComplexityConfig,
123}
124
125impl ExpressionAnalyzer {
126    /// Create a new analyzer with default configuration
127    #[must_use]
128    pub fn new() -> Self {
129        Self {
130            config: ExpressionComplexityConfig::default(),
131        }
132    }
133
134    /// Create analyzer with custom configuration
135    #[must_use]
136    pub fn with_config(config: ExpressionComplexityConfig) -> Self {
137        Self { config }
138    }
139
140    /// Analyze expression complexity
141    #[must_use]
142    pub fn analyze(&self, expression: &str) -> ExpressionComplexity {
143        let length = expression.len();
144
145        // Check length first
146        if length > self.config.max_length {
147            return ExpressionComplexity {
148                length,
149                max_depth: 0,
150                operation_count: 0,
151                function_count: 0,
152                is_safe: false,
153                unsafe_reason: Some(format!(
154                    "Expression too long: {} > {} characters",
155                    length, self.config.max_length
156                )),
157            };
158        }
159
160        // Calculate nesting depth
161        let max_depth = Self::calculate_max_depth(expression);
162        if max_depth > self.config.max_depth {
163            return ExpressionComplexity {
164                length,
165                max_depth,
166                operation_count: 0,
167                function_count: 0,
168                is_safe: false,
169                unsafe_reason: Some(format!(
170                    "Nesting too deep: {} > {} levels",
171                    max_depth, self.config.max_depth
172                )),
173            };
174        }
175
176        // Count operations
177        let operation_count = Self::count_operations(expression);
178        if operation_count > self.config.max_operations {
179            return ExpressionComplexity {
180                length,
181                max_depth,
182                operation_count,
183                function_count: 0,
184                is_safe: false,
185                unsafe_reason: Some(format!(
186                    "Too many operations: {} > {}",
187                    operation_count, self.config.max_operations
188                )),
189            };
190        }
191
192        // Count function calls
193        let function_count = Self::count_functions(expression);
194        if function_count > self.config.max_functions {
195            return ExpressionComplexity {
196                length,
197                max_depth,
198                operation_count,
199                function_count,
200                is_safe: false,
201                unsafe_reason: Some(format!(
202                    "Too many function calls: {} > {}",
203                    function_count, self.config.max_functions
204                )),
205            };
206        }
207
208        // Check for potentially dangerous patterns
209        if let Some(reason) = Self::check_dangerous_patterns(expression) {
210            return ExpressionComplexity {
211                length,
212                max_depth,
213                operation_count,
214                function_count,
215                is_safe: false,
216                unsafe_reason: Some(reason),
217            };
218        }
219
220        ExpressionComplexity {
221            length,
222            max_depth,
223            operation_count,
224            function_count,
225            is_safe: true,
226            unsafe_reason: None,
227        }
228    }
229
230    /// Calculate maximum nesting depth
231    fn calculate_max_depth(expression: &str) -> usize {
232        let mut current_depth: usize = 0;
233        let mut max_depth: usize = 0;
234
235        for ch in expression.chars() {
236            match ch {
237                '(' | '[' | '{' => {
238                    current_depth += 1;
239                    max_depth = max_depth.max(current_depth);
240                }
241                ')' | ']' | '}' => {
242                    current_depth = current_depth.saturating_sub(1);
243                }
244                _ => {}
245            }
246        }
247
248        max_depth
249    }
250
251    /// Count operations in expression
252    fn count_operations(expression: &str) -> usize {
253        let operators = ['+', '-', '*', '/', '%', '^', '&', '|', '!', '<', '>', '='];
254        let mut count = 0;
255        let mut prev_was_operator = false;
256
257        for ch in expression.chars() {
258            if operators.contains(&ch) {
259                if !prev_was_operator {
260                    count += 1;
261                }
262                prev_was_operator = true;
263            } else if !ch.is_whitespace() {
264                prev_was_operator = false;
265            }
266        }
267
268        count
269    }
270
271    /// Count function calls in expression
272    fn count_functions(expression: &str) -> usize {
273        // Common mathematical functions
274        let functions = [
275            "sin", "cos", "tan", "asin", "acos", "atan", "sinh", "cosh", "tanh", "asinh", "acosh",
276            "atanh", "sqrt", "exp", "ln", "log", "abs", "sign", "int", "ceil", "floor", "round",
277            "min", "max", "pi", "e",
278        ];
279
280        let mut count = 0;
281        for func in &functions {
282            count += expression.matches(func).count();
283        }
284
285        count
286    }
287
288    /// Check for potentially dangerous patterns
289    fn check_dangerous_patterns(expression: &str) -> Option<String> {
290        // Check for excessive consecutive operations
291        if expression.contains("+++")
292            || expression.contains("---")
293            || expression.contains("***")
294            || expression.contains("///")
295        {
296            return Some("Excessive consecutive operations detected".to_string());
297        }
298
299        // Check for potential stack overflow patterns
300        let repeated_opens = expression.matches("((((").count();
301        if repeated_opens > 0 {
302            return Some("Potential stack overflow pattern detected".to_string());
303        }
304
305        // Check for extremely large numbers that could cause issues
306        if let Some(captures) = regex::Regex::new(r"\d{10,}").unwrap().captures(expression) {
307            if let Some(num_str) = captures.get(0) {
308                return Some(format!(
309                    "Extremely large number detected: {}",
310                    num_str.as_str()
311                ));
312            }
313        }
314
315        None
316    }
317
318    /// Get the configured maximum evaluation time
319    #[must_use]
320    pub fn max_evaluation_time(&self) -> Duration {
321        self.config.max_evaluation_time
322    }
323}

336        let analyzer = ExpressionAnalyzer::with_config(ExpressionComplexityConfig {
