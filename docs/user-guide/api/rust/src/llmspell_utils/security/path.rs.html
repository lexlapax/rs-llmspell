4//! Path security utilities
5//!
6//! This module provides enhanced path security validation to prevent:
7//! - Path traversal attacks
8//! - Symlink attacks
9//! - Directory escape attempts
10//! - Hidden file access (optional)
11//! - System directory access
12
13use llmspell_core::{LLMSpellError, Result as LLMResult};
14use std::path::{Path, PathBuf};

16/// Configuration for path security validation
17#[derive(Debug, Clone)]
18#[allow(clippy::struct_excessive_bools)]
19pub struct PathSecurityConfig {
20    /// Allow access to hidden files (starting with .)
21    pub allow_hidden: bool,
22    /// Allow following symlinks
23    pub allow_symlinks: bool,
24    /// Jail directory to restrict access
25    pub jail_directory: Option<PathBuf>,
26    /// Disallowed path prefixes (e.g., /etc, /sys)
27    pub disallowed_prefixes: Vec<PathBuf>,
28    /// Maximum path depth allowed
29    pub max_depth: Option<usize>,
30    /// Enhanced symlink detection (recursive checking)
31    pub enhanced_symlink_detection: bool,
32    /// Enable chroot jail enforcement
33    pub enable_chroot: bool,
34    /// Check permission inheritance from parent directories
35    pub check_permission_inheritance: bool,
36    /// Maximum number of symlinks to follow in a path
37    pub max_symlinks: usize,
38    /// Enable cross-platform path validation
39    pub cross_platform_validation: bool,
40}

123impl PathSecurityValidator {

128            config: PathSecurityConfig::default(),

138    /// Validate a path for security concerns
139    ///
140    /// # Errors
141    ///
142    /// Returns `LLMSpellError::Validation` if the path fails security checks
143    pub fn validate(&self, path: &Path) -> LLMResult<PathBuf> {
144        // First, convert to absolute path if relative
145        let abs_path = if path.is_absolute() {
146            path.to_path_buf()
147        } else {
148            std::env::current_dir()
149                .map_err(|e| LLMSpellError::Validation {
150                    message: format!("Failed to get current directory: {e}"),
151                    field: Some("path".to_string()),
152                })?
153                .join(path)
154        };
155
156        // Check for path traversal patterns BEFORE normalization
157        Self::check_path_traversal(&abs_path)?;
158
159        // Normalize the path to resolve . and .. components
160        let normalized = Self::normalize_path_internal(&abs_path)?;
161
162        // Check depth
163        self.check_path_depth(&normalized)?;
164
165        // Check for hidden files
166        self.check_hidden_files(&normalized)?;
167
168        // Check against disallowed prefixes
169        self.check_disallowed_prefixes(&normalized)?;
170
171        // Check jail directory
172        self.check_jail_directory(&normalized)?;
173
174        // Check symlinks (enhanced if enabled)
175        if self.config.enhanced_symlink_detection {
176            self.check_symlinks_enhanced(&normalized)?;
177        } else {
178            self.check_symlinks(&normalized)?;
179        }
180
181        // Check cross-platform compatibility
182        if self.config.cross_platform_validation {
183            self.check_cross_platform(&normalized)?;
184        }
185
186        // Check permission inheritance
187        if self.config.check_permission_inheritance {
188            self.check_permission_inheritance(&normalized)?;
189        }
190
191        // Apply chroot jail if enabled
192        let final_path = if self.config.enable_chroot {
193            self.apply_chroot_jail(&normalized)?
194        } else {
195            normalized
196        };
197
198        Ok(final_path)
199    }
200
201    /// Normalize a path, resolving . and .. components
202    fn normalize_path_internal(path: &Path) -> LLMResult<PathBuf> {
203        use crate::file_utils::normalize_path;
204
205        let normalized = normalize_path(path);
206
207        // Additional check: ensure the normalized path doesn't contain any .. components
208        if normalized
209            .components()
210            .any(|c| matches!(c, std::path::Component::ParentDir))
211        {
212            return Err(LLMSpellError::Validation {
213                message: "Path normalization failed to resolve all .. components".to_string(),
214                field: Some("path".to_string()),
215            });
216        }
217
218        Ok(normalized)
219    }
220
221    /// Check for path traversal attempts
222    fn check_path_traversal(path: &Path) -> LLMResult<()> {
223        let path_str = path.to_string_lossy();
224
225        // Check for various path traversal patterns
226        let dangerous_patterns = [
227            "..",
228            "..\\",
229            "../",
230            "%2e%2e",
231            "%252e%252e",
232            "..%2f",
233            "..%5c",
234            "..%252f",
235            "..%255c",
236        ];
237
238        for pattern in &dangerous_patterns {
239            if path_str.to_lowercase().contains(pattern) {
240                return Err(LLMSpellError::Validation {
241                    message: format!("Path contains dangerous pattern: {pattern}"),
242                    field: Some("path".to_string()),
243                });
244            }
245        }
246
247        Ok(())
248    }
249
250    /// Check path depth
251    fn check_path_depth(&self, path: &Path) -> LLMResult<()> {
252        if let Some(max_depth) = self.config.max_depth {
253            let depth = path.components().count();
254            if depth > max_depth {
255                return Err(LLMSpellError::Validation {
256                    message: format!("Path depth {depth} exceeds maximum allowed {max_depth}"),
257                    field: Some("path".to_string()),
258                });
259            }
260        }
261        Ok(())
262    }
263
264    /// Check for hidden files
265    fn check_hidden_files(&self, path: &Path) -> LLMResult<()> {
266        if !self.config.allow_hidden {
267            for component in path.components() {
268                if let std::path::Component::Normal(name) = component {
269                    if let Some(name_str) = name.to_str() {
270                        if name_str.starts_with('.') && name_str != "." && name_str != ".." {
271                            return Err(LLMSpellError::Validation {
272                                message: "Access to hidden files is not allowed".to_string(),
273                                field: Some("path".to_string()),
274                            });
275                        }
276                    }
277                }
278            }
279        }
280        Ok(())
281    }
282
283    /// Check against disallowed prefixes
284    fn check_disallowed_prefixes(&self, path: &Path) -> LLMResult<()> {
285        for prefix in &self.config.disallowed_prefixes {
286            if path.starts_with(prefix) {
287                return Err(LLMSpellError::Validation {
288                    message: format!("Access to {} is not allowed", prefix.display()),
289                    field: Some("path".to_string()),
290                });
291            }
292        }
293        Ok(())
294    }
295
296    /// Check jail directory constraint
297    fn check_jail_directory(&self, path: &Path) -> LLMResult<()> {
298        if let Some(jail) = &self.config.jail_directory {
299            let jail_normalized = Self::normalize_path_internal(jail)?;
300            if !path.starts_with(&jail_normalized) {
301                return Err(LLMSpellError::Validation {
302                    message: format!(
303                        "Path '{}' is outside the allowed directory '{}'",
304                        path.display(),
305                        jail.display()
306                    ),
307                    field: Some("path".to_string()),
308                });
309            }
310        }
311        Ok(())
312    }
313
314    /// Check for symlinks
315    fn check_symlinks(&self, path: &Path) -> LLMResult<()> {
316        if !self.config.allow_symlinks {
317            // Check each component of the path for symlinks
318            let mut current = PathBuf::new();
319            for component in path.components() {
320                current.push(component);
321                if current.exists() && current.read_link().is_ok() {
322                    return Err(LLMSpellError::Validation {
323                        message: format!(
324                            "Symlink detected at '{}', symlinks are not allowed",
325                            current.display()
326                        ),
327                        field: Some("path".to_string()),
328                    });
329                }
330            }
331        }
332        Ok(())
333    }
334
335    /// Enhanced symlink detection with recursive checking
336    fn check_symlinks_enhanced(&self, path: &Path) -> LLMResult<()> {
337        if !self.config.allow_symlinks {
338            return self.check_symlinks(path);
339        }
340
341        let mut symlink_count = 0;
342        let mut current = PathBuf::new();
343
344        for component in path.components() {
345            current.push(component);
346
347            if current.exists() {
348                if let Ok(target) = current.read_link() {
349                    symlink_count += 1;
350
351                    if symlink_count > self.config.max_symlinks {
352                        return Err(LLMSpellError::Validation {
353                            message: format!(
354                                "Too many symlinks in path: {} > {}",
355                                symlink_count, self.config.max_symlinks
356                            ),
357                            field: Some("path".to_string()),
358                        });
359                    }
360
361                    // Check if the symlink target is safe
362                    let target_path = if target.is_absolute() {
363                        target
364                    } else {
365                        current.parent().unwrap_or(Path::new("/")).join(target)
366                    };
367
368                    // Recursively check the symlink target
369                    self.check_symlinks_enhanced(&target_path)?;
370
371                    // Check for symlink loops
372                    if target_path == current {
373                        return Err(LLMSpellError::Validation {
374                            message: "Symlink loop detected".to_string(),
375                            field: Some("path".to_string()),
376                        });
377                    }
378                }
379            }
380        }
381
382        Ok(())
383    }

385    /// Apply chroot jail constraints
386    fn apply_chroot_jail(&self, path: &Path) -> LLMResult<PathBuf> {
387        if let Some(jail) = &self.config.jail_directory {
388            let jail_normalized = Self::normalize_path_internal(jail)?;
389
390            // Ensure the path is within the jail
391            if !path.starts_with(&jail_normalized) {
392                return Err(LLMSpellError::Validation {
393                    message: format!(
394                        "Path '{}' is outside the chroot jail '{}'",
395                        path.display(),
396                        jail.display()
397                    ),
398                    field: Some("path".to_string()),
399                });
400            }
401
402            // Return the path relative to the jail root
403            let relative_path =
404                path.strip_prefix(&jail_normalized)
405                    .map_err(|e| LLMSpellError::Validation {
406                        message: format!("Failed to create relative path: {e}"),
407                        field: Some("path".to_string()),
408                    })?;
409
410            Ok(PathBuf::from("/").join(relative_path))
411        } else {
412            Ok(path.to_path_buf())
413        }
414    }

416    /// Check permission inheritance from parent directories
417    fn check_permission_inheritance(&self, path: &Path) -> LLMResult<()> {
418        use std::fs;
419
420        // Early return if permission inheritance checking is disabled
421        if !self.config.check_permission_inheritance {
422            return Ok(());
423        }
424
425        let mut current = path.to_path_buf();
426
427        // Check permissions for each parent directory
428        while let Some(parent) = current.parent() {
429            if parent.exists() {
430                match fs::metadata(parent) {
431                    Ok(metadata) => {
432                        // Check if parent directory is readable
433                        #[cfg(unix)]
434                        {
435                            use std::os::unix::fs::PermissionsExt;
436                            let mode = metadata.permissions().mode();
437
438                            // Check if parent directory has read permission for owner
439                            if mode & 0o400 == 0 {
440                                return Err(LLMSpellError::Validation {
441                                    message: format!(
442                                        "Parent directory '{}' is not readable",
443                                        parent.display()
444                                    ),
445                                    field: Some("path".to_string()),
446                                });
447                            }
448                        }
449
450                        // Check if parent directory is not writable by others (security risk)
451                        #[cfg(unix)]
452                        {
453                            use std::os::unix::fs::PermissionsExt;
454                            let mode = metadata.permissions().mode();
455
456                            // Check if parent directory is writable by others
457                            if mode & 0o002 != 0 {
458                                return Err(LLMSpellError::Validation {
459                                    message: format!(
460                                        "Parent directory '{}' is writable by others (security risk)",
461                                        parent.display()
462                                    ),
463                                    field: Some("path".to_string()),
464                                });
465                            }
466                        }
467                    }
468                    Err(e) => {
469                        return Err(LLMSpellError::Validation {
470                            message: format!(
471                                "Failed to check permissions for parent directory '{}': {}",
472                                parent.display(),
473                                e
474                            ),
475                            field: Some("path".to_string()),
476                        });
477                    }
478                }
479            }
480
481            current = parent.to_path_buf();
482
483            // Don't check beyond the root directory
484            if current == Path::new("/") {
485                break;
486            }
487        }
488
489        Ok(())
490    }

492    /// Check cross-platform path compatibility
493    fn check_cross_platform(&self, path: &Path) -> LLMResult<()> {
494        // Early return if cross-platform validation is disabled
495        if !self.config.cross_platform_validation {
496            return Ok(());
497        }
498        let path_str = path.to_string_lossy();
499
500        // Check for Windows-specific issues
501        let windows_invalid_chars = ['<', '>', ':', '"', '|', '?', '*'];
502        for &ch in &windows_invalid_chars {
503            if path_str.contains(ch) {
504                return Err(LLMSpellError::Validation {
505                    message: format!("Path contains Windows-invalid character: '{ch}'"),
506                    field: Some("path".to_string()),
507                });
508            }
509        }
510
511        // Check for reserved Windows names
512        let reserved_names = [
513            "CON", "PRN", "AUX", "NUL", "COM1", "COM2", "COM3", "COM4", "COM5", "COM6", "COM7",
514            "COM8", "COM9", "LPT1", "LPT2", "LPT3", "LPT4", "LPT5", "LPT6", "LPT7", "LPT8", "LPT9",
515        ];
516
517        for component in path.components() {
518            if let std::path::Component::Normal(name) = component {
519                if let Some(name_str) = name.to_str() {
520                    let name_upper = name_str.to_uppercase();
521                    if reserved_names.contains(&name_upper.as_str()) {
522                        return Err(LLMSpellError::Validation {
523                            message: format!("Path contains reserved Windows name: '{name_str}'"),
524                            field: Some("path".to_string()),
525                        });
526                    }
527
528                    // Check for names ending with spaces or dots (Windows issue)
529                    if name_str.ends_with(' ') || name_str.ends_with('.') {
530                        return Err(LLMSpellError::Validation {
531                            message: format!(
532                                "Path component '{name_str}' ends with space or dot (Windows incompatible)"
533                            ),
534                            field: Some("path".to_string()),
535                        });
536                    }
537                }
538            }
539        }
540
541        // Check path length limits
542        if path_str.len() > 260 {
543            return Err(LLMSpellError::Validation {
544                message: format!(
545                    "Path length {} exceeds Windows limit of 260 characters",
546                    path_str.len()
547                ),
548                field: Some("path".to_string()),
549            });
550        }
551
552        Ok(())
553    }
