69/// File validation result
70#[derive(Debug, Clone)]
71pub struct FileValidationResult {
72    /// Whether the file is valid
73    pub is_valid: bool,
74    /// Detected file type
75    pub file_type: Option<String>,
76    /// Detected MIME type
77    pub mime_type: Option<String>,
78    /// File size in bytes
79    pub size: u64,
80    /// Validation errors
81    pub errors: Vec<FileValidationError>,
82    /// Security warnings
83    pub warnings: Vec<String>,
84}

86/// File validation errors
87#[derive(Debug, Clone, PartialEq)]
88pub enum FileValidationError {
89    /// File size exceeds limit
90    FileTooLarge {
91        /// Actual file size in bytes
92        size: u64,
93        /// Maximum allowed size in bytes
94        limit: u64,
95    },
96    /// Invalid file extension
97    InvalidExtension {
98        /// The invalid extension
99        extension: String,
100    },
101    /// Invalid MIME type
102    InvalidMimeType {
103        /// The invalid MIME type
104        mime_type: String,
105    },
106    /// Magic number mismatch
107    MagicNumberMismatch {
108        /// Expected magic number
109        expected: String,
110        /// Found magic number
111        found: String,
112    },
113    /// Malicious content detected
114    MaliciousContent {
115        /// Reason for detection
116        reason: String,
117    },
118    /// Invalid filename
119    InvalidFilename {
120        /// Reason for invalidity
121        reason: String,
122    },
123    /// File not found
124    FileNotFound {
125        /// Path to the missing file
126        path: String,
127    },
128    /// IO error
129    IoError {
130        /// Error message
131        message: String,
132    },
133}
134
135impl std::fmt::Display for FileValidationError {
136    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
137        match self {
138            Self::FileTooLarge { size, limit } => {
139                write!(f, "File too large: {size} bytes (limit: {limit} bytes)")
140            }
141            Self::InvalidExtension { extension } => {
142                write!(f, "Invalid file extension: {extension}")
143            }
144            Self::InvalidMimeType { mime_type } => {
145                write!(f, "Invalid MIME type: {mime_type}")
146            }
147            Self::MagicNumberMismatch { expected, found } => {
148                write!(
149                    f,
150                    "Magic number mismatch: expected {expected}, found {found}"
151                )
152            }
153            Self::MaliciousContent { reason } => {
154                write!(f, "Malicious content detected: {reason}")
155            }
156            Self::InvalidFilename { reason } => {
157                write!(f, "Invalid filename: {reason}")
158            }
159            Self::FileNotFound { path } => {
160                write!(f, "File not found: {path}")
161            }
162            Self::IoError { message } => {
163                write!(f, "IO error: {message}")
164            }
165        }
166    }
167}

174impl MagicNumbers {
175    fn get_magic_numbers() -> HashMap<&'static str, Vec<u8>> {
176        let mut magic = HashMap::new();
177
178        // Images
179        magic.insert("png", vec![0x89, 0x50, 0x4E, 0x47, 0x0D, 0x0A, 0x1A, 0x0A]);
180        magic.insert("jpg", vec![0xFF, 0xD8, 0xFF]);
181        magic.insert("jpeg", vec![0xFF, 0xD8, 0xFF]);
182        magic.insert("gif87", vec![0x47, 0x49, 0x46, 0x38, 0x37, 0x61]);
183        magic.insert("gif89", vec![0x47, 0x49, 0x46, 0x38, 0x39, 0x61]);
184
185        // Documents
186        magic.insert("pdf", vec![0x25, 0x50, 0x44, 0x46]); // %PDF
187        magic.insert("doc", vec![0xD0, 0xCF, 0x11, 0xE0, 0xA1, 0xB1, 0x1A, 0xE1]);
188        magic.insert("docx", vec![0x50, 0x4B, 0x03, 0x04]); // ZIP format
189
190        // Archives
191        magic.insert("zip", vec![0x50, 0x4B, 0x03, 0x04]);
192        magic.insert("rar", vec![0x52, 0x61, 0x72, 0x21]);
193        magic.insert("7z", vec![0x37, 0x7A, 0xBC, 0xAF, 0x27, 0x1C]);
194
195        // Executables (dangerous)
196        magic.insert("exe", vec![0x4D, 0x5A]); // MZ
197        magic.insert("elf", vec![0x7F, 0x45, 0x4C, 0x46]); // ELF
198        magic.insert("mach-o", vec![0xFE, 0xED, 0xFA, 0xCE]);
199
200        // Scripts (potentially dangerous)
201        magic.insert("script_shebang", vec![0x23, 0x21]); // #!
202
203        // Audio/Video
204        magic.insert("mp3_id3", vec![0x49, 0x44, 0x33]); // ID3
205        magic.insert("mp3_sync", vec![0xFF, 0xFB]);
206        magic.insert("mp4", vec![0x00, 0x00, 0x00, 0x20, 0x66, 0x74, 0x79, 0x70]);
207
208        magic
209    }

230    /// Validate a file upload
231    ///
232    /// # Errors
233    ///
234    /// Returns validation errors if the file fails any security checks
235    pub fn validate_file(
236        &self,
237        file_path: &Path,
238    ) -> Result<FileValidationResult, FileValidationError> {
239        let mut result = FileValidationResult {
240            is_valid: true,
241            file_type: None,
242            mime_type: None,
243            size: 0,
244            errors: Vec::new(),
245            warnings: Vec::new(),
246        };
247
248        // Check if file exists
249        if !file_path.exists() {
250            return Err(FileValidationError::FileNotFound {
251                path: file_path.display().to_string(),
252            });
253        }
254
255        // Get file metadata
256        let metadata = std::fs::metadata(file_path).map_err(|e| FileValidationError::IoError {
257            message: e.to_string(),
258        })?;
259
260        result.size = metadata.len();
261
262        // Check file size
263        if result.size > self.config.max_file_size {
264            result.errors.push(FileValidationError::FileTooLarge {
265                size: result.size,
266                limit: self.config.max_file_size,
267            });
268            result.is_valid = false;
269        }
270
271        // Validate filename
272        if let Some(filename) = file_path.file_name() {
273            if let Some(filename_str) = filename.to_str() {
274                self.validate_filename(filename_str, &mut result);
275            }
276        }
277
278        // Check file extension
279        if let Some(extension) = file_path.extension() {
280            if let Some(ext_str) = extension.to_str() {
281                if !self
282                    .config
283                    .allowed_extensions
284                    .contains(&ext_str.to_lowercase())
285                {
286                    result.errors.push(FileValidationError::InvalidExtension {
287                        extension: ext_str.to_string(),
288                    });
289                    result.is_valid = false;
290                }
291                result.file_type = Some(ext_str.to_lowercase());
292            }
293        }
294
295        // Verify magic numbers
296        if self.config.verify_magic_numbers {
297            self.verify_magic_numbers(file_path, &mut result)?;
298        }
299
300        // Scan content for malicious patterns
301        if self.config.scan_content {
302            self.scan_content(file_path, &mut result);
303        }
304
305        Ok(result)
306    }

308    /// Validate filename for security issues
309    fn validate_filename(&self, filename: &str, result: &mut FileValidationResult) {
310        // Check filename length
311        if filename.len() > self.config.max_filename_length {
312            result.errors.push(FileValidationError::InvalidFilename {
313                reason: format!("Filename too long: {} characters", filename.len()),
314            });
315            result.is_valid = false;
316        }
317
318        // Check for directory traversal attempts
319        if filename.contains("..") || filename.contains('/') || filename.contains('\\') {
320            result.errors.push(FileValidationError::InvalidFilename {
321                reason: "Filename contains directory traversal characters".to_string(),
322            });
323            result.is_valid = false;
324        }
325
326        // Check for null bytes
327        if filename.contains('\0') {
328            result.errors.push(FileValidationError::InvalidFilename {
329                reason: "Filename contains null bytes".to_string(),
330            });
331            result.is_valid = false;
332        }
333
334        // Check for control characters
335        if filename.chars().any(char::is_control) {
336            result.errors.push(FileValidationError::InvalidFilename {
337                reason: "Filename contains control characters".to_string(),
338            });
339            result.is_valid = false;
340        }
341
342        // Warn about special characters
343        if filename
344            .chars()
345            .any(|c| matches!(c, '<' | '>' | ':' | '"' | '|' | '?' | '*'))
346        {
347            result
348                .warnings
349                .push("Filename contains special characters that may cause issues".to_string());
350        }
351    }

353    /// Verify file magic numbers
354    #[allow(clippy::unused_self)]
355    fn verify_magic_numbers(
356        &self,
357        file_path: &Path,
358        result: &mut FileValidationResult,
359    ) -> Result<(), FileValidationError> {
360        let mut file =
361            std::fs::File::open(file_path).map_err(|e| FileValidationError::IoError {
362                message: e.to_string(),
363            })?;
364
365        let mut buffer = vec![0u8; 16]; // Read first 16 bytes
366        let bytes_read = file
367            .read(&mut buffer)
368            .map_err(|e| FileValidationError::IoError {
369                message: e.to_string(),
370            })?;
371
372        if bytes_read == 0 {
373            result.warnings.push("Empty file".to_string());
374            return Ok(());
375        }
376
377        buffer.truncate(bytes_read);
378
379        let magic_numbers = MagicNumbers::get_magic_numbers();
380        let mut found_match = false;
381
382        // Check against known magic numbers
383        for (file_type, magic) in &magic_numbers {
384            if buffer.starts_with(magic) {
385                // Check for dangerous file types
386                if matches!(*file_type, "exe" | "elf" | "mach-o" | "script_shebang") {
387                    result.errors.push(FileValidationError::MaliciousContent {
388                        reason: format!("Executable file type detected: {file_type}"),
389                    });
390                    result.is_valid = false;
391                }
392
393                // Verify extension matches magic number
394                if let Some(ref ext) = result.file_type {
395                    if !file_type.starts_with(ext) && ext != "txt" {
396                        result.warnings.push(format!(
397                            "File extension '{ext}' doesn't match detected type '{file_type}'"
398                        ));
399                    }
400                }
401
402                found_match = true;
403                break;
404            }
405        }
406
407        if !found_match && result.file_type.is_some() {
408            // For text files, we don't require magic numbers
409            let text_extensions = ["txt", "csv", "json", "xml", "log"];
410            if !text_extensions.contains(&result.file_type.as_ref().unwrap().as_str()) {
411                result
412                    .warnings
413                    .push("Unable to verify file type from magic numbers".to_string());
414            }
415        }
416
417        Ok(())
418    }

420    /// Scan file content for malicious patterns
421    #[allow(clippy::unused_self)]
422    fn scan_content(&self, file_path: &Path, result: &mut FileValidationResult) {
423        let content = std::fs::read_to_string(file_path).unwrap_or_default();
424
425        // Check for suspicious patterns in text files
426        if !content.is_empty() {
427            // Check for script injections
428            let script_patterns = [
429                "<script",
430                "javascript:",
431                "onerror=",
432                "onclick=",
433                "onload=",
434                "eval(",
435                "document.write",
436                "innerHTML",
437            ];
438
439            for pattern in &script_patterns {
440                if content.to_lowercase().contains(pattern) {
441                    result
442                        .warnings
443                        .push(format!("Suspicious pattern found: {pattern}"));
444                }
445            }
446
447            // Check for PHP code
448            if content.contains("<?php") || content.contains("<?=") {
449                result.errors.push(FileValidationError::MaliciousContent {
450                    reason: "PHP code detected".to_string(),
451                });
452                result.is_valid = false;
453            }
454
455            // Check for shell commands
456            let shell_patterns = ["rm -rf", "chmod", "chown", "/bin/sh", "/bin/bash"];
457            for pattern in &shell_patterns {
458                if content.contains(pattern) {
459                    result
460                        .warnings
461                        .push(format!("Shell command pattern found: {pattern}"));
462                }
463            }
464        }
465    }

467    /// Sanitize a filename for safe storage
468    #[must_use]
469    pub fn sanitize_filename(&self, filename: &str) -> String {
470        let mut sanitized = filename.to_string();
471
472        // Remove directory traversal attempts
473        sanitized = sanitized.replace("..", "");
474        sanitized = sanitized.replace('/', "_");
475        sanitized = sanitized.replace('\\', "_");
476
477        // Remove control characters and null bytes
478        sanitized = sanitized
479            .chars()
480            .filter(|c| !c.is_control() && *c != '\0')
481            .collect();
482
483        // Replace special characters
484        for c in ['<', '>', ':', '"', '|', '?', '*'] {
485            sanitized = sanitized.replace(c, "_");
486        }
487
488        // Truncate if too long
489        if sanitized.len() > self.config.max_filename_length {
490            let extension = Path::new(&sanitized)
491                .extension()
492                .and_then(|ext| ext.to_str())
493                .unwrap_or("");
494
495            let base_len = self.config.max_filename_length - extension.len() - 1;
496            sanitized = format!(
497                "{}.{}",
498                &sanitized[..base_len.min(sanitized.len())],
499                extension
500            );
501        }
502
503        // Ensure filename is not empty
504        if sanitized.trim().is_empty() {
505            sanitized = "unnamed_file".to_string();
506        }
507
508        sanitized
509    }

511    /// Get safe file storage path
512    #[must_use]
513    pub fn get_safe_storage_path(&self, base_dir: &Path, filename: &str) -> PathBuf {
514        let sanitized_filename = self.sanitize_filename(filename);
515
516        // Add timestamp to prevent collisions
517        let timestamp = chrono::Utc::now().timestamp();
518        let unique_filename = format!("{timestamp}_{sanitized_filename}");
519
520        base_dir.join(unique_filename)
521    }

524/// File processing sandbox for isolating file operations
525pub struct FileProcessingSandbox {
526    /// Temporary directory for processing
527    temp_dir: PathBuf,
528    /// Maximum processing time
529    #[allow(dead_code)]
530    max_processing_time: std::time::Duration,
531    /// Memory limit for processing
532    #[allow(dead_code)]
533    memory_limit: Option<u64>,
534}
535
536impl FileProcessingSandbox {
537    /// Create a new file processing sandbox
538    ///
539    /// # Errors
540    ///
541    /// Returns an error if the temporary directory cannot be created
542    pub fn new() -> Result<Self, std::io::Error> {
543        let temp_dir =
544            std::env::temp_dir().join(format!("llmspell_sandbox_{}", uuid::Uuid::new_v4()));
545        std::fs::create_dir_all(&temp_dir)?;
546
547        Ok(Self {
548            temp_dir,
549            max_processing_time: std::time::Duration::from_secs(300), // 5 minutes
550            memory_limit: Some(1024 * 1024 * 1024),                   // 1GB
551        })
552    }
553
554    /// Get the sandbox directory
555    #[must_use]
556    pub fn sandbox_dir(&self) -> &Path {
557        &self.temp_dir
558    }
559
560    /// Copy file to sandbox for processing
561    ///
562    /// # Errors
563    ///
564    /// Returns an error if the file cannot be copied
565    pub fn copy_to_sandbox(&self, source: &Path) -> Result<PathBuf, std::io::Error> {
566        let filename = source.file_name().ok_or_else(|| {
567            std::io::Error::new(std::io::ErrorKind::InvalidInput, "Invalid filename")
568        })?;
569
570        let dest = self.temp_dir.join(filename);
571        std::fs::copy(source, &dest)?;
572
573        Ok(dest)
574    }
575
576    /// Clean up sandbox directory
577    ///
578    /// # Errors
579    ///
580    /// Returns an error if the directory cannot be removed
581    pub fn cleanup(&self) -> Result<(), std::io::Error> {
582        if self.temp_dir.exists() {
583            std::fs::remove_dir_all(&self.temp_dir)?;
584        }
585        Ok(())
586    }
587}
588
589impl Drop for FileProcessingSandbox {
590    fn drop(&mut self) {
591        let _ = self.cleanup();
592    }
