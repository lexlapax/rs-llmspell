40/// Ensure a directory exists, creating it if necessary
41///
42/// This function creates the directory and all parent directories if they don't exist.
43/// It's safe to call even if the directory already exists.
44///
45/// # Examples
46///
47/// ```rust,no_run
48/// use llmspell_utils::file_utils::ensure_dir;
49/// use std::path::Path;
50///
51/// # fn main() -> std::io::Result<()> {
52/// ensure_dir(Path::new("/tmp/myapp/data"))?;
53/// # Ok(())
54/// # }
55/// ```
56///
57/// # Errors
58///
59/// Returns an error if:
60/// - The path exists but is not a directory
61/// - Directory creation fails due to permissions or other OS errors
62pub fn ensure_dir(path: &Path) -> io::Result<()> {
63    if path.exists() {
64        if path.is_dir() {
65            Ok(())
66        } else {
67            Err(io::Error::new(
68                io::ErrorKind::AlreadyExists,
69                format!("Path exists but is not a directory: {}", path.display()),
70            ))
71        }
72    } else {
73        fs::create_dir_all(path).map_err(|e| {
74            io::Error::new(
75                e.kind(),
76                format!("Failed to create directory '{}': {}", path.display(), e),
77            )
78        })
79    }
80}

82/// Expand path with environment variables and tilde expansion
83///
84/// Supports:
85/// - Environment variable expansion: `$HOME`, `${HOME}`, `%HOME%` (Windows)
86/// - Tilde expansion: `~` expands to home directory
87/// - Relative path resolution to absolute paths
88///
89/// # Examples
90///
91/// ```rust,no_run
92/// use llmspell_utils::file_utils::expand_path;
93///
94/// # fn main() -> Result<(), std::io::Error> {
95/// let path = expand_path("~/Documents/config.json")?;
96/// let path2 = expand_path("$HOME/.config/app")?;
97/// let path3 = expand_path("${TMPDIR}/cache")?;
98/// # Ok(())
99/// # }
100/// ```
101///
102/// # Errors
103///
104/// Returns an error if:
105/// - Home directory cannot be determined
106/// - Environment variable is not found
107/// - Path contains invalid UTF-8
108pub fn expand_path(path: &str) -> Result<PathBuf, io::Error> {
109    let mut expanded = path.to_string();
110
111    // Handle tilde expansion
112    if expanded.starts_with("~/") || expanded == "~" {
113        let home = env::var("HOME")
114            .or_else(|_| env::var("USERPROFILE"))
115            .map_err(|_| {
116                io::Error::new(
117                    io::ErrorKind::NotFound,
118                    "Could not determine home directory",
119                )
120            })?;
121
122        expanded = if expanded == "~" {
123            home
124        } else {
125            format!("{}{}", home, &expanded[1..])
126        };
127    }
128
129    // Handle environment variable expansion
130    // Support both $VAR and ${VAR} syntax
131    let mut result = String::new();
132    let mut chars = expanded.chars();
133
134    while let Some(ch) = chars.next() {
135        if ch == '$' {
136            let mut var_name = String::new();
137
138            // Check if it's ${VAR} syntax
139            let next_char = chars.clone().next();
140            if next_char == Some('{') {
141                chars.next(); // consume '{'
142                              // Collect until we find '}'
143                for c in chars.by_ref() {
144                    if c == '}' {
145                        break;
146                    }
147                    var_name.push(c);
148                }
149            } else {
150                // $VAR syntax - collect alphanumeric and underscore chars
151                for c in chars.clone() {
152                    if c.is_alphanumeric() || c == '_' {
153                        var_name.push(c);
154                        chars.next(); // consume the character
155                    } else {
156                        break;
157                    }
158                }
159            }
160
161            if var_name.is_empty() {
162                result.push('$');
163            } else {
164                match env::var(&var_name) {
165                    Ok(value) => result.push_str(&value),
166                    Err(_) => {
167                        return Err(io::Error::new(
168                            io::ErrorKind::NotFound,
169                            format!("Environment variable '{var_name}' not found"),
170                        ));
171                    }
172                }
173            }
174        } else if cfg!(windows) && ch == '%' {
175            // Windows %VAR% syntax
176            let mut var_name = String::new();
177            for c in chars.by_ref() {
178                if c == '%' {
179                    break;
180                }
181                var_name.push(c);
182            }
183
184            if var_name.is_empty() {
185                result.push('%');
186            } else {
187                match env::var(&var_name) {
188                    Ok(value) => result.push_str(&value),
189                    Err(_) => {
190                        return Err(io::Error::new(
191                            io::ErrorKind::NotFound,
192                            format!("Environment variable '{var_name}' not found"),
193                        ));
194                    }
195                }
196            }
197        } else {
198            result.push(ch);
199        }
200    }
201
202    // Convert to PathBuf and clean the path
203    Ok(PathBuf::from(result).clean())
204}

228/// Safely read a file with size limits
229///
230/// Reads the entire contents of a file into memory, with a size limit
231/// to prevent memory exhaustion.
232///
233/// # Examples
234///
235/// ```rust,no_run
236/// use llmspell_utils::file_utils::read_file;
237/// use std::path::Path;
238///
239/// # fn main() -> Result<(), Box<dyn std::error::Error>> {
240/// let content = read_file(Path::new("/etc/hosts"))?;
241/// println!("File content: {}", String::from_utf8_lossy(&content));
242/// # Ok(())
243/// # }
244/// ```
245///
246/// # Errors
247///
248/// Returns an error if:
249/// - File does not exist
250/// - File is larger than 100MB
251/// - Read permissions are denied
252/// - I/O error occurs
253pub fn read_file(path: &Path) -> Result<Vec<u8>> {
254    // Check file size first
255    let metadata = fs::metadata(path)
256        .with_context(|| format!("Failed to read metadata for file: {}", path.display()))?;
257
258    if metadata.len() > MAX_FILE_SIZE {
259        anyhow::bail!(
260            "File '{}' is too large ({} bytes, max {} bytes)",
261            path.display(),
262            metadata.len(),
263            MAX_FILE_SIZE
264        );
265    }
266
267    let mut file =
268        File::open(path).with_context(|| format!("Failed to open file: {}", path.display()))?;
269
270    // Safe cast: we've already checked the file size is <= MAX_FILE_SIZE (100MB)
271    #[allow(clippy::cast_possible_truncation)]
272    let capacity = metadata.len().min(usize::MAX as u64) as usize;
273    let mut contents = Vec::with_capacity(capacity);
274    file.read_to_end(&mut contents)
275        .with_context(|| format!("Failed to read file: {}", path.display()))?;
276
277    Ok(contents)
278}

280/// Write data to a file with proper error handling
281///
282/// This is a simple write operation. For critical data, use `write_file_atomic`.
283///
284/// # Examples
285///
286/// ```rust,no_run
287/// use llmspell_utils::file_utils::write_file;
288/// use std::path::Path;
289///
290/// # fn main() -> Result<(), Box<dyn std::error::Error>> {
291/// write_file(Path::new("/tmp/output.txt"), b"Hello, world!")?;
292/// # Ok(())
293/// # }
294/// ```
295///
296/// # Errors
297///
298/// Returns an error if:
299/// - Parent directory does not exist
300/// - Write permissions are denied
301/// - Disk is full
302/// - I/O error occurs
303pub fn write_file(path: &Path, data: &[u8]) -> Result<()> {
304    // Ensure parent directory exists
305    if let Some(parent) = path.parent() {
306        ensure_dir(parent).with_context(|| {
307            format!("Failed to create parent directory for: {}", path.display())
308        })?;
309    }
310
311    let mut file =
312        File::create(path).with_context(|| format!("Failed to create file: {}", path.display()))?;
313
314    file.write_all(data)
315        .with_context(|| format!("Failed to write to file: {}", path.display()))?;
316
317    file.sync_all()
318        .with_context(|| format!("Failed to sync file to disk: {}", path.display()))?;
319
320    Ok(())
321}

323/// Atomically write data to a file
324///
325/// This function writes data to a temporary file and then atomically renames it
326/// to the target path. This ensures that the file is either fully written or
327/// not modified at all, preventing partial writes.
328///
329/// # Examples
330///
331/// ```rust,no_run
332/// use llmspell_utils::file_utils::write_file_atomic;
333/// use std::path::Path;
334///
335/// # fn main() -> Result<(), Box<dyn std::error::Error>> {
336/// // This write is atomic - the file will either be fully written or unchanged
337/// write_file_atomic(Path::new("/tmp/important.json"), b"{\"status\": \"ok\"}")?;
338/// # Ok(())
339/// # }
340/// ```
341///
342/// # Errors
343///
344/// Returns an error if:
345/// - Parent directory does not exist
346/// - Write permissions are denied
347/// - Disk is full
348/// - I/O error occurs
349pub fn write_file_atomic(path: &Path, data: &[u8]) -> Result<()> {
350    // Ensure parent directory exists
351    if let Some(parent) = path.parent() {
352        ensure_dir(parent).with_context(|| {
353            format!("Failed to create parent directory for: {}", path.display())
354        })?;
355    }
356
357    // Create temporary file in the same directory
358    let temp_path = {
359        let mut temp = path.to_path_buf();
360        let file_name = path
361            .file_name()
362            .ok_or_else(|| anyhow::anyhow!("Path has no file name: {}", path.display()))?;
363
364        let file_name_str = file_name.to_string_lossy();
365        let uuid_str = uuid::Uuid::new_v4().simple();
366        let temp_name = format!(".{file_name_str}.tmp.{uuid_str}");
367
368        temp.set_file_name(temp_name);
369        temp
370    };
371
372    // Write to temporary file
373    let result = (|| -> Result<()> {
374        let mut file = File::create(&temp_path)
375            .with_context(|| format!("Failed to create temporary file: {}", temp_path.display()))?;
376
377        file.write_all(data).with_context(|| {
378            format!("Failed to write to temporary file: {}", temp_path.display())
379        })?;
380
381        file.sync_all()
382            .with_context(|| format!("Failed to sync temporary file: {}", temp_path.display()))?;
383
384        Ok(())
385    })();
386
387    // If write failed, clean up temp file
388    if let Err(e) = result {
389        let _ = fs::remove_file(&temp_path);
390        return Err(e);
391    }
392
393    // Atomic rename
394    fs::rename(&temp_path, path).with_context(|| {
395        format!(
396            "Failed to rename {} to {}",
397            temp_path.display(),
398            path.display()
399        )
400    })?;
401
402    Ok(())
403}

431/// Join paths safely, handling platform differences
432///
433/// This function safely joins path components, handling:
434/// - Empty components
435/// - Absolute paths in components (which reset the path)
436/// - Platform-specific separators
437///
438/// # Examples
439///
440/// ```rust
441/// use llmspell_utils::file_utils::join_paths;
442/// use std::path::Path;
443///
444/// let joined = join_paths(&[Path::new("/home"), Path::new("user"), Path::new("docs")]);
445/// assert_eq!(joined.to_str().unwrap(), "/home/user/docs");
446/// ```
447#[must_use]
448pub fn join_paths(paths: &[&Path]) -> PathBuf {
449    let mut result = PathBuf::new();
450
451    for path in paths {
452        if path.is_absolute() {
453            result = path.to_path_buf();
454        } else {
455            result.push(path);
456        }
457    }
458
459    result.clean()
460}

518/// Remove a file if it exists
519///
520/// This function does not return an error if the file doesn't exist.
521///
522/// # Examples
523///
524/// ```rust,no_run
525/// use llmspell_utils::file_utils::remove_file_if_exists;
526/// use std::path::Path;
527///

532/// ```

534/// # Errors
535///
536/// Returns an error if:
537/// - The path exists but is a directory

542        Ok(()) => Ok(()),
543        Err(e) if e.kind() == io::ErrorKind::NotFound => Ok(()),

545    }

578/// Append data to a file
579///
580/// This function appends data to an existing file or creates it if it doesn't exist.
581/// The parent directory is created if necessary.
582///
583/// # Examples
584///
585/// ```rust,no_run
586/// use llmspell_utils::file_utils::append_file;
587/// use std::path::Path;
588///
589/// # fn main() -> Result<(), Box<dyn std::error::Error>> {
590/// append_file(Path::new("/tmp/log.txt"), b"\nNew log entry")?;
591/// # Ok(())
592/// # }
593/// ```
594///
595/// # Errors
596///
597/// Returns an error if:
598/// - Parent directory cannot be created
599/// - Write permissions are denied
600/// - I/O error occurs
601pub fn append_file(path: &Path, data: &[u8]) -> Result<()> {
602    // Ensure parent directory exists
603    if let Some(parent) = path.parent() {
604        ensure_dir(parent).with_context(|| {
605            format!("Failed to create parent directory for: {}", path.display())
606        })?;
607    }
608
609    let mut file = std::fs::OpenOptions::new()
610        .create(true)
611        .append(true)
612        .open(path)
613        .with_context(|| format!("Failed to open file for appending: {}", path.display()))?;
614
615    file.write_all(data)
616        .with_context(|| format!("Failed to append to file: {}", path.display()))?;
617
618    file.sync_all()
619        .with_context(|| format!("Failed to sync file to disk: {}", path.display()))?;
620
621    Ok(())
622}

624/// Move or rename a file
625///
626/// This function moves a file from one location to another, creating parent
627/// directories as needed. It's equivalent to rename on the same filesystem.
628///
629/// # Examples
630///
631/// ```rust,no_run
632/// use llmspell_utils::file_utils::move_file;
633/// use std::path::Path;
634///
635/// # fn main() -> Result<(), Box<dyn std::error::Error>> {
636/// move_file(Path::new("/tmp/old.txt"), Path::new("/tmp/new.txt"))?;
637/// # Ok(())
638/// # }
639/// ```
640///
641/// # Errors
642///
643/// Returns an error if:
644/// - Source file does not exist
645/// - Destination directory cannot be created
646/// - Insufficient permissions
647/// - Files are on different filesystems and copy fails
648pub fn move_file(from: &Path, to: &Path) -> Result<()> {
649    // Ensure destination directory exists
650    if let Some(parent) = to.parent() {
651        ensure_dir(parent).with_context(|| {
652            format!(
653                "Failed to create destination directory for: {}",
654                to.display()
655            )
656        })?;
657    }
658
659    // Try to rename first (most efficient)
660    match fs::rename(from, to) {
661        Ok(()) => Ok(()),
662        Err(e) if e.raw_os_error() == Some(18) => {
663            // Cross-device link error (EXDEV on Unix) - need to copy and delete
664            copy_file(from, to)?;
665            remove_file_if_exists(from)?;
666            Ok(())
667        }
668        Err(e) => Err(e).with_context(|| {
669            format!(
670                "Failed to move file from {} to {}",
671                from.display(),
672                to.display()
673            )
674        }),
675    }
676}

724/// Get file or directory metadata
725///
726/// Returns detailed metadata about a file or directory.
727///
728/// # Examples
729///
730/// ```rust,no_run
731/// use llmspell_utils::file_utils::get_metadata;
732/// use std::path::Path;
733///
734/// # fn main() -> Result<(), Box<dyn std::error::Error>> {

742///

746/// - I/O error occurs
747pub fn get_metadata(path: &Path) -> Result<FileMetadata> {

754        is_file: metadata.is_file(),

756        readonly: metadata.permissions().readonly(),

778/// List directory contents with metadata
779///
780/// Returns a vector of directory entries with basic metadata.
781/// Entries are sorted alphabetically by name.
782///
783/// # Examples
784///
785/// ```rust,no_run
786/// use llmspell_utils::file_utils::list_dir;
787/// use std::path::Path;
788///
789/// # fn main() -> Result<(), Box<dyn std::error::Error>> {
790/// let entries = list_dir(Path::new("/tmp"))?;
791/// for entry in entries {
792///     println!("{}: {} bytes", entry.name, entry.size);
793/// }
794/// # Ok(())
795/// # }
796/// ```
797///
798/// # Errors
799///
800/// Returns an error if:
801/// - Path is not a directory
802/// - Insufficient permissions
803/// - I/O error occurs
804#[derive(Debug, Clone)]
805pub struct DirEntry {
806    /// File or directory name (not full path)
807    pub name: String,
808    /// Full path to the entry
809    pub path: PathBuf,
810    /// File size in bytes
811    pub size: u64,
812    /// Whether this is a directory
813    pub is_dir: bool,
814    /// Whether this is a regular file
815    pub is_file: bool,
816    /// Whether this is a symbolic link
817    pub is_symlink: bool,
818    /// Last modification time
819    pub modified: Option<std::time::SystemTime>,
820}
821
822/// List directory contents with metadata
823///
824/// Returns a vector of directory entries sorted alphabetically.
825///
826/// # Examples
827///
828/// ```rust,no_run
829/// use llmspell_utils::file_utils::list_dir;
830/// use std::path::Path;
831///
832/// # fn main() -> Result<(), Box<dyn std::error::Error>> {
833/// let entries = list_dir(Path::new("/tmp"))?;
834/// for entry in entries {
835///     println!("{}: {} bytes", entry.name, entry.size);
836/// }
837/// # Ok(())
838/// # }
839/// ```
840///
841/// # Errors
842///
843/// Returns an error if:
844/// - Path is not a directory
845/// - Insufficient permissions
846/// - I/O error occurs
847pub fn list_dir(path: &Path) -> Result<Vec<DirEntry>> {
848    let mut entries = Vec::new();
849
850    let dir = fs::read_dir(path)
851        .with_context(|| format!("Failed to read directory: {}", path.display()))?;
852
853    for entry in dir {
854        let entry = entry
855            .with_context(|| format!("Failed to read directory entry in: {}", path.display()))?;
856
857        let file_name = entry.file_name();
858        let file_name_str = file_name.to_string_lossy().to_string();
859        let entry_path = entry.path();
860
861        // Get metadata - if it fails, create minimal entry
862        let (size, is_dir, is_file, is_symlink, modified) = match entry.metadata() {
863            Ok(metadata) => (
864                metadata.len(),
865                metadata.is_dir(),
866                metadata.is_file(),
867                metadata.is_symlink(),
868                metadata.modified().ok(),
869            ),
870            Err(_) => (0, false, false, false, None),
871        };
872
873        entries.push(DirEntry {
874            name: file_name_str,
875            path: entry_path,
876            size,
877            is_dir,
878            is_file,
879            is_symlink,
880            modified,
881        });
882    }
883
884    // Sort entries by name
885    entries.sort_by(|a, b| a.name.cmp(&b.name));
886
887    Ok(entries)
888}
