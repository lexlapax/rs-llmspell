1//! Enhanced module filtering system with pattern matching and wildcard support
2//!
3//! Provides flexible module-based filtering for debug output with glob patterns,
4//! hierarchical matching, and regex support for advanced use cases.
5
6use regex::Regex;
7use serde::{Deserialize, Serialize};
8use std::collections::HashMap;
9
10/// Enhanced module filter with pattern matching capabilities
11#[derive(Debug, Clone)]
12pub struct EnhancedModuleFilter {
13    /// Simple exact matches for fast lookup
14    exact_matches: HashMap<String, bool>,
15    /// Wildcard patterns compiled to regex
16    pattern_cache: HashMap<String, (Regex, bool)>,
17    /// Hierarchical rules (parent.child.*)
18    hierarchical_rules: Vec<(String, bool)>,
19    /// Default action when no patterns match
20    default_enabled: bool,
21}
22
23/// Pattern type for module filtering
24#[derive(Debug, Clone, Serialize, Deserialize)]
25pub enum FilterPattern {
26    /// Exact module name match
27    Exact(String),
28    /// Wildcard pattern (*, ?, [])
29    Wildcard(String),
30    /// Regular expression pattern
31    Regex(String),
32    /// Hierarchical pattern (module.submodule.*)
33    Hierarchical(String),
34}
35
36/// Filter rule with pattern and enable/disable action
37#[derive(Debug, Clone, Serialize, Deserialize)]
38pub struct FilterRule {
39    /// Pattern to match against
40    pub pattern: FilterPattern,
41    /// Whether to enable (true) or disable (false) logging
42    pub enabled: bool,
43    /// Optional description of the rule
44    pub description: Option<String>,
45}
46
47impl Default for EnhancedModuleFilter {
48    fn default() -> Self {
49        Self::new()
50    }
51}
52
53impl EnhancedModuleFilter {
54    /// Create a new enhanced module filter
55    #[must_use]
56    pub fn new() -> Self {
57        Self {
58            exact_matches: HashMap::new(),
59            pattern_cache: HashMap::new(),
60            hierarchical_rules: Vec::new(),
61            default_enabled: true,
62        }
63    }
64
65    /// Set the default action when no patterns match
66    pub fn set_default_enabled(&mut self, enabled: bool) {
67        self.default_enabled = enabled;
68    }
69
70    /// Add a filter rule
71    pub fn add_rule(&mut self, rule: FilterRule) {
72        // When adding an enabled rule to an empty filter, switch to allow-list behavior
73        // This provides intuitive behavior: adding enabled rules creates an allow-list
74        if rule.enabled && self.is_empty() {
75            self.default_enabled = false;
76        }
77
78        match rule.pattern {
79            FilterPattern::Exact(module) => {
80                self.exact_matches.insert(module, rule.enabled);
81            }
82            FilterPattern::Wildcard(pattern) => {
83                if let Ok(regex) = Self::wildcard_to_regex(&pattern) {
84                    self.pattern_cache.insert(pattern, (regex, rule.enabled));
85                }
86            }
87            FilterPattern::Regex(regex_str) => {
88                if let Ok(regex) = Regex::new(&regex_str) {
89                    self.pattern_cache.insert(regex_str, (regex, rule.enabled));
90                }
91            }
92            FilterPattern::Hierarchical(pattern) => {
93                self.hierarchical_rules.push((pattern, rule.enabled));
94                // Sort by specificity (longer patterns first)
95                self.hierarchical_rules
96                    .sort_by(|a, b| b.0.len().cmp(&a.0.len()));
97            }
98        }
99    }
100
101    /// Check if the filter has no rules
102    fn is_empty(&self) -> bool {
103        self.exact_matches.is_empty()
104            && self.pattern_cache.is_empty()
105            && self.hierarchical_rules.is_empty()
106    }
107
108    /// Add a simple pattern filter
109    pub fn add_filter(&mut self, pattern: &str, enabled: bool) {
110        let rule = if pattern.ends_with(".*") {
111            // Hierarchical pattern takes precedence over wildcard
112            FilterRule {
113                pattern: FilterPattern::Hierarchical(pattern.trim_end_matches(".*").to_string()),
114                enabled,
115                description: None,
116            }
117        } else if pattern.contains('*') || pattern.contains('?') || pattern.contains('[') {
118            FilterRule {
119                pattern: FilterPattern::Wildcard(pattern.to_string()),
120                enabled,
121                description: None,
122            }
123        } else {
124            FilterRule {
125                pattern: FilterPattern::Exact(pattern.to_string()),
126                enabled,
127                description: None,
128            }
129        };
130
131        self.add_rule(rule);
132    }
133
134    /// Check if a module should be logged
135    #[must_use]
136    pub fn should_log(&self, module: &str) -> bool {
137        // First check exact matches (fastest)
138        if let Some(&enabled) = self.exact_matches.get(module) {
139            return enabled;
140        }
141
142        // Check hierarchical rules (ordered by specificity)
143        for (pattern, enabled) in &self.hierarchical_rules {
144            if Self::matches_hierarchical(module, pattern) {
145                return *enabled;
146            }
147        }
148
149        // Check compiled regex patterns
150        for (regex, enabled) in self.pattern_cache.values() {
151            if regex.is_match(module) {
152                return *enabled;
153            }
154        }
155
156        // Default action
157        self.default_enabled
158    }
159
160    /// Clear all filters
161    pub fn clear(&mut self) {
162        self.exact_matches.clear();
163        self.pattern_cache.clear();
164        self.hierarchical_rules.clear();
165    }
166
167    /// Get all current filter rules as a summary
168    #[must_use]
169    pub fn get_filter_summary(&self) -> FilterSummary {
170        let mut rules = Vec::new();
171
172        // Add exact matches
173        for (pattern, enabled) in &self.exact_matches {
174            rules.push(FilterRule {
175                pattern: FilterPattern::Exact(pattern.clone()),
176                enabled: *enabled,
177                description: None,
178            });
179        }
180
181        // Add hierarchical rules
182        for (pattern, enabled) in &self.hierarchical_rules {
183            rules.push(FilterRule {
184                pattern: FilterPattern::Hierarchical(pattern.clone()),
185                enabled: *enabled,
186                description: None,
187            });
188        }
189
190        // Add pattern rules (we store the original pattern as key)
191        for pattern in self.pattern_cache.keys() {
192            if let Some((_, enabled)) = self.pattern_cache.get(pattern) {
193                let filter_pattern = if Self::is_regex_pattern(pattern) {
194                    FilterPattern::Regex(pattern.clone())
195                } else {
196                    FilterPattern::Wildcard(pattern.clone())
197                };
198
199                rules.push(FilterRule {
200                    pattern: filter_pattern,
201                    enabled: *enabled,
202                    description: None,
203                });
204            }
205        }
206
207        let total_rules = rules.len();
208        FilterSummary {
209            rules,
210            default_enabled: self.default_enabled,
211            total_rules,
212        }
213    }

215    /// Convert wildcard pattern to regex
216    fn wildcard_to_regex(pattern: &str) -> Result<Regex, regex::Error> {
217        let mut regex_pattern = String::new();
218        let chars: Vec<char> = pattern.chars().collect();
219        let mut i = 0;
220
221        regex_pattern.push('^');
222
223        while i < chars.len() {
224            match chars.get(i).copied() {
225                Some('*') => regex_pattern.push_str(".*"),
226                Some('?') => regex_pattern.push('.'),
227                Some('[') => {
228                    // Handle character classes
229                    regex_pattern.push('[');
230                    i += 1;
231                    while i < chars.len() && chars.get(i) != Some(&']') {
232                        if chars.get(i) == Some(&'\\') {
233                            regex_pattern.push('\\');
234                            i += 1;
235                            if let Some(&escaped_char) = chars.get(i) {
236                                regex_pattern.push(escaped_char);
237                            }
238                        } else if let Some(&ch) = chars.get(i) {
239                            regex_pattern.push(ch);
240                        }
241                        i += 1;
242                    }
243                    if chars.get(i) == Some(&']') {
244                        regex_pattern.push(']');
245                    }
246                }
247                Some('\\') => {
248                    regex_pattern.push('\\');
249                    i += 1;
250                    if let Some(&escaped_char) = chars.get(i) {
251                        regex_pattern.push(escaped_char);
252                    }
253                }
254                Some(c) if c.is_ascii_alphanumeric() || c == '_' || c == '-' => {
255                    regex_pattern.push(c);
256                }
257                Some(c) => {
258                    regex_pattern.push('\\');
259                    regex_pattern.push(c);
260                }
261                None => break,
262            }
263            i += 1;
264        }
265
266        regex_pattern.push('$');
267        Regex::new(&regex_pattern)
268    }

270    /// Check if a module matches a hierarchical pattern
271    fn matches_hierarchical(module: &str, pattern: &str) -> bool {
272        if module == pattern {
273            return true;
274        }
275
276        // Check if module starts with pattern followed by a dot
277        if let Some(remaining) = module.strip_prefix(pattern) {
278            remaining.starts_with('.')
279        } else {
280            false
281        }
282    }
283
284    /// Check if a pattern looks like a regex (heuristic)
285    fn is_regex_pattern(pattern: &str) -> bool {
286        // Simple heuristic: if it contains regex special chars that aren't wildcards
287        pattern.contains('(')
288            || pattern.contains(')')
289            || pattern.contains('+')
290            || pattern.contains('{')
291            || pattern.contains('}')
292            || pattern.contains('^')
293            || pattern.contains('$')
294            || pattern.contains('|')
295    }
296
297    /// Remove a specific filter pattern
298    pub fn remove_filter(&mut self, pattern: &str) -> bool {
299        // Try exact match first
300        if self.exact_matches.remove(pattern).is_some() {
301            return true;
302        }
303
304        // Try pattern cache
305        if self.pattern_cache.remove(pattern).is_some() {
306            return true;
307        }
308
309        // Try hierarchical rules
310        if let Some(pos) = self
311            .hierarchical_rules
312            .iter()
313            .position(|(p, _)| p == pattern)
314        {
315            self.hierarchical_rules.remove(pos);
316            return true;
317        }
318
319        false
320    }
321
322    /// Get performance statistics
323    #[must_use]
324    pub fn get_stats(&self) -> FilterStats {
325        FilterStats {
326            exact_matches: self.exact_matches.len(),
327            pattern_cache_size: self.pattern_cache.len(),
328            hierarchical_rules: self.hierarchical_rules.len(),
329            total_rules: self.exact_matches.len()
330                + self.pattern_cache.len()
331                + self.hierarchical_rules.len(),
332        }
333    }
334}

360/// Preset filter configurations for common use cases
361impl EnhancedModuleFilter {
362    /// Create a filter that only logs errors and warnings
363    #[must_use]
364    pub fn errors_only() -> Self {
365        let mut filter = Self::new();
366        filter.set_default_enabled(false);
367        filter.add_filter("error", true);
368        filter.add_filter("warn", true);
369        filter
370    }
371
372    /// Create a filter for development debugging
373    #[must_use]
374    pub fn development() -> Self {
375        let mut filter = Self::new();
376        filter.add_filter("workflow.*", true);
377        filter.add_filter("agent.*", true);
378        filter.add_filter("tool.*", true);
379        filter.add_filter("*.test", false);
380        filter.add_filter("*.benchmark", false);
381        filter
382    }
383
384    /// Create a filter for production logging
385    #[must_use]
386    pub fn production() -> Self {
387        let mut filter = Self::new();
388        filter.set_default_enabled(false);
389        filter.add_filter("security.*", true);
390        filter.add_filter("performance.*", true);
391        filter.add_filter("error.*", true);
392        filter.add_filter("audit.*", true);
393        filter
394    }
395
396    /// Create a filter for specific component debugging
397    #[must_use]
398    pub fn component(component: &str) -> Self {
399        let mut filter = Self::new();
400        filter.set_default_enabled(false);
401        filter.add_filter(&format!("{component}.*"), true);
402        filter
403    }

410    #[test]
411    fn test_exact_matching() {
412        let mut filter = EnhancedModuleFilter::new();
413
414        // For fine-grained control with both enabled and disabled rules,
415        // explicitly set default behavior before adding rules
416        filter.add_filter("agent.internal", false); // Add disable rule first
417        filter.add_filter("workflow.step1", true); // Then add enable rule
418
419        // Re-enable default for unmatched modules (fine-grained control mode)
420        filter.set_default_enabled(true);
421
422        assert!(filter.should_log("workflow.step1"));
423        assert!(!filter.should_log("agent.internal"));
424        assert!(filter.should_log("other.module")); // default enabled
425    }
