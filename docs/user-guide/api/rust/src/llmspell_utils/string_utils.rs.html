12/// Truncate a string to a maximum length with ellipsis
13///
14/// If the string is longer than `max_len`, it will be truncated and ellipsis added.
15/// The ellipsis counts towards the maximum length.
16///
17/// # Examples
18///
19/// ```rust
20/// use llmspell_utils::string_utils::truncate;
21///
22/// assert_eq!(truncate("Hello, world!", 5), "He...");
23/// assert_eq!(truncate("Short", 10), "Short");
24/// assert_eq!(truncate("", 5), "");
25/// ```
26#[must_use]
27pub fn truncate(s: &str, max_len: usize) -> String {
28    truncate_with_ellipsis(s, max_len, DEFAULT_ELLIPSIS)
29}

31/// Truncate a string to a maximum length with custom ellipsis
32///
33/// # Examples
34///
35/// ```rust
36/// use llmspell_utils::string_utils::truncate_with_ellipsis;
37///
38/// assert_eq!(truncate_with_ellipsis("Hello, world!", 7, "..."), "Hell...");
39/// assert_eq!(truncate_with_ellipsis("Hello, world!", 7, "…"), "Hell…");
40/// ```
41#[must_use]
42pub fn truncate_with_ellipsis(s: &str, max_len: usize, ellipsis: &str) -> String {
43    if s.len() <= max_len {
44        return s.to_string();
45    }
46
47    if max_len <= ellipsis.len() {
48        // If max_len is too small for ellipsis, just return the ellipsis truncated
49        return ellipsis.chars().take(max_len).collect();
50    }
51
52    let truncate_at = max_len - ellipsis.len();
53
54    // Find a char boundary
55    let mut boundary = truncate_at;
56    while boundary > 0 && !s.is_char_boundary(boundary) {
57        boundary -= 1;
58    }
59
60    format!("{}{}", &s[..boundary], ellipsis)
61}

63/// Wrap text to fit within a specified width
64///
65/// Breaks text at word boundaries when possible.
66///
67/// # Examples
68///
69/// ```rust
70/// use llmspell_utils::string_utils::word_wrap;
71///
72/// let text = "This is a long line that needs to be wrapped";
73/// let wrapped = word_wrap(text, 20);
74/// assert_eq!(wrapped, vec!["This is a long line", "that needs to be", "wrapped"]);
75/// ```
76#[must_use]
77pub fn word_wrap(text: &str, width: usize) -> Vec<String> {
78    if width == 0 {
79        return vec![];
80    }
81
82    let mut lines = Vec::new();
83
84    for line in text.lines() {
85        if line.len() <= width {
86            lines.push(line.to_string());
87            continue;
88        }
89
90        let words: Vec<&str> = line.split_whitespace().collect();
91        let mut current_line = String::new();
92
93        for word in words {
94            if current_line.is_empty() {
95                // First word on the line
96                if word.len() > width {
97                    // Word is too long, must be broken
98                    let mut chars = word.chars();
99                    while !chars.as_str().is_empty() {
100                        let chunk: String = chars.by_ref().take(width).collect();
101                        lines.push(chunk);
102                    }
103                } else {
104                    current_line = word.to_string();
105                }
106            } else if current_line.len() + 1 + word.len() <= width {
107                // Word fits on current line
108                current_line.push(' ');
109                current_line.push_str(word);
110            } else {
111                // Word doesn't fit, start new line
112                lines.push(current_line);
113                if word.len() > width {
114                    // Word is too long for any line, must be broken
115                    current_line = String::new();
116                    let mut chars = word.chars();
117                    while !chars.as_str().is_empty() {
118                        let chunk: String = chars.by_ref().take(width).collect();
119                        if chars.as_str().is_empty() {
120                            current_line = chunk;
121                        } else {
122                            lines.push(chunk);
123                        }
124                    }
125                } else {
126                    current_line = word.to_string();
127                }
128            }
129        }
130
131        if !current_line.is_empty() {
132            lines.push(current_line);
133        }
134    }
135
136    lines
137}

139/// Convert a string to `snake_case`
140///
141/// # Examples
142///
143/// ```rust
144/// use llmspell_utils::string_utils::to_snake_case;
145///
146/// assert_eq!(to_snake_case("HelloWorld"), "hello_world");
147/// assert_eq!(to_snake_case("HTTPResponse"), "httpresponse");
148/// assert_eq!(to_snake_case("already_snake_case"), "already_snake_case");
149/// assert_eq!(to_snake_case("PascalCase"), "pascal_case");
150/// ```
151#[must_use]
152pub fn to_snake_case(s: &str) -> String {
153    let mut result = String::new();
154    let mut prev_upper = false;
155
156    for (i, ch) in s.chars().enumerate() {
157        if ch.is_uppercase() {
158            if i > 0 && !prev_upper {
159                result.push('_');
160            }
161            // Safe: to_lowercase() always produces at least one char for valid Unicode
162            if let Some(lower_ch) = ch.to_lowercase().next() {
163                result.push(lower_ch);
164            }
165            prev_upper = true;
166        } else {
167            result.push(ch);
168            prev_upper = false;
169        }
170    }
171
172    result
173}

175/// Convert a string to `camelCase`
176///
177/// # Examples
178///
179/// ```rust
180/// use llmspell_utils::string_utils::to_camel_case;
181///
182/// assert_eq!(to_camel_case("hello_world"), "helloWorld");
183/// assert_eq!(to_camel_case("http_response"), "httpResponse");
184/// assert_eq!(to_camel_case("already_camelCase"), "alreadyCamelCase");
185/// ```
186#[must_use]
187pub fn to_camel_case(s: &str) -> String {
188    let mut result = String::new();
189    let mut capitalize_next = false;
190
191    for (i, ch) in s.chars().enumerate() {
192        if ch == '_' || ch == '-' {
193            capitalize_next = true;
194        } else if capitalize_next {
195            // Safe: to_uppercase() always produces at least one char for valid Unicode
196            if let Some(upper_ch) = ch.to_uppercase().next() {
197                result.push(upper_ch);
198            }
199            capitalize_next = false;
200        } else if i == 0 {
201            // Safe: to_lowercase() always produces at least one char for valid Unicode
202            if let Some(lower_ch) = ch.to_lowercase().next() {
203                result.push(lower_ch);
204            }
205        } else {
206            result.push(ch);
207        }
208    }
209
210    result
211}

246/// Sanitize a string for safe usage
247///
248/// Removes or escapes potentially dangerous characters:
249/// - Control characters (except newline and tab)
250/// - Non-ASCII characters
251/// - Leading/trailing spaces (but not tabs)
252///
253/// # Examples
254///
255/// ```rust
256/// use llmspell_utils::string_utils::sanitize;
257///
258/// assert_eq!(sanitize("  Hello\x00World  "), "HelloWorld");
259/// assert_eq!(sanitize("Normal text"), "Normal text");
260/// assert_eq!(sanitize("Line1\nLine2\t"), "Line1\nLine2\t");
261/// ```
262#[must_use]
263pub fn sanitize(s: &str) -> String {
264    // Trim only regular spaces, not tabs or newlines
265    let trimmed = s.trim_matches(' ');
266    trimmed
267        .chars()
268        .filter(|&ch| ch == '\n' || ch == '\t' || (ch.is_ascii_graphic() || ch == ' '))
269        .collect()
270}

272/// Escape special characters for safe display
273///
274/// Escapes characters that might cause issues in various contexts
275///
276/// # Examples
277///
278/// ```rust
279/// use llmspell_utils::string_utils::escape_special;
280///
281/// assert_eq!(escape_special("Hello\nWorld"), "Hello\\nWorld");
282/// assert_eq!(escape_special("Path\\to\\file"), "Path\\\\to\\\\file");
283/// assert_eq!(escape_special("Quote\"test\""), "Quote\\\"test\\\"");
284/// ```
285#[must_use]
286pub fn escape_special(s: &str) -> String {
287    let mut result = String::with_capacity(s.len());
288
289    for ch in s.chars() {
290        match ch {
291            '\n' => result.push_str("\\n"),
292            '\r' => result.push_str("\\r"),
293            '\t' => result.push_str("\\t"),
294            '\\' => result.push_str("\\\\"),
295            '"' => result.push_str("\\\""),
296            '\'' => result.push_str("\\'"),
297            '\0' => result.push_str("\\0"),
298            _ => result.push(ch),
299        }
300    }
301
302    result
303}

321/// Add custom indentation to each line of a string
322///
323/// # Examples
324///
325/// ```rust
326/// use llmspell_utils::string_utils::indent_with;
327///
328/// let text = "Line 1\nLine 2";
329/// let indented = indent_with(text, "> ");
330/// assert_eq!(indented, "> Line 1\n> Line 2");
331/// ```
332#[must_use]
333pub fn indent_with(s: &str, prefix: &str) -> String {
334    let lines: Vec<&str> = s.lines().collect();
335    let mut result = String::new();
336
337    for (i, line) in lines.iter().enumerate() {
338        result.push_str(prefix);
339        result.push_str(line);
340        if i < lines.len() - 1 || s.ends_with('\n') {
341            result.push('\n');
342        }
343    }
344
345    result
346}

348/// Remove common leading whitespace from lines
349///
350/// # Examples
351///
352/// ```rust
353/// use llmspell_utils::string_utils::dedent;
354///
355/// let text = "    Line 1\n    Line 2\n      Line 3";
356/// let dedented = dedent(text);
357/// assert_eq!(dedented, "Line 1\nLine 2\n  Line 3");
358/// ```
359#[must_use]
360pub fn dedent(s: &str) -> String {
361    let lines: Vec<&str> = s.lines().collect();
362
363    // Find minimum indentation (ignoring empty lines)
364    let min_indent = lines
365        .iter()
366        .filter(|line| !line.trim().is_empty())
367        .map(|line| line.len() - line.trim_start().len())
368        .min()
369        .unwrap_or(0);
370
371    // Remove common indentation
372    let mut result = String::new();
373    for (i, line) in lines.iter().enumerate() {
374        if line.len() >= min_indent {
375            result.push_str(&line[min_indent..]);
376        } else {
377            result.push_str(line);
378        }
379        if i < lines.len() - 1 || s.ends_with('\n') {
380            result.push('\n');
381        }
382    }
383
384    result
385}

387/// Check if a string is a valid identifier (alphanumeric + underscore, not starting with digit)
388///
389/// # Examples
390///
391/// ```rust
392/// use llmspell_utils::string_utils::is_valid_identifier;
393///
394/// assert!(is_valid_identifier("valid_name"));
395/// assert!(is_valid_identifier("_private"));
396/// assert!(!is_valid_identifier("123invalid"));
397/// assert!(!is_valid_identifier("invalid-name"));
398/// assert!(!is_valid_identifier(""));
399/// ```
400#[must_use]
401pub fn is_valid_identifier(s: &str) -> bool {
402    if s.is_empty() {
403        return false;
404    }
405
406    let mut chars = s.chars();
407
408    // First character must be alphabetic or underscore
409    if let Some(first) = chars.next() {
410        if !first.is_alphabetic() && first != '_' {
411            return false;
412        }
413    }
414
415    // Rest must be alphanumeric or underscore
416    chars.all(|ch| ch.is_alphanumeric() || ch == '_')
417}

419/// Split a string by lines, preserving line endings
420///
421/// Unlike `str::lines()`, this preserves the line ending characters
422///
423/// # Examples
424///
425/// ```rust
426/// use llmspell_utils::string_utils::lines_with_endings;
427///
428/// let text = "Line 1\nLine 2\r\nLine 3";
429/// let lines: Vec<_> = lines_with_endings(text).collect();
430/// assert_eq!(lines, vec!["Line 1\n", "Line 2\r\n", "Line 3"]);
431/// ```
432pub fn lines_with_endings(s: &str) -> impl Iterator<Item = &str> {
433    let mut start = 0;
434    let bytes = s.as_bytes();
435
436    std::iter::from_fn(move || {
437        if start >= bytes.len() {
438            return None;
439        }
440
441        let mut end = start;
442        while end < bytes.len() {
443            if bytes[end] == b'\n' {
444                end += 1;
445                break;
446            } else if end + 1 < bytes.len() && bytes[end] == b'\r' && bytes[end + 1] == b'\n' {
447                end += 2;
448                break;
449            }
450            end += 1;
451        }
452
453        let line = &s[start..end];
454        start = end;
455        Some(line)
456    })
457}

496/// Find common prefix of multiple strings
497///
498/// # Examples
499///
500/// ```rust
501/// use llmspell_utils::string_utils::common_prefix;
502///
503/// assert_eq!(common_prefix(&["prefix_a", "prefix_b", "prefix_c"]), "prefix_");
504/// assert_eq!(common_prefix(&["hello", "help", "hero"]), "he");
505/// assert_eq!(common_prefix(&["abc", "xyz"]), "");
506/// assert_eq!(common_prefix(&[]), "");
507/// ```
508#[must_use]
509pub fn common_prefix(strings: &[&str]) -> String {
510    if strings.is_empty() {
511        return String::new();
512    }
513
514    let mut prefix = String::new();
515    let first = strings[0];
516
517    'outer: for (i, ch) in first.chars().enumerate() {
518        for s in &strings[1..] {
519            if s.chars().nth(i) != Some(ch) {
520                break 'outer;
521            }
522        }
523        prefix.push(ch);
524    }
525
526    prefix
527}

630/// Extract substring by start and end indices
631///
632/// Returns empty string if indices are invalid
633///
634/// # Examples
635///
636/// ```rust
637/// use llmspell_utils::string_utils::substring;
638///
639/// assert_eq!(substring("hello world", 0, 5), "hello");
640/// assert_eq!(substring("hello world", 6, 11), "world");
641/// assert_eq!(substring("hello", 10, 20), "");
642/// ```
643#[must_use]
644pub fn substring(s: &str, start: usize, end: usize) -> String {
645    if start >= s.len() || end > s.len() || start >= end {
646        return String::new();
647    }
648
649    // Find char boundaries
650    let mut start_byte = start;
651    while start_byte > 0 && !s.is_char_boundary(start_byte) {
652        start_byte -= 1;
653    }
654
655    let mut end_byte = end;
656    while end_byte < s.len() && !s.is_char_boundary(end_byte) {
657        end_byte += 1;
658    }
659
660    s[start_byte..end_byte].to_string()
661}
