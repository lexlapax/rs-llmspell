1//! ABOUTME: Persistent storage backend for API keys using sled database
2//! ABOUTME: Provides encrypted key storage with metadata persistence
3
4#![allow(clippy::missing_errors_doc)]
5
6use crate::api_key_manager::{ApiKeyMetadata, ApiKeyStorage};
7use serde::{Deserialize, Serialize};
8use std::path::Path;
9
10/// Persistent storage implementation using sled
11pub struct PersistentApiKeyStorage {
12    db: sled::Db,
13    encryption_key: [u8; 32],
14}
15
16/// Key-value pair for storage
17#[derive(Serialize, Deserialize)]
18struct StoredKeyData {
19    encrypted_key: Vec<u8>,
20    metadata: ApiKeyMetadata,
21}
22
23impl PersistentApiKeyStorage {
24    /// Create a new persistent storage backend
25    pub fn new<P: AsRef<Path>>(path: P, encryption_key: [u8; 32]) -> Result<Self, String> {
26        let db = sled::open(path).map_err(|e| format!("Failed to open database: {e}"))?;
27        Ok(Self { db, encryption_key })
28    }
29
30    /// Encrypt a key using simple XOR (for demonstration)
31    /// In production, use proper encryption like AES-256-GCM
32    fn encrypt(&self, key: &str) -> Vec<u8> {
33        key.bytes()
34            .enumerate()
35            .map(|(i, b)| b ^ self.encryption_key[i % 32])
36            .collect()
37    }
38
39    /// Decrypt a key
40    fn decrypt(&self, encrypted: &[u8]) -> String {
41        encrypted
42            .iter()
43            .enumerate()
44            .map(|(i, &b)| (b ^ self.encryption_key[i % 32]) as char)
45            .collect()
46    }
47}
48
49impl ApiKeyStorage for PersistentApiKeyStorage {
50    fn store(&mut self, key_id: &str, key: &str, metadata: &ApiKeyMetadata) -> Result<(), String> {
51        let encrypted_key = self.encrypt(key);
52        let stored_data = StoredKeyData {
53            encrypted_key,
54            metadata: metadata.clone(),
55        };
56
57        let serialized = serde_json::to_vec(&stored_data)
58            .map_err(|e| format!("Failed to serialize key data: {e}"))?;
59
60        self.db
61            .insert(key_id.as_bytes(), serialized)
62            .map_err(|e| format!("Failed to store key: {e}"))?;
63
64        self.db
65            .flush()
66            .map_err(|e| format!("Failed to flush database: {e}"))?;
67
68        Ok(())
69    }
70
71    fn get(&self, key_id: &str) -> Result<Option<String>, String> {
72        match self.db.get(key_id.as_bytes()) {
73            Ok(Some(data)) => {
74                let stored_data: StoredKeyData = serde_json::from_slice(&data)
75                    .map_err(|e| format!("Failed to deserialize key data: {e}"))?;
76                Ok(Some(self.decrypt(&stored_data.encrypted_key)))
77            }
78            Ok(None) => Ok(None),
79            Err(e) => Err(format!("Failed to retrieve key: {e}")),
80        }
81    }
82
83    fn get_metadata(&self, key_id: &str) -> Result<Option<ApiKeyMetadata>, String> {
84        match self.db.get(key_id.as_bytes()) {
85            Ok(Some(data)) => {
86                let stored_data: StoredKeyData = serde_json::from_slice(&data)
87                    .map_err(|e| format!("Failed to deserialize key data: {e}"))?;
88                Ok(Some(stored_data.metadata))
89            }
90            Ok(None) => Ok(None),
91            Err(e) => Err(format!("Failed to retrieve metadata: {e}")),
92        }
93    }
94
95    fn update_metadata(&mut self, key_id: &str, metadata: &ApiKeyMetadata) -> Result<(), String> {
96        match self.db.get(key_id.as_bytes()) {
97            Ok(Some(data)) => {
98                let mut stored_data: StoredKeyData = serde_json::from_slice(&data)
99                    .map_err(|e| format!("Failed to deserialize key data: {e}"))?;
100
101                stored_data.metadata = metadata.clone();
102
103                let serialized = serde_json::to_vec(&stored_data)
104                    .map_err(|e| format!("Failed to serialize key data: {e}"))?;
105
106                self.db
107                    .insert(key_id.as_bytes(), serialized)
108                    .map_err(|e| format!("Failed to update metadata: {e}"))?;
109
110                self.db
111                    .flush()
112                    .map_err(|e| format!("Failed to flush database: {e}"))?;
113
114                Ok(())
115            }
116            Ok(None) => Err("Key not found".to_string()),
117            Err(e) => Err(format!("Failed to retrieve key for update: {e}")),
118        }
119    }
120
121    fn delete(&mut self, key_id: &str) -> Result<(), String> {
122        self.db
123            .remove(key_id.as_bytes())
124            .map_err(|e| format!("Failed to delete key: {e}"))?;
125
126        self.db
127            .flush()
128            .map_err(|e| format!("Failed to flush database: {e}"))?;
129
130        Ok(())
131    }
132
133    /// List all key IDs
134    fn list_keys(&self) -> Result<Vec<String>, String> {
135        let mut keys = Vec::new();
136        for item in self.db.iter() {
137            match item {
138                Ok((key, _)) => {
139                    if let Ok(key_str) = String::from_utf8(key.to_vec()) {
140                        keys.push(key_str);
141                    }
142                }
143                Err(e) => return Err(format!("Failed to iterate keys: {e}")),
144            }
145        }
146        Ok(keys)
147    }
148}

150#[cfg(test)]
151mod tests {
152    use super::*;
153    use chrono::Utc;
154    use tempfile::TempDir;
155
156    fn create_test_metadata(service: &str) -> ApiKeyMetadata {
157        ApiKeyMetadata {
158            key_id: format!("{service}_key"),
159            service: service.to_string(),
160            created_at: Utc::now(),
161            last_used: None,
162            expires_at: None,
163            is_active: true,
164            usage_count: 0,
165        }
166    }
167    #[test]
168    fn test_persistent_storage() {
169        let temp_dir = TempDir::new().unwrap();
170        let encryption_key = [42u8; 32];
171
172        let mut storage = PersistentApiKeyStorage::new(temp_dir.path(), encryption_key).unwrap();
173
174        // Store a key
175        let metadata = create_test_metadata("test_service");
176        storage
177            .store("test_key", "my_secret_key", &metadata)
178            .unwrap();
179
180        // Retrieve the key
181        let retrieved = storage.get("test_key").unwrap();
182        assert_eq!(retrieved, Some("my_secret_key".to_string()));
183
184        // Retrieve metadata
185        let retrieved_metadata = storage.get_metadata("test_key").unwrap().unwrap();
186        assert_eq!(retrieved_metadata.service, "test_service");
187
188        // Update metadata
189        let mut updated_metadata = metadata.clone();
190        updated_metadata.usage_count = 5;
191        storage
192            .update_metadata("test_key", &updated_metadata)
193            .unwrap();
194
195        let retrieved_metadata = storage.get_metadata("test_key").unwrap().unwrap();
196        assert_eq!(retrieved_metadata.usage_count, 5);
197
198        // Delete the key
199        storage.delete("test_key").unwrap();
200        assert_eq!(storage.get("test_key").unwrap(), None);
201    }
202    #[test]
203    fn test_encryption_decryption() {
204        let temp_dir = TempDir::new().unwrap();
205        let encryption_key = [123u8; 32];
206
207        let storage = PersistentApiKeyStorage::new(temp_dir.path(), encryption_key).unwrap();
208
209        let original = "my_super_secret_api_key";
210        let encrypted = storage.encrypt(original);
211        let decrypted = storage.decrypt(&encrypted);
212
213        assert_eq!(original, decrypted);
214        assert_ne!(original.as_bytes(), encrypted.as_slice());
215    }
