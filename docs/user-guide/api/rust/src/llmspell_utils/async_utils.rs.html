21/// Errors that can occur during async operations
22#[derive(Debug, Error)]
23pub enum AsyncError {
24    /// Operation timed out
25    #[error("Operation timed out after {0:?}")]
26    Timeout(Duration),
27
28    /// Retry limit exceeded
29    #[error("Retry limit exceeded after {attempts} attempts")]
30    RetryLimitExceeded {
31        /// Number of attempts made before giving up
32        attempts: usize,
33    },
34
35    /// Operation was cancelled
36    #[error("Operation was cancelled")]
37    Cancelled,
38}
39
40/// Result type for async operations
41pub type AsyncResult<T> = Result<T, AsyncError>;
42
43/// Configuration for retry operations
44#[derive(Debug, Clone)]
45pub struct RetryConfig {
46    /// Maximum number of attempts
47    pub max_attempts: usize,
48    /// Initial delay between retries
49    pub initial_delay: Duration,
50    /// Factor to multiply delay by after each attempt
51    pub backoff_factor: f64,
52    /// Maximum delay between retries
53    pub max_delay: Duration,
54    /// Add jitter to retry delays
55    pub jitter: bool,
56}
57
58impl Default for RetryConfig {
59    fn default() -> Self {
60        Self {
61            max_attempts: 3,
62            initial_delay: Duration::from_millis(100),
63            backoff_factor: 2.0,
64            max_delay: Duration::from_secs(10),
65            jitter: true,
66        }
67    }
68}

70/// Execute an async operation with a timeout
71///
72/// # Errors
73///
74/// Returns an error if the operation times out
75///
76/// # Examples
77///
78/// ```
79/// use llmspell_utils::async_utils::timeout;
80/// use std::time::Duration;
81///
82/// # async fn example() -> Result<(), Box<dyn std::error::Error>> {
83/// let result = timeout(
84///     Duration::from_secs(1),
85///     async { 42 }
86/// ).await?;
87/// assert_eq!(result, 42);
88/// # Ok(())
89/// # }
90/// ```
91pub async fn timeout<T, F>(duration: Duration, future: F) -> Result<T, time::error::Elapsed>
92where
93    F: Future<Output = T>,
94{
95    time::timeout(duration, future).await
96}

98/// Execute an async operation with a timeout, returning a default value on timeout
99///
100/// # Examples
101///
102/// ```
103/// use llmspell_utils::async_utils::timeout_with_default;
104/// use std::time::Duration;
105///
106/// # async fn example() {
107/// let result = timeout_with_default(
108///     Duration::from_secs(1),
109///     async { 42 },
110///     0
111/// ).await;
112/// assert_eq!(result, 42);
113/// # }
114/// ```
115pub async fn timeout_with_default<T, F>(duration: Duration, future: F, default: T) -> T
116where
117    F: Future<Output = T>,
118{
119    match time::timeout(duration, future).await {
120        Ok(value) => value,
121        Err(_) => default,
122    }
123}

125/// Retry an async operation with exponential backoff
126///
127/// # Errors
128///
129/// Returns `AsyncError::RetryLimitExceeded` if all attempts fail
130///
131/// # Examples
132///
133/// ```
134/// use llmspell_utils::async_utils::{retry_async, RetryConfig};
135/// use std::sync::atomic::{AtomicUsize, Ordering};
136/// use std::sync::Arc;
137///
138/// # async fn example() -> Result<(), Box<dyn std::error::Error>> {
139/// let counter = Arc::new(AtomicUsize::new(0));
140/// let counter_clone = counter.clone();
141///
142/// let result = retry_async(RetryConfig::default(), || {
143///     let count = counter_clone.fetch_add(1, Ordering::SeqCst);
144///     async move {
145///         if count < 2 {
146///             Err("Not ready yet")
147///         } else {
148///             Ok("Success!")
149///         }
150///     }
151/// }).await?;
152///
153/// assert_eq!(result, "Success!");
154/// assert_eq!(counter.load(Ordering::SeqCst), 3);
155/// # Ok(())
156/// # }
157/// ```
158#[allow(
159    clippy::cast_precision_loss,
160    clippy::cast_possible_truncation,
161    clippy::cast_sign_loss
162)]
163pub async fn retry_async<T, E, F, Fut>(config: RetryConfig, mut f: F) -> Result<T, AsyncError>
164where
165    F: FnMut() -> Fut,
166    Fut: Future<Output = Result<T, E>>,
167    E: Debug,
168{
169    let mut attempt = 0;
170    let mut delay = config.initial_delay;
171
172    loop {
173        attempt += 1;
174
175        match f().await {
176            Ok(value) => return Ok(value),
177            Err(e) => {
178                if attempt >= config.max_attempts {
179                    tracing::error!(
180                        "Retry limit exceeded after {} attempts. Last error: {:?}",
181                        attempt,
182                        e
183                    );
184                    return Err(AsyncError::RetryLimitExceeded { attempts: attempt });
185                }
186
187                tracing::debug!(
188                    "Attempt {} failed with error: {:?}. Retrying in {:?}",
189                    attempt,
190                    e,
191                    delay
192                );
193
194                // Add jitter if configured
195                let actual_delay = if config.jitter {
196                    let jitter_range = delay.as_millis() as f64 * 0.1;
197                    let jitter = (rand::random::<f64>() - 0.5) * 2.0 * jitter_range;
198                    Duration::from_millis((delay.as_millis() as f64 + jitter).max(0.0) as u64)
199                } else {
200                    delay
201                };
202
203                time::sleep(actual_delay).await;
204
205                // Calculate next delay with exponential backoff
206                delay = Duration::from_millis(
207                    ((delay.as_millis() as f64 * config.backoff_factor) as u64)
208                        .min(config.max_delay.as_millis() as u64),
209                );
210            }
211        }
212    }
213}

215/// Map a function over async operations with a concurrency limit
216///
217/// # Examples
218///
219/// ```
220/// use llmspell_utils::async_utils::concurrent_map;
221///
222/// # async fn example() -> Result<(), Box<dyn std::error::Error>> {
223/// let numbers = vec![1, 2, 3, 4, 5];
224/// let results = concurrent_map(
225///     numbers.into_iter(),
226///     2, // Max 2 concurrent operations
227///     |n| async move { n * 2 }
228/// ).await;
229///
230/// assert_eq!(results, vec![2, 4, 6, 8, 10]);
231/// # Ok(())
232/// # }
233/// ```
234pub async fn concurrent_map<I, F, Fut, T, U>(items: I, concurrency_limit: usize, f: F) -> Vec<U>
235where
236    I: IntoIterator<Item = T>,
237    F: Fn(T) -> Fut + Clone,
238    Fut: Future<Output = U>,
239    T: Send + 'static,
240    U: Send + 'static,
241{
242    let mut futures = FuturesUnordered::new();
243    let mut items = items.into_iter();
244    let mut results = Vec::new();
245    let mut active_count = 0;
246    let mut item_index = 0;
247    let mut result_map = std::collections::HashMap::new();
248
249    loop {
250        // Fill up to concurrency limit
251        while active_count < concurrency_limit {
252            if let Some(item) = items.next() {
253                let fut = f.clone()(item);
254                let index = item_index;
255                item_index += 1;
256                active_count += 1;
257
258                futures.push(async move {
259                    let result = fut.await;
260                    (index, result)
261                });
262            } else {
263                break;
264            }
265        }
266
267        // If no futures are active and no more items, we're done
268        if active_count == 0 {
269            break;
270        }
271
272        // Wait for a future to complete
273        if let Some((index, result)) = futures.next().await {
274            active_count -= 1;
275            result_map.insert(index, result);
276        }
277    }
278
279    // Reconstruct results in original order
280    for i in 0..item_index {
281        if let Some(result) = result_map.remove(&i) {
282            results.push(result);
283        }
284    }
285
286    results
287}

289/// A future that can be cancelled
290pub struct CancellableFuture<F> {
291    future: F,
292    cancelled: bool,
293}
294
295impl<F> CancellableFuture<F> {
296    /// Create a new cancellable future
297    pub fn new(future: F) -> Self {
298        Self {
299            future,
300            cancelled: false,
301        }
302    }
303
304    /// Cancel this future
305    pub fn cancel(&mut self) {
306        self.cancelled = true;
307    }
308
309    /// Check if this future has been cancelled
310    pub fn is_cancelled(&self) -> bool {
311        self.cancelled
312    }
313}
314
315impl<F> Future for CancellableFuture<F>
316where
317    F: Future,
318{
319    type Output = Result<F::Output, AsyncError>;
320
321    fn poll(self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Self::Output> {
322        let this = unsafe { self.get_unchecked_mut() };
323
324        if this.cancelled {
325            return Poll::Ready(Err(AsyncError::Cancelled));
326        }
327
328        let future = unsafe { Pin::new_unchecked(&mut this.future) };
329        match future.poll(cx) {
330            Poll::Ready(value) => Poll::Ready(Ok(value)),
331            Poll::Pending => Poll::Pending,
332        }
333    }
334}

337pub type BoxedResultFuture<T, E> = Pin<Box<dyn Future<Output = Result<T, E>> + Send>>;

339/// Execute multiple futures concurrently and return the first successful result
340///
341/// # Errors
342///
343/// Returns `AsyncError::RetryLimitExceeded` if all futures fail
344///
345/// # Examples
346///
347/// ```
348/// use llmspell_utils::async_utils::{race_to_success, BoxedResultFuture};
349/// use std::time::Duration;
350/// use tokio::time::sleep;
351///
352/// # async fn example() -> Result<(), Box<dyn std::error::Error>> {
353/// let futures: Vec<BoxedResultFuture<&str, &str>> = vec![
354///     Box::pin(async {
355///         sleep(Duration::from_millis(100)).await;
356///         Ok("slow")
357///     }),
358///     Box::pin(async {
359///         sleep(Duration::from_millis(10)).await;
360///         Ok("fast")
361///     }),
362/// ];
363///
364/// let result = race_to_success(futures).await?;
365/// assert_eq!(result, "fast");
366/// # Ok(())
367/// # }
368/// ```
369pub async fn race_to_success<T, E>(futures: Vec<BoxedResultFuture<T, E>>) -> Result<T, AsyncError>
370where
371    T: Send + 'static,
372    E: Debug + Send + 'static,
373{
374    use futures::future::select_all;
375
376    if futures.is_empty() {
377        return Err(AsyncError::RetryLimitExceeded { attempts: 0 });
378    }
379
380    let (result, _index, _remaining) = select_all(futures).await;
381
382    match result {
383        Ok(value) => Ok(value),
384        Err(e) => {
385            tracing::error!("All futures failed. Last error: {:?}", e);
386            Err(AsyncError::RetryLimitExceeded { attempts: 1 })
387        }
388    }
389}

391/// Add jitter for backoff calculations
392#[cfg(test)]
393#[allow(
394    clippy::cast_precision_loss,
395    clippy::cast_possible_truncation,
396    clippy::cast_sign_loss
397)]
398fn add_jitter(duration: Duration, jitter_fraction: f64) -> Duration {
399    let millis = duration.as_millis() as f64;
400    let jitter = millis * jitter_fraction * (rand::random::<f64>() - 0.5) * 2.0;
401    Duration::from_millis((millis + jitter).max(0.0) as u64)
402}
