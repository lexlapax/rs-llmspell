11/// Production-safe error response
12#[derive(Debug, Clone, serde::Serialize)]
13pub struct SafeErrorResponse {
14    /// Error message safe for client
15    pub error: String,
16    /// Error code for reference
17    pub code: String,
18    /// Error category if allowed
19    #[serde(skip_serializing_if = "Option::is_none")]
20    pub category: Option<String>,
21    /// Whether the client should retry
22    pub retry: bool,
23    /// Request ID for correlation
24    #[serde(skip_serializing_if = "Option::is_none")]
25    pub request_id: Option<String>,
26    /// Timestamp
27    pub timestamp: String,
28}

30/// Error context builder for rich error information
31#[derive(Debug, Default, Clone)]
32pub struct ErrorContext {
33    /// Operation being performed
34    pub operation: Option<String>,
35    /// Resource involved
36    pub resource: Option<String>,
37    /// User or session ID (sanitized)
38    pub user_id: Option<String>,
39    /// Additional context
40    pub metadata: HashMap<String, String>,
41}
42
43impl ErrorContext {
44    /// Create a new error context
45    #[must_use]
46    pub fn new() -> Self {
47        Self::default()
48    }
49
50    /// Set the operation
51    #[must_use]
52    pub fn with_operation(mut self, operation: impl Into<String>) -> Self {
53        self.operation = Some(operation.into());
54        self
55    }
56
57    /// Set the resource
58    #[must_use]
59    pub fn with_resource(mut self, resource: impl Into<String>) -> Self {
60        self.resource = Some(resource.into());
61        self
62    }
63
64    /// Set the user ID
65    #[must_use]
66    pub fn with_user_id(mut self, user_id: impl Into<String>) -> Self {
67        self.user_id = Some(user_id.into());
68        self
69    }
70
71    /// Add metadata
72    #[must_use]
73    pub fn with_metadata(mut self, key: impl Into<String>, value: impl Into<String>) -> Self {
74        self.metadata.insert(key.into(), value.into());
75        self
76    }
77}

79/// Safe error handler for production use
80pub struct SafeErrorHandler {
81    /// Information disclosure preventer
82    preventer: Arc<InfoDisclosurePreventer>,
83    /// Production error handler
84    handler: ProductionErrorHandler,
85    /// Whether we're in production mode
86    #[allow(dead_code)]
87    is_production: bool,
88}
89
90impl SafeErrorHandler {
91    /// Create a new safe error handler
92    #[must_use]
93    pub fn new(is_production: bool) -> Self {
94        let preventer = if is_production {
95            Arc::new(InfoDisclosurePreventer::production())
96        } else {
97            Arc::new(InfoDisclosurePreventer::development())
98        };
99
100        let handler = ProductionErrorHandler::new(Arc::clone(&preventer));
101
102        Self {
103            preventer,
104            handler,
105            is_production,
106        }
107    }
108
109    /// Handle an `LLMSpellError` safely
110    #[must_use]
111    pub fn handle_llmspell_error(
112        &self,
113        error: &LLMSpellError,
114        context: &ErrorContext,
115    ) -> SafeErrorResponse {
116        // Create error info with context
117        let mut error_info = ErrorInfo {
118            message: error.to_string(),
119            kind: Some(Self::categorize_llmspell_error(error)),
120            stack_trace: None, // Backtrace support requires nightly or specific feature flags
121            context: HashMap::new(),
122            source_location: None,
123        };
124
125        // Add context to error info
126        if let Some(ref op) = context.operation {
127            error_info
128                .context
129                .insert("operation".to_string(), op.clone());
130        }
131        if let Some(ref res) = context.resource {
132            error_info
133                .context
134                .insert("resource".to_string(), res.clone());
135        }
136
137        // Log the full error internally
138        self.log_error(&error_info, context);
139
140        // Sanitize for external consumption
141        let sanitized = self.preventer.sanitize_error(&error_info);
142
143        SafeErrorResponse {
144            error: sanitized.message,
145            code: sanitized.error_code,
146            category: sanitized.category,
147            retry: sanitized.retriable,
148            request_id: context.metadata.get("request_id").cloned(),
149            timestamp: chrono::Utc::now().to_rfc3339(),
150        }
151    }
152
153    /// Handle any error safely
154    pub fn handle_error(
155        &self,
156        error: &dyn std::error::Error,
157        context: &ErrorContext,
158    ) -> SafeErrorResponse {
159        let sanitized = self.handler.handle_error(error);
160
161        SafeErrorResponse {
162            error: sanitized.message,
163            code: sanitized.error_code,
164            category: sanitized.category,
165            retry: sanitized.retriable,
166            request_id: context.metadata.get("request_id").cloned(),
167            timestamp: chrono::Utc::now().to_rfc3339(),
168        }
169    }
170
171    /// Categorize `LLMSpellError`
172    fn categorize_llmspell_error(error: &LLMSpellError) -> String {
173        match error {
174            LLMSpellError::Validation { .. } => "validation".to_string(),
175            LLMSpellError::Tool { .. } => "tool_error".to_string(),
176            LLMSpellError::Provider { .. } => "provider_error".to_string(),
177            LLMSpellError::Storage { .. } => "storage_error".to_string(),
178            LLMSpellError::Timeout { .. } => "timeout".to_string(),
179            LLMSpellError::RateLimit { .. } => "rate_limit".to_string(),
180            LLMSpellError::Network { .. } => "network_error".to_string(),
181            LLMSpellError::Script { .. } => "script_error".to_string(),
182            _ => "internal".to_string(),
183        }
184    }
185
186    /// Log error internally with full details
187    fn log_error(&self, error_info: &ErrorInfo, context: &ErrorContext) {
188        let mut fields = vec![];
189
190        if let Some(ref op) = context.operation {
191            fields.push(("operation", op.as_str()));
192        }
193        if let Some(ref res) = context.resource {
194            fields.push(("resource", res.as_str()));
195        }
196        if let Some(ref user) = context.user_id {
197            fields.push(("user_id", user.as_str()));
198        }
199
200        tracing::error!(
201            error_code = %self.preventer.generate_error_code(&error_info.message),
202            error_type = ?error_info.kind,
203            ?fields,
204            "Error occurred: {}",
205            error_info.message
206        );
207    }
208}

210/// Debug information manager for development
211pub struct DebugInfoManager {
212    /// Whether to include debug info
213    enabled: bool,
214    /// Maximum debug info size
215    max_size: usize,
216}
217
218impl DebugInfoManager {
219    /// Create a new debug info manager
220    #[must_use]
221    pub fn new(enabled: bool) -> Self {
222        Self {
223            enabled,
224            max_size: 10000, // 10KB max
225        }
226    }
227
228    /// Add debug information to response if enabled
229    pub fn add_debug_info<T: serde::Serialize + for<'de> serde::Deserialize<'de>>(
230        &self,
231        response: T,
232        debug_info: HashMap<String, String>,
233    ) -> T {
234        if !self.enabled || debug_info.is_empty() {
235            return response;
236        }
237
238        // Only in development mode, add debug info
239        if let Ok(mut value) = serde_json::to_value(&response) {
240            if let Some(obj) = value.as_object_mut() {
241                let mut safe_debug = HashMap::new();
242                let mut total_size = 0;
243
244                for (key, val) in debug_info {
245                    let size = key.len() + val.len();
246                    if total_size + size <= self.max_size {
247                        safe_debug.insert(key, val);
248                        total_size += size;
249                    }
250                }
251
252                obj.insert("_debug".to_string(), serde_json::json!(safe_debug));
253            }
254
255            if let Ok(updated) = serde_json::from_value(value) {
256                return updated;
257            }
258        }
259
260        response
261    }
262}

264/// Stack trace remover for production
265pub struct StackTraceRemover;
266
267impl StackTraceRemover {
268    /// Remove stack traces from error strings
269    #[must_use]
270    pub fn remove_stack_traces(error_str: &str) -> String {
271        let lines: Vec<&str> = error_str.lines().collect();
272        let mut result = Vec::new();
273        let mut in_stack_trace = false;
274
275        for line in lines {
276            if line.trim_start().starts_with("at ")
277                || line.contains("stack backtrace:")
278                || line.trim_start().starts_with("note: run with")
279            {
280                in_stack_trace = true;
281                continue;
282            }
283
284            if in_stack_trace && line.trim().is_empty() {
285                in_stack_trace = false;
286                continue;
287            }
288
289            if !in_stack_trace {
290                result.push(line);
291            }
292        }
293
294        result.join("\n").trim().to_string()
295    }
296}

298/// Logging standards enforcer
299pub struct LoggingStandards {
300    /// Minimum log level for production
301    pub min_level: tracing::Level,
302    /// Fields that must be included
303    pub required_fields: Vec<String>,
304    /// Fields that must be excluded
305    pub forbidden_fields: Vec<String>,
306}
307
308impl Default for LoggingStandards {
309    fn default() -> Self {
310        Self {
311            min_level: tracing::Level::INFO,
312            required_fields: vec![
313                "timestamp".to_string(),
314                "level".to_string(),
315                "target".to_string(),
316            ],
317            forbidden_fields: vec![
318                "password".to_string(),
319                "secret".to_string(),
320                "token".to_string(),
321                "api_key".to_string(),
322                "private_key".to_string(),
323            ],
324        }
325    }
326}
327
328impl LoggingStandards {
329    /// Validate a log record
330    #[must_use]
331    pub fn validate_log_record(&self, record: &tracing::Event) -> bool {
332        // Check level
333        if record.metadata().level() > &self.min_level {
334            return false;
335        }
336
337        // Would check fields here in a real implementation
338        true
339    }
