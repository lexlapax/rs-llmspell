11/// Real-time event metrics with windowed analytics
12#[derive(Debug, Default)]
13pub struct EventMetrics {
14    /// Total events processed
15    pub total_events: u64,
16    /// Events by type
17    pub events_by_type: HashMap<String, u64>,
18    /// Events by language
19    pub events_by_language: HashMap<Language, u64>,
20    /// Average processing time
21    pub avg_processing_time: Duration,
22    /// Peak events per second
23    pub peak_events_per_second: f64,
24    /// Current events per second
25    pub current_events_per_second: f64,
26    /// Real-time analytics
27    pub real_time_analytics: RealTimeAnalytics,
28}

30/// Real-time analytics data
31#[derive(Debug)]
32pub struct RealTimeAnalytics {
33    /// Events per second over last 60 seconds (sliding window)
34    pub eps_history: VecDeque<f64>,
35    /// Memory usage over time
36    pub memory_usage_bytes: u64,
37    /// Processing latency percentiles
38    pub latency_percentiles: LatencyPercentiles,
39    /// Error rate tracking
40    pub error_rate: f64,
41    /// Throughput trend (increasing/decreasing)
42    pub throughput_trend: ThroughputTrend,
43    /// Last update timestamp
44    pub last_updated: SystemTime,
45}

47/// Latency percentile metrics
48#[derive(Debug, Default)]
49pub struct LatencyPercentiles {
50    pub p50: Duration,
51    pub p90: Duration,
52    pub p95: Duration,
53    pub p99: Duration,
54}

56/// Throughput trend indicator
57#[derive(Debug, PartialEq, Clone, Copy)]
58pub enum ThroughputTrend {
59    Increasing,
60    Stable,
61    Decreasing,
62    Unknown,
63}

84/// Enhanced metrics collector with real-time analytics
85pub struct MetricsCollector {
86    metrics: Arc<RwLock<EventMetrics>>,
87    start_time: Instant,
88    /// Processing time samples for percentile calculation
89    processing_samples: Arc<RwLock<VecDeque<Duration>>>,
90    /// Error count tracking
91    error_count: Arc<RwLock<u64>>,
92    /// Real-time update task handle
93    _update_task: Option<tokio::task::JoinHandle<()>>,
94}
95
96impl MetricsCollector {
97    pub fn new() -> Self {
98        let metrics = Arc::new(RwLock::new(EventMetrics::default()));
99        let processing_samples = Arc::new(RwLock::new(VecDeque::with_capacity(10000)));
100        let error_count = Arc::new(RwLock::new(0u64));
101
102        // Start real-time analytics update task
103        let update_task = Self::start_analytics_task(
104            Arc::clone(&metrics),
105            Arc::clone(&processing_samples),
106            Arc::clone(&error_count),
107        );
108
109        Self {
110            metrics,
111            start_time: Instant::now(),
112            processing_samples,
113            error_count,
114            _update_task: Some(update_task),
115        }
116    }
117
118    /// Start background task for real-time analytics updates
119    fn start_analytics_task(
120        metrics: Arc<RwLock<EventMetrics>>,
121        processing_samples: Arc<RwLock<VecDeque<Duration>>>,
122        error_count: Arc<RwLock<u64>>,
123    ) -> tokio::task::JoinHandle<()> {
124        tokio::spawn(async move {
125            let mut interval = interval(Duration::from_secs(1));
126            let mut last_event_count = 0u64;
127            let mut last_update = Instant::now();
128
129            loop {
130                interval.tick().await;
131
132                let current_time = Instant::now();
133                let elapsed = current_time.duration_since(last_update).as_secs_f64();
134
135                let mut metrics_guard = metrics.write();
136                let current_event_count = metrics_guard.total_events;
137
138                // Calculate current EPS
139                let current_eps = if elapsed > 0.0 {
140                    #[allow(clippy::cast_precision_loss)]
141                    let event_diff = (current_event_count - last_event_count) as f64 / elapsed;
142                    event_diff
143                } else {
144                    0.0
145                };
146
147                // Update sliding window (keep last 60 seconds)
148                metrics_guard
149                    .real_time_analytics
150                    .eps_history
151                    .push_back(current_eps);
152                if metrics_guard.real_time_analytics.eps_history.len() > 60 {
153                    metrics_guard.real_time_analytics.eps_history.pop_front();
154                }
155
156                // Calculate throughput trend
157                let trend = if metrics_guard.real_time_analytics.eps_history.len() >= 10 {
158                    let recent: f64 = metrics_guard
159                        .real_time_analytics
160                        .eps_history
161                        .iter()
162                        .rev()
163                        .take(5)
164                        .sum::<f64>()
165                        / 5.0;
166                    let older: f64 = metrics_guard
167                        .real_time_analytics
168                        .eps_history
169                        .iter()
170                        .rev()
171                        .skip(5)
172                        .take(5)
173                        .sum::<f64>()
174                        / 5.0;
175
176                    if recent > older * 1.1 {
177                        ThroughputTrend::Increasing
178                    } else if recent < older * 0.9 {
179                        ThroughputTrend::Decreasing
180                    } else {
181                        ThroughputTrend::Stable
182                    }
183                } else {
184                    ThroughputTrend::Unknown
185                };
186
187                metrics_guard.real_time_analytics.throughput_trend = trend;
188
189                // Update latency percentiles
190                let samples = processing_samples.read();
191                if !samples.is_empty() {
192                    let mut sorted_samples: Vec<Duration> = samples.iter().cloned().collect();
193                    sorted_samples.sort();
194
195                    let len = sorted_samples.len();
196                    metrics_guard.real_time_analytics.latency_percentiles = LatencyPercentiles {
197                        p50: sorted_samples[(len * 50) / 100],
198                        p90: sorted_samples[(len * 90) / 100],
199                        p95: sorted_samples[(len * 95) / 100],
200                        p99: sorted_samples[(len * 99) / 100],
201                    };
202                }
203
204                // Calculate error rate
205                let total_errors = *error_count.read();
206                metrics_guard.real_time_analytics.error_rate = if current_event_count > 0 {
207                    #[allow(clippy::cast_precision_loss)]
208                    let error_rate = total_errors as f64 / current_event_count as f64 * 100.0;
209                    error_rate
210                } else {
211                    0.0
212                };
213
214                // Update memory usage (approximate)
215                #[allow(clippy::cast_possible_truncation)]
216                let memory_bytes = (samples.len() * std::mem::size_of::<Duration>()) as u64;
217                metrics_guard.real_time_analytics.memory_usage_bytes = memory_bytes;
218
219                metrics_guard.real_time_analytics.last_updated = SystemTime::now();
220
221                last_event_count = current_event_count;
222                last_update = current_time;
223            }
224        })
225    }

227    /// Record an event with enhanced analytics
228    pub fn record_event(&self, event: &UniversalEvent, processing_time: Duration) {
229        let mut metrics = self.metrics.write();
230
231        metrics.total_events += 1;
232        *metrics
233            .events_by_type
234            .entry(event.event_type.clone())
235            .or_insert(0) += 1;
236        *metrics
237            .events_by_language
238            .entry(event.language)
239            .or_insert(0) += 1;
240
241        // Update average processing time (exponential moving average)
242        let alpha = 0.1; // Smoothing factor
243        #[allow(clippy::cast_precision_loss)]
244        let current_avg = metrics.avg_processing_time.as_nanos() as f64;
245        #[allow(clippy::cast_precision_loss)]
246        let new_time = processing_time.as_nanos() as f64;
247        let new_avg = alpha * new_time + (1.0 - alpha) * current_avg;
248        #[allow(clippy::cast_possible_truncation, clippy::cast_sign_loss)]
249        let new_avg_nanos = new_avg as u64;
250        metrics.avg_processing_time = Duration::from_nanos(new_avg_nanos);
251
252        // Update events per second
253        let elapsed = self.start_time.elapsed().as_secs_f64();
254        if elapsed > 0.0 {
255            #[allow(clippy::cast_precision_loss)]
256            let eps = metrics.total_events as f64 / elapsed;
257            metrics.current_events_per_second = eps;
258            if metrics.current_events_per_second > metrics.peak_events_per_second {
259                metrics.peak_events_per_second = metrics.current_events_per_second;
260            }
261        }
262
263        // Store processing time sample for percentile calculation
264        let mut samples = self.processing_samples.write();
265        samples.push_back(processing_time);
266        // Keep only recent samples (last 10,000)
267        if samples.len() > 10000 {
268            samples.pop_front();
269        }
270    }

272    /// Record an error for error rate calculation
273    pub fn record_error(&self) {
274        let mut error_count = self.error_count.write();
275        *error_count += 1;
276    }

278    /// Get real-time dashboard data
279    pub fn get_dashboard_metrics(&self) -> DashboardMetrics {
280        let metrics = self.metrics.read();
281
282        DashboardMetrics {
283            total_events: metrics.total_events,
284            current_eps: metrics.current_events_per_second,
285            peak_eps: metrics.peak_events_per_second,
286            avg_latency: metrics.avg_processing_time,
287            p99_latency: metrics.real_time_analytics.latency_percentiles.p99,
288            error_rate: metrics.real_time_analytics.error_rate,
289            throughput_trend: metrics.real_time_analytics.throughput_trend,
290            memory_usage_mb: {
291                #[allow(clippy::cast_precision_loss)]
292                let memory_mb =
293                    metrics.real_time_analytics.memory_usage_bytes as f64 / 1024.0 / 1024.0;
294                memory_mb
295            },
296            uptime_seconds: self.start_time.elapsed().as_secs(),
297        }
298    }
299
300    /// Get current metrics snapshot
301    pub fn get_metrics(&self) -> EventMetrics {
302        self.metrics.read().clone()
303    }
304
305    /// Reset all metrics
306    pub fn reset(&self) {
307        *self.metrics.write() = EventMetrics::default();
308    }
