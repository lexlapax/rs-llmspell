11/// Performance metrics for a hook
12#[derive(Debug, Clone)]
13pub struct PerformanceMetrics {
14    pub execution_count: u64,
15    pub total_duration: Duration,
16    pub min_duration: Duration,
17    pub max_duration: Duration,
18    pub avg_duration: Duration,
19    pub p50_duration: Duration,
20    pub p95_duration: Duration,
21    pub p99_duration: Duration,
22    pub last_execution: Option<DateTime<Utc>>,
23}

41/// Performance sample for percentile calculations
42#[derive(Debug, Clone)]
43struct PerformanceSample {
44    duration: Duration,
45    timestamp: DateTime<Utc>,
46}

48/// Performance monitor for hook execution
49pub struct PerformanceMonitor {
50    samples: Arc<RwLock<HashMap<String, Vec<PerformanceSample>>>>,
51    counters: Arc<RwLock<HashMap<String, AtomicU64>>>,
52    config: PerformanceConfig,
53}

55/// Configuration for performance monitoring
56#[derive(Debug, Clone)]
57pub struct PerformanceConfig {
58    /// Maximum samples to keep per hook
59    pub max_samples: usize,
60    /// Sample retention duration
61    pub retention_period: Duration,
62    /// Enable percentile calculations
63    pub calculate_percentiles: bool,
64}
65
66impl Default for PerformanceConfig {
67    fn default() -> Self {
68        Self {
69            max_samples: 1000,
70            retention_period: Duration::from_secs(3600), // 1 hour
71            calculate_percentiles: true,
72        }
73    }
74}

82impl PerformanceMonitor {
83    /// Create a new performance monitor
84    pub fn new() -> Self {
85        Self::with_config(PerformanceConfig::default())
86    }
87
88    /// Create with custom configuration
89    pub fn with_config(config: PerformanceConfig) -> Self {
90        Self {
91            samples: Arc::new(RwLock::new(HashMap::new())),
92            counters: Arc::new(RwLock::new(HashMap::new())),
93            config,
94        }
95    }

97    /// Start tracking a hook execution
98    pub fn start_execution(&self, hook_name: &str) -> ExecutionTimer {
99        ExecutionTimer {
100            hook_name: hook_name.to_string(),
101            start_time: Instant::now(),
102            monitor: self.clone(),
103            completed: false,
104        }
105    }
106
107    /// Record an execution duration
108    pub fn record_execution(&self, hook_name: &str, duration: Duration) {
109        let mut samples = self.samples.write();
110        let hook_samples = samples.entry(hook_name.to_string()).or_default();
111
112        // Add new sample
113        hook_samples.push(PerformanceSample {
114            duration,
115            timestamp: Utc::now(),
116        });
117
118        // Clean up old samples
119        let cutoff = Utc::now() - chrono::Duration::from_std(self.config.retention_period).unwrap();
120        hook_samples.retain(|s| s.timestamp > cutoff);
121
122        // Limit sample count
123        if hook_samples.len() > self.config.max_samples {
124            let to_remove = hook_samples.len() - self.config.max_samples;
125            hook_samples.drain(0..to_remove);
126        }
127
128        // Update counter
129        let counters = self.counters.read();
130        if let Some(counter) = counters.get(hook_name) {
131            counter.fetch_add(1, Ordering::Relaxed);
132        } else {
133            drop(counters);
134            let mut counters = self.counters.write();
135            counters
136                .entry(hook_name.to_string())
137                .or_insert_with(|| AtomicU64::new(0))
138                .fetch_add(1, Ordering::Relaxed);
139        }
140    }
141
142    /// Get performance metrics for a hook
143    pub fn get_metrics(&self, hook_name: &str) -> Option<PerformanceMetrics> {
144        let samples = self.samples.read();
145        let hook_samples = samples.get(hook_name)?;
146
147        if hook_samples.is_empty() {
148            return None;
149        }
150
151        let execution_count = self
152            .counters
153            .read()
154            .get(hook_name)
155            .map(|c| c.load(Ordering::Relaxed))
156            .unwrap_or(0);
157
158        let total_duration: Duration = hook_samples.iter().map(|s| s.duration).sum();
159
160        let min_duration = hook_samples
161            .iter()
162            .map(|s| s.duration)
163            .min()
164            .unwrap_or(Duration::ZERO);
165
166        let max_duration = hook_samples
167            .iter()
168            .map(|s| s.duration)
169            .max()
170            .unwrap_or(Duration::ZERO);
171
172        let avg_duration = if !hook_samples.is_empty() {
173            #[allow(clippy::cast_possible_truncation)]
174            let len_u32 = hook_samples.len() as u32;
175            total_duration / len_u32
176        } else {
177            Duration::ZERO
178        };
179
180        let (p50, p95, p99) = if self.config.calculate_percentiles {
181            let mut durations: Vec<_> = hook_samples.iter().map(|s| s.duration).collect();
182            durations.sort();
183
184            let p50_idx = durations.len() / 2;
185            #[allow(
186                clippy::cast_precision_loss,
187                clippy::cast_possible_truncation,
188                clippy::cast_sign_loss
189            )]
190            let p95_idx = (durations.len() as f64 * 0.95) as usize;
191            #[allow(
192                clippy::cast_precision_loss,
193                clippy::cast_possible_truncation,
194                clippy::cast_sign_loss
195            )]
196            let p99_idx = (durations.len() as f64 * 0.99) as usize;
197
198            (
199                durations.get(p50_idx).copied().unwrap_or(Duration::ZERO),
200                durations.get(p95_idx).copied().unwrap_or(Duration::ZERO),
201                durations.get(p99_idx).copied().unwrap_or(Duration::ZERO),
202            )
203        } else {
204            (Duration::ZERO, Duration::ZERO, Duration::ZERO)
205        };
206
207        let last_execution = hook_samples.last().map(|s| s.timestamp);
208
209        Some(PerformanceMetrics {
210            execution_count,
211            total_duration,
212            min_duration,
213            max_duration,
214            avg_duration,
215            p50_duration: p50,
216            p95_duration: p95,
217            p99_duration: p99,
218            last_execution,
219        })
220    }
221
222    /// Get all hook metrics
223    pub fn get_all_metrics(&self) -> HashMap<String, PerformanceMetrics> {
224        let samples = self.samples.read();
225        samples
226            .keys()
227            .filter_map(|name| {
228                self.get_metrics(name)
229                    .map(|metrics| (name.clone(), metrics))
230            })
231            .collect()
232    }
233
234    /// Clear metrics for a specific hook
235    pub fn clear_metrics(&self, hook_name: &str) {
236        self.samples.write().remove(hook_name);
237        self.counters.write().remove(hook_name);
238    }
239
240    /// Clear all metrics
241    pub fn clear_all_metrics(&self) {
242        self.samples.write().clear();
243        self.counters.write().clear();
244    }
245
246    /// Check if a hook is performing within threshold
247    pub fn is_within_threshold(&self, hook_name: &str, threshold: Duration) -> bool {
248        if let Some(metrics) = self.get_metrics(hook_name) {
249            metrics.avg_duration <= threshold
250        } else {
251            true // No data means we assume it's fine
252        }
253    }
254
255    /// Get hooks exceeding performance threshold
256    pub fn get_slow_hooks(&self, threshold: Duration) -> Vec<(String, PerformanceMetrics)> {
257        self.get_all_metrics()
258            .into_iter()
259            .filter(|(_, metrics)| metrics.avg_duration > threshold)
260            .collect()
261    }

274/// Timer for tracking execution duration
275pub struct ExecutionTimer {
276    hook_name: String,
277    start_time: Instant,
278    monitor: PerformanceMonitor,
279    completed: bool,
280}
281
282impl ExecutionTimer {
283    /// Complete the timing and record the duration
284    pub fn complete(mut self) -> Duration {
285        let duration = self.start_time.elapsed();
286        self.monitor.record_execution(&self.hook_name, duration);
287        self.completed = true;
288        duration
289    }
290}
291
292impl Drop for ExecutionTimer {
293    fn drop(&mut self) {
294        // Auto-complete if not explicitly completed
295        if !self.completed {
296            let duration = self.start_time.elapsed();
297            self.monitor.record_execution(&self.hook_name, duration);
298        }
299    }
300}

302/// Performance report for diagnostics
303#[derive(Debug)]
304pub struct PerformanceReport {
305    pub total_hooks: usize,
306    pub total_executions: u64,
307    pub avg_duration: Duration,
308    pub slowest_hook: Option<(String, Duration)>,
309    pub most_frequent_hook: Option<(String, u64)>,
310    pub slow_hooks: Vec<(String, PerformanceMetrics)>,
311}
312
313impl PerformanceMonitor {
314    /// Generate a performance report
315    pub fn generate_report(&self, slow_threshold: Duration) -> PerformanceReport {
316        let all_metrics = self.get_all_metrics();
317
318        let total_hooks = all_metrics.len();
319        let total_executions: u64 = all_metrics.values().map(|m| m.execution_count).sum();
320
321        let total_duration: Duration = all_metrics.values().map(|m| m.total_duration).sum();
322
323        let avg_duration = if total_executions > 0 {
324            #[allow(clippy::cast_possible_truncation)]
325            let total_exec_u32 = total_executions as u32;
326            total_duration / total_exec_u32
327        } else {
328            Duration::ZERO
329        };
330
331        let slowest_hook = all_metrics
332            .iter()
333            .max_by_key(|(_, m)| m.max_duration)
334            .map(|(name, m)| (name.clone(), m.max_duration));
335
336        let most_frequent_hook = all_metrics
337            .iter()
338            .max_by_key(|(_, m)| m.execution_count)
339            .map(|(name, m)| (name.clone(), m.execution_count));
340
341        let slow_hooks = self.get_slow_hooks(slow_threshold);
342
343        PerformanceReport {
344            total_hooks,
345            total_executions,
346            avg_duration,
347            slowest_hook,
348            most_frequent_hook,
349            slow_hooks,
350        }
351    }
352}

359    fn test_performance_monitoring() {
360        let monitor = PerformanceMonitor::new();
361
362        // Record some executions
363        monitor.record_execution("hook1", Duration::from_millis(10));
364        monitor.record_execution("hook1", Duration::from_millis(20));
365        monitor.record_execution("hook1", Duration::from_millis(30));
366
367        monitor.record_execution("hook2", Duration::from_millis(5));
368        monitor.record_execution("hook2", Duration::from_millis(15));
369
370        // Get metrics

372        assert_eq!(metrics1.execution_count, 3);
373        assert_eq!(metrics1.min_duration, Duration::from_millis(10));
374        assert_eq!(metrics1.max_duration, Duration::from_millis(30));
375        assert_eq!(metrics1.avg_duration, Duration::from_millis(20));
376
377        let metrics2 = monitor.get_metrics("hook2").unwrap();
