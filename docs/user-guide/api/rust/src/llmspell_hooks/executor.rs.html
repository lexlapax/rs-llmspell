114    /// Execute a hook with protection
115    pub async fn execute_hook(
116        &self,
117        hook: &dyn Hook,
118        context: &mut HookContext,
119    ) -> Result<HookResult> {
120        // OPTIMIZATION: Cache metadata to avoid repeated calls
121        let metadata = hook.metadata();
122        let hook_name = &metadata.name; // Use reference instead of clone
123
124        // Check if hook should execute
125        if !hook.should_execute(context) {
126            return Ok(HookResult::Skipped("Hook conditions not met".to_string()));
127        }
128
129        // OPTIMIZATION: Get config and circuit breaker in single operation
130        let (hook_config, breaker_opt) = {
131            let configs = self.hook_configs.read();
132            let config = configs.get(hook_name).cloned().unwrap_or_default();
133            let breaker = if self.config.enable_circuit_breaker && config.use_circuit_breaker {
134                Some(self.get_circuit_breaker(hook_name, &config))
135            } else {
136                None
137            };
138            (config, breaker)
139        };
140
141        // Check circuit breaker if enabled
142        if let Some(ref breaker) = breaker_opt {
143            if !breaker.can_execute() {
144                warn!("Circuit breaker open for hook: {}", hook_name);
145                return Ok(HookResult::Skipped(format!(
146                    "Circuit breaker open for hook: {}",
147                    hook_name
148                )));
149            }
150        }
151
152        // OPTIMIZATION: Combine timer start and instant measurement
153        let (timer, start) = if self.config.enable_performance_monitoring {
154            let start = Instant::now();
155            let timer = Some(self.performance_monitor.start_execution(hook_name));
156            (timer, start)
157        } else {
158            (None, Instant::now())
159        };
160
161        // Create hook execution event for correlation tracking
162        let execution_event = if let Some(ref tracker) = self.correlation_tracker {
163            let event_data = serde_json::json!({
164                "hook_name": hook_name,
165                "hook_language": metadata.language,
166                "component_id": context.component_id.name,
167                "component_type": context.component_id.component_type,
168                "hook_point": context.point,
169                "operation": "hook_execution_start"
170            });
171
172            let event = UniversalEvent::new(
173                "hook.execution.start",
174                event_data,
175                llmspell_events::Language::Rust,
176            )
177            .with_correlation_id(context.correlation_id)
178            .with_source("hook_executor")
179            .with_tag("hook_execution");
180
181            tracker.track_event(event.clone());
182            Some(event)
183        } else {
184            None
185        };
186
187        // Execute the hook
188        let result = hook.execute(context).await;
189
190        let duration = start.elapsed();
191
192        // Create hook completion event and link to start event
193        if let Some(ref tracker) = self.correlation_tracker {
194            let success = result.is_ok();
195            let event_data = serde_json::json!({
196                "hook_name": hook_name,
197                "success": success,
198                "duration_ms": duration.as_millis(),
199                "operation": "hook_execution_complete"
200            });
201
202            let completion_event = UniversalEvent::new(
203                "hook.execution.complete",
204                event_data,
205                llmspell_events::Language::Rust,
206            )
207            .with_correlation_id(context.correlation_id)
208            .with_source("hook_executor")
209            .with_tag("hook_execution");
210
211            tracker.track_event(completion_event.clone());
212
213            // Create link between start and completion events
214            if let Some(start_event) = execution_event {
215                let link = EventLink::new(
216                    start_event.id,
217                    completion_event.id,
218                    EventRelationship::PartOf,
219                )
220                .with_strength(1.0)
221                .with_metadata("relationship", "execution_pair")
222                .with_metadata("hook_name", hook_name.to_string());
223
224                tracker.add_link(link);
225            }
226        }
227
228        // Complete performance tracking
229        if let Some(timer) = timer {
230            timer.complete();
231        }
232
233        // OPTIMIZATION: Update circuit breaker using cached reference
234        if let Some(breaker) = breaker_opt {
235            match &result {
236                Ok(_) => breaker.record_success(duration),
237                Err(e) => breaker.record_failure(e),
238            }
239        }
240
241        // OPTIMIZATION: Pre-compute timeout check
242        let timeout = hook_config
243            .timeout
244            .unwrap_or(self.config.max_execution_time);
245        if duration > timeout {
246            warn!(
247                "Hook {} execution took {:?}, exceeding timeout of {:?}",
248                hook_name, duration, timeout
249            );
250        }
251
252        // Persist hook execution if enabled
253        if self.config.enable_persistence && hook_config.persist_executions {
254            if let Some(ref _persistence_manager) = self.persistence_manager {
255                // For now, we can't persist hooks that don't implement ReplayableHook
256                // This would require all hooks to implement ReplayableHook or a different approach
257                // TODO: Consider alternative persistence strategies
258                /*
259                if let Ok(ref hook_result) = result {
260                        // Create hook metadata
261                        let mut metadata = HookMetadata::new(
262                            hook_name.clone(),
263                            context.component_id.component_type.clone(),
264                            context.component_id.name.clone(),
265                        );

268                        metadata.retention_priority = hook_config.retention_priority;

318    /// Execute multiple hooks in sequence
319    pub async fn execute_hooks(
320        &self,
321        hooks: &[Arc<dyn Hook>],
322        context: &mut HookContext,
323    ) -> Result<Vec<HookResult>> {
324        let mut results = Vec::with_capacity(hooks.len());
325
326        // Create batch execution event for correlation tracking
327        if let Some(ref tracker) = self.correlation_tracker {
328            let event_data = serde_json::json!({
329                "hook_count": hooks.len(),
330                "component_id": context.component_id.name,
331                "component_type": context.component_id.component_type,
332                "hook_point": context.point,
333                "operation": "batch_hook_execution_start"
334            });
335
336            let batch_event = UniversalEvent::new(
337                "hook.batch_execution.start",
338                event_data,
339                llmspell_events::Language::Rust,
340            )
341            .with_correlation_id(context.correlation_id)
342            .with_source("hook_executor")
343            .with_tag("batch_execution");
344
345            tracker.track_event(batch_event);
346        }
347
348        for hook in hooks {
349            let result = self.execute_hook(hook.as_ref(), context).await?;
350
351            // Check if we should stop execution
352            match &result {
353                HookResult::Cancel(reason) => {
354                    info!("Hook execution cancelled: {}", reason);
355                    results.push(result);
356                    break;
357                }
358                HookResult::Replace(_) => {
359                    // Replace stops further execution
360                    results.push(result);
361                    break;
362                }
363                _ => {
364                    results.push(result);
365                }
366            }
367        }
368
369        // Create batch completion event for correlation tracking
370        if let Some(ref tracker) = self.correlation_tracker {
371            let event_data = serde_json::json!({
372                "hook_count": hooks.len(),
373                "results_count": results.len(),
374                "operation": "batch_hook_execution_complete"
375            });
376
377            let batch_completion_event = UniversalEvent::new(
378                "hook.batch_execution.complete",
379                event_data,
380                llmspell_events::Language::Rust,
381            )
382            .with_correlation_id(context.correlation_id)
383            .with_source("hook_executor")
384            .with_tag("batch_execution");
385
386            tracker.track_event(batch_completion_event);
387        }
388
389        Ok(results)
390    }

423    /// Check if overhead is within target
424    pub fn is_within_overhead_target(&self) -> bool {
425        let metrics = self.get_all_metrics();
426        if metrics.is_empty() {
427            return true;
428        }
429
430        let total_hook_time: Duration = metrics.values().map(|m| m.total_duration).sum();
431
432        let total_executions: u64 = metrics.values().map(|m| m.execution_count).sum();
433
434        if total_executions == 0 {
435            return true;
436        }
437
438        #[allow(clippy::cast_precision_loss)]
439        let avg_hook_time = total_hook_time.as_secs_f64() / total_executions as f64;
440        let overhead_ratio = avg_hook_time / self.config.max_execution_time.as_secs_f64();
441
442        overhead_ratio <= self.config.performance_overhead_target
443    }

494    /// Execute hooks for artifact events
495    pub async fn execute_artifact_hooks(
496        &self,
497        artifact_event: &llmspell_core::events::ArtifactEvent,
498        hooks: &[Arc<dyn Hook>],
499    ) -> Result<Vec<HookResult>> {
500        use crate::artifact_hooks::{add_artifact_to_context, event_to_hook_point};
501
502        // Convert artifact event to hook point
503        let hook_point = event_to_hook_point(artifact_event);
504
505        // Create context for artifact hooks
506        let component_id = crate::types::ComponentId::new(
507            crate::types::ComponentType::Custom("artifact_manager".to_string()),
508            artifact_event.source.to_string(),
509        );
510
511        let mut context = HookContext::new(hook_point, component_id);
512        if let Some(correlation_id) = artifact_event.metadata.correlation_id() {
513            // Parse UUID from string
514            if let Ok(uuid) = uuid::Uuid::parse_str(correlation_id) {
515                context = context.with_correlation_id(uuid);
516            }
517        }
518
519        // Add artifact-specific metadata
520        match &artifact_event.event_type {
521            llmspell_core::events::ArtifactEventType::Created(e) => {
522                add_artifact_to_context(&mut context, &e.artifact.id, "created");
523                context.insert_metadata("artifact_name".to_string(), e.artifact.name.clone());
524                context.insert_metadata(
525                    "artifact_type".to_string(),
526                    e.artifact.artifact_type.clone(),
527                );
528            }
529            llmspell_core::events::ArtifactEventType::Modified(e) => {
530                add_artifact_to_context(&mut context, &e.artifact_id, "modified");
531                context.insert_metadata(
532                    "modifications".to_string(),
533                    format!("{:?}", e.modifications),
534                );
535            }
536            llmspell_core::events::ArtifactEventType::Deleted(e) => {
537                add_artifact_to_context(&mut context, &e.artifact_id, "deleted");
538                context.insert_metadata("permanent".to_string(), e.permanent.to_string());
539            }
540            llmspell_core::events::ArtifactEventType::Validated(e) => {
541                add_artifact_to_context(&mut context, &e.artifact_id, "validated");
542                context.insert_metadata(
543                    "validation_type".to_string(),
544                    format!("{:?}", e.validation_type),
545                );
546                context.insert_metadata("valid".to_string(), e.results.valid.to_string());
547            }
548            llmspell_core::events::ArtifactEventType::ValidationFailed(e) => {
549                add_artifact_to_context(&mut context, &e.artifact_id, "validation_failed");
550                context.insert_metadata(
551                    "validation_type".to_string(),
552                    format!("{:?}", e.validation_type),
553                );
554                context.insert_metadata("failures".to_string(), format!("{:?}", e.failures));
555            }
556            llmspell_core::events::ArtifactEventType::Derived(e) => {
557                add_artifact_to_context(&mut context, &e.derived_artifact.id, "derived");
558                context.insert_metadata("parent_artifact".to_string(), e.parent_id.to_string());
559                context.insert_metadata(
560                    "derivation_type".to_string(),
561                    format!("{:?}", e.derivation_type),
562                );
563            }
564            llmspell_core::events::ArtifactEventType::Accessed(e) => {
565                add_artifact_to_context(&mut context, &e.artifact_id, "accessed");
566                context.insert_metadata("access_type".to_string(), format!("{:?}", e.access_type));
567            }
568            _ => {}
569        }
570
571        // Execute all hooks for this artifact event
572        self.execute_hooks(hooks, &mut context).await
573    }
