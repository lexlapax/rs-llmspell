4//! # LLMSpell Hooks
5//!
6//! A comprehensive hook and event system for rs-llmspell with cross-language support,
7//! automatic performance protection, and production-ready patterns.
8//!
9//! ## Features
10//!
11//! - **Cross-Language Support**: Hooks work across Lua, JavaScript, and Python
12//! - **Performance Protection**: Built-in CircuitBreaker prevents slow hooks
13//! - **Production Patterns**: Caching, rate limiting, retry, and cost tracking
14//! - **Event System**: High-performance event bus with backpressure handling
15//! - **Future Ready**: Prepared for persistence, distribution, and library mode
16//!
17//! ## Example
18//!
19//! ```rust,no_run
20//! use llmspell_hooks::{Hook, HookContext, HookResult, HookPoint, ComponentId, ComponentType};
21//! use async_trait::async_trait;
22//! use anyhow::Result;
23//!
24//! struct MyHook;
25//!
26//! #[async_trait]
27//! impl Hook for MyHook {
28//!     fn as_any(&self) -> &dyn std::any::Any {
29//!         self
30//!     }
31//!     
32//!     async fn execute(&self, context: &mut HookContext) -> Result<HookResult> {
33//!         println!("Hook executed at {:?}", context.point);
34//!         Ok(HookResult::Continue)
35//!     }
36//! }
37//! ```
38//!
39//! ## Performance Characteristics
40//!
41//! - **Hook Execution**: <1ms overhead with CircuitBreaker protection
42//! - **Event Bus Throughput**: >90K events/sec
43//! - **Circuit Breaker**: Opens after 50ms latency (configurable)
44//! - **Cache Hit Rate**: >95% for repeated patterns
45//! - **Rate Limiting**: Token bucket with configurable rates
46//!
47//! ## Integration Example
48//!
49//! ```rust,no_run
50//! use llmspell_hooks::{
51//!     HookRegistry, HookPoint, CachingHook, CachingConfig,
52//!     RateLimitHook, LoggingHook, MetricsHook
53//! };
54//! use std::time::Duration;
55//!
56//! # fn example() -> Result<(), Box<dyn std::error::Error>> {
57//! // Create registry with production-ready hooks
58//! let registry = HookRegistry::new();
59//!
60//! // Add caching for performance
61//! let mut cache_config = CachingConfig::default();
62//! cache_config.default_ttl = Duration::from_secs(60);
63//! cache_config.max_entries = 1000;
64//! let cache = CachingHook::with_config(cache_config);
65//!
66//! // Add rate limiting for cost control  
67//! let limiter = RateLimitHook::new()
68//!     .with_rate_per_second(100.0) // 100 requests per second
69//!     .with_burst(10);
70//!
71//! // Add logging hook
72//! let logger = LoggingHook::new();
73//!
74//! // Add metrics collection
75//! let metrics = MetricsHook::new();
76//!
77//! // Register hooks for specific hook points
78//! registry.register(HookPoint::BeforeAgentExecution, cache)?;
79//! registry.register(HookPoint::BeforeAgentExecution, limiter)?;
80//! registry.register(HookPoint::BeforeAgentExecution, logger)?;
81//! registry.register(HookPoint::AfterAgentExecution, metrics)?;
82//! # Ok(())
83//! # }
84//! ```
