22/// Replay request for a single hook execution
23#[derive(Debug, Clone, Serialize, Deserialize)]
24pub struct ReplayRequest {
25    /// Execution ID to replay
26    pub execution_id: Uuid,
27    /// Replay configuration
28    pub config: ReplayConfig,
29    /// Optional correlation ID for tracking
30    pub correlation_id: Option<Uuid>,
31}

33/// Replay response
34#[derive(Debug, Clone, Serialize, Deserialize)]
35pub struct ReplayResponse {
36    /// Replay result
37    pub result: ReplayResult,
38    /// Any warnings during replay
39    pub warnings: Vec<String>,
40}

42/// Enhanced replay manager
43pub struct ReplayManager {
44    /// Hook persistence manager
45    #[allow(dead_code)]
46    persistence_manager: Arc<HookPersistenceManager>,
47    /// Storage backend
48    storage_backend: Arc<dyn StorageBackend>,
49    /// Registered hooks for replay
50    hook_registry: Arc<RwLock<HashMap<String, Arc<dyn ReplayableHook>>>>,
51    /// Result comparator
52    comparator: HookResultComparator,
53    /// Replay scheduler
54    scheduler: Arc<ReplayScheduler>,
55    /// Active replays
56    active_replays: Arc<RwLock<HashMap<Uuid, ReplayStatus>>>,
57}

59/// Status of an active replay
60#[derive(Debug, Clone)]
61struct ReplayStatus {
62    #[allow(dead_code)]
63    request: ReplayRequest,
64    start_time: Instant,
65    status: ReplayState,
66}

68#[derive(Debug, Clone, PartialEq)]
69pub enum ReplayState {
70    Running,
71    Completed,
72    Failed(String),
73    Cancelled,
74}

97    /// Replay a single hook execution
98    pub async fn replay(&self, request: ReplayRequest) -> Result<ReplayResponse> {
99        let replay_id = Uuid::new_v4();
100
101        // Track active replay
102        {
103            let mut active = self.active_replays.write();
104            active.insert(
105                replay_id,
106                ReplayStatus {
107                    request: request.clone(),
108                    start_time: Instant::now(),
109                    status: ReplayState::Running,
110                },
111            );
112        }
113
114        // Execute replay
115        let result = self.execute_replay(replay_id, request).await;
116
117        // Update status
118        {
119            let mut active = self.active_replays.write();
120            if let Some(status) = active.get_mut(&replay_id) {
121                status.status = match &result {
122                    Ok(_) => ReplayState::Completed,
123                    Err(e) => ReplayState::Failed(e.to_string()),
124                };
125            }
126        }
127
128        result
129    }

131    /// Execute the actual replay
132    async fn execute_replay(
133        &self,
134        replay_id: Uuid,
135        request: ReplayRequest,
136    ) -> Result<ReplayResponse> {
137        let start_time = SystemTime::now();
138        let mut warnings = Vec::new();
139
140        // Load original execution
141        let original = self
142            .storage_backend
143            .load_execution(&request.execution_id)
144            .await?
145            .ok_or_else(|| anyhow::anyhow!("Execution not found: {}", request.execution_id))?;
146
147        // Get the hook
148        let hook = self
149            .get_hook(&original.hook_id)?
150            .ok_or_else(|| anyhow::anyhow!("Hook not registered: {}", original.hook_id))?;
151
152        // Deserialize and modify context
153        let mut context = self.deserialize_context(&hook, &original.hook_context)?;
154
155        // Apply parameter modifications
156        let applied_modifications =
157            self.apply_modifications(&mut context, &request.config.modifications, &mut warnings)?;
158
159        // Execute based on mode
160        let hook_result = match request.config.mode {
161            ReplayMode::Exact => {
162                // Execute with original context
163                self.execute_hook_with_timeout(&hook, &mut context, request.config.timeout)
164                    .await
165            }
166            ReplayMode::Modified => {
167                // Execute with modified context
168                self.execute_hook_with_timeout(&hook, &mut context, request.config.timeout)
169                    .await
170            }
171            ReplayMode::Simulate => {
172                // Don't actually execute, just return original result
173                let result: HookResult = serde_json::from_str(&original.result)
174                    .context("Failed to deserialize original result")?;
175                Ok(result)
176            }
177            ReplayMode::Debug => {
178                // Execute with debug logging enabled
179                debug!("Replay debug mode - Context: {:?}", context);
180                self.execute_hook_with_timeout(&hook, &mut context, request.config.timeout)
181                    .await
182            }
183        };
184
185        let duration = start_time.elapsed().unwrap_or_default();
186
187        // Compare results if requested
188        let comparison = if request.config.compare_results {
189            let original_result: HookResult = serde_json::from_str(&original.result)?;
190            Some(self.comparator.compare(
191                &original_result,
192                hook_result.as_ref().unwrap_or(&HookResult::Continue),
193            ))
194        } else {
195            None
196        };
197
198        // Build result
199        let result = ReplayResult {
200            replay_id,
201            original_execution_id: request.execution_id,
202            hook_name: original.hook_id.clone(),
203            start_time,
204            duration,
205            hook_result: hook_result.map_err(|e| e.to_string()),
206            comparison,
207            applied_modifications,
208            metadata: HashMap::new(),
209        };
210
211        Ok(ReplayResponse { result, warnings })
212    }

214    /// Apply parameter modifications to context
215    fn apply_modifications(
216        &self,
217        context: &mut HookContext,
218        modifications: &[ParameterModification],
219        warnings: &mut Vec<String>,
220    ) -> Result<Vec<ParameterModification>> {
221        let mut applied = Vec::new();
222
223        for modification in modifications {
224            if !modification.enabled {
225                continue;
226            }
227
228            match self.apply_single_modification(context, modification) {
229                Ok(()) => applied.push(modification.clone()),
230                Err(e) => {
231                    warnings.push(format!(
232                        "Failed to apply modification to {}: {}",
233                        modification.path, e
234                    ));
235                    if applied.is_empty() {
236                        // Fail if no modifications succeeded yet
237                        return Err(e);
238                    }
239                }
240            }
241        }
242
243        Ok(applied)
244    }

246    /// Apply a single parameter modification
247    fn apply_single_modification(
248        &self,
249        context: &mut HookContext,
250        modification: &ParameterModification,
251    ) -> Result<()> {
252        let parts: Vec<&str> = modification.path.split('.').collect();
253
254        match parts.as_slice() {
255            ["context", "data", key] => {
256                context
257                    .data
258                    .insert(key.to_string(), modification.value.clone());
259                Ok(())
260            }
261            ["context", "metadata", key] => {
262                if let Value::String(value) = &modification.value {
263                    context.metadata.insert(key.to_string(), value.clone());
264                    Ok(())
265                } else {
266                    Err(anyhow::anyhow!("Metadata values must be strings"))
267                }
268            }
269            _ => Err(anyhow::anyhow!(
270                "Unsupported modification path: {}",
271                modification.path
272            )),
273        }
274    }

276    /// Execute hook with timeout
277    async fn execute_hook_with_timeout(
278        &self,
279        hook: &Arc<dyn ReplayableHook>,
280        context: &mut HookContext,
281        timeout: Duration,
282    ) -> Result<HookResult> {
283        tokio::time::timeout(timeout, hook.execute(context))
284            .await
285            .context("Hook execution timed out")?
286    }

288    /// Batch replay multiple executions
289    pub async fn batch_replay(&self, request: BatchReplayRequest) -> Result<BatchReplayResponse> {
290        let start_time = Instant::now();
291        let mut results = Vec::new();
292        let mut success_count = 0;
293        let mut failure_count = 0;
294
295        if request.parallel && request.max_concurrent > 1 {
296            // Parallel execution
297            use futures::stream::{self, StreamExt};
298
299            let futures = request.executions.into_iter().map(|execution| {
300                let config = request.config.clone();
301                let execution_id = execution.execution_id;
302                async move {
303                    self.replay(ReplayRequest {
304                        execution_id,
305                        config,
306                        correlation_id: None,
307                    })
308                    .await
309                }
310            });
311
312            let mut stream = stream::iter(futures).buffer_unordered(request.max_concurrent);
313
314            while let Some(result) = stream.next().await {
315                match result {
316                    Ok(response) => {
317                        if response.result.hook_result.is_ok() {
318                            success_count += 1;
319                        } else {
320                            failure_count += 1;
321                        }
322                        results.push(response.result);
323                    }
324                    Err(e) => {
325                        failure_count += 1;
326                        error!("Batch replay error: {}", e);
327                        if request.config.stop_on_error {
328                            break;
329                        }
330                    }
331                }
332            }
333        } else {
334            // Sequential execution
335            for execution in request.executions {
336                match self
337                    .replay(ReplayRequest {
338                        execution_id: execution.execution_id,
339                        config: request.config.clone(),
340                        correlation_id: None,
341                    })
342                    .await
343                {
344                    Ok(response) => {
345                        if response.result.hook_result.is_ok() {
346                            success_count += 1;
347                        } else {
348                            failure_count += 1;
349                        }
350                        results.push(response.result);
351                    }
352                    Err(e) => {
353                        failure_count += 1;
354                        error!("Batch replay error: {}", e);
355                        if request.config.stop_on_error {
356                            break;
357                        }
358                    }
359                }
360            }
361        }
362
363        Ok(BatchReplayResponse {
364            results,
365            total_duration: start_time.elapsed(),
366            success_count,
367            failure_count,
368            metadata: HashMap::new(),
369        })
370    }

381    /// Get active replay status
382    pub fn get_active_replays(&self) -> Vec<(Uuid, ReplayState)> {
383        self.active_replays
384            .read()
385            .iter()
386            .map(|(id, status)| (*id, status.status.clone()))
387            .collect()
388    }

390    /// Cancel an active replay
391    pub fn cancel_replay(&self, replay_id: Uuid) -> Result<()> {
392        let mut active = self.active_replays.write();
393        if let Some(status) = active.get_mut(&replay_id) {
394            if status.status == ReplayState::Running {
395                status.status = ReplayState::Cancelled;
396                Ok(())
397            } else {
398                Err(anyhow::anyhow!("Replay is not running"))
399            }
400        } else {
401            Err(anyhow::anyhow!("Replay not found"))
402        }
403    }

405    /// Get a hook from the registry
406    fn get_hook(&self, hook_id: &str) -> Result<Option<Arc<dyn ReplayableHook>>> {
407        Ok(self.hook_registry.read().get(hook_id).cloned())
408    }

419    /// Clean up completed replays
420    pub fn cleanup_completed_replays(&self, older_than: Duration) {
421        let cutoff = Instant::now() - older_than;
422        self.active_replays.write().retain(|_, status| {
423            status.start_time > cutoff || status.status == ReplayState::Running
424        });
425    }
