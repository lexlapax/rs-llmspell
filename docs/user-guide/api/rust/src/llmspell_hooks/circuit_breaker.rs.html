11/// Circuit breaker state
12#[derive(Debug, Clone, Copy, PartialEq, Eq)]
13pub enum BreakerState {
14    /// Circuit is closed, normal operation
15    Closed,
16    /// Circuit is open, blocking calls
17    Open,
18    /// Circuit is half-open, testing recovery
19    HalfOpen,
20}
21
22/// Circuit breaker configuration
23#[derive(Debug, Clone)]
24pub struct BreakerConfig {
25    /// Maximum failures before opening circuit
26    pub failure_threshold: u32,
27    /// Success threshold to close circuit from half-open
28    pub success_threshold: u32,
29    /// Time window for failure counting
30    pub failure_window: Duration,
31    /// How long to stay open before trying half-open
32    pub open_duration: Duration,
33    /// Maximum consecutive slow operations
34    pub slow_call_threshold: u32,
35    /// Duration that defines a slow call
36    pub slow_call_duration: Duration,
37}
38
39impl Default for BreakerConfig {
40    fn default() -> Self {
41        Self {
42            // OPTIMIZATION: Reduced failure threshold for faster detection
43            failure_threshold: 3,
44            success_threshold: 2,
45            // OPTIMIZATION: Shorter window for faster recovery
46            failure_window: Duration::from_secs(30),
47            // OPTIMIZATION: Shorter open duration for faster recovery
48            open_duration: Duration::from_secs(15),
49            // OPTIMIZATION: Lower slow call threshold for <5% overhead target
50            slow_call_threshold: 2,
51            // OPTIMIZATION: Stricter slow call duration to maintain performance
52            slow_call_duration: Duration::from_millis(50),
53        }
54    }
55}

170/// Statistics for circuit breaker
171#[derive(Debug)]
172struct BreakerStats {
173    failures: AtomicU64,
174    successes: AtomicU64,
175    slow_calls: AtomicU64,
176    total_calls: AtomicU64,
177    last_failure_time: RwLock<Option<Instant>>,
178    state_changed_at: RwLock<Instant>,
179}
180
181impl Default for BreakerStats {
182    fn default() -> Self {
183        Self {
184            failures: AtomicU64::new(0),
185            successes: AtomicU64::new(0),
186            slow_calls: AtomicU64::new(0),
187            total_calls: AtomicU64::new(0),
188            last_failure_time: RwLock::new(None),
189            state_changed_at: RwLock::new(Instant::now()),
190        }
191    }
192}

194/// Circuit breaker for hook execution
195pub struct CircuitBreaker {
196    config: BreakerConfig,
197    state: Arc<RwLock<BreakerState>>,
198    stats: Arc<BreakerStats>,
199    name: String,
200}
201
202impl CircuitBreaker {
203    /// Create a new circuit breaker
204    pub fn new(name: String) -> Self {
205        Self::with_config(name, BreakerConfig::default())
206    }
207
208    /// Create with custom configuration
209    pub fn with_config(name: String, config: BreakerConfig) -> Self {
210        Self {
211            config,
212            state: Arc::new(RwLock::new(BreakerState::Closed)),
213            stats: Arc::new(BreakerStats::default()),
214            name,
215        }
216    }
217
218    /// Get current state
219    pub fn state(&self) -> BreakerState {
220        *self.state.read()
221    }
222
223    /// Check if circuit allows execution
224    pub fn can_execute(&self) -> bool {
225        let current_state = *self.state.read();
226
227        match current_state {
228            BreakerState::Closed => true,
229            BreakerState::Open => {
230                // Check if we should transition to half-open
231                let state_changed_at = *self.stats.state_changed_at.read();
232                if state_changed_at.elapsed() >= self.config.open_duration {
233                    self.transition_to(BreakerState::HalfOpen);
234                    true
235                } else {
236                    false
237                }
238            }
239            BreakerState::HalfOpen => true,
240        }
241    }
242
243    /// Record a successful execution
244    pub fn record_success(&self, duration: Duration) {
245        self.stats.successes.fetch_add(1, Ordering::Relaxed);
246        self.stats.total_calls.fetch_add(1, Ordering::Relaxed);
247
248        let current_state = *self.state.read();
249
250        // Check for slow call
251        if duration >= self.config.slow_call_duration {
252            self.stats.slow_calls.fetch_add(1, Ordering::Relaxed);
253
254            // Check if we should open due to slow calls
255            if current_state == BreakerState::Closed {
256                let slow_calls = self.stats.slow_calls.load(Ordering::Relaxed);
257                if slow_calls >= u64::from(self.config.slow_call_threshold) {
258                    self.transition_to(BreakerState::Open);
259                    return;
260                }
261            }
262        }
263
264        // Handle state transitions on success
265        if current_state == BreakerState::HalfOpen {
266            let successes = self.stats.successes.load(Ordering::Relaxed);
267            if successes >= u64::from(self.config.success_threshold) {
268                self.transition_to(BreakerState::Closed);
269            }
270        }
271    }
272
273    /// Record a failed execution
274    pub fn record_failure(&self, _error: &anyhow::Error) {
275        self.stats.failures.fetch_add(1, Ordering::Relaxed);
276        self.stats.total_calls.fetch_add(1, Ordering::Relaxed);
277        *self.stats.last_failure_time.write() = Some(Instant::now());
278
279        let current_state = *self.state.read();
280
281        match current_state {
282            BreakerState::Closed => {
283                // Check if we should open the circuit
284                let failures = self.stats.failures.load(Ordering::Relaxed);
285                if failures >= u64::from(self.config.failure_threshold) {
286                    self.transition_to(BreakerState::Open);
287                }
288            }
289            BreakerState::HalfOpen => {
290                // Any failure in half-open state reopens the circuit
291                self.transition_to(BreakerState::Open);
292            }
293            BreakerState::Open => {
294                // Already open, nothing to do
295            }
296        }
297    }
298
299    /// Transition to a new state
300    fn transition_to(&self, new_state: BreakerState) {
301        let mut state = self.state.write();
302        if *state != new_state {
303            info!(
304                "Circuit breaker '{}' transitioning from {:?} to {:?}",
305                self.name, *state, new_state
306            );
307
308            *state = new_state;
309            *self.stats.state_changed_at.write() = Instant::now();
310
311            // Reset counters on state change
312            if new_state == BreakerState::Closed {
313                self.stats.failures.store(0, Ordering::Relaxed);
314                self.stats.successes.store(0, Ordering::Relaxed);
315                self.stats.slow_calls.store(0, Ordering::Relaxed);
316            } else if new_state == BreakerState::HalfOpen {
317                self.stats.successes.store(0, Ordering::Relaxed);
318            }
319        }
320    }
321
322    /// Get circuit breaker statistics
323    pub fn stats(&self) -> CircuitBreakerStats {
324        CircuitBreakerStats {
325            state: self.state(),
326            failures: self.stats.failures.load(Ordering::Relaxed),
327            successes: self.stats.successes.load(Ordering::Relaxed),
328            slow_calls: self.stats.slow_calls.load(Ordering::Relaxed),
329            total_calls: self.stats.total_calls.load(Ordering::Relaxed),
330        }
331    }
332
333    /// Reset the circuit breaker
334    pub fn reset(&self) {
335        self.transition_to(BreakerState::Closed);
336        self.stats.failures.store(0, Ordering::Relaxed);
337        self.stats.successes.store(0, Ordering::Relaxed);
338        self.stats.slow_calls.store(0, Ordering::Relaxed);
339        self.stats.total_calls.store(0, Ordering::Relaxed);
340        *self.stats.last_failure_time.write() = None;
341    }
342}

354/// Circuit breaker manager for multiple breakers
355pub struct CircuitBreakerManager {
356    breakers: Arc<DashMap<String, Arc<CircuitBreaker>>>,
357    default_config: BreakerConfig,
358}
359
360impl CircuitBreakerManager {
361    /// Create a new manager
362    pub fn new() -> Self {
363        Self::with_config(BreakerConfig::default())
364    }
365
366    /// Create with default configuration
367    pub fn with_config(default_config: BreakerConfig) -> Self {
368        Self {
369            breakers: Arc::new(DashMap::new()),
370            default_config,
371        }
372    }
373
374    /// Get or create a circuit breaker
375    pub fn get_or_create(&self, name: &str) -> Arc<CircuitBreaker> {
376        self.breakers
377            .entry(name.to_string())
378            .or_insert_with(|| {
379                Arc::new(CircuitBreaker::with_config(
380                    name.to_string(),
381                    self.default_config.clone(),
382                ))
383            })
384            .clone()
385    }
386
387    /// Create a custom circuit breaker
388    pub fn create_custom(&self, name: &str, config: BreakerConfig) -> Arc<CircuitBreaker> {
389        let breaker = Arc::new(CircuitBreaker::with_config(name.to_string(), config));
390        self.breakers.insert(name.to_string(), breaker.clone());
391        breaker
392    }
393
394    /// Get all circuit breaker stats
395    pub fn all_stats(&self) -> Vec<(String, CircuitBreakerStats)> {
396        self.breakers
397            .iter()
398            .map(|entry| (entry.key().clone(), entry.value().stats()))
399            .collect()
400    }
401
402    /// Reset all circuit breakers
403    pub fn reset_all(&self) {
404        for breaker in self.breakers.iter() {
405            breaker.value().reset();
406        }
407    }
408}

418    use super::*;
419    use std::thread;
420    #[test]
421    fn test_circuit_breaker_states() {
422        let config = BreakerConfig {
423            failure_threshold: 2,
424            success_threshold: 2,
425            open_duration: Duration::from_millis(100),
426            ..Default::default()
427        };
428
429        let breaker = CircuitBreaker::with_config("test".to_string(), config);
430
431        // Initially closed
432        assert_eq!(breaker.state(), BreakerState::Closed);
433        assert!(breaker.can_execute());
434
435        // Record failures to open circuit
436        breaker.record_failure(&anyhow::anyhow!("error 1"));
437        assert_eq!(breaker.state(), BreakerState::Closed);
438
439        breaker.record_failure(&anyhow::anyhow!("error 2"));
440        assert_eq!(breaker.state(), BreakerState::Open);
441        assert!(!breaker.can_execute());
442
443        // Wait for open duration
444        thread::sleep(Duration::from_millis(150));
445
446        // Should transition to half-open
447        assert!(breaker.can_execute());
448        assert_eq!(breaker.state(), BreakerState::HalfOpen);
449
450        // Success in half-open
451        breaker.record_success(Duration::from_millis(10));
452        breaker.record_success(Duration::from_millis(10));
453        assert_eq!(breaker.state(), BreakerState::Closed);
454    }
