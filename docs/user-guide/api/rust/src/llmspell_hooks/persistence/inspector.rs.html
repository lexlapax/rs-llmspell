101    /// Query executions based on filters
102    pub async fn query_executions(
103        &self,
104        query: InspectionQuery,
105    ) -> Result<Vec<SerializedHookExecution>> {
106        let mut results = Vec::new();
107
108        // Start with correlation ID if provided
109        if let Some(correlation_id) = query.correlation_id {
110            let executions = self
111                .storage_backend
112                .load_executions_by_correlation(&correlation_id)
113                .await?;
114            results.extend(executions);
115        } else {
116            // For broader queries, we'd need to implement a search method in storage
117            // For now, return empty
118            return Ok(Vec::new());
119        }
120
121        // Apply filters
122        results = self.apply_filters(results, &query);
123
124        // Apply limit
125        if let Some(limit) = query.limit {
126            results.truncate(limit);
127        }
128
129        Ok(results)
130    }

132    /// Analyze executions
133    pub async fn analyze_executions(
134        &self,
135        executions: &[SerializedHookExecution],
136    ) -> Result<ExecutionAnalysis> {
137        let total_executions = executions.len();
138        let mut unique_hooks = HashSet::new();
139        let mut execution_by_hook = HashMap::new();
140        let execution_by_point = HashMap::new();
141        let mut total_duration = Duration::from_secs(0);
142        let mut durations_by_hook: HashMap<String, Vec<Duration>> = HashMap::new();
143        let mut result_counts = HashMap::new();
144        let mut hourly_dist = HashMap::new();
145        let mut daily_dist = HashMap::new();
146
147        for execution in executions {
148            // Track unique hooks
149            unique_hooks.insert(execution.hook_id.clone());
150
151            // Count by hook
152            *execution_by_hook
153                .entry(execution.hook_id.clone())
154                .or_insert(0) += 1;
155
156            // Track durations
157            total_duration += execution.duration;
158            durations_by_hook
159                .entry(execution.hook_id.clone())
160                .or_default()
161                .push(execution.duration);
162
163            // Count result types
164            let result_type = execution.result.split("::").last().unwrap_or("Unknown");
165            *result_counts.entry(result_type.to_string()).or_insert(0) += 1;
166
167            // Time distribution (would need proper date parsing in real implementation)
168            if let Ok(duration_since_epoch) =
169                execution.timestamp.duration_since(SystemTime::UNIX_EPOCH)
170            {
171                let hours = (duration_since_epoch.as_secs() / 3600) % 24;
172                let days = duration_since_epoch.as_secs() / 86400;
173                #[allow(clippy::cast_possible_truncation)]
174                let hours_u32 = hours as u32;
175                #[allow(clippy::cast_possible_truncation)]
176                let days_u32 = days as u32;
177                *hourly_dist.entry(hours_u32).or_insert(0) += 1;
178                *daily_dist.entry(days_u32).or_insert(0) += 1;
179            }
180        }
181
182        // Calculate average duration
183        let average_duration = if total_executions > 0 {
184            #[allow(clippy::cast_possible_truncation)]
185            let total_exec_u32 = total_executions as u32;
186            total_duration / total_exec_u32
187        } else {
188            Duration::from_secs(0)
189        };
190
191        // Find slowest hooks
192        let mut slowest_hooks: Vec<(String, Duration)> = durations_by_hook
193            .iter()
194            .map(|(hook, durations)| {
195                #[allow(clippy::cast_possible_truncation)]
196                let len_u32 = durations.len() as u32;
197                let avg = durations.iter().sum::<Duration>() / len_u32;
198                (hook.clone(), avg)
199            })
200            .collect();
201        slowest_hooks.sort_by(|a, b| b.1.cmp(&a.1));
202        slowest_hooks.truncate(10);
203
204        // Calculate rates
205        #[allow(clippy::cast_precision_loss)]
206        let total_exec_f64 = total_executions as f64;
207        #[allow(clippy::cast_precision_loss)]
208        #[allow(clippy::cast_lossless)]
209        let cancel_count_f64 = *result_counts.get("Cancel").unwrap_or(&0) as f64;
210        let error_rate = cancel_count_f64 / total_exec_f64;
211
212        #[allow(clippy::cast_lossless)]
213        let modified_count_f64 = *result_counts.get("Modified").unwrap_or(&0) as f64;
214        let modification_rate = modified_count_f64 / total_exec_f64;
215
216        #[allow(clippy::cast_lossless)]
217        let cancel_count_f64_2 = *result_counts.get("Cancel").unwrap_or(&0) as f64;
218        let cancellation_rate = cancel_count_f64_2 / total_exec_f64;
219
220        // Find peak times
221        let peak_hour = hourly_dist
222            .iter()
223            .max_by_key(|(_, count)| *count)
224            .map(|(hour, _)| *hour);
225        let peak_day = daily_dist
226            .iter()
227            .max_by_key(|(_, count)| *count)
228            .map(|(day, _)| *day);
229
230        Ok(ExecutionAnalysis {
231            total_executions,
232            unique_hooks,
233            execution_by_hook,
234            execution_by_point,
235            average_duration,
236            slowest_hooks,
237            error_rate,
238            modification_rate,
239            cancellation_rate,
240            time_distribution: TimeDistribution {
241                hourly: hourly_dist,
242                daily: daily_dist,
243                peak_hour,
244                peak_day,
245            },
246        })
247    }

249    /// Detect execution patterns
250    pub async fn detect_patterns(
251        &self,
252        executions: &[SerializedHookExecution],
253    ) -> Result<Vec<ExecutionPattern>> {
254        let mut patterns = Vec::new();
255
256        // Detect sequential patterns
257        if let Some(pattern) = self.detect_sequential_pattern(executions) {
258            patterns.push(pattern);
259        }
260
261        // Detect error cascades
262        if let Some(pattern) = self.detect_error_cascade(executions) {
263            patterns.push(pattern);
264        }
265
266        // Detect performance bottlenecks
267        if let Some(pattern) = self.detect_bottlenecks(executions) {
268            patterns.push(pattern);
269        }
270
271        Ok(patterns)
272    }

274    /// Compare two sets of executions
275    pub async fn compare_executions(
276        &self,
277        set_a: &[SerializedHookExecution],
278        set_b: &[SerializedHookExecution],
279    ) -> Result<ComparisonResult> {
280        let analysis_a = self.analyze_executions(set_a).await?;
281        let analysis_b = self.analyze_executions(set_b).await?;
282
283        let duration_change = if analysis_a.average_duration > Duration::from_secs(0) {
284            (analysis_b.average_duration.as_secs_f64() / analysis_a.average_duration.as_secs_f64()
285                - 1.0)
286                * 100.0
287        } else {
288            0.0
289        };
290
291        let error_rate_change = analysis_b.error_rate - analysis_a.error_rate;
292
293        Ok(ComparisonResult {
294            #[allow(clippy::cast_possible_wrap)]
295            total_executions_change: set_b.len() as i64 - set_a.len() as i64,
296            #[allow(clippy::cast_possible_wrap)]
297            unique_hooks_change: analysis_b.unique_hooks.len() as i64
298                - analysis_a.unique_hooks.len() as i64,
299            average_duration_change_percent: duration_change,
300            error_rate_change,
301            new_hooks: analysis_b
302                .unique_hooks
303                .difference(&analysis_a.unique_hooks)
304                .cloned()
305                .collect(),
306            removed_hooks: analysis_a
307                .unique_hooks
308                .difference(&analysis_b.unique_hooks)
309                .cloned()
310                .collect(),
311        })
312    }

314    /// Apply filters to executions
315    fn apply_filters(
316        &self,
317        mut executions: Vec<SerializedHookExecution>,
318        query: &InspectionQuery,
319    ) -> Vec<SerializedHookExecution> {
320        // Filter by hook IDs
321        if let Some(ref hook_ids) = query.hook_ids {
322            executions.retain(|e| hook_ids.contains(&e.hook_id));
323        }
324
325        // Filter by time range
326        if let Some(ref time_range) = query.time_range {
327            executions.retain(|e| e.timestamp >= time_range.start && e.timestamp <= time_range.end);
328        }
329
330        // Filter by result type
331        if let Some(ref result_type) = query.result_type {
332            executions.retain(|e| match result_type {
333                ResultTypeFilter::Continue => e.result.contains("Continue"),
334                ResultTypeFilter::Modified => e.result.contains("Modified"),
335                ResultTypeFilter::Cancelled => e.result.contains("Cancel"),
336                ResultTypeFilter::Replaced => e.result.contains("Replace"),
337                ResultTypeFilter::Skipped => e.result.contains("Skipped"),
338                ResultTypeFilter::Any => true,
339            });
340        }
341
342        executions
343    }

345    /// Detect sequential execution pattern
346    fn detect_sequential_pattern(
347        &self,
348        executions: &[SerializedHookExecution],
349    ) -> Option<ExecutionPattern> {
350        if executions.len() < 3 {
351            return None;
352        }
353
354        // Look for repeated sequences
355        let mut sequences: HashMap<Vec<String>, usize> = HashMap::new();
356
357        for window in executions.windows(3) {
358            let sequence: Vec<String> = window.iter().map(|e| e.hook_id.clone()).collect();
359            *sequences.entry(sequence).or_insert(0) += 1;
360        }
361
362        // Find most common sequence
363        if let Some((sequence, count)) = sequences.iter().max_by_key(|(_, count)| *count) {
364            if *count > 1 {
365                return Some(ExecutionPattern {
366                    pattern_type: PatternType::Sequential,
367                    occurrences: *count,
368                    hook_sequence: sequence.clone(),
369                    average_interval: Duration::from_secs(0), // Would calculate actual interval
370                    components_involved: HashSet::new(),
371                });
372            }
373        }
374
375        None
376    }

378    /// Detect error cascade pattern
379    fn detect_error_cascade(
380        &self,
381        executions: &[SerializedHookExecution],
382    ) -> Option<ExecutionPattern> {
383        let mut error_sequences = Vec::new();
384        let mut current_sequence = Vec::new();
385
386        for execution in executions {
387            if execution.result.contains("Cancel") || execution.result.contains("Error") {
388                current_sequence.push(execution.hook_id.clone());
389            } else if !current_sequence.is_empty() {
390                if current_sequence.len() > 2 {
391                    error_sequences.push(current_sequence.clone());
392                }
393                current_sequence.clear();
394            }
395        }
396
397        if !error_sequences.is_empty() {
398            let longest = error_sequences.iter().max_by_key(|s| s.len()).unwrap();
399            return Some(ExecutionPattern {
400                pattern_type: PatternType::ErrorCascade,
401                occurrences: error_sequences.len(),
402                hook_sequence: longest.clone(),
403                average_interval: Duration::from_secs(0),
404                components_involved: HashSet::new(),
405            });
406        }
407
408        None
409    }

411    /// Detect performance bottlenecks
412    fn detect_bottlenecks(
413        &self,
414        executions: &[SerializedHookExecution],
415    ) -> Option<ExecutionPattern> {
416        // Find hooks that consistently take longer than average
417        #[allow(clippy::cast_possible_truncation)]
418        let exec_len_u32 = executions.len() as u32;
419        let avg_duration = executions.iter().map(|e| e.duration).sum::<Duration>() / exec_len_u32;
420        let threshold = avg_duration * 3; // 3x average is considered a bottleneck
421
422        let slow_hooks: Vec<String> = executions
423            .iter()
424            .filter(|e| e.duration > threshold)
425            .map(|e| e.hook_id.clone())
426            .collect();
427
428        if !slow_hooks.is_empty() {
429            return Some(ExecutionPattern {
430                pattern_type: PatternType::PerformanceBottleneck,
431                occurrences: slow_hooks.len(),
432                hook_sequence: slow_hooks,
433                average_interval: Duration::from_secs(0),
434                components_involved: HashSet::new(),
435            });
436        }
437
438        None
439    }
