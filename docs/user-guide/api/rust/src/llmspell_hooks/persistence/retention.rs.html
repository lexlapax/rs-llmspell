11/// Retention policy configuration for hooks
12#[derive(Debug, Clone, Serialize, Deserialize)]
13pub struct RetentionPolicy {
14    /// Maximum age of hook executions to retain
15    pub max_age: Option<Duration>,
16    /// Maximum number of executions to retain per hook type
17    pub max_count: Option<usize>,
18    /// Minimum number to always keep regardless of age
19    pub min_count: usize,
20    /// Priority threshold - keep hooks with priority >= this value longer
21    pub priority_threshold: i32,
22    /// Additional time to retain high-priority hooks
23    pub priority_bonus_time: Duration,
24    /// Whether to archive hooks before deletion
25    pub archive_before_delete: bool,
26    /// Compression settings for archived hooks
27    pub compress_archives: bool,
28}
29
30impl Default for RetentionPolicy {
31    fn default() -> Self {
32        Self {
33            max_age: Some(Duration::from_secs(7 * 24 * 60 * 60)), // 7 days
34            max_count: Some(10000),
35            min_count: 100,
36            priority_threshold: 5,
37            priority_bonus_time: Duration::from_secs(7 * 24 * 60 * 60), // Extra 7 days
38            archive_before_delete: true,
39            compress_archives: true,
40        }
41    }
42}

85/// Manages retention policies for different hook types
86pub struct RetentionManager {
87    /// Per-hook-type retention policies
88    policies: RwLock<HashMap<String, RetentionPolicy>>,
89    /// Default policy for hooks without specific configuration
90    default_policy: RetentionPolicy,
91    /// Global retention statistics
92    stats: RwLock<RetentionStatistics>,
93}
94
95#[derive(Debug, Default)]
96pub struct RetentionStatistics {
97    total_cleaned: u64,
98    total_archived: u64,
99    last_cleanup_time: Option<std::time::SystemTime>,
100    space_reclaimed_bytes: u64,
101}
102
103impl Default for RetentionManager {
104    fn default() -> Self {
105        Self::new(RetentionPolicy::default())
106    }
107}
108
109impl RetentionManager {
110    /// Create a new retention manager with default policy
111    pub fn new(default_policy: RetentionPolicy) -> Self {
112        Self {
113            policies: RwLock::new(HashMap::new()),
114            default_policy,
115            stats: RwLock::new(RetentionStatistics::default()),
116        }
117    }
118
119    /// Set retention policy for a specific hook type
120    pub fn set_policy(&self, hook_id: &str, policy: RetentionPolicy) {
121        self.policies.write().insert(hook_id.to_string(), policy);
122    }
123
124    /// Get retention policy for a hook type
125    pub fn get_policy(&self, hook_id: &str) -> RetentionPolicy {
126        self.policies
127            .read()
128            .get(hook_id)
129            .cloned()
130            .unwrap_or_else(|| self.default_policy.clone())
131    }
132
133    /// Apply retention policies to a specific hook type
134    pub async fn apply_retention_policies(&self, hook_id: &str) -> Result<()> {
135        let policy = self.get_policy(hook_id);
136
137        info!(
138            hook_id = hook_id,
139            max_age_secs = ?policy.max_age.map(|d| d.as_secs()),
140            max_count = ?policy.max_count,
141            "Applying retention policy"
142        );
143
144        // In a real implementation, this would:
145        // 1. Query hook executions for the given hook_id
146        // 2. Apply age-based retention
147        // 3. Apply count-based retention
148        // 4. Archive if configured
149        // 5. Delete or compress as needed
150
151        // Update statistics
152        let mut stats = self.stats.write();
153        stats.last_cleanup_time = Some(std::time::SystemTime::now());
154
155        Ok(())
156    }
157
158    /// Apply retention policies to all hook types
159    pub async fn apply_all_retention_policies(&self) -> Result<()> {
160        let hook_ids: Vec<String> = {
161            let policies = self.policies.read();
162            policies.keys().cloned().collect()
163        };
164
165        for hook_id in hook_ids {
166            self.apply_retention_policies(&hook_id)
167                .await
168                .context(format!("Failed to apply retention for hook: {}", hook_id))?;
169        }
170
171        Ok(())
172    }
173
174    /// Check if a hook execution should be retained based on age and priority
175    pub fn should_retain(&self, hook_id: &str, age: Duration, priority: i32) -> bool {
176        let policy = self.get_policy(hook_id);
177
178        // Check age-based retention
179        if let Some(max_age) = policy.max_age {
180            let effective_max_age = if priority >= policy.priority_threshold {
181                max_age + policy.priority_bonus_time
182            } else {
183                max_age
184            };
185
186            if age > effective_max_age {
187                debug!(
188                    hook_id = hook_id,
189                    age_secs = age.as_secs(),
190                    max_age_secs = effective_max_age.as_secs(),
191                    "Hook execution too old"
192                );
193                return false;
194            }
195        }
196
197        true
198    }
199
200    /// Get retention statistics
201    pub fn get_statistics(&self) -> RetentionStatistics {
202        self.stats.read().clone()
203    }
204
205    /// Clear all custom policies (revert to default)
206    pub fn clear_policies(&self) {
207        self.policies.write().clear();
208    }
209}

222#[cfg(test)]
223mod tests {
224    use super::*;
225    #[test]
226    fn test_retention_policy_presets() {
227        let short = RetentionPolicy::short_term();
228        assert_eq!(short.max_age, Some(Duration::from_secs(24 * 60 * 60)));
229        assert_eq!(short.max_count, Some(1000));
230
231        let long = RetentionPolicy::long_term();
232        assert_eq!(long.max_age, Some(Duration::from_secs(30 * 24 * 60 * 60)));
233        assert_eq!(long.max_count, Some(100000));
234
235        let all = RetentionPolicy::retain_all();
236        assert_eq!(all.max_age, None);
237        assert_eq!(all.max_count, None);
238    }
239    #[test]
240    fn test_retention_manager() {
241        let manager = RetentionManager::default();
242
243        // Set custom policy
244        let custom_policy = RetentionPolicy::short_term();
245        manager.set_policy("test_hook", custom_policy.clone());
246
247        // Retrieve policy
248        let retrieved = manager.get_policy("test_hook");
249        assert_eq!(retrieved.max_age, custom_policy.max_age);
250
251        // Default policy for unknown hook
252        let default = manager.get_policy("unknown_hook");
253        assert_eq!(default.max_age, Some(Duration::from_secs(7 * 24 * 60 * 60)));
254    }
255    #[test]
256    fn test_should_retain() {
257        let manager = RetentionManager::default();
258
259        // Test with default policy
260        assert!(manager.should_retain(
261            "test_hook",
262            Duration::from_secs(3 * 24 * 60 * 60), // 3 days
263            0,                                     // Low priority
264        ));
265
266        assert!(!manager.should_retain(
267            "test_hook",
268            Duration::from_secs(10 * 24 * 60 * 60), // 10 days
269            0,                                      // Low priority
270        ));
271
272        // High priority gets bonus time
273        assert!(manager.should_retain(
274            "test_hook",
275            Duration::from_secs(10 * 24 * 60 * 60), // 10 days
276            10,                                     // High priority
277        ));
278    }
