12/// Options for replaying hook executions
13#[derive(Debug, Clone, Serialize, Deserialize)]
14pub struct ReplayOptions {
15    /// Whether to modify parameters during replay
16    pub modify_parameters: bool,
17    /// Custom parameters to use during replay
18    pub custom_parameters: Option<serde_json::Value>,
19    /// Whether to simulate the original timing
20    pub simulate_timing: bool,
21    /// Whether to skip side effects
22    pub dry_run: bool,
23    /// Maximum executions to replay
24    pub max_executions: Option<usize>,
25    /// Filter by hook types
26    pub hook_type_filter: Option<Vec<String>>,
27}
28
29impl Default for ReplayOptions {
30    fn default() -> Self {
31        Self {
32            modify_parameters: false,
33            custom_parameters: None,
34            simulate_timing: false,
35            dry_run: true,
36            max_executions: None,
37            hook_type_filter: None,
38        }
39    }
40}

42/// Hook replay engine for reconstructing executions
43pub struct HookReplayEngine {
44    /// Replay statistics
45    stats: ReplayStatistics,
46}
47
48#[derive(Debug, Default)]
49struct ReplayStatistics {
50    total_replayed: u64,
51    successful_replays: u64,
52    failed_replays: u64,
53    total_replay_time: Duration,
54}
55
56impl HookReplayEngine {
57    /// Create a new replay engine
58    pub fn new() -> Self {
59        Self {
60            stats: ReplayStatistics::default(),
61        }
62    }
63
64    /// Replay a single hook execution
65    pub async fn replay_execution(
66        &mut self,
67        hook: &dyn ReplayableHook,
68        execution: &SerializedHookExecution,
69        options: &ReplayOptions,
70    ) -> Result<HookResult> {
71        info!(
72            hook_id = execution.hook_id,
73            execution_id = %execution.execution_id,
74            "Replaying hook execution"
75        );
76
77        let start = std::time::Instant::now();
78
79        // Deserialize the hook context
80        let mut context = hook
81            .deserialize_context(&execution.hook_context)
82            .context("Failed to deserialize hook context")?;
83
84        // Modify parameters if requested
85        if options.modify_parameters {
86            if let Some(ref params) = options.custom_parameters {
87                context.insert_data("custom_params".to_string(), params.clone());
88            }
89        }
90
91        // Simulate timing if requested
92        if options.simulate_timing {
93            debug!(
94                "Simulating original execution time: {:?}",
95                execution.duration
96            );
97            tokio::time::sleep(execution.duration).await;
98        }
99
100        // Execute the hook
101        let result = if options.dry_run {
102            // In dry run mode, return the original result
103            serde_json::from_str(&execution.result)
104                .context("Failed to deserialize original result")?
105        } else {
106            // Actually execute the hook
107            hook.execute(&mut context).await?
108        };
109
110        // Update statistics
111        self.stats.total_replayed += 1;
112        match &result {
113            HookResult::Continue | HookResult::Modified(_) | HookResult::Replace(_) => {
114                self.stats.successful_replays += 1;
115            }
116            _ => {}
117        }
118        self.stats.total_replay_time += start.elapsed();
119
120        Ok(result)
121    }
122
123    /// Replay multiple hook executions
124    pub async fn replay_executions(
125        &mut self,
126        hooks: &[(Box<dyn ReplayableHook>, SerializedHookExecution)],
127        options: &ReplayOptions,
128    ) -> Result<Vec<HookResult>> {
129        let mut results = Vec::new();
130        let max = options.max_executions.unwrap_or(hooks.len());
131
132        for (hook, execution) in hooks.iter().take(max) {
133            // Apply hook type filter if specified
134            if let Some(ref filter) = options.hook_type_filter {
135                if !filter.contains(&execution.hook_id) {
136                    continue;
137                }
138            }
139
140            match self
141                .replay_execution(hook.as_ref(), execution, options)
142                .await
143            {
144                Ok(result) => results.push(result),
145                Err(e) => {
146                    self.stats.failed_replays += 1;
147                    return Err(e.context(format!("Failed to replay hook: {}", execution.hook_id)));
148                }
149            }
150        }
151
152        Ok(results)
153    }
154
155    /// Get replay statistics
156    pub fn get_statistics(&self) -> (u64, u64, u64, Duration) {
157        (
158            self.stats.total_replayed,
159            self.stats.successful_replays,
160            self.stats.failed_replays,
161            self.stats.total_replay_time,
162        )
163    }
164
165    /// Reset statistics
166    pub fn reset_statistics(&mut self) {
167        self.stats = ReplayStatistics::default();
168    }
169}

177#[cfg(test)]
178mod tests {
179    use super::*;
180    #[test]
181    fn test_replay_options() {
182        let options = ReplayOptions::default();
183        assert!(!options.modify_parameters);
184        assert!(options.dry_run);
185        assert!(!options.simulate_timing);
186    }
187    #[test]
188    fn test_replay_engine_creation() {
189        let engine = HookReplayEngine::new();
190        let (total, success, failed, _) = engine.get_statistics();
191        assert_eq!(total, 0);
192        assert_eq!(success, 0);
193        assert_eq!(failed, 0);
194    }
