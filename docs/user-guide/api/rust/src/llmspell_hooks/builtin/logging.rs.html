14// OPTIMIZATION: Pre-allocate common strings to reduce allocations
15const BUILTIN_TAG: &str = "builtin";
16const LOGGING_TAG: &str = "logging";
17const HOOK_NAME: &str = "logging_hook";
18const HOOK_DESCRIPTION: &str = "Built-in hook for logging hook execution events";
19const HOOK_VERSION: &str = "1.0.0";
20const EMPTY_JSON: &str = "{}";
21const SERIALIZATION_ERROR: &str = "{serialization_error}";

23/// Configuration for the logging hook
24#[derive(Debug, Clone, Serialize, Deserialize)]
25pub struct LoggingConfig {
26    /// Minimum log level to capture
27    pub level: LogLevel,
28    /// Whether to include detailed context data
29    pub include_context_data: bool,
30    /// Whether to include metadata
31    pub include_metadata: bool,
32    /// Whether to log performance metrics
33    pub log_performance: bool,
34    /// Maximum data size to log (bytes)
35    pub max_data_size: usize,
36}
37
38impl Default for LoggingConfig {
39    fn default() -> Self {
40        Self {
41            level: LogLevel::Info,
42            include_context_data: true,
43            include_metadata: true,
44            log_performance: true,
45            max_data_size: 1024, // 1KB default limit
46        }
47    }
48}

50/// Log levels for the logging hook
51#[derive(Debug, Clone, Copy, Serialize, Deserialize, PartialEq, Eq, PartialOrd, Ord)]
52pub enum LogLevel {
53    Trace,
54    Debug,
55    Info,
56    Warn,
57    Error,
58}

60/// Built-in logging hook for comprehensive hook execution logging
61pub struct LoggingHook {
62    config: LoggingConfig,
63    metadata: HookMetadata,
64}
65
66impl LoggingHook {
67    /// Create a new logging hook with default configuration
68    pub fn new() -> Self {
69        Self {
70            config: LoggingConfig::default(),
71            metadata: HookMetadata {
72                // OPTIMIZATION: Use pre-allocated constants to reduce allocations
73                name: HOOK_NAME.to_owned(),
74                description: Some(HOOK_DESCRIPTION.to_owned()),
75                priority: Priority::LOW, // Run after other hooks
76                language: Language::Native,
77                tags: vec![BUILTIN_TAG.to_owned(), LOGGING_TAG.to_owned()],
78                version: HOOK_VERSION.to_owned(),
79            },
80        }
81    }
82
83    /// Create a new logging hook with custom configuration
84    pub fn with_config(config: LoggingConfig) -> Self {
85        Self {
86            config,
87            metadata: HookMetadata {
88                name: "LoggingHook".to_string(),
89                description: Some("Built-in hook for logging hook execution".to_string()),
90                priority: Priority::LOW,
91                language: Language::Native,
92                tags: vec!["builtin".to_string(), "logging".to_string()],
93                version: "1.0.0".to_string(),
94            },
95        }
96    }
97
98    /// Set the log level
99    pub fn with_level(mut self, level: LogLevel) -> Self {
100        self.config.level = level;
101        self
102    }
103
104    /// Enable or disable context data logging
105    pub fn with_context_data(mut self, include: bool) -> Self {
106        self.config.include_context_data = include;
107        self
108    }
109
110    /// Enable or disable performance logging
111    pub fn with_performance_logging(mut self, enable: bool) -> Self {
112        self.config.log_performance = enable;
113        self
114    }
115
116    /// Set maximum data size to log
117    pub fn with_max_data_size(mut self, size: usize) -> Self {
118        self.config.max_data_size = size;
119        self
120    }
121
122    /// Truncate data if it exceeds the maximum size
123    /// OPTIMIZATION: Use Cow to avoid unnecessary allocations
124    fn truncate_data<'a>(&self, data: &'a str) -> std::borrow::Cow<'a, str> {
125        if data.len() > self.config.max_data_size {
126            std::borrow::Cow::Owned(format!(
127                "{}... (truncated)",
128                &data[..self.config.max_data_size]
129            ))
130        } else {
131            std::borrow::Cow::Borrowed(data)
132        }
133    }
134
135    /// Format context data as a log-friendly string
136    /// OPTIMIZATION: Use constants and Cow to reduce allocations
137    fn format_context_data(&self, context: &HookContext) -> String {
138        if !self.config.include_context_data || context.data.is_empty() {
139            return EMPTY_JSON.to_owned();
140        }
141
142        match serde_json::to_string(&context.data) {
143            Ok(json) => self.truncate_data(&json).into_owned(),
144            Err(_) => SERIALIZATION_ERROR.to_owned(),
145        }
146    }
147
148    /// Format metadata as a log-friendly string
149    /// OPTIMIZATION: Use constant for empty case
150    fn format_metadata(&self, context: &HookContext) -> String {
151        if !self.config.include_metadata || context.metadata.is_empty() {
152            return EMPTY_JSON.to_owned();
153        }
154
155        format!("{:?}", context.metadata)
156    }
157}

159impl Default for LoggingHook {
160    fn default() -> Self {
161        Self::new()
162    }
163}

165#[async_trait]
166impl Hook for LoggingHook {
167    async fn execute(&self, context: &mut HookContext) -> Result<HookResult> {
168        let start_time = std::time::Instant::now();
169
170        // Format log message components
171        let context_data = self.format_context_data(context);
172        let metadata = self.format_metadata(context);
173
174        let base_message = format!(
175            "Hook execution: point={:?}, component={:?}:{}, language={:?}, correlation_id={}",
176            context.point,
177            context.component_id.component_type,
178            context.component_id.name,
179            context.language,
180            context.correlation_id
181        );
182
183        let detailed_message = if self.config.include_context_data || self.config.include_metadata {
184            format!(
185                "{}, data={}, metadata={}",
186                base_message, context_data, metadata
187            )
188        } else {
189            base_message
190        };
191
192        // Log based on configured level
193        match self.config.level {
194            LogLevel::Trace => trace!("{}", detailed_message),
195            LogLevel::Debug => debug!("{}", detailed_message),
196            LogLevel::Info => info!("{}", detailed_message),
197            LogLevel::Warn => warn!("{}", detailed_message),
198            LogLevel::Error => error!("{}", detailed_message),
199        }
200
201        // Log performance if enabled
202        if self.config.log_performance {
203            let duration = start_time.elapsed();
204            debug!(
205                "LoggingHook performance: {:?} for hook point {:?}",
206                duration, context.point
207            );
208        }
209
210        // Add logging metadata to context
211        context.insert_metadata("logged_at".to_string(), Utc::now().to_rfc3339());
212        context.insert_metadata(
213            "logging_hook_version".to_string(),
214            self.metadata.version.clone(),
215        );
216
217        Ok(HookResult::Continue)
218    }
219
220    fn metadata(&self) -> HookMetadata {
221        self.metadata.clone()
222    }
223
224    fn should_execute(&self, _context: &HookContext) -> bool {
225        // Always execute logging hook unless explicitly disabled
226        true
227    }
228
229    fn as_any(&self) -> &dyn std::any::Any {
230        self
231    }
232}

234#[async_trait]
235impl MetricHook for LoggingHook {
236    async fn record_pre_execution(&self, context: &HookContext) -> Result<()> {
237        if self.config.log_performance {
238            trace!("LoggingHook starting for hook point {:?}", context.point);
239        }
240        Ok(())
241    }
242
243    async fn record_post_execution(
244        &self,
245        context: &HookContext,
246        result: &HookResult,
247        duration: std::time::Duration,
248    ) -> Result<()> {
249        if self.config.log_performance {
250            debug!(
251                "LoggingHook completed for hook point {:?}: result={}, duration={:?}",
252                context.point,
253                result.description(),
254                duration
255            );
256        }
257        Ok(())
258    }
259}

261#[async_trait]
262impl ReplayableHook for LoggingHook {
263    fn is_replayable(&self) -> bool {
264        true
265    }
266
267    fn serialize_context(&self, ctx: &HookContext) -> Result<Vec<u8>> {
268        // Create a serializable version of the context with logging config
269        let mut context_data = ctx.data.clone();
270
271        // Add logging configuration to the context for replay
272        context_data.insert(
273            "_logging_config".to_string(),
274            serde_json::to_value(&self.config)?,
275        );
276
277        // Create a modified context with the config data
278        let mut replay_context = ctx.clone();
279        replay_context.data = context_data;
280
281        Ok(serde_json::to_vec(&replay_context)?)
282    }
283
284    fn deserialize_context(&self, data: &[u8]) -> Result<HookContext> {
285        let mut context: HookContext = serde_json::from_slice(data)?;
286
287        // Remove the logging config from the context data
288        // (it was only needed for serialization)
289        context.data.remove("_logging_config");
290
291        Ok(context)
292    }
293
294    fn replay_id(&self) -> String {
295        format!("{}:{}", self.metadata.name, self.metadata.version)
296    }
297}
