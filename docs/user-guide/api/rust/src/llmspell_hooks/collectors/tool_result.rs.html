32#[async_trait]
33impl Hook for ToolResultCollector {
34    async fn execute(&self, context: &mut HookContext) -> Result<HookResult> {
35        // Only process AfterToolExecution hooks
36        if context.point != HookPoint::AfterToolExecution {
37            return Ok(HookResult::Continue);
38        }
39
40        // Check if we should collect
41        if !self.should_collect(context).await {
42            return Ok(HookResult::Continue);
43        }
44
45        // Extract artifact data
46        match self.extract_artifact_data(context).await {
47            Ok(artifact_data) => {
48                // Add artifact data to context for session to process
49                context.insert_data(
50                    "collected_artifact".to_string(),
51                    json!({
52                        "type": "tool_result",
53                        "data": artifact_data,
54                    }),
55                );
56
57                // Add metadata
58                context.insert_metadata("artifact_collected".to_string(), "true".to_string());
59                context.insert_metadata("artifact_type".to_string(), "tool_result".to_string());
60
61                Ok(HookResult::Continue)
62            }
63            Err(e) => {
64                // Log error but don't fail the hook chain
65                tracing::warn!("Failed to collect tool result artifact: {}", e);
66                Ok(HookResult::Continue)
67            }
68        }
69    }
70
71    fn metadata(&self) -> HookMetadata {
72        HookMetadata {
73            name: "ToolResultCollector".to_string(),
74            version: "1.0.0".to_string(),
75            description: Some("Collects tool execution results as artifacts".to_string()),
76            priority: crate::Priority::NORMAL,
77            language: crate::Language::Native,
78            tags: vec!["collector".to_string(), "tool".to_string()],
79        }
80    }
81
82    fn should_execute(&self, context: &HookContext) -> bool {
83        context.point == HookPoint::AfterToolExecution
84    }
85
86    fn as_any(&self) -> &dyn std::any::Any {
87        self
88    }
89}

91#[async_trait]
92impl ArtifactCollector for ToolResultCollector {
93    async fn should_collect(&self, context: &HookContext) -> bool {
94        // Check if there's an operation result
95        let has_result = context
96            .operation
97            .as_ref()
98            .and_then(|op| op.result.as_ref())
99            .is_some();
100
101        // Check if it's an error and we should collect errors
102        let is_error = context
103            .operation
104            .as_ref()
105            .and_then(|op| op.error.as_ref())
106            .is_some();
107
108        if is_error && !self.config.collect_errors {
109            return false;
110        }
111
112        // Must have either result or error
113        if !has_result && !is_error {
114            return false;
115        }
116
117        // Check sampling
118        if !should_sample(&self.config) {
119            return false;
120        }
121
122        true
123    }
124
125    async fn extract_artifact_data(&self, context: &HookContext) -> Result<ArtifactData> {
126        let operation = context
127            .operation
128            .as_ref()
129            .ok_or_else(|| anyhow!("No operation context available"))?;
130
131        // Get tool name from component
132        let tool_name = &context.component_id.name;
133
134        // Determine if this is an error result
135        let is_error = operation.error.is_some();
136
137        // Extract content
138        let content = if let Some(result) = &operation.result {
139            serde_json::to_vec_pretty(result)?
140        } else if let Some(error) = &operation.error {
141            json!({
142                "error": error,
143                "tool": tool_name,
144                "timestamp": Utc::now().to_rfc3339(),
145            })
146            .to_string()
147            .into_bytes()
148        } else {
149            return Err(anyhow!("No result or error to collect"));
150        };
151
152        // Check size limits
153        if !is_size_acceptable(content.len(), &self.config) {
154            return Err(anyhow!(
155                "Content size {} is outside configured limits [{}, {}]",
156                content.len(),
157                self.config.min_size,
158                self.config.max_size
159            ));
160        }
161
162        // Generate artifact name
163        let timestamp = Utc::now().format("%Y%m%d_%H%M%S");
164        let name = if is_error {
165            format!("{}_error_{}.json", tool_name, timestamp)
166        } else {
167            format!("{}_result_{}.json", tool_name, timestamp)
168        };
169
170        // Build metadata
171        let metadata = json!({
172            "tool_name": tool_name,
173            "operation_type": operation.operation_type,
174            "operation_id": operation.operation_id.to_string(),
175            "is_error": is_error,
176            "collected_at": Utc::now().to_rfc3339(),
177            "parameters": operation.parameters,
178        });
179
180        // Build tags
181        let mut tags = self.config.auto_tags.clone();
182        tags.push("tool_result".to_string());
183        tags.push(tool_name.clone());
184        if is_error {
185            tags.push("error".to_string());
186        }
187
188        Ok(ArtifactData {
189            name,
190            content,
191            mime_type: "application/json".to_string(),
192            metadata,
193            tags,
194        })
195    }
196
197    fn artifact_type(&self) -> &str {
198        "tool_result"
199    }
200
201    fn config(&self) -> &CollectionConfig {
202        &self.config
203    }
204}

206#[cfg(test)]
207mod tests {
208    use super::*;
209    use crate::context::OperationContext;
210    use crate::types::{ComponentId, ComponentType};
211    use uuid::Uuid;
212    #[tokio::test]
213    async fn test_tool_result_collection() {
214        let config = CollectionConfig {
215            min_size: 10, // Lower minimum for tests
216            ..Default::default()
217        };
218        let collector = ToolResultCollector::with_config(config);
219        let component_id = ComponentId::new(ComponentType::Tool, "calculator".to_string());
220        let mut context = HookContext::new(HookPoint::AfterToolExecution, component_id);
221
222        // Add operation with result
223        context.operation = Some(OperationContext {
224            operation_type: "calculate".to_string(),
225            operation_id: Uuid::new_v4(),
226            parameters: json!({"expression": "2+2"}),
227            result: Some(json!({"value": 4})),
228            error: None,
229        });
230
231        // Should collect
232        assert!(collector.should_collect(&context).await);
233
234        // Extract artifact data
235        let artifact_data = collector.extract_artifact_data(&context).await.unwrap();
236        assert!(artifact_data.name.contains("calculator_result_"));
237        assert_eq!(artifact_data.mime_type, "application/json");
238        assert!(artifact_data.tags.contains(&"tool_result".to_string()));
239
240        // Execute hook
241        let result = collector.execute(&mut context).await.unwrap();
242        assert!(matches!(result, HookResult::Continue));
243
244        // Check that artifact was added to context
245        assert!(context.data.contains_key("collected_artifact"));
246    }
247    #[tokio::test]
248    async fn test_error_collection() {
249        let collector = ToolResultCollector::new();
250        let component_id = ComponentId::new(ComponentType::Tool, "failing_tool".to_string());
251        let mut context = HookContext::new(HookPoint::AfterToolExecution, component_id);
252
253        // Add operation with error
254        context.operation = Some(OperationContext {
255            operation_type: "process".to_string(),
256            operation_id: Uuid::new_v4(),
257            parameters: json!({"input": "bad data"}),
258            result: None,
259            error: Some("Invalid input format".to_string()),
260        });
261
262        // Should collect errors by default
263        assert!(collector.should_collect(&context).await);
264
265        // Extract artifact data
266        let artifact_data = collector.extract_artifact_data(&context).await.unwrap();
267        assert!(artifact_data.name.contains("failing_tool_error_"));
268        assert!(artifact_data.tags.contains(&"error".to_string()));
269    }
