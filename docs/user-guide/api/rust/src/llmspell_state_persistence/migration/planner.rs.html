16#[derive(Debug, Error)]
17pub enum MigrationPlannerError {
18    #[error("Planning failed: {reason}")]
19    PlanningFailed { reason: String },
20
21    #[error("Schema incompatible: {details}")]
22    IncompatibleSchema { details: String },
23
24    #[error("Migration path not found: {from} -> {to}")]
25    NoMigrationPath {
26        from: SemanticVersion,
27        to: SemanticVersion,
28    },
29}
30
31impl From<MigrationPlannerError> for StateError {
32    fn from(err: MigrationPlannerError) -> Self {
33        StateError::MigrationError(err.to_string())
34    }
35}

37/// Enhanced migration step with more detail than legacy MigrationStep
38#[derive(Debug, Clone, Serialize, Deserialize)]
39pub struct MigrationStep {
40    pub id: String,
41    pub from_version: SemanticVersion,
42    pub to_version: SemanticVersion,
43    pub migration_type: String,
44    pub description: String,
45    pub estimated_duration: Duration,
46    pub risk_level: crate::schema::compatibility::RiskLevel,
47    pub requires_backup: bool,
48    pub validation_rules: Vec<String>,
49}

51impl MigrationStep {
52    pub fn from_legacy(legacy: &LegacyMigrationStep) -> Self {
53        Self {
54            id: format!("{}_{}", legacy.from_version, legacy.to_version),
55            from_version: SemanticVersion::new(legacy.from_version, 0, 0),
56            to_version: SemanticVersion::new(legacy.to_version, 0, 0),
57            migration_type: legacy.migration_type.clone(),
58            description: legacy.description.clone(),
59            estimated_duration: Duration::from_secs(60), // Default estimate
60            risk_level: crate::schema::compatibility::RiskLevel::Medium,
61            requires_backup: legacy.migration_type.contains("breaking"),
62            validation_rules: vec!["basic_validation".to_string()],
63        }
64    }
65
66    pub fn to_legacy(&self) -> LegacyMigrationStep {
67        LegacyMigrationStep {
68            from_version: self.from_version.major,
69            to_version: self.to_version.major,
70            migration_type: self.migration_type.clone(),
71            description: self.description.clone(),
72        }
73    }
74}

76/// Enhanced migration plan that integrates with StateManager
77#[derive(Debug, Clone, Serialize, Deserialize)]
78pub struct MigrationPlan {
79    pub id: String,
80    pub from_version: SemanticVersion,
81    pub to_version: SemanticVersion,
82    pub steps: Vec<MigrationStep>,
83    pub estimated_duration: Duration,
84    pub total_risk_level: crate::schema::compatibility::RiskLevel,
85    pub requires_backup: bool,
86    pub compatibility_analysis: CompatibilityResult,
87    pub warnings: Vec<String>,
88    pub metadata: HashMap<String, serde_json::Value>,
89    pub created_at: std::time::SystemTime,
90}
91
92impl MigrationPlan {
93    pub fn from_schema_plan(
94        schema_plan: SchemaMigrationPlan,
95        compatibility: CompatibilityResult,
96    ) -> Self {
97        let steps: Vec<MigrationStep> = schema_plan
98            .steps
99            .iter()
100            .map(|step| MigrationStep {
101                id: format!("{}_{}", step.from_version, step.to_version),
102                from_version: SemanticVersion::new(step.from_version, 0, 0),
103                to_version: SemanticVersion::new(step.to_version, 0, 0),
104                migration_type: step.migration_type.clone(),
105                description: step.description.clone(),
106                estimated_duration: Duration::from_secs(60),
107                risk_level: schema_plan.risk_level.clone(),
108                requires_backup: schema_plan.requires_backup,
109                validation_rules: vec!["schema_validation".to_string()],
110            })
111            .collect();
112
113        Self {
114            id: uuid::Uuid::new_v4().to_string(),
115            from_version: schema_plan.from_version,
116            to_version: schema_plan.to_version,
117            steps,
118            estimated_duration: schema_plan.estimated_duration,
119            total_risk_level: schema_plan.risk_level,
120            requires_backup: schema_plan.requires_backup,
121            compatibility_analysis: compatibility,
122            warnings: schema_plan.warnings,
123            metadata: HashMap::new(),
124            created_at: std::time::SystemTime::now(),
125        }
126    }
127
128    pub fn is_safe(&self) -> bool {
129        self.total_risk_level <= crate::schema::compatibility::RiskLevel::Medium
130            && self.compatibility_analysis.compatible
131    }
132
133    pub fn has_breaking_changes(&self) -> bool {
134        !self.compatibility_analysis.breaking_changes.is_empty()
135    }
136
137    pub fn get_affected_fields(&self) -> Vec<String> {
138        self.compatibility_analysis
139            .field_changes
140            .keys()
141            .cloned()
142            .collect()
143    }
144}

152impl MigrationPlanner {
153    pub fn new() -> Self {
154        Self {
155            schema_planner: SchemaMigrationPlanner::new(),
156            schema_registry: SchemaRegistry::new(),
157        }
158    }

160    pub fn with_registry(schema_registry: SchemaRegistry) -> Self {
161        Self {

164        }

167    /// Register a schema for migration planning
168    pub fn register_schema(&mut self, schema: EnhancedStateSchema) -> StateResult<()> {
169        // Register with both the schema registry and planner
170        self.schema_registry
171            .register_schema(schema.clone(), None)
172            .map_err(|e| StateError::MigrationError(e.to_string()))?;
173
174        self.schema_planner.register_schema(schema);
175        Ok(())
176    }
177
178    /// Create a migration plan between versions
179    pub fn create_migration_plan(
180        &mut self,
181        from_version: &SemanticVersion,
182        to_version: &SemanticVersion,
183    ) -> StateResult<MigrationPlan> {
184        // Get schemas for compatibility analysis
185        let from_schema = self
186            .schema_registry
187            .get_schema(from_version)
188            .ok_or_else(|| MigrationPlannerError::PlanningFailed {
189                reason: format!("Source schema {} not found", from_version),
190            })?;
191
192        let to_schema = self.schema_registry.get_schema(to_version).ok_or_else(|| {
193            MigrationPlannerError::PlanningFailed {
194                reason: format!("Target schema {} not found", to_version),
195            }
196        })?;
197
198        // Analyze compatibility
199        let compatibility = CompatibilityChecker::check_compatibility(&from_schema, &to_schema);
200
201        // Create schema migration plan
202        let schema_plan = self
203            .schema_planner
204            .create_migration_plan(from_version, to_version)
205            .map_err(|e| MigrationPlannerError::PlanningFailed {
206                reason: e.to_string(),
207            })?;
208
209        // Convert to high-level migration plan
210        let migration_plan = MigrationPlan::from_schema_plan(schema_plan, compatibility);
211
212        // Validate the plan
213        self.validate_plan(&migration_plan)?;
214
215        Ok(migration_plan)
216    }
217
218    /// Validate a migration plan
219    pub fn validate_plan(&self, plan: &MigrationPlan) -> StateResult<()> {
220        // Check for empty plan
221        if plan.steps.is_empty() && plan.from_version != plan.to_version {
222            return Err(MigrationPlannerError::PlanningFailed {
223                reason: "Migration plan has no steps but versions differ".to_string(),
224            }
225            .into());
226        }
227
228        // Check version sequence
229        if !plan.steps.is_empty() {
230            let first_step = &plan.steps[0];
231            if first_step.from_version != plan.from_version {
232                return Err(MigrationPlannerError::PlanningFailed {
233                    reason: "First step doesn't match plan source version".to_string(),
234                }
235                .into());
236            }
237
238            let last_step = &plan.steps[plan.steps.len() - 1];
239            if last_step.to_version != plan.to_version {
240                return Err(MigrationPlannerError::PlanningFailed {
241                    reason: "Last step doesn't match plan target version".to_string(),
242                }
243                .into());
244            }
245
246            // Check step sequence
247            for i in 1..plan.steps.len() {
248                if plan.steps[i - 1].to_version != plan.steps[i].from_version {
249                    return Err(MigrationPlannerError::PlanningFailed {
250                        reason: format!("Step sequence broken between steps {} and {}", i - 1, i),
251                    }
252                    .into());
253                }
254            }
255        }
256
257        // Check for breaking changes without backup
258        if plan.has_breaking_changes() && !plan.requires_backup {
259            return Err(MigrationPlannerError::IncompatibleSchema {
260                details: "Breaking changes detected but backup not configured".to_string(),
261            }
262            .into());
263        }
264
265        Ok(())
266    }

276    /// Check if migration is possible between versions
277    pub fn is_migration_possible(
278        &self,
279        from_version: &SemanticVersion,
280        to_version: &SemanticVersion,
281    ) -> bool {
282        if let (Some(from_schema), Some(to_schema)) = (
283            self.schema_registry.get_schema(from_version),
284            self.schema_registry.get_schema(to_version),
285        ) {
286            CompatibilityChecker::is_compatible(&from_schema, &to_schema)
287                || from_version < to_version // Allow forward migration with plan
288        } else {
289            false
290        }
291    }

293    /// Get migration complexity estimate
294    pub fn estimate_complexity(
295        &self,
296        from_version: &SemanticVersion,
297        to_version: &SemanticVersion,
298    ) -> StateResult<MigrationComplexity> {
299        if let (Some(from_schema), Some(to_schema)) = (
300            self.schema_registry.get_schema(from_version),
301            self.schema_registry.get_schema(to_version),
302        ) {
303            let compatibility = CompatibilityChecker::check_compatibility(&from_schema, &to_schema);
304
305            #[allow(clippy::cast_possible_truncation)]
306            let field_changes_count = compatibility.field_changes.len() as u64;
307            let complexity = MigrationComplexity {
308                risk_level: compatibility.risk_level.clone(),
309                field_changes: compatibility.field_changes.len(),
310                breaking_changes: compatibility.breaking_changes.len(),
311                estimated_duration: Duration::from_secs(field_changes_count * 10 + 60),
312                requires_backup: compatibility.risk_level
313                    >= crate::schema::compatibility::RiskLevel::High,
314                complexity_score: self.calculate_complexity_score(&compatibility),
315            };
316
317            Ok(complexity)
318        } else {
319            Err(MigrationPlannerError::PlanningFailed {
320                reason: "One or both schemas not found".to_string(),
321            }
322            .into())
323        }
324    }
325
326    /// Calculate a numeric complexity score
327    fn calculate_complexity_score(&self, compatibility: &CompatibilityResult) -> u32 {
328        let mut score = 0u32;
329
330        // Base score for field changes
331        #[allow(clippy::cast_possible_truncation)]
332        let field_changes_u32 = compatibility.field_changes.len() as u32;
333        score += field_changes_u32 * 10;
334
335        // Penalty for breaking changes
336        score += compatibility.breaking_changes.len() as u32 * 50;
337
338        // Risk level multiplier
339        let risk_multiplier = match compatibility.risk_level {
340            crate::schema::compatibility::RiskLevel::Low => 1,
341            crate::schema::compatibility::RiskLevel::Medium => 2,
342            crate::schema::compatibility::RiskLevel::High => 4,
343            crate::schema::compatibility::RiskLevel::Critical => 8,
344        };
345
346        score * risk_multiplier
347    }

361/// Migration complexity assessment
362#[derive(Debug, Clone, Serialize, Deserialize)]
363pub struct MigrationComplexity {
364    pub risk_level: crate::schema::compatibility::RiskLevel,
365    pub field_changes: usize,
366    pub breaking_changes: usize,
367    pub estimated_duration: Duration,
368    pub requires_backup: bool,
369    pub complexity_score: u32,
370}
371
372impl MigrationComplexity {
373    pub fn is_simple(&self) -> bool {
374        self.complexity_score < 100
375            && self.risk_level <= crate::schema::compatibility::RiskLevel::Low
376    }
377
378    pub fn is_complex(&self) -> bool {
379        self.complexity_score > 500
380            || self.risk_level >= crate::schema::compatibility::RiskLevel::High
381    }
382}
