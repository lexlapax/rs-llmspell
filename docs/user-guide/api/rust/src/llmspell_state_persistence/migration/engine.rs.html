22#[derive(Debug, Error)]
23pub enum MigrationEngineError {
24    #[error("Migration failed: {reason}")]
25    MigrationFailed { reason: String },
26
27    #[error("Schema not found: {version}")]
28    SchemaNotFound { version: SemanticVersion },
29
30    #[error("Validation failed: {details}")]
31    ValidationFailed { details: String },
32
33    #[error("Rollback failed: {reason}")]
34    RollbackFailed { reason: String },
35
36    #[error("Migration timeout after {timeout:?}")]
37    Timeout { timeout: Duration },
38
39    #[error("Migration cancelled: {reason}")]
40    Cancelled { reason: String },
41}

96    /// Execute migration from one schema version to another
97    pub async fn migrate(
98        &self,
99        from_version: &SemanticVersion,
100        to_version: &SemanticVersion,
101        config: MigrationConfig,
102    ) -> StateResult<MigrationResult> {
103        let migration_id = Uuid::new_v4();
104        let correlation_id = Uuid::new_v4();
105        let event_builder = MigrationEventBuilder::new(migration_id);
106
107        info!(
108            "Starting migration {} from {} to {}",
109            migration_id, from_version, to_version
110        );
111
112        // Create migration plan
113        let plan = {
114            let mut planner = self.migration_planner.write();
115            let registry = self.schema_registry.read();
116
117            // Register schemas in planner if needed
118            if let Some(from_schema) = registry.get_schema(from_version) {
119                planner.register_schema((*from_schema).clone());
120            } else {
121                return Err(MigrationEngineError::SchemaNotFound {
122                    version: from_version.clone(),
123                }
124                .into());
125            }
126
127            if let Some(to_schema) = registry.get_schema(to_version) {
128                planner.register_schema((*to_schema).clone());
129            } else {
130                return Err(MigrationEngineError::SchemaNotFound {
131                    version: to_version.clone(),
132                }
133                .into());
134            }
135
136            planner
137                .create_migration_plan(from_version, to_version)
138                .map_err(|e| StateError::MigrationError(e.to_string()))?
139        };
140
141        // Create migration context
142        let total_steps = plan.steps.len();
143        let registry = self.schema_registry.read().clone();
144        let mut context = MigrationContext::new(config.clone(), registry, total_steps);
145        context.set_metadata("migration_id".to_string(), serde_json::json!(migration_id));
146        context.set_metadata(
147            "correlation_id".to_string(),
148            serde_json::json!(correlation_id),
149        );
150
151        // Track active migration
152        {
153            let mut active = self.active_migrations.write();
154            active.insert(migration_id, context.clone());
155        }
156
157        let mut result =
158            MigrationResult::new(from_version.clone(), to_version.clone(), total_steps);
159
160        // Execute pre-migration hooks
161        self.execute_migration_hooks(&plan, "pre_migration", correlation_id)
162            .await?;
163
164        // Emit migration started event using new event system
165        let migration_started_event = event_builder.migration_started(
166            from_version.clone(),
167            to_version.clone(),
168            total_steps,
169            config.dry_run,
170        );
171        self.emit_typed_migration_event(&migration_started_event, correlation_id)
172            .await?;
173
174        result.mark_in_progress();
175
176        // Execute migration steps
177        match self.execute_migration_plan(&plan, &mut context).await {
178            Ok(items_migrated) => {
179                let duration = context.elapsed_time();
180                result.mark_completed(items_migrated, duration);
181
182                // Execute post-migration hooks
183                self.execute_migration_hooks(&plan, "post_migration", correlation_id)
184                    .await?;
185
186                // Emit migration completed event using new event system
187                let migration_completed_event = event_builder.migration_completed(
188                    from_version.clone(),
189                    to_version.clone(),
190                    duration,
191                    items_migrated,
192                    total_steps,
193                );
194                self.emit_typed_migration_event(&migration_completed_event, correlation_id)
195                    .await?;
196
197                info!(
198                    "Migration {} completed successfully: {} items in {:?}",
199                    migration_id, items_migrated, duration
200                );
201            }
202            Err(e) => {
203                result.mark_failed(e.to_string());
204                result.add_error(e.to_string());
205
206                error!("Migration {} failed: {}", migration_id, e);
207
208                // Rollback if configured to do so
209                if config.rollback_on_error {
210                    warn!("Attempting rollback for migration {}", migration_id);
211                    match self.rollback_migration(&plan, &context).await {
212                        Ok(_) => {
213                            result.mark_rolled_back();
214                            info!("Migration {} rolled back successfully", migration_id);
215                        }
216                        Err(rollback_err) => {
217                            error!(
218                                "Rollback failed for migration {}: {}",
219                                migration_id, rollback_err
220                            );
221                            result.add_error(format!("Rollback failed: {}", rollback_err));
222                        }
223                    }
224                }
225
226                // Emit migration failed event using new event system
227                let migration_failed_event = event_builder.migration_failed(
228                    from_version.clone(),
229                    to_version.clone(),
230                    e.to_string(),
231                    result.items_migrated,
232                    config.rollback_on_error,
233                );
234                self.emit_typed_migration_event(&migration_failed_event, correlation_id)
235                    .await?;
236
237                return Err(StateError::MigrationError(e.to_string()));
238            }
239        }
240
241        // Remove from active migrations
242        {
243            let mut active = self.active_migrations.write();
244            active.remove(&migration_id);
245        }
246
247        Ok(result)
248    }

250    /// Execute the migration plan step by step
251    async fn execute_migration_plan(
252        &self,
253        plan: &MigrationPlan,
254        context: &mut MigrationContext,
255    ) -> Result<usize, MigrationEngineError> {
256        let mut total_items_migrated = 0;
257        let start_time = Instant::now();
258
259        for (step_index, step) in plan.steps.iter().enumerate() {
260            // Check timeout
261            if start_time.elapsed() > context.config.timeout {
262                return Err(MigrationEngineError::Timeout {
263                    timeout: context.config.timeout,
264                });
265            }
266
267            debug!(
268                "Executing migration step {}/{}: {}",
269                step_index + 1,
270                plan.steps.len(),
271                step.description
272            );
273
274            context.current_step = step_index + 1;
275
276            // Create transformation for this step
277            let transformation = super::transforms::StateTransformation::new(
278                format!("migration_step_{}", step_index),
279                step.description.clone(),
280                step.from_version,
281                step.to_version,
282            );
283
284            // Execute transformation
285            let items_migrated = self
286                .execute_transformation(&transformation, &context.config)
287                .await?;
288
289            total_items_migrated += items_migrated;
290
291            debug!(
292                "Step {}/{} completed: {} items migrated",
293                step_index + 1,
294                plan.steps.len(),
295                items_migrated
296            );
297        }
298
299        Ok(total_items_migrated)
300    }

302    /// Execute a single transformation step
303    async fn execute_transformation(
304        &self,
305        transformation: &super::transforms::StateTransformation,
306        config: &MigrationConfig,
307    ) -> Result<usize, MigrationEngineError> {
308        if config.dry_run {
309            debug!("Dry run: Skipping actual transformation");
310            return Ok(0);
311        }
312
313        let mut items_processed = 0;
314
315        // Get all state keys to migrate
316        let all_keys = self.storage_adapter.list_keys("").await.map_err(|e| {
317            MigrationEngineError::MigrationFailed {
318                reason: format!("Failed to list keys: {}", e),
319            }
320        })?;
321
322        // Process in batches
323        for chunk in all_keys.chunks(config.batch_size) {
324            for key in chunk {
325                // Skip non-state keys (like hook history)
326                if key.starts_with("hook_history:") || key.starts_with("agent_state:") {
327                    continue;
328                }
329
330                match self.transform_state_item(key, transformation).await {
331                    Ok(transformed) => {
332                        if transformed {
333                            items_processed += 1;
334                        }
335                    }
336                    Err(e) => {
337                        warn!("Failed to transform state item '{}': {}", key, e);
338                        if config.validation_level == ValidationLevel::Strict {
339                            return Err(MigrationEngineError::MigrationFailed {
340                                reason: format!("Transformation failed for key '{}': {}", key, e),
341                            });
342                        }
343                    }
344                }
345            }
346        }
347
348        Ok(items_processed)
349    }

351    /// Transform a single state item
352    async fn transform_state_item(
353        &self,
354        key: &str,
355        transformation: &super::transforms::StateTransformation,
356    ) -> StateResult<bool> {
357        // Load the current state item
358        let current_state: Option<SerializableState> = self.storage_adapter.load(key).await?;
359
360        if let Some(mut state) = current_state {
361            // Use the DataTransformer to transform the state
362            let transformer = super::transforms::DataTransformer::new();
363
364            match transformer.transform_state(&mut state, transformation) {
365                Ok(result) => {
366                    if result.success && result.fields_transformed > 0 {
367                        // Store the transformed state
368                        self.storage_adapter.store(key, &state).await?;
369                        Ok(true)
370                    } else {
371                        Ok(false)
372                    }
373                }
374                Err(e) => {
375                    warn!("State transformation failed for key '{}': {}", key, e);
376                    Ok(false)
377                }
378            }
379        } else {
380            Ok(false)
381        }
382    }

384    /// Execute migration hooks
385    async fn execute_migration_hooks(
386        &self,
387        plan: &MigrationPlan,
388        hook_point: &str,
389        correlation_id: Uuid,
390    ) -> StateResult<()> {
391        let component_id = ComponentId::new(
392            ComponentType::Custom("migration".to_string()),
393            "migration_engine".to_string(),
394        );
395
396        let mut hook_context =
397            HookContext::new(HookPoint::Custom(hook_point.to_string()), component_id);
398        hook_context = hook_context.with_correlation_id(correlation_id);
399
400        hook_context.insert_metadata("from_version".to_string(), plan.from_version.to_string());
401        hook_context.insert_metadata("to_version".to_string(), plan.to_version.to_string());
402        hook_context.insert_metadata("risk_level".to_string(), format!("{:?}", plan.risk_level));
403        hook_context.insert_metadata(
404            "requires_backup".to_string(),
405            plan.requires_backup.to_string(),
406        );
407        hook_context.insert_metadata("steps_count".to_string(), plan.steps.len().to_string());
408
409        // Execute hooks through the existing hook executor
410        // For now, we use an empty hooks list since hooks would be registered externally
411        let hooks: Vec<std::sync::Arc<dyn llmspell_hooks::Hook>> = vec![];
412        let start_time = Instant::now();
413
414        // Make context mutable for hook execution
415        let mut mutable_context = hook_context;
416
417        match self
418            .hook_executor
419            .execute_hooks(&hooks, &mut mutable_context)
420            .await
421        {
422            Ok(results) => {
423                let execution_time = start_time.elapsed();
424                debug!(
425                    "Migration hooks executed successfully for {}: {} hooks in {:?}",
426                    hook_point,
427                    results.len(),
428                    execution_time
429                );
430
431                // Check for hook results
432                for result in results {
433                    match result {
434                        llmspell_hooks::HookResult::Continue => {
435                            debug!("Hook executed successfully for {}", hook_point);
436                        }
437                        llmspell_hooks::HookResult::Modified(_) => {
438                            debug!("Hook modified context for {}", hook_point);
439                        }
440                        llmspell_hooks::HookResult::Cancel(reason) => {
441                            warn!("Migration hook cancelled during {}: {}", hook_point, reason);
442                            // Continue with migration even if hooks fail (configurable behavior)
443                        }
444                        llmspell_hooks::HookResult::Redirect(_) => {
445                            debug!("Hook redirected for {}", hook_point);
446                        }
447                        llmspell_hooks::HookResult::Replace(_) => {
448                            debug!("Hook replaced data for {}", hook_point);
449                        }
450                        llmspell_hooks::HookResult::Retry { .. } => {
451                            debug!("Hook requested retry for {}", hook_point);
452                        }
453                        llmspell_hooks::HookResult::Fork { .. } => {
454                            debug!("Hook requested fork for {}", hook_point);
455                        }
456                        llmspell_hooks::HookResult::Cache { .. } => {
457                            debug!("Hook requested caching for {}", hook_point);
458                        }
459                        llmspell_hooks::HookResult::Skipped(reason) => {
460                            debug!("Hook skipped for {}: {}", hook_point, reason);
461                        }
462                    }
463                }
464            }
465            Err(hook_error) => {
466                let execution_time = start_time.elapsed();
467                error!(
468                    "Migration hook execution failed for {} after {:?}: {}",
469                    hook_point, execution_time, hook_error
470                );
471
472                // Decide whether to fail the migration or continue
473                // For now, we continue but log the error
474                warn!(
475                    "Continuing migration despite hook failure in phase: {}",
476                    hook_point
477                );
478            }
479        }
480
481        Ok(())
482    }

484    /// Emit migration event for correlation tracking (legacy method)
485    #[allow(dead_code)]
486    async fn emit_migration_event(
487        &self,
488        event_type: &str,
489        plan: &MigrationPlan,
490        correlation_id: Uuid,
491        result: Option<&MigrationResult>,
492    ) -> StateResult<()> {
493        let mut event_data = serde_json::json!({
494            "from_version": plan.from_version,
495            "to_version": plan.to_version,
496            "risk_level": plan.risk_level,
497            "requires_backup": plan.requires_backup,
498            "steps": plan.steps.len(),
499        });
500
501        if let Some(result) = result {
502            event_data["status"] = serde_json::json!(result.status);
503            event_data["items_migrated"] = serde_json::json!(result.items_migrated);
504            event_data["duration_ms"] = serde_json::json!(result.duration.as_millis());
505        }
506
507        let event = UniversalEvent::new(event_type, event_data, llmspell_events::Language::Rust)
508            .with_correlation_id(correlation_id)
509            .with_source("migration_engine")
510            .with_tag("migration");
511
512        // Track for correlation
513        self.correlation_tracker.track_event(event.clone());
514
515        // Publish event
516        self.event_bus
517            .publish(event)
518            .await
519            .map_err(|e| StateError::storage(e.to_string()))?;
520
521        Ok(())
522    }

524    /// Emit typed migration event using new MigrationEvent system
525    async fn emit_typed_migration_event(
526        &self,
527        migration_event: &MigrationEvent,
528        correlation_id: Uuid,
529    ) -> StateResult<()> {
530        // Convert to UniversalEvent for existing event system integration
531        let universal_event: UniversalEvent = migration_event.clone().into();
532
533        // Create event metadata for correlation tracking
534        let _event_metadata = migration_event.create_metadata(correlation_id);
535
536        // Track for correlation
537        self.correlation_tracker
538            .track_event(universal_event.clone());
539
540        // Publish event to event bus
541        self.event_bus
542            .publish(universal_event)
543            .await
544            .map_err(|e| StateError::storage(e.to_string()))?;
545
546        // Log event for debugging
547        match migration_event {
548            MigrationEvent::MigrationStarted { .. } => {
549                debug!(
550                    "Migration started event emitted with correlation_id: {}",
551                    correlation_id
552                );
553            }
554            MigrationEvent::MigrationCompleted { .. } => {
555                debug!(
556                    "Migration completed event emitted with correlation_id: {}",
557                    correlation_id
558                );
559            }
560            MigrationEvent::MigrationFailed { .. } => {
561                error!(
562                    "Migration failed event emitted with correlation_id: {}",
563                    correlation_id
564                );
565            }
566            _ => {
567                debug!(
568                    "Migration event {:?} emitted with correlation_id: {}",
569                    migration_event.event_type(),
570                    correlation_id
571                );
572            }
573        }
574
575        Ok(())
576    }

596    /// Cancel an active migration
597    pub async fn cancel_migration(&self, migration_id: Uuid) -> StateResult<()> {
598        let mut active = self.active_migrations.write();
599        if active.remove(&migration_id).is_some() {
600            info!("Migration {} cancelled", migration_id);
601            Ok(())
602        } else {
603            Err(StateError::MigrationError(format!(
604                "Migration {} not found or already completed",
605                migration_id
606            )))
607        }
