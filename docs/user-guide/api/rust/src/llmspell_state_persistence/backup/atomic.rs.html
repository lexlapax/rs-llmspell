115    /// Capture atomic snapshot of current state
116    pub async fn capture(&self) -> Result<Vec<u8>, StateError> {
117        let _lock = self.operation_lock.lock().await;
118        info!("Starting atomic backup capture: {}", self.backup_id);
119
120        let start_time = Instant::now();
121        let _operation = BackupOperation {
122            backup_id: self.backup_id.clone(),
123            status: OperationStatus::InProgress,
124            started_at: self.snapshot_time,
125            completed_at: None,
126            entries_processed: 0,
127            bytes_processed: 0,
128            errors: Vec::new(),
129        };
130
131        // Create snapshot using read lock for consistency
132        let snapshot = self.create_snapshot().await?;
133
134        // Serialize snapshot
135        let serialized = self.serialize_snapshot(&snapshot)?;
136
137        let duration = start_time.elapsed();
138        info!(
139            "Atomic backup captured: {} entries, {} bytes in {:?}",
140            snapshot.metadata.entry_count,
141            serialized.len(),
142            duration
143        );
144
145        Ok(serialized)
146    }

148    /// Create consistent snapshot of state data
149    async fn create_snapshot(&self) -> Result<StateSnapshot, StateError> {
150        let mut entries = HashMap::new();
151        let mut total_size = 0u64;
152
153        // Discover all scopes by listing keys with empty prefix to get all data
154        let scopes = self.discover_scopes().await?;
155
156        debug!(
157            "Discovered {} scopes for backup: {:?}",
158            scopes.len(),
159            scopes
160        );
161
162        // Capture state for each scope
163        for scope in scopes {
164            match self.capture_scope_data(&scope).await {
165                Ok(scope_entries) => {
166                    for (key, entry) in scope_entries {
167                        // Estimate size of JSON value
168                        if let Ok(serialized) = serde_json::to_vec(&entry.data) {
169                            total_size += serialized.len() as u64;
170                        }
171                        entries.insert(key, entry);
172                    }
173                }
174                Err(e) => {
175                    error!("Failed to capture scope {:?}: {}", scope, e);
176                    // Continue with other scopes
177                }
178            }
179        }
180
181        let metadata = SnapshotMetadata {
182            backup_id: self.backup_id.clone(),
183            parent_id: self.parent_backup.clone(),
184            created_at: self.snapshot_time,
185            entry_count: entries.len(),
186            total_size,
187            schema_version: "1.0.0".to_string(), // TODO: Get from state manager
188        };
189
190        Ok(StateSnapshot {
191            timestamp: self.snapshot_time,
192            entries,
193            metadata,
194        })
195    }

197    /// Capture data for a specific scope
198    async fn capture_scope_data(
199        &self,
200        scope: &StateScope,
201    ) -> Result<HashMap<String, SnapshotEntry>, StateError> {
202        let mut entries = HashMap::new();
203
204        // For incremental backups, only capture changes since parent
205        if let Some(ref parent_id) = self.parent_backup {
206            debug!("Capturing incremental changes since: {}", parent_id);
207            // TODO: Implement change tracking
208        }
209
210        // Get all keys for scope
211        let keys = self.state_manager.list_keys(scope.clone()).await?;
212
213        for key in keys {
214            match self.state_manager.get(scope.clone(), &key).await {
215                Ok(Some(data)) => {
216                    let entry = SnapshotEntry {
217                        scope: scope.clone(),
218                        key: key.clone(),
219                        data,
220                        version: 1,                       // TODO: Get actual version
221                        last_modified: SystemTime::now(), // TODO: Track modification time
222                    };
223
224                    let entry_key = format!("{}:{}", scope, key);
225                    entries.insert(entry_key, entry);
226                }
227                Ok(None) => {
228                    // Key no longer exists, skip
229                }
230                Err(e) => {
231                    error!("Failed to load key {} in scope {:?}: {}", key, scope, e);
232                    // Continue with other keys
233                }
234            }
235        }
236
237        Ok(entries)
238    }

240    /// Discover all scopes that contain data by examining storage keys
241    async fn discover_scopes(&self) -> Result<Vec<StateScope>, StateError> {
242        use std::collections::HashSet;
243
244        let mut unique_scopes = HashSet::new();
245
246        // Get all storage keys using StateManager's method
247        let all_keys = self.state_manager.get_all_storage_keys().await?;
248
249        debug!("Found {} total keys for scope discovery", all_keys.len());
250
251        // Use StateScope's built-in parsing to extract scope information
252        for key in all_keys {
253            if let Some((scope, _key_part)) = StateScope::parse_storage_key(&key) {
254                unique_scopes.insert(scope);
255            }
256        }
257
258        let scopes: Vec<StateScope> = unique_scopes.into_iter().collect();
259        debug!("Discovered {} unique scopes: {:?}", scopes.len(), scopes);
260
261        Ok(scopes)
262    }

270    /// Restore state from snapshot data
271    pub async fn restore(&self, snapshot_data: &[u8]) -> Result<(), StateError> {
272        let _lock = self.operation_lock.lock().await;
273        info!("Starting atomic restore from backup: {}", self.backup_id);
274
275        // Deserialize snapshot
276        let snapshot: StateSnapshot = rmp_serde::from_slice(snapshot_data)
277            .map_err(|e| StateError::serialization(e.to_string()))?;
278
279        // Validate snapshot
280        self.validate_snapshot(&snapshot)?;
281
282        // Clear existing state in the scopes we're restoring to ensure clean restore
283        // Extract unique scopes from the snapshot data
284        let scopes_to_clear: HashSet<StateScope> = snapshot
285            .entries
286            .values()
287            .map(|entry| entry.scope.clone())
288            .collect();
289
290        for scope in &scopes_to_clear {
291            match self.state_manager.clear_scope(scope.clone()).await {
292                Ok(_) => {
293                    debug!("Cleared scope: {}", scope);
294                }
295                Err(e) => {
296                    error!("Failed to clear scope {}: {}", scope, e);
297                    return Err(e);
298                }
299            }
300        }
301
302        // Restore state atomically
303        let mut restored_count = 0;
304
305        for (key, entry) in snapshot.entries {
306            match self
307                .state_manager
308                .set(entry.scope, &entry.key, entry.data)
309                .await
310            {
311                Ok(_) => {
312                    restored_count += 1;
313                }
314                Err(e) => {
315                    error!("Failed to restore key {}: {}", key, e);
316                    return Err(e);
317                }
318            }
319        }
320
321        info!(
322            "Atomic restore completed: {} entries restored",
323            restored_count
324        );
325        Ok(())
326    }

328    /// Restore state from snapshot data with progress tracking
329    pub async fn restore_with_progress<F>(
330        &self,
331        snapshot_data: &[u8],
332        mut progress_callback: F,
333    ) -> Result<(), StateError>
334    where
335        F: FnMut(usize, usize) + Send,
336    {
337        let _lock = self.operation_lock.lock().await;
338        info!(
339            "Starting atomic restore from backup: {} (with progress)",
340            self.backup_id
341        );
342
343        // Deserialize snapshot
344        let snapshot: StateSnapshot = rmp_serde::from_slice(snapshot_data)
345            .map_err(|e| StateError::serialization(e.to_string()))?;
346
347        // Validate snapshot
348        self.validate_snapshot(&snapshot)?;
349
350        // Clear existing state in the scopes we're restoring to ensure clean restore
351        // Extract unique scopes from the snapshot data
352        let scopes_to_clear: HashSet<StateScope> = snapshot
353            .entries
354            .values()
355            .map(|entry| entry.scope.clone())
356            .collect();
357
358        for scope in &scopes_to_clear {
359            match self.state_manager.clear_scope(scope.clone()).await {
360                Ok(_) => {
361                    debug!("Cleared scope: {}", scope);
362                }
363                Err(e) => {
364                    error!("Failed to clear scope {}: {}", scope, e);
365                    return Err(e);
366                }
367            }
368        }
369
370        let total_entries = snapshot.entries.len();
371        debug!("Restoring {} entries", total_entries);
372
373        // Restore state atomically with progress reporting
374        let mut restored_count = 0;
375
376        for (idx, (key, entry)) in snapshot.entries.into_iter().enumerate() {
377            match self
378                .state_manager
379                .set(entry.scope, &entry.key, entry.data)
380                .await
381            {
382                Ok(_) => {
383                    restored_count += 1;
384                    // Report progress every 10 entries or on last entry
385                    if restored_count % 10 == 0 || idx == total_entries - 1 {
386                        progress_callback(restored_count, total_entries);
387                    }
388                }
389                Err(e) => {
390                    error!("Failed to restore key {}: {}", key, e);
391                    return Err(e);
392                }
393            }
394        }
395
396        info!(
397            "Atomic restore completed: {} entries restored",
398            restored_count
399        );
400        Ok(())
401    }

403    /// Validate snapshot before restore
404    fn validate_snapshot(&self, snapshot: &StateSnapshot) -> Result<(), StateError> {
405        // Basic validation
406        if snapshot.metadata.backup_id != self.backup_id {
407            return Err(StateError::validation_error(
408                "Backup validation failed: Backup ID mismatch".to_string(),
409            ));
410        }
411
412        // Empty snapshots are valid - they represent an empty state
413
414        Ok(())
415    }
