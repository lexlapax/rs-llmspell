175    /// Create a new backup
176    pub async fn create_backup(&self, incremental: bool) -> BackupResult<BackupStatus> {
177        let backup_id = self.generate_backup_id();
178        info!(
179            "Creating {} backup: {}",
180            if incremental { "incremental" } else { "full" },
181            backup_id
182        );
183
184        // TODO: Emit backup started event when StateEvent is available
185
186        // Determine parent backup for incremental
187        let parent_id = if incremental {
188            self.find_parent_backup().await?
189        } else {
190            None
191        };
192
193        // Create atomic backup operation
194        let atomic_backup = AtomicBackup::new(
195            backup_id.clone(),
196            self.state_manager.clone(),
197            parent_id.clone(),
198        )
199        .map_err(|e| StateError::storage(e.to_string()))?;
200
201        // Perform the backup
202        let start_time = std::time::Instant::now();
203        let backup_data = atomic_backup.capture().await?;
204        let duration = start_time.elapsed();
205
206        // Extract entry count from snapshot metadata
207        let snapshot: crate::backup::atomic::StateSnapshot = rmp_serde::from_slice(&backup_data)
208            .map_err(|e| {
209                StateError::serialization(format!(
210                    "Failed to deserialize snapshot for metadata: {}",
211                    e
212                ))
213            })?;
214        let entry_count = snapshot.metadata.entry_count;
215
216        // Compress if enabled
217        let (data, compression_info) = if self.config.compression_enabled {
218            let compressed = self.compress_backup(&backup_data).await?;
219            let info = CompressionInfo {
220                algorithm: self.config.compression_type.to_string(),
221                level: self.config.compression_level as u32,
222                original_size: backup_data.len() as u64,
223                compressed_size: compressed.len() as u64,
224            };
225            (compressed, Some(info))
226        } else {
227            (backup_data.clone(), None)
228        };
229
230        // Save backup to disk
231        let backup_path = self.get_backup_path(&backup_id);
232        tokio::fs::write(&backup_path, &data)
233            .await
234            .map_err(|e| StateError::storage(format!("Failed to write backup file: {}", e)))?;
235
236        // Create metadata
237        let metadata = BackupMetadata {
238            id: backup_id.clone(),
239            created_at: SystemTime::now(),
240            backup_type: if incremental {
241                BackupType::Incremental
242            } else {
243                BackupType::Full
244            },
245            parent_id: parent_id.clone(),
246            schema_version: self.get_current_schema_version().await?,
247            checksums: self.calculate_checksums(&data),
248            compression: compression_info.clone(),
249            encryption: None, // TODO: Implement encryption
250            stats: BackupStats {
251                total_entries: entry_count,
252                total_size: data.len() as u64,
253                #[allow(clippy::cast_possible_truncation)]
254                duration_ms: duration.as_millis() as u64,
255                scopes_backed_up: vec!["global".to_string()], // TODO: Get actual scopes
256            },
257        };
258
259        // Save metadata
260        self.save_backup_metadata(&metadata).await?;
261
262        // Update backup index
263        {
264            let mut index = self.backup_index.write().await;
265            index.insert(backup_id.clone(), metadata.clone());
266        }
267
268        // Create status
269        let status = BackupStatus {
270            id: backup_id.clone(),
271            created_at: metadata.created_at,
272            size_bytes: metadata.stats.total_size,
273            is_incremental: incremental,
274            parent_id,
275            state_version: metadata.schema_version,
276            entry_count: metadata.stats.total_entries,
277            validation: None,
278            metadata: HashMap::new(),
279        };
280
281        // TODO: Emit backup completed event when StateEvent is available
282
283        // Clean up old backups if needed
284        self.cleanup_old_backups().await?;
285
286        Ok(status)
287    }

289    /// Restore from a backup
290    pub async fn restore_backup(
291        &self,
292        backup_id: &str,
293        options: RestoreOptions,
294    ) -> BackupResult<()> {
295        info!(
296            "Restoring backup: {} (dry_run: {})",
297            backup_id, options.dry_run
298        );
299
300        // TODO: Emit restore started event when StateEvent is available
301
302        // Validate backup if requested
303        if options.verify_checksums {
304            let validation = self.validate_backup(backup_id).await?;
305            if !validation.is_valid {
306                return Err(StateError::validation_error(format!(
307                    "Backup validation failed: {}",
308                    validation.errors.join(", ")
309                )));
310            }
311        }
312
313        // Create backup of current state if requested
314        if options.backup_current && !options.dry_run {
315            info!("Creating backup of current state before restore");
316            self.create_backup(false).await?;
317        }
318
319        // Load backup metadata
320        let metadata = self.get_backup_metadata(backup_id).await?;
321
322        // Build restore chain for incremental backups
323        let restore_chain = if metadata.backup_type == BackupType::Incremental {
324            self.build_restore_chain(backup_id).await?
325        } else {
326            vec![backup_id.to_string()]
327        };
328
329        // Perform restore
330        if !options.dry_run {
331            for backup_id in &restore_chain {
332                self.restore_single_backup(backup_id).await?;
333            }
334        }
335
336        // TODO: Emit restore completed event when StateEvent is available
337
338        Ok(())
339    }

343        let index = self.backup_index.read().await;
344        let mut backups = Vec::new();
345
346        for (id, metadata) in index.iter() {
347            backups.push(BackupStatus {

357            });
358        }
359
360        // Sort by creation time (newest first)
361        backups.sort_by(|a, b| b.created_at.cmp(&a.created_at));
362
363        Ok(backups)
364    }

366    /// Validate a backup
367    pub async fn validate_backup(&self, backup_id: &str) -> BackupResult<BackupValidation> {
368        debug!("Validating backup: {}", backup_id);
369
370        let mut errors = Vec::new();
371        let warnings = Vec::new();
372
373        // Load metadata
374        let metadata = match self.get_backup_metadata(backup_id).await {
375            Ok(m) => m,
376            Err(e) => {
377                errors.push(format!("Failed to load metadata: {}", e));
378                return Ok(BackupValidation {
379                    is_valid: false,
380                    validated_at: SystemTime::now(),
381                    checksum_valid: false,
382                    integrity_valid: false,
383                    errors,
384                    warnings,
385                });
386            }
387        };
388
389        // Load backup data
390        let backup_path = self.get_backup_path(backup_id);
391        let data = match tokio::fs::read(&backup_path).await {
392            Ok(d) => d,
393            Err(e) => {
394                errors.push(format!("Failed to read backup file: {}", e));
395                return Ok(BackupValidation {
396                    is_valid: false,
397                    validated_at: SystemTime::now(),
398                    checksum_valid: false,
399                    integrity_valid: false,
400                    errors,
401                    warnings,
402                });
403            }
404        };
405
406        // Verify checksums
407        let calculated_checksums = self.calculate_checksums(&data);
408        let checksum_valid = calculated_checksums == metadata.checksums;
409        if !checksum_valid {
410            errors.push("Checksum verification failed".to_string());
411        }
412
413        // Basic integrity checks
414        #[allow(clippy::cast_possible_truncation)]
415        let expected_size = metadata.stats.total_size as usize;
416        let integrity_valid = !data.is_empty() && data.len() == expected_size;
417        if !integrity_valid {
418            errors.push("Data integrity check failed".to_string());
419        }
420
421        let is_valid = errors.is_empty();
422
423        Ok(BackupValidation {
424            is_valid,
425            validated_at: SystemTime::now(),
426            checksum_valid,
427            integrity_valid,
428            errors,
429            warnings,
430        })
431    }

435    fn generate_backup_id(&self) -> BackupId {
436        let timestamp = SystemTime::now()
437            .duration_since(UNIX_EPOCH)
438            .unwrap()
439            .as_secs();
440        let random_suffix: String = (0..8)
441            .map(|_| format!("{:x}", rand::random::<u8>()))
442            .collect();
443        format!("backup_{}_{}", timestamp, random_suffix)
444    }

446    async fn find_parent_backup(&self) -> BackupResult<Option<BackupId>> {
447        let index = self.backup_index.read().await;
448
449        // Find the most recent backup
450        let mut backups: Vec<_> = index.iter().collect();
451        backups.sort_by(|a, b| b.1.created_at.cmp(&a.1.created_at));
452
453        Ok(backups.first().map(|(id, _)| (*id).clone()))
454    }

479    fn calculate_checksums(&self, data: &[u8]) -> HashMap<String, String> {
480        use sha2::{Digest, Sha256};
481        let mut checksums = HashMap::new();
482
483        let hash = Sha256::digest(data);
484        checksums.insert("sha256".to_string(), format!("{:x}", hash));
485
486        checksums
487    }

520    async fn build_restore_chain(&self, backup_id: &str) -> BackupResult<Vec<BackupId>> {
521        let mut chain = Vec::new();
522        let mut current_id = backup_id.to_string();
523
524        loop {
525            let metadata = self.get_backup_metadata(&current_id).await?;
526            chain.push(current_id.clone());
527
528            // If this is a full backup or has no parent, we're done
529            if metadata.backup_type == BackupType::Full || metadata.parent_id.is_none() {
530                break;
531            }
532
533            // Move to parent backup
534            if let Some(parent_id) = metadata.parent_id {
535                current_id = parent_id;
536            } else {
537                break;
538            }
539        }
540
541        // Reverse to get oldest-first order for restoration
542        chain.reverse();
543
544        debug!(
545            "Built restore chain of {} backups for {}",
546            chain.len(),
547            backup_id
548        );
549
550        Ok(chain)
551    }

553    async fn restore_single_backup(&self, backup_id: &str) -> BackupResult<()> {
554        info!("Restoring single backup: {}", backup_id);
555
556        // TODO: Execute pre-restore hooks when hook infrastructure supports custom points
557        // For now, the restore process will emit events that can trigger hooks
558
559        // Load backup metadata
560        let metadata = self.get_backup_metadata(backup_id).await?;
561
562        // Load backup data from file
563        let backup_path = self.get_backup_path(backup_id);
564        let compressed_data = tokio::fs::read(&backup_path)
565            .await
566            .map_err(|e| StateError::storage(format!("Failed to read backup: {}", e)))?;
567
568        // Decompress if needed
569        let backup_data = if let Some(ref compression_info) = metadata.compression {
570            let reduction_percent = if compression_info.original_size > 0
571                && compression_info.compressed_size < compression_info.original_size
572            {
573                #[allow(clippy::cast_precision_loss)]
574                let size_diff =
575                    (compression_info.original_size - compression_info.compressed_size) as f64;
576                #[allow(clippy::cast_precision_loss)]
577                let original_size = compression_info.original_size as f64;
578                (size_diff / original_size) * 100.0
579            } else {
580                0.0
581            };
582            debug!(
583                "Decompressing backup with {} ({:.1}% reduction)",
584                compression_info.algorithm, reduction_percent
585            );
586
587            let compressor = BackupCompression::new(
588                self.config.compression_type,
589                CompressionLevel::new(compression_info.level)?,
590            );
591            compressor.decompress(&compressed_data)?
592        } else {
593            compressed_data
594        };
595
596        // Create atomic backup instance for restoration
597        let atomic_backup = AtomicBackup::new(
598            backup_id.to_string(),
599            self.state_manager.clone(),
600            metadata.parent_id.clone(),
601        )
602        .map_err(|e| StateError::storage(e.to_string()))?;
603
604        // Perform the restoration
605        atomic_backup.restore(&backup_data).await?;
606
607        info!(
608            "Successfully restored {} entries from backup {}",
609            metadata.stats.total_entries, backup_id
610        );
611
612        // TODO: Execute post-restore hooks when hook infrastructure supports custom points
613        // The atomic restore process already triggers state change hooks for each restored entry
614
615        Ok(())
616    }

618    fn load_backup_index(&mut self) -> Result<()> {
619        // Load existing backup metadata from disk
620        let mut loaded_backups = Vec::new();
621
622        if let Ok(entries) = std::fs::read_dir(&self.config.backup_dir) {
623            for entry in entries.flatten() {
624                let path = entry.path();
625                if path.extension().and_then(|s| s.to_str()) == Some("meta") {
626                    // Load metadata file
627                    if let Ok(contents) = std::fs::read_to_string(&path) {
628                        if let Ok(metadata) = serde_json::from_str::<BackupMetadata>(&contents) {
629                            loaded_backups.push((metadata.id.clone(), metadata));
630                        }
631                    }
632                }
633            }
634        }
635
636        let loaded_count = loaded_backups.len();
637
638        // Update index with loaded backups
639        // Since we're in the constructor, we can use try_write().unwrap() safely
640        if let Ok(mut index) = self.backup_index.try_write() {
641            for (id, metadata) in loaded_backups {
642                index.insert(id, metadata);
643            }
644        }
645
646        if loaded_count > 0 {
647            info!("Loaded {} existing backups from disk", loaded_count);
648        }
649
650        Ok(())
651    }

653    /// Apply retention policies to manage backup storage
654    pub async fn apply_retention_policies(&self) -> BackupResult<RetentionReport> {
655        let start_time = Instant::now();
656        let event_builder = BackupEventBuilder::new();
657
658        // Build composite retention policy based on configuration
659        let mut composite_policy = CompositePolicy::new();
660
661        // Add time-based policy if configured
662        if let Some(max_age) = self.config.max_backup_age {
663            composite_policy = composite_policy.add_policy(Box::new(TimeBasedPolicy::new(max_age)));
664        }
665
666        // Add count-based policy if configured
667        if let Some(max_count) = self.config.max_backups {
668            composite_policy =
669                composite_policy.add_policy(Box::new(CountBasedPolicy::new(max_count)));
670        }
671
672        // Don't add importance-based policy by default - it overrides count/time limits
673        // Users can add it explicitly if they want importance-based retention
674
675        // Get all backups
676        let index = self.backup_index.read().await;
677        let all_backups: Vec<BackupMetadata> = index.values().cloned().collect();
678
679        // Emit cleanup started event
680        let cleanup_started_event = event_builder.cleanup_started(
681            all_backups.len(),
682            format!(
683                "max_backups={:?}, max_age={:?}",
684                self.config.max_backups, self.config.max_backup_age
685            ),
686            false,
687        );
688        if let Err(e) = self
689            .state_manager
690            .event_bus()
691            .publish(cleanup_started_event.to_universal_event())
692            .await
693        {
694            debug!("Failed to emit cleanup started event: {}", e);
695        }
696
697        // Calculate total storage usage
698        let total_size: u64 = all_backups.iter().map(|b| b.stats.total_size).sum();
699
700        // Create retention context
701        let context = RetentionContext {
702            all_backups: all_backups.clone(),
703            total_size,
704            storage_limit: None, // Could be configured in the future
705            current_time: SystemTime::now(),
706        };
707
708        // Evaluate retention for each backup
709        let mut decisions = Vec::new();
710        for backup in &all_backups {
711            let decision = composite_policy.evaluate(backup, &context);
712            decisions.push(decision);
713        }
714
715        // Execute cleanup
716        let cleanup = BackupCleanup::new(self.config.backup_dir.clone(), false);
717        let cleanup_result = cleanup.execute_cleanup(decisions.clone(), &index).await?;
718
719        // Remove deleted backups from index
720        if cleanup_result.deleted_count > 0 {
721            drop(index); // Release read lock
722            let mut index = self.backup_index.write().await;
723            for decision in &decisions {
724                if !decision.should_retain {
725                    // Emit backup deleted event
726                    if let Some(metadata) = index.get(&decision.backup_id) {
727                        let deleted_event = event_builder.backup_deleted(
728                            decision.backup_id.clone(),
729                            metadata.stats.total_size,
730                            decision.reason.clone(),
731                        );
732                        if let Err(e) = self
733                            .state_manager
734                            .event_bus()
735                            .publish(deleted_event.to_universal_event())
736                            .await
737                        {
738                            debug!("Failed to emit backup deleted event: {}", e);
739                        }
740                    }
741                    index.remove(&decision.backup_id);
742                }
743            }
744        }
745
746        let report = RetentionReport {
747            evaluated_count: all_backups.len(),
748            retained_count: decisions.iter().filter(|d| d.should_retain).count(),
749            deleted_count: cleanup_result.deleted_count,
750            space_freed: cleanup_result.space_freed,
751            decisions,
752            execution_time: start_time.elapsed(),
753        };
754
755        // Emit cleanup completed event
756        let cleanup_completed_event = event_builder.cleanup_completed(
757            report.evaluated_count,
758            report.deleted_count,
759            report.retained_count,
760            report.space_freed,
761            report.execution_time,
762        );
763        if let Err(e) = self
764            .state_manager
765            .event_bus()
766            .publish(cleanup_completed_event.to_universal_event())
767            .await
768        {
769            debug!("Failed to emit cleanup completed event: {}", e);
770        }
771
772        Ok(report)
773    }
774
775    /// Manually trigger retention policy application
776    pub async fn cleanup_backups(&self) -> BackupResult<RetentionReport> {
777        info!("Manually triggering backup cleanup");
778        self.apply_retention_policies().await
779    }
