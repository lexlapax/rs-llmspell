4//! # State Persistence Module
5//!
6//! This module provides persistent state management with the following features:
7//! - Multiple storage backends (Memory, Sled, RocksDB)
8//! - Hook integration for state change events
9//! - State scoping and isolation
10//! - Schema versioning and migration support
11//! - Backup and recovery capabilities
12//!
13//! ## Example Usage
14//!
15//! ```rust,ignore
16//! use llmspell_state_persistence::{StateManager, StateScope, PersistenceConfig};
17//! use serde_json::json;
18//!
19//! #[tokio::main]
20//! async fn main() -> Result<(), Box<dyn std::error::Error>> {
21//!     // Create state manager with default in-memory backend
22//!     let state_manager = StateManager::new().await?;
23//!
24//!     // Set state in global scope
25//!     state_manager.set(StateScope::Global, "key", json!("value")).await?;
26//!
27//!     // Get state
28//!     let value = state_manager.get(StateScope::Global, "key").await?;
29//!     assert_eq!(value, Some(json!("value")));
30//!
31//!     // Use agent-scoped state
32//!     let agent_scope = StateScope::Agent("agent123".to_string());
33//!     state_manager.set(agent_scope.clone(), "config", json!({"model": "gpt-4"})).await?;
34//!
35//!     Ok(())
36//! }
37//! ```
