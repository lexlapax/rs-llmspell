18/// Enhanced state schema with semantic versioning
19#[derive(Debug, Clone, Serialize, Deserialize)]
20pub struct EnhancedStateSchema {
21    pub version: SemanticVersion,
22    pub hash: String,
23    pub created_at: std::time::SystemTime,
24    pub fields: HashMap<String, crate::config::FieldSchema>,
25    pub compatibility: CompatibilityLevel,
26    pub migration_path: Vec<MigrationStep>,
27    pub dependencies: Vec<SemanticVersion>,
28    pub metadata: HashMap<String, serde_json::Value>,
29}
30
31impl EnhancedStateSchema {
32    pub fn new(version: SemanticVersion) -> Self {
33        let mut schema = Self {
34            version,
35            hash: String::new(), // Will be calculated after initialization
36            created_at: std::time::SystemTime::now(),
37            fields: HashMap::new(),
38            compatibility: CompatibilityLevel::BackwardCompatible,
39            migration_path: Vec::new(),
40            dependencies: Vec::new(),
41            metadata: HashMap::new(),
42        };
43        schema.hash = schema.calculate_content_hash();
44        schema
45    }
46
47    pub fn from_legacy(legacy: StateSchema) -> Self {
48        Self {
49            version: SemanticVersion::new(legacy.version, 0, 0),
50            hash: legacy.hash,
51            created_at: std::time::SystemTime::now(),
52            fields: legacy.fields,
53            compatibility: legacy.compatibility,
54            migration_path: legacy.migration_path,
55            dependencies: Vec::new(),
56            metadata: HashMap::new(),
57        }
58    }
59
60    pub fn to_legacy(&self) -> StateSchema {
61        StateSchema {
62            version: self.version.major,
63            hash: self.hash.clone(),
64            fields: self.fields.clone(),
65            compatibility: self.compatibility.clone(),
66            migration_path: self.migration_path.clone(),
67        }
68    }
69
70    pub fn add_field(&mut self, name: String, field: crate::config::FieldSchema) {
71        self.fields.insert(name, field);
72        self.hash = self.calculate_content_hash();
73    }
74
75    pub fn remove_field(&mut self, name: &str) -> Option<crate::config::FieldSchema> {
76        let result = self.fields.remove(name);
77        if result.is_some() {
78            self.hash = self.calculate_content_hash();
79        }
80        result
81    }
82
83    pub fn add_dependency(&mut self, version: SemanticVersion) {
84        if !self.dependencies.contains(&version) {
85            self.dependencies.push(version);
86        }
87    }
88
89    pub fn set_metadata(&mut self, key: String, value: serde_json::Value) {
90        self.metadata.insert(key, value);
91    }
92
93    fn calculate_content_hash(&self) -> String {
94        use std::collections::hash_map::DefaultHasher;
95        use std::hash::{Hash, Hasher};
96
97        let mut hasher = DefaultHasher::new();
98
99        // Hash the version
100        self.version.major.hash(&mut hasher);
101        self.version.minor.hash(&mut hasher);
102        self.version.patch.hash(&mut hasher);
103
104        // Hash field names in sorted order for consistency
105        let mut field_names: Vec<_> = self.fields.keys().cloned().collect();
106        field_names.sort();
107        for name in field_names {
108            name.hash(&mut hasher);
109            if let Some(field) = self.fields.get(&name) {
110                field.field_type.hash(&mut hasher);
111                field.required.hash(&mut hasher);
112            }
113        }
114
115        // Hash dependencies
116        for dep in &self.dependencies {
117            dep.major.hash(&mut hasher);
118            dep.minor.hash(&mut hasher);
119            dep.patch.hash(&mut hasher);
120        }
121
122        format!("{:x}", hasher.finish())
123    }
124}

126#[cfg(test)]
127mod tests {
128    use super::*;
129    use crate::config::FieldSchema;
130    #[test]
131    fn test_enhanced_schema_creation() {
132        let version = SemanticVersion::new(1, 2, 3);
133        let schema = EnhancedStateSchema::new(version.clone());
134
135        assert_eq!(schema.version, version);
136        assert!(!schema.hash.is_empty());
137        assert!(schema.fields.is_empty());
138        assert_eq!(schema.compatibility, CompatibilityLevel::BackwardCompatible);
139        assert!(schema.migration_path.is_empty());
140        assert!(schema.dependencies.is_empty());
141    }
142    #[test]
143    fn test_legacy_conversion() {
144        let legacy = StateSchema::v1();
145        let enhanced = EnhancedStateSchema::from_legacy(legacy.clone());
146        let converted_back = enhanced.to_legacy();
147
148        assert_eq!(converted_back.version, legacy.version);
149        assert_eq!(converted_back.hash, legacy.hash);
150        assert_eq!(converted_back.fields, legacy.fields);
151        assert_eq!(converted_back.compatibility, legacy.compatibility);
152        assert_eq!(converted_back.migration_path, legacy.migration_path);
153    }
154    #[test]
155    fn test_field_operations() {
156        let mut schema = EnhancedStateSchema::new(SemanticVersion::new(1, 0, 0));
157        let original_hash = schema.hash.clone();
158
159        let field = FieldSchema {
160            field_type: "string".to_string(),
161            required: true,
162            default_value: None,
163            validators: vec![],
164        };
165
166        schema.add_field("test_field".to_string(), field.clone());
167        assert_ne!(schema.hash, original_hash);
168        assert!(schema.fields.contains_key("test_field"));
169
170        let removed = schema.remove_field("test_field");
171        assert!(removed.is_some());
172        assert!(!schema.fields.contains_key("test_field"));
173    }
174    #[test]
175    fn test_dependency_management() {
176        let mut schema = EnhancedStateSchema::new(SemanticVersion::new(2, 0, 0));
177        let dep1 = SemanticVersion::new(1, 0, 0);
178        let dep2 = SemanticVersion::new(1, 5, 0);
179
180        schema.add_dependency(dep1.clone());
181        schema.add_dependency(dep2.clone());
182        schema.add_dependency(dep1.clone()); // Duplicate - should not be added
183
184        assert_eq!(schema.dependencies.len(), 2);
185        assert!(schema.dependencies.contains(&dep1));
186        assert!(schema.dependencies.contains(&dep2));
187    }
188    #[test]
189    fn test_metadata_operations() {
190        let mut schema = EnhancedStateSchema::new(SemanticVersion::new(1, 0, 0));
191
192        schema.set_metadata("author".to_string(), serde_json::json!("Test Author"));
193        schema.set_metadata("description".to_string(), serde_json::json!("Test schema"));
194
195        assert_eq!(schema.metadata.len(), 2);
196        assert_eq!(
197            schema.metadata.get("author"),
198            Some(&serde_json::json!("Test Author"))
199        );
200        assert_eq!(
201            schema.metadata.get("description"),
202            Some(&serde_json::json!("Test schema"))
203        );
204    }
