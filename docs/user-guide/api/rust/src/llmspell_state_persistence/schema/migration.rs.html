12#[derive(Debug, Error)]
13pub enum MigrationPlannerError {
14    #[error("No migration path found from {from} to {to}")]
15    NoMigrationPath {
16        from: Box<SemanticVersion>,
17        to: Box<SemanticVersion>,
18    },
19
20    #[error("Migration step validation failed: {reason}")]
21    ValidationFailed { reason: String },
22
23    #[error("Circular dependency detected in migration path")]
24    CircularDependency,
25
26    #[error("Invalid migration step: {step_id}")]
27    InvalidMigrationStep { step_id: String },
28
29    #[error("Migration planning failed: {details}")]
30    PlanningFailed { details: String },
31}
32
33impl From<MigrationPlannerError> for StateError {
34    fn from(err: MigrationPlannerError) -> Self {
35        StateError::MigrationError(err.to_string())
36    }
37}

113    /// Create a migration plan between two schema versions
114    pub fn create_migration_plan(
115        &mut self,
116        from_version: &SemanticVersion,
117        to_version: &SemanticVersion,
118    ) -> Result<MigrationPlan, MigrationPlannerError> {
119        if from_version == to_version {
120            return Ok(Self::create_no_op_plan(from_version.clone()));
121        }
122
123        let _from_schema = self.schema_registry.get(from_version).ok_or_else(|| {
124            MigrationPlannerError::PlanningFailed {
125                details: format!("Source schema {} not found", from_version),
126            }
127        })?;
128
129        let _to_schema = self.schema_registry.get(to_version).ok_or_else(|| {
130            MigrationPlannerError::PlanningFailed {
131                details: format!("Target schema {} not found", to_version),
132            }
133        })?;
134
135        // Find the optimal migration path
136        let path = self.find_migration_path(from_version, to_version)?;
137
138        // Create migration steps for the path
139        let mut steps = Vec::new();
140        let mut data_transformations = Vec::new();
141        let mut warnings = Vec::new();
142        let mut overall_risk = RiskLevel::Low;
143        let mut requires_backup = false;
144
145        for i in 0..path.len() - 1 {
146            let step_from = &path[i];
147            let step_to = &path[i + 1];
148
149            let compatibility = self.get_or_compute_compatibility(step_from, step_to)?;
150
151            // Update overall risk level
152            if compatibility.risk_level > overall_risk {
153                overall_risk = compatibility.risk_level.clone();
154            }
155
156            if compatibility.risk_level >= RiskLevel::High {
157                requires_backup = true;
158            }
159
160            // Create migration step
161            let migration_step = MigrationStep {
162                from_version: step_from.major, // Convert to legacy format for compatibility
163                to_version: step_to.major,
164                migration_type: Self::determine_migration_type(&compatibility),
165                description: format!("Migrate from {} to {}", step_from, step_to),
166            };
167            steps.push(migration_step);
168
169            // Create data transformations
170            let transformation =
171                self.create_data_transformation(step_from, step_to, &compatibility)?;
172            data_transformations.push(transformation);
173
174            // Collect warnings
175            warnings.extend(compatibility.warnings.clone());
176        }
177
178        // Estimate duration based on complexity
179        let estimated_duration = Self::estimate_migration_duration(&steps, &data_transformations);
180
181        // Create rollback plan if needed
182        let rollback_plan = if requires_backup {
183            Some(Box::new(
184                self.create_rollback_plan(to_version, from_version)?,
185            ))
186        } else {
187            None
188        };
189
190        Ok(MigrationPlan {
191            from_version: from_version.clone(),
192            to_version: to_version.clone(),
193            steps,
194            estimated_duration,
195            risk_level: overall_risk,
196            requires_backup,
197            data_transformations,
198            rollback_plan,
199            warnings,
200        })
201    }

203    /// Find the optimal migration path between two versions
204    fn find_migration_path(
205        &self,
206        from_version: &SemanticVersion,
207        to_version: &SemanticVersion,
208    ) -> Result<Vec<SemanticVersion>, MigrationPlannerError> {
209        // Use breadth-first search to find the shortest path
210        let mut queue = VecDeque::new();
211        let mut visited = std::collections::HashSet::new();
212        let mut parent: HashMap<SemanticVersion, SemanticVersion> = HashMap::new();
213
214        queue.push_back(from_version.clone());
215        visited.insert(from_version.clone());
216
217        while let Some(current) = queue.pop_front() {
218            if current == *to_version {
219                // Reconstruct path
220                let mut path = Vec::new();
221                let mut node = to_version.clone();
222
223                while let Some(prev) = parent.get(&node) {
224                    path.push(node.clone());
225                    node = prev.clone();
226                }
227                path.push(from_version.clone());
228                path.reverse();
229
230                return Ok(path);
231            }
232
233            // Find all reachable versions from current
234            for next_version in self.get_reachable_versions(&current) {
235                if !visited.contains(&next_version) {
236                    visited.insert(next_version.clone());
237                    parent.insert(next_version.clone(), current.clone());
238                    queue.push_back(next_version);
239                }
240            }
241        }
242
243        Err(MigrationPlannerError::NoMigrationPath {
244            from: Box::new(from_version.clone()),
245            to: Box::new(to_version.clone()),
246        })
247    }

249    /// Get all versions reachable from a given version
250    fn get_reachable_versions(&self, from_version: &SemanticVersion) -> Vec<SemanticVersion> {
251        self.schema_registry
252            .keys()
253            .filter(|version| {
254                *version > from_version && self.is_migration_possible(from_version, version)
255            })
256            .cloned()
257            .collect()
258    }

260    /// Check if migration is possible between two versions
261    fn is_migration_possible(&self, from: &SemanticVersion, to: &SemanticVersion) -> bool {
262        // Allow migration within same major version or to next major version
263        to.major <= from.major + 1
264    }

266    /// Get or compute compatibility between two versions
267    fn get_or_compute_compatibility(
268        &mut self,
269        from: &SemanticVersion,
270        to: &SemanticVersion,
271    ) -> Result<CompatibilityResult, MigrationPlannerError> {
272        let key = (from.clone(), to.clone());
273
274        if let Some(cached) = self.compatibility_cache.get(&key) {
275            return Ok(cached.clone());
276        }
277
278        let from_schema = self.schema_registry.get(from).ok_or_else(|| {
279            MigrationPlannerError::PlanningFailed {
280                details: format!("Schema {} not found", from),
281            }
282        })?;
283
284        let to_schema =
285            self.schema_registry
286                .get(to)
287                .ok_or_else(|| MigrationPlannerError::PlanningFailed {
288                    details: format!("Schema {} not found", to),
289                })?;
290
291        let compatibility = CompatibilityChecker::check_compatibility(from_schema, to_schema);
292        self.compatibility_cache.insert(key, compatibility.clone());
293
294        Ok(compatibility)
295    }

297    /// Create data transformation for a migration step
298    fn create_data_transformation(
299        &self,
300        from_version: &SemanticVersion,
301        to_version: &SemanticVersion,
302        compatibility: &CompatibilityResult,
303    ) -> Result<DataTransformation, MigrationPlannerError> {
304        let transformation_id = format!("transform_{}_{}", from_version, to_version);
305        let description = format!("Transform data from {} to {}", from_version, to_version);
306
307        let mut field_mappings = HashMap::new();
308        let mut validation_rules = Vec::new();
309
310        // Create field mappings based on compatibility analysis
311        for (field_name, field_change) in &compatibility.field_changes {
312            match field_change {
313                super::compatibility::FieldChange::Added { new_field } => {
314                    if let Some(default_value) = &new_field.default_value {
315                        field_mappings.insert(
316                            field_name.clone(),
317                            FieldMapping::Default {
318                                to_field: field_name.clone(),
319                                default_value: default_value.clone(),
320                            },
321                        );
322                    }
323                }
324                super::compatibility::FieldChange::Removed { .. } => {
325                    field_mappings.insert(
326                        field_name.clone(),
327                        FieldMapping::Remove {
328                            field: field_name.clone(),
329                        },
330                    );
331                }
332                super::compatibility::FieldChange::Modified {
333                    old_field,
334                    new_field,
335                    ..
336                } => {
337                    if old_field.field_type != new_field.field_type {
338                        field_mappings.insert(
339                            field_name.clone(),
340                            FieldMapping::Transform {
341                                from_field: field_name.clone(),
342                                to_field: field_name.clone(),
343                                transformation: format!(
344                                    "convert_{}_to_{}",
345                                    old_field.field_type, new_field.field_type
346                                ),
347                            },
348                        );
349                    } else {
350                        field_mappings.insert(
351                            field_name.clone(),
352                            FieldMapping::Direct {
353                                from_field: field_name.clone(),
354                                to_field: field_name.clone(),
355                            },
356                        );
357                    }
358                }
359                super::compatibility::FieldChange::TypeChanged { old_type, new_type } => {
360                    field_mappings.insert(
361                        field_name.clone(),
362                        FieldMapping::Transform {
363                            from_field: field_name.clone(),
364                            to_field: field_name.clone(),
365                            transformation: format!("convert_{}_to_{}", old_type, new_type),
366                        },
367                    );
368                }
369                super::compatibility::FieldChange::RequiredChanged { .. } => {
370                    field_mappings.insert(
371                        field_name.clone(),
372                        FieldMapping::Direct {
373                            from_field: field_name.clone(),
374                            to_field: field_name.clone(),
375                        },
376                    );
377                }
378            }
379        }
380
381        // Add validation rules based on risk level
382        match compatibility.risk_level {
383            RiskLevel::High | RiskLevel::Critical => {
384                validation_rules.push("validate_all_required_fields".to_string());
385                validation_rules.push("validate_data_integrity".to_string());
386            }
387            RiskLevel::Medium => {
388                validation_rules.push("validate_required_fields".to_string());
389            }
390            _ => {}
391        }
392
393        Ok(DataTransformation {
394            transformation_id,
395            description,
396            field_mappings,
397            custom_logic: None,
398            validation_rules,
399        })
400    }

402    /// Create a rollback plan
403    fn create_rollback_plan(
404        &mut self,
405        from_version: &SemanticVersion,
406        to_version: &SemanticVersion,
407    ) -> Result<MigrationPlan, MigrationPlannerError> {
408        // Create a simple rollback plan (reverse migration)
409        Ok(MigrationPlan {
410            from_version: from_version.clone(),
411            to_version: to_version.clone(),
412            steps: vec![MigrationStep {
413                from_version: from_version.major,
414                to_version: to_version.major,
415                migration_type: "rollback".to_string(),
416                description: format!("Rollback from {} to {}", from_version, to_version),
417            }],
418            estimated_duration: std::time::Duration::from_secs(300), // 5 minutes default
419            risk_level: RiskLevel::High,                             // Rollbacks are always risky
420            requires_backup: true,
421            data_transformations: vec![],
422            rollback_plan: None, // No nested rollback plans
423            warnings: vec!["Rollback may cause data loss".to_string()],
424        })
425    }

442    /// Determine migration type based on compatibility
443    fn determine_migration_type(compatibility: &CompatibilityResult) -> String {
444        match compatibility.risk_level {
445            RiskLevel::Low => "minor_upgrade".to_string(),
446            RiskLevel::Medium => "standard_migration".to_string(),
447            RiskLevel::High => "breaking_migration".to_string(),
448            RiskLevel::Critical => "major_migration".to_string(),
449        }
450    }

452    /// Estimate migration duration
453    fn estimate_migration_duration(
454        steps: &[MigrationStep],
455        transformations: &[DataTransformation],
456    ) -> std::time::Duration {
457        let base_duration = std::time::Duration::from_secs(60); // 1 minute base
458        #[allow(clippy::cast_possible_truncation)]
459        let steps_len_u64 = steps.len() as u64;
460        let step_duration = std::time::Duration::from_secs(30 * steps_len_u64);
461        let transform_duration = std::time::Duration::from_secs(
462            transformations
463                .iter()
464                .map(|t| {
465                    #[allow(clippy::cast_possible_truncation)]
466                    let field_mappings_len_u64 = t.field_mappings.len() as u64;
467                    field_mappings_len_u64 * 10
468                })
469                .sum(),
470        );
471
472        base_duration + step_duration + transform_duration
473    }

475    /// Validate a migration plan
476    pub fn validate_plan(&self, plan: &MigrationPlan) -> Result<(), MigrationPlannerError> {
477        // Check that all schemas in the plan exist
478        if !self.schema_registry.contains_key(&plan.from_version) {
479            return Err(MigrationPlannerError::ValidationFailed {
480                reason: format!("Source schema {} not found", plan.from_version),
481            });
482        }
483
484        if !self.schema_registry.contains_key(&plan.to_version) {
485            return Err(MigrationPlannerError::ValidationFailed {
486                reason: format!("Target schema {} not found", plan.to_version),
487            });
488        }
489
490        // Validate step sequence
491        if !plan.steps.is_empty() {
492            let first_step = &plan.steps[0];
493            if first_step.from_version != plan.from_version.major {
494                return Err(MigrationPlannerError::ValidationFailed {
495                    reason: "First step doesn't match plan source version".to_string(),
496                });
497            }
498
499            let last_step = &plan.steps[plan.steps.len() - 1];
500            if last_step.to_version != plan.to_version.major {
501                return Err(MigrationPlannerError::ValidationFailed {
502                    reason: "Last step doesn't match plan target version".to_string(),
503                });
504            }
505        }
506
507        // Validate data transformations
508        for (i, transformation) in plan.data_transformations.iter().enumerate() {
509            if transformation.field_mappings.is_empty() && plan.steps.len() > i {
510                return Err(MigrationPlannerError::ValidationFailed {
511                    reason: format!("Transformation {} has no field mappings", i),
512                });
513            }
514        }
515
516        Ok(())
517    }

519    /// Get migration statistics
520    pub fn get_migration_stats(&self) -> MigrationStats {
521        let total_schemas = self.schema_registry.len();
522        let total_compatibility_checks = self.compatibility_cache.len();
523
524        let risk_distribution = {
525            let mut distribution = HashMap::new();
526            for result in self.compatibility_cache.values() {
527                *distribution.entry(result.risk_level.clone()).or_insert(0) += 1;
528            }
529            distribution
530        };
531
532        let average_migration_complexity = if !self.compatibility_cache.is_empty() {
533            #[allow(clippy::cast_precision_loss)]
534            let sum_f64 = self
535                .compatibility_cache
536                .values()
537                .map(|r| r.field_changes.len())
538                .sum::<usize>() as f64;
539            #[allow(clippy::cast_precision_loss)]
540            let len_f64 = self.compatibility_cache.len() as f64;
541            sum_f64 / len_f64
542        } else {
543            0.0
544        };
545
546        MigrationStats {
547            total_schemas,
548            total_compatibility_checks,
549            risk_distribution,
550            average_migration_complexity,
551        }
552    }
