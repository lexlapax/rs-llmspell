10#[derive(Debug, Error)]
11pub enum CompatibilityError {
12    #[error("Incompatible schema versions: {from} -> {to}")]
13    IncompatibleVersions {
14        from: SemanticVersion,
15        to: SemanticVersion,
16    },
17
18    #[error("Breaking change detected: {change}")]
19    BreakingChange { change: String },
20
21    #[error("Missing required field: {field}")]
22    MissingRequiredField { field: String },
23
24    #[error("Field type mismatch: {field} ({expected} -> {actual})")]
25    FieldTypeMismatch {
26        field: String,
27        expected: String,
28        actual: String,
29    },
30}

32/// Result of compatibility analysis between two schemas
33#[derive(Debug, Clone, Serialize, Deserialize)]
34pub struct CompatibilityResult {
35    pub compatible: bool,
36    pub compatibility_level: CompatibilityLevel,
37    pub breaking_changes: Vec<String>,
38    pub warnings: Vec<String>,
39    pub field_changes: HashMap<String, FieldChange>,
40    pub migration_required: bool,
41    pub risk_level: RiskLevel,
42}

44/// Type of change made to a field
45#[derive(Debug, Clone, Serialize, Deserialize)]
46pub enum FieldChange {
47    Added {
48        new_field: FieldSchema,
49    },
50    Removed {
51        old_field: FieldSchema,
52    },
53    Modified {
54        old_field: FieldSchema,
55        new_field: FieldSchema,
56        changes: Vec<String>,
57    },
58    TypeChanged {
59        old_type: String,
60        new_type: String,
61    },
62    RequiredChanged {
63        was_required: bool,
64        now_required: bool,
65    },
66}

68/// Risk level of schema migration
69#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq, PartialOrd, Ord, Hash)]
70pub enum RiskLevel {
71    Low,      // Only additions, backward compatible
72    Medium,   // Some modifications, migration recommended
73    High,     // Breaking changes, migration required
74    Critical, // Major structural changes, data loss possible
75}

77/// Schema compatibility checker
78pub struct CompatibilityChecker;

80impl CompatibilityChecker {
81    /// Check compatibility between two schemas
82    pub fn check_compatibility(
83        from_schema: &EnhancedStateSchema,
84        to_schema: &EnhancedStateSchema,
85    ) -> CompatibilityResult {
86        let mut result = CompatibilityResult {
87            compatible: true,
88            compatibility_level: CompatibilityLevel::BackwardCompatible,
89            breaking_changes: Vec::new(),
90            warnings: Vec::new(),
91            field_changes: HashMap::new(),
92            migration_required: false,
93            risk_level: RiskLevel::Low,
94        };
95
96        // Check version compatibility
97        Self::check_version_compatibility(&from_schema.version, &to_schema.version, &mut result);
98
99        // Check field compatibility
100        Self::check_field_compatibility(&from_schema.fields, &to_schema.fields, &mut result);
101
102        // Check dependency compatibility
103        Self::check_dependency_compatibility(
104            &from_schema.dependencies,
105            &to_schema.dependencies,
106            &mut result,
107        );
108
109        // Determine overall compatibility and risk level
110        Self::determine_overall_compatibility(&mut result);
111
112        result
113    }
114
115    /// Quick compatibility check (boolean result only)
116    pub fn is_compatible(
117        from_schema: &EnhancedStateSchema,
118        to_schema: &EnhancedStateSchema,
119    ) -> bool {
120        let result = Self::check_compatibility(from_schema, to_schema);
121        result.compatible
122    }
123
124    /// Check if upgrade is safe (no data loss)
125    pub fn is_safe_upgrade(
126        from_schema: &EnhancedStateSchema,
127        to_schema: &EnhancedStateSchema,
128    ) -> bool {
129        let result = Self::check_compatibility(from_schema, to_schema);
130        result.risk_level == RiskLevel::Low || result.risk_level == RiskLevel::Medium
131    }
132
133    fn check_version_compatibility(
134        from_version: &SemanticVersion,
135        to_version: &SemanticVersion,
136        result: &mut CompatibilityResult,
137    ) {
138        if to_version.is_breaking_change_from(from_version) {
139            result.breaking_changes.push(format!(
140                "Major version change: {} -> {}",
141                from_version, to_version
142            ));
143            result.compatibility_level = CompatibilityLevel::BreakingChange;
144            result.migration_required = true;
145        } else if to_version < from_version {
146            result.breaking_changes.push(format!(
147                "Downgrade detected: {} -> {}",
148                from_version, to_version
149            ));
150            result.compatibility_level = CompatibilityLevel::BreakingChange;
151        } else if to_version.minor > from_version.minor {
152            result.warnings.push(format!(
153                "Minor version upgrade: {} -> {}",
154                from_version, to_version
155            ));
156        }
157    }
158
159    fn check_field_compatibility(
160        from_fields: &HashMap<String, FieldSchema>,
161        to_fields: &HashMap<String, FieldSchema>,
162        result: &mut CompatibilityResult,
163    ) {
164        let from_field_names: HashSet<String> = from_fields.keys().cloned().collect();
165        let to_field_names: HashSet<String> = to_fields.keys().cloned().collect();
166
167        // Check for removed fields
168        for removed_field in from_field_names.difference(&to_field_names) {
169            let old_field = from_fields.get(removed_field).unwrap();
170
171            if old_field.required {
172                result
173                    .breaking_changes
174                    .push(format!("Required field '{}' was removed", removed_field));
175            } else {
176                result
177                    .warnings
178                    .push(format!("Optional field '{}' was removed", removed_field));
179            }
180
181            result.field_changes.insert(
182                removed_field.clone(),
183                FieldChange::Removed {
184                    old_field: old_field.clone(),
185                },
186            );
187            result.migration_required = true;
188        }
189
190        // Check for added fields
191        for added_field in to_field_names.difference(&from_field_names) {
192            let new_field = to_fields.get(added_field).unwrap();
193
194            if new_field.required && new_field.default_value.is_none() {
195                result.breaking_changes.push(format!(
196                    "Required field '{}' was added without default value",
197                    added_field
198                ));
199            } else {
200                result
201                    .warnings
202                    .push(format!("Field '{}' was added", added_field));
203            }
204
205            result.field_changes.insert(
206                added_field.clone(),
207                FieldChange::Added {
208                    new_field: new_field.clone(),
209                },
210            );
211            result.migration_required = true; // Adding fields requires migration
212        }
213
214        // Check for modified fields
215        for common_field in from_field_names.intersection(&to_field_names) {
216            let old_field = from_fields.get(common_field).unwrap();
217            let new_field = to_fields.get(common_field).unwrap();
218
219            Self::check_field_modification(common_field, old_field, new_field, result);
220        }
221    }
222
223    fn check_field_modification(
224        field_name: &str,
225        old_field: &FieldSchema,
226        new_field: &FieldSchema,
227        result: &mut CompatibilityResult,
228    ) {
229        let mut changes = Vec::new();
230
231        // Check type changes
232        if old_field.field_type != new_field.field_type {
233            result.breaking_changes.push(format!(
234                "Field '{}' type changed: {} -> {}",
235                field_name, old_field.field_type, new_field.field_type
236            ));
237            changes.push(format!(
238                "Type: {} -> {}",
239                old_field.field_type, new_field.field_type
240            ));
241
242            result.field_changes.insert(
243                field_name.to_string(),
244                FieldChange::TypeChanged {
245                    old_type: old_field.field_type.clone(),
246                    new_type: new_field.field_type.clone(),
247                },
248            );
249        }
250
251        // Check required changes
252        if old_field.required != new_field.required {
253            if new_field.required && !old_field.required {
254                // Optional -> Required is breaking
255                result.breaking_changes.push(format!(
256                    "Field '{}' changed from optional to required",
257                    field_name
258                ));
259            } else {
260                // Required -> Optional is safe
261                result.warnings.push(format!(
262                    "Field '{}' changed from required to optional",
263                    field_name
264                ));
265            }
266            changes.push(format!(
267                "Required: {} -> {}",
268                old_field.required, new_field.required
269            ));
270
271            result.field_changes.insert(
272                field_name.to_string(),
273                FieldChange::RequiredChanged {
274                    was_required: old_field.required,
275                    now_required: new_field.required,
276                },
277            );
278        }
279
280        // Check default value changes
281        if old_field.default_value != new_field.default_value {
282            result
283                .warnings
284                .push(format!("Field '{}' default value changed", field_name));
285            changes.push("Default value changed".to_string());
286        }
287
288        // Check validator changes
289        if old_field.validators != new_field.validators {
290            result
291                .warnings
292                .push(format!("Field '{}' validators changed", field_name));
293            changes.push("Validators changed".to_string());
294        }
295
296        // If any changes were detected, record them
297        if !changes.is_empty() {
298            result.field_changes.insert(
299                field_name.to_string(),
300                FieldChange::Modified {
301                    old_field: old_field.clone(),
302                    new_field: new_field.clone(),
303                    changes,
304                },
305            );
306        }
307    }
308
309    fn check_dependency_compatibility(
310        from_deps: &[SemanticVersion],
311        to_deps: &[SemanticVersion],
312        result: &mut CompatibilityResult,
313    ) {
314        let from_deps_set: HashSet<_> = from_deps.iter().collect();
315        let to_deps_set: HashSet<_> = to_deps.iter().collect();
316
317        // Check for removed dependencies
318        for removed_dep in from_deps_set.difference(&to_deps_set) {
319            result
320                .warnings
321                .push(format!("Dependency on schema {} was removed", removed_dep));
322        }
323
324        // Check for added dependencies
325        for added_dep in to_deps_set.difference(&from_deps_set) {
326            result
327                .warnings
328                .push(format!("Dependency on schema {} was added", added_dep));
329        }
330    }
331
332    fn determine_overall_compatibility(result: &mut CompatibilityResult) {
333        // Determine compatibility based on breaking changes
334        if !result.breaking_changes.is_empty() {
335            result.compatible = false;
336            result.compatibility_level = CompatibilityLevel::BreakingChange;
337        }
338
339        // Determine risk level
340        result.risk_level = if !result.breaking_changes.is_empty() {
341            let critical_changes = result
342                .breaking_changes
343                .iter()
344                .any(|change| change.contains("removed") || change.contains("type changed"));
345
346            if critical_changes {
347                RiskLevel::Critical
348            } else {
349                RiskLevel::High
350            }
351        } else if result.migration_required || !result.warnings.is_empty() {
352            RiskLevel::Medium
353        } else {
354            RiskLevel::Low
355        };
356    }
357}

359/// Compatibility matrix for managing multiple schema versions
360#[derive(Debug, Clone)]
361pub struct CompatibilityMatrix {
362    results: HashMap<(SemanticVersion, SemanticVersion), CompatibilityResult>,
363}
364
365impl CompatibilityMatrix {
366    pub fn new() -> Self {
367        Self {
368            results: HashMap::new(),
369        }
370    }
371
372    /// Add a compatibility result to the matrix
373    pub fn add_result(
374        &mut self,
375        from: SemanticVersion,
376        to: SemanticVersion,
377        result: CompatibilityResult,
378    ) {
379        self.results.insert((from, to), result);
380    }
381
382    /// Get compatibility result between two versions
383    pub fn get_result(
384        &self,
385        from: &SemanticVersion,
386        to: &SemanticVersion,
387    ) -> Option<&CompatibilityResult> {
388        self.results.get(&(from.clone(), to.clone()))
389    }
390
391    /// Check if upgrade path exists between versions
392    pub fn has_upgrade_path(&self, from: &SemanticVersion, to: &SemanticVersion) -> bool {
393        if let Some(result) = self.get_result(from, to) {
394            return result.compatible || result.migration_required;
395        }
396
397        // Try to find indirect path through intermediate versions
398        self.find_upgrade_path(from, to).is_some()
399    }
400
401    /// Find upgrade path between versions
402    pub fn find_upgrade_path(
403        &self,
404        from: &SemanticVersion,
405        to: &SemanticVersion,
406    ) -> Option<Vec<SemanticVersion>> {
407        // Simple implementation - could be enhanced with graph algorithms
408        let mut path = vec![from.clone()];
409        let mut current = from.clone();
410
411        while current < *to {
412            let mut next_version = None;
413            let mut min_risk = RiskLevel::Critical;
414
415            // Find the best next step
416            for ((from_ver, to_ver), result) in &self.results {
417                if from_ver == &current
418                    && to_ver <= to
419                    && result.compatible
420                    && result.risk_level <= min_risk
421                {
422                    min_risk = result.risk_level.clone();
423                    next_version = Some(to_ver.clone());
424                }
425            }
426
427            if let Some(next) = next_version {
428                path.push(next.clone());
429                current = next;
430
431                if current == *to {
432                    return Some(path);
433                }
434            } else {
435                break;
436            }
437        }
438
439        None
440    }

442    /// Get all compatible target versions for a source version
443    pub fn get_compatible_targets(&self, from: &SemanticVersion) -> Vec<SemanticVersion> {
444        self.results
445            .iter()
446            .filter_map(|((from_ver, to_ver), result)| {
447                if from_ver == from && result.compatible {
448                    Some(to_ver.clone())
449                } else {
450                    None
451                }
452            })
453            .collect()
454    }

456    /// Get statistics about the compatibility matrix
457    pub fn get_stats(&self) -> CompatibilityStats {
458        let total_pairs = self.results.len();
459        let compatible_pairs = self.results.values().filter(|r| r.compatible).count();
460        let migration_required = self
461            .results
462            .values()
463            .filter(|r| r.migration_required)
464            .count();
465
466        let risk_distribution = {
467            let mut distribution = HashMap::new();
468            for result in self.results.values() {
469                *distribution.entry(result.risk_level.clone()).or_insert(0) += 1;
470            }
471            distribution
472        };
473
474        CompatibilityStats {
475            total_pairs,
476            compatible_pairs,
477            migration_required,
478            risk_distribution,
479        }
480    }
