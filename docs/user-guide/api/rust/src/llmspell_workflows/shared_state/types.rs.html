4use serde::{Deserialize, Serialize};
5use std::collections::HashMap;

7/// State scope for isolation between workflows
8#[derive(Debug, Clone, PartialEq, Eq, Hash, Serialize, Deserialize)]
9pub enum StateScope {
10    /// Global scope - accessible by all workflows
11    Global,
12
13    /// Workflow-specific scope
14    Workflow(String),
15
16    /// Step-specific scope within a workflow
17    Step {
18        workflow_id: String,
19        step_name: String,
20    },
21
22    /// Custom namespace
23    Custom(String),
24}
25
26impl StateScope {
27    /// Create a scoped key combining scope and key
28    pub fn scoped_key(&self, key: &str) -> String {
29        match self {
30            StateScope::Global => key.to_string(),
31            StateScope::Workflow(id) => format!("workflow:{}:{}", id, key),
32            StateScope::Step {
33                workflow_id,
34                step_name,
35            } => {
36                format!("step:{}:{}:{}", workflow_id, step_name, key)
37            }
38            StateScope::Custom(namespace) => format!("{}:{}", namespace, key),
39        }
40    }
41}

43/// State entry with metadata
44#[derive(Debug, Clone, Serialize, Deserialize)]
45pub struct StateEntry {
46    /// The actual value
47    pub value: serde_json::Value,
48
49    /// When this entry was created
50    pub created_at: chrono::DateTime<chrono::Utc>,
51
52    /// When this entry was last updated
53    pub updated_at: chrono::DateTime<chrono::Utc>,
54
55    /// Optional metadata
56    pub metadata: HashMap<String, serde_json::Value>,
57}
58
59impl StateEntry {
60    /// Create a new state entry
61    pub fn new(value: serde_json::Value) -> Self {
62        let now = chrono::Utc::now();
63        Self {
64            value,
65            created_at: now,
66            updated_at: now,
67            metadata: HashMap::new(),
68        }
69    }
70
71    /// Update the value and timestamp
72    pub fn update(&mut self, value: serde_json::Value) {
73        self.value = value;
74        self.updated_at = chrono::Utc::now();
75    }
76}

78/// Trait for types that can access workflow state
79pub trait StateAccess {
80    /// Get a value from state
81    fn get_state(&self, scope: &StateScope, key: &str) -> Option<serde_json::Value>;
82
83    /// Set a value in state
84    fn set_state(&self, scope: &StateScope, key: &str, value: serde_json::Value);
85
86    /// Delete a value from state
87    fn delete_state(&self, scope: &StateScope, key: &str);
88
89    /// List all keys in a scope
90    fn list_keys(&self, scope: &StateScope) -> Vec<String>;
91
92    /// Get all values in a scope
93    fn get_scope(&self, scope: &StateScope) -> HashMap<String, serde_json::Value>;
94
95    /// Clear all values in a scope
96    fn clear_scope(&self, scope: &StateScope);
97}
