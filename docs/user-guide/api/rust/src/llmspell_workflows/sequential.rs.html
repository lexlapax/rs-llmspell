28/// Sequential workflow that executes steps one after another
29pub struct SequentialWorkflow {
30    name: String,
31    steps: Vec<WorkflowStep>,
32    state_manager: StateManager,
33    step_executor: StepExecutor,
34    error_handler: ErrorHandler,
35    error_strategy: ErrorStrategy,
36    /// Optional workflow executor for hook integration
37    workflow_executor: Option<Arc<WorkflowExecutor>>,
38    /// Workflow metadata
39    metadata: ComponentMetadata,
40    /// Core workflow configuration for Workflow trait
41    core_config: CoreWorkflowConfig,
42    /// Core workflow steps for Workflow trait
43    core_steps: Arc<RwLock<Vec<CoreWorkflowStep>>>,
44    /// Core workflow results for Workflow trait
45    core_results: Arc<RwLock<Vec<CoreStepResult>>>,
46}

56        name: String,
57        config: WorkflowConfig,
58        registry: Option<Arc<dyn ComponentLookup>>,
59    ) -> Self {
60        let error_strategy = config.default_error_strategy.clone();
61        let error_handler = ErrorHandler::new(error_strategy.clone());
62        let state_manager = StateManager::new(config.clone());

65        } else {
66            StepExecutor::new(config.clone())

70
71        // Create core workflow config from our config
72        let core_config = CoreWorkflowConfig::new()
73            .with_max_parallel(Some(1)) // Sequential execution
74            .with_continue_on_error(matches!(error_strategy, ErrorStrategy::Continue))
75            .with_timeout(config.max_execution_time);
76
77        Self {
78            name,
79            steps: Vec::new(),
80            state_manager,
81            step_executor,
82            error_handler,
83            error_strategy,
84            workflow_executor: None,

198#[async_trait]
199impl BaseAgent for SequentialWorkflow {
200    fn metadata(&self) -> &ComponentMetadata {
201        &self.metadata
202    }

204    async fn execute_impl(
205        &self,
206        input: AgentInput,
207        context: ExecutionContext,
208    ) -> Result<AgentOutput> {
209        // Validate input
210        self.validate_input(&input).await?;
211
212        // Debug: Check state availability
213        debug!(
214            "SequentialWorkflow::execute - State available in context: {}",
215            context.state.is_some()
216        );
217
218        // Execute workflow with state (inlined from execute_with_state)
219        let start_time = Instant::now();
220        // Generate ComponentId once and use it consistently
221        let execution_component_id = ComponentId::new();
222        let execution_id = execution_component_id.to_string();
223
224        // Debug: Double-check state availability
225        debug!(
226            "execute_impl - State in context: {}, type: {:?}",
227            context.state.is_some(),
228            context.state.as_ref().map(|_| "StateAccess")
229        );
230
231        info!(
232            "Starting sequential workflow: {} (execution: {}) - State available: {}",
233            self.name,
234            execution_id,
235            context.state.is_some()
236        );
237
238        // Note: workflow.started event is now emitted by execute() wrapper in BaseAgent
239
240        // Execute workflow start hooks
241        if let Some(workflow_executor) = &self.workflow_executor {
242            let component_id = llmspell_hooks::ComponentId::new(
243                llmspell_hooks::ComponentType::Workflow,
244                format!("workflow_{}", self.name),
245            );
246            let workflow_state = self.state_manager.get_state_snapshot().await?;
247            let hook_ctx = WorkflowHookContext::new(
248                component_id,
249                self.metadata.clone(),
250                workflow_state,
251                "sequential".to_string(),
252                WorkflowExecutionPhase::WorkflowStart,
253            );
254            let _ = workflow_executor.execute_workflow_hooks(hook_ctx).await;
255        }
256
257        // Start execution tracking
258        self.state_manager.start_execution().await?;
259
260        let mut state_keys = Vec::new();
261        let mut steps_executed = 0usize;
262        let mut steps_failed = 0usize;
263        let mut steps_skipped = 0usize;
264
265        for (index, step) in self.steps.iter().enumerate() {
266            // Check for execution timeout
267            if self.state_manager.check_execution_timeout().await? {
268                error!("Workflow '{}' exceeded maximum execution time", self.name);
269                self.state_manager.complete_execution(false).await?;
270
271                // Note: workflow.failed event is now emitted by execute() wrapper in BaseAgent
272
273                return Err(LLMSpellError::Workflow {
274                    message: format!("Workflow '{}' exceeded maximum execution time", self.name),
275                    step: None,
276                    source: None,
277                });
278            }
279
280            debug!(
281                "Executing step {} of {}: {}",
282                index + 1,
283                self.steps.len(),
284                step.name
285            );
286
287            // Create execution context for step
288            let shared_data = self.state_manager.get_all_shared_data().await?;
289            let mut workflow_state = crate::types::WorkflowState::new();
290            // CRITICAL: Use the workflow's execution_component_id, not a new one!
291            workflow_state.execution_id = execution_component_id;
292            workflow_state.shared_data = shared_data;
293            workflow_state.current_step = index;
294            let mut step_context = StepExecutionContext::new(workflow_state.clone(), None);
295
296            // Pass events to step context if available
297            if let Some(ref events) = context.events {
298                step_context = step_context.with_events(events.clone());
299            }
300
301            // Pass state to step context if available
302            if let Some(ref state) = context.state {
303                info!(
304                    "Passing state to StepExecutionContext for step {} - state exists: true",
305                    step.name
306                );
307                step_context = step_context.with_state(state.clone());
308                info!(
309                    "After with_state: step_context.state is_some: {}",
310                    step_context.state.is_some()
311                );
312            } else {
313                warn!(
314                    "No state available in ExecutionContext for step {} - state is None!",
315                    step.name
316                );
317            }
318
319            // Execute step with retry logic
320            let step_result = if self.workflow_executor.is_some() {
321                self.step_executor
322                    .execute_step_with_retry_and_metadata(
323                        step,
324                        step_context,
325                        &self.error_strategy,
326                        Some(self.metadata.clone()),
327                        Some("sequential".to_string()),
328                    )
329                    .await?
330            } else {
331                self.step_executor
332                    .execute_step_with_retry(step, step_context, &self.error_strategy)
333                    .await?
334            };
335
336            // Record the result
337            self.state_manager
338                .record_step_result(step_result.clone())
339                .await?;
340
341            if step_result.success {
342                steps_executed += 1;
343
344                // Write step output to state if state is available
345                if let Some(ref state) = context.state {
346                    let state_key = WorkflowResult::generate_state_key(&execution_id, &step.name);
347                    let output_value = serde_json::json!({
348                        "step_name": step.name,
349                        "step_id": step_result.step_id.to_string(),
350                        "output": step_result.output,
351                        "duration_ms": step_result.duration.as_millis(),
352                        "retry_count": step_result.retry_count,
353                    });
354
355                    state.write(&state_key, output_value).await.map_err(|e| {
356                        LLMSpellError::Component {
357                            message: format!("Failed to write step output to state: {}", e),
358                            source: None,
359                        }
360                    })?;
361
362                    state_keys.push(state_key);
363                    debug!("Wrote step output to state for step: {}", step.name);
364                }
365
366                self.state_manager.advance_step().await?;
367            } else {
368                steps_failed += 1;
369
370                // Handle the failure based on error strategy
371                let error_action = self
372                    .error_handler
373                    .handle_step_failure(&step_result, Some(&self.error_strategy))
374                    .await?;
375
376                match error_action {
377                    ErrorAction::StopWorkflow => {
378                        warn!("Stopping workflow '{}' due to step failure", self.name);
379                        self.state_manager.complete_execution(false).await?;
380
381                        // Note: workflow.failed event is now emitted by execute() wrapper in BaseAgent
382
383                        return Err(LLMSpellError::Workflow {
384                            message: format!(
385                                "Step '{}' failed: {}",
386                                step.name,
387                                step_result
388                                    .error
389                                    .unwrap_or_else(|| "Unknown error".to_string())
390                            ),
391                            step: Some(step.name.clone()),
392                            source: None,
393                        });
394                    }
395                    ErrorAction::ContinueToNext => {
396                        warn!(
397                            "Continuing to next step after failure in step: {}",
398                            step.name
399                        );
400                        steps_skipped += 1;
401                        self.state_manager.advance_step().await?;
402                    }
403                    ErrorAction::RetryStep => {
404                        // This is handled by execute_step_with_retry, so if we're here,
405                        // all retries have been exhausted and we should continue based on strategy
406                        if matches!(self.error_strategy, ErrorStrategy::Continue) {
407                            warn!("All retries exhausted for step {}, continuing", step.name);
408                            steps_skipped += 1;
409                            self.state_manager.advance_step().await?;
410                            continue;
411                        } else {
412                            warn!(
413                                "All retries exhausted for step {}, stopping workflow",
414                                step.name
415                            );
416                            self.state_manager.complete_execution(false).await?;
417
418                            // Note: workflow.failed event is now emitted by execute() wrapper in BaseAgent
419
420                            return Err(LLMSpellError::Workflow {
421                                message: format!("All retries exhausted for step {}", step.name),
422                                step: Some(step.name.clone()),
423                                source: None,
424                            });
425                        }
426                    }
427                }
428            }
429        }
430
431        // All steps completed
432        let duration = start_time.elapsed();
433        self.state_manager.complete_execution(true).await?;
434
435        // Note: workflow.completed event is now emitted by execute() wrapper in BaseAgent
436
437        // Execute workflow completion hooks
438        if let Some(workflow_executor) = &self.workflow_executor {
439            let component_id = llmspell_hooks::ComponentId::new(
440                llmspell_hooks::ComponentType::Workflow,
441                format!("workflow_{}", self.name),
442            );
443            let workflow_state = self.state_manager.get_state_snapshot().await?;
444            let hook_ctx = WorkflowHookContext::new(
445                component_id,
446                self.metadata.clone(),
447                workflow_state,
448                "sequential".to_string(),
449                WorkflowExecutionPhase::WorkflowComplete,
450            );
451            let _ = workflow_executor.execute_workflow_hooks(hook_ctx).await;
452        }
453
454        // Return metadata-only result
455        let result = if steps_failed > 0 {
456            WorkflowResult::partial(
457                execution_id,
458                WorkflowType::Sequential,
459                self.name.clone(),
460                state_keys,
461                steps_executed,
462                steps_failed,
463                steps_skipped,
464                duration,
465                None,
466            )
467        } else {
468            WorkflowResult::success(
469                execution_id,
470                WorkflowType::Sequential,
471                self.name.clone(),
472                state_keys,
473                steps_executed,
474                duration,
475            )
476        };
477
478        // Build output text
479        let output_text = if result.success {
480            format!(
481                "Sequential workflow '{}' completed successfully. {} steps executed. Duration: {:?}",
482                result.workflow_name,
483                result.steps_executed,
484                result.duration
485            )
486        } else {
487            format!(
488                "Sequential workflow '{}' failed: {}. {} steps executed, {} failed. Duration: {:?}",
489                result.workflow_name,
490                result
491                    .error
492                    .as_ref()
493                    .map(|e| e.to_string())
494                    .unwrap_or_else(|| "Unknown error".to_string()),
495                result.steps_executed,
496                result.steps_failed,
497                result.duration
498            )
499        };
500
501        // Build metadata
502        let mut metadata = llmspell_core::types::OutputMetadata {
503            #[allow(clippy::cast_possible_truncation)]
504            execution_time_ms: Some(result.duration.as_millis() as u64),
505            ..Default::default()
506        };
507
508        metadata
509            .extra
510            .insert("workflow_type".to_string(), serde_json::json!("sequential"));
511        metadata.extra.insert(
512            "workflow_name".to_string(),
513            serde_json::json!(result.workflow_name),
514        );
515        metadata.extra.insert(
516            "execution_id".to_string(),
517            serde_json::json!(result.execution_id),
518        );
519        metadata.extra.insert(
520            "workflow_id".to_string(),
521            serde_json::json!(result.execution_id),
522        );
523        metadata.extra.insert(
524            "total_steps".to_string(),
525            serde_json::json!(self.steps.len()),
526        );
527        metadata.extra.insert(
528            "successful_steps".to_string(),
529            serde_json::json!(result.steps_executed.saturating_sub(result.steps_failed)),
530        );
531        metadata.extra.insert(
532            "failed_steps".to_string(),
533            serde_json::json!(result.steps_failed),
534        );
535
536        #[allow(clippy::cast_precision_loss)]
537        let success_rate = if result.steps_executed > 0 {
538            ((result.steps_executed - result.steps_failed) as f64 / result.steps_executed as f64)
539                * 100.0
540        } else {
541            0.0
542        };
543        metadata
544            .extra
545            .insert("success_rate".to_string(), serde_json::json!(success_rate));
546
547        // If workflow failed, return an error so BaseAgent emits workflow.failed event
548        if !result.success {
549            return Err(LLMSpellError::Workflow {
550                message: output_text.clone(),
551                step: result.error.as_ref().and_then(|e| {
552                    if let WorkflowError::StepExecutionFailed { step_name, .. } = e {
553                        Some(step_name.clone())
554                    } else {
555                        None
556                    }
557                }),
558                source: None,
559            });
560        }
561
562        Ok(AgentOutput::text(output_text).with_metadata(metadata))
563    }
564
565    async fn validate_input(&self, input: &AgentInput) -> Result<()> {
566        // Basic validation - workflow can accept any non-empty text input
567        if input.text.is_empty() {
568            return Err(LLMSpellError::Validation {
569                message: "Workflow input text cannot be empty".to_string(),
570                field: Some("text".to_string()),
571            });
572        }
573
574        // Validate that we have steps to execute
575        if self.steps.is_empty() {
576            return Err(LLMSpellError::Validation {
577                message: "Cannot execute workflow without steps".to_string(),
578                field: Some("steps".to_string()),
579            });
580        }
581
582        Ok(())
583    }
584
585    async fn handle_error(&self, error: LLMSpellError) -> Result<AgentOutput> {
586        // Handle workflow-specific errors gracefully
587        let error_text = match &error {
588            LLMSpellError::Workflow { message, step, .. } => {
589                if let Some(step_name) = step {
590                    format!("Workflow error in step '{}': {}", step_name, message)
591                } else {
592                    format!("Workflow error: {}", message)
593                }
594            }
595            LLMSpellError::Validation { message, field } => {
596                if let Some(field_name) = field {
597                    format!("Validation error in field '{}': {}", field_name, message)
598                } else {
599                    format!("Validation error: {}", message)
600                }
601            }
602            _ => format!("Sequential workflow error: {}", error),
603        };
604
605        let mut metadata = llmspell_core::types::OutputMetadata::default();
606        metadata.extra.insert(
607            "error_type".to_string(),
608            serde_json::json!("workflow_error"),
609        );
610        metadata
611            .extra
612            .insert("workflow_type".to_string(), serde_json::json!("sequential"));
613        metadata
614            .extra
615            .insert("workflow_name".to_string(), serde_json::json!(self.name));
616
617        Ok(AgentOutput::text(error_text).with_metadata(metadata))
618    }
619}

645        let status = self.state_manager.get_status().await?;
646
647        // Convert our WorkflowStatus to CoreWorkflowStatus
648        let core_status = match status {
649            WorkflowStatus::Pending => CoreWorkflowStatus::Pending,
650            WorkflowStatus::Running => CoreWorkflowStatus::Running,
651            WorkflowStatus::Completed => CoreWorkflowStatus::Completed,
652            WorkflowStatus::Failed => CoreWorkflowStatus::Failed,
653            WorkflowStatus::Cancelled => CoreWorkflowStatus::Cancelled,
654            WorkflowStatus::PartiallyCompleted => CoreWorkflowStatus::Completed,
655        };
656
657        Ok(core_status)
658    }

661        let results = self.core_results.read().await;

714    pub fn with_hooks(mut self, workflow_executor: Arc<WorkflowExecutor>) -> Self {
715        self.workflow_executor = Some(workflow_executor);
716        self
717    }
718
719    /// Set the component registry for component lookup
720    pub fn with_registry(mut self, registry: Arc<dyn ComponentLookup>) -> Self {
721        self.registry = Some(registry);
722        self
723    }
724
725    /// Build the sequential workflow
726    pub fn build(mut self) -> SequentialWorkflow {
727        // Apply error strategy if provided
728        if let Some(strategy) = self.error_strategy {
729            self.config.default_error_strategy = strategy;
730        }
731
732        let mut workflow = match (self.workflow_executor, self.registry) {
733            (Some(executor), Some(registry)) => SequentialWorkflow::new_with_hooks_and_registry(
734                self.name,
735                self.config,
736                executor,
737                Some(registry),
738            ),
739            (Some(executor), None) => {
740                SequentialWorkflow::new_with_hooks(self.name, self.config, executor)
741            }
742            (None, Some(registry)) => {
743                SequentialWorkflow::new_with_registry(self.name, self.config, Some(registry))
744            }
745            (None, None) => SequentialWorkflow::new(self.name, self.config),
746        };
747        workflow.add_steps(self.steps);
748        workflow
749    }
750}
