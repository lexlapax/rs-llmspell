17/// Configuration for workflow lifecycle hook integration
18#[derive(Debug, Clone)]
19pub struct WorkflowLifecycleConfig {
20    /// Enable hook execution (can be disabled for performance)
21    pub enable_hooks: bool,
22    /// Enable circuit breaker protection
23    pub enable_circuit_breaker: bool,
24    /// Maximum time allowed for hook execution
25    pub max_hook_execution_time: Duration,
26    /// Circuit breaker configuration
27    pub circuit_breaker_failure_threshold: u32,
28    pub circuit_breaker_recovery_time: Duration,
29    /// Enable comprehensive audit logging
30    pub enable_audit_logging: bool,
31    /// Maximum security level allowed for workflow execution
32    pub max_security_level: llmspell_core::traits::tool::SecurityLevel,
33}
34
35impl Default for WorkflowLifecycleConfig {
36    fn default() -> Self {
37        Self {
38            enable_hooks: true,
39            enable_circuit_breaker: true,
40            max_hook_execution_time: Duration::from_millis(200), // 200ms max for hooks
41            circuit_breaker_failure_threshold: 5,
42            circuit_breaker_recovery_time: Duration::from_secs(30),
43            enable_audit_logging: true,
44            max_security_level: llmspell_core::traits::tool::SecurityLevel::Privileged,
45        }
46    }
47}

49/// Workflow-specific hook context with execution metadata
50#[derive(Debug, Clone)]
51pub struct WorkflowHookContext {
52    /// Base hook context
53    pub base_context: HookContext,
54    /// Workflow metadata
55    pub workflow_metadata: ComponentMetadata,
56    /// Current workflow state
57    pub workflow_state: WorkflowState,
58    /// Workflow type (Sequential, Conditional, Loop, Parallel)
59    pub workflow_type: String,
60    /// Current step information (if applicable)
61    pub step_context: Option<StepContext>,
62    /// Shared workflow data
63    pub shared_data: HashMap<String, JsonValue>,
64    /// Workflow execution phase
65    pub execution_phase: WorkflowExecutionPhase,
66    /// Pattern-specific context (e.g., iteration count for loops)
67    pub pattern_context: HashMap<String, JsonValue>,
68}

70/// Workflow execution phases for hook context
71#[derive(Debug, Clone)]
72pub enum WorkflowExecutionPhase {
73    WorkflowStart,
74    WorkflowComplete,
75    StepBoundary,
76    StateChange,
77    SharedDataAccess,
78    ConditionEvaluation,
79    BranchSelection,
80    LoopIterationStart,
81    LoopIterationComplete,
82    LoopTermination,
83    ParallelFork,
84    ParallelJoin,
85    ParallelSynchronization,
86    ErrorHandling,
87}
88
89impl WorkflowHookContext {
90    /// Create a new workflow hook context
91    pub fn new(
92        component_id: HookComponentId,
93        workflow_metadata: ComponentMetadata,
94        workflow_state: WorkflowState,
95        workflow_type: String,
96        execution_phase: WorkflowExecutionPhase,
97    ) -> Self {
98        let hook_point = match &execution_phase {
99            WorkflowExecutionPhase::WorkflowStart => {
100                HookPoint::Custom("workflow_start".to_string())
101            }
102            WorkflowExecutionPhase::WorkflowComplete => {
103                HookPoint::Custom("workflow_complete".to_string())
104            }
105            WorkflowExecutionPhase::StepBoundary => {
106                HookPoint::Custom("workflow_step_boundary".to_string())
107            }
108            WorkflowExecutionPhase::StateChange => {
109                HookPoint::Custom("workflow_state_change".to_string())
110            }
111            WorkflowExecutionPhase::SharedDataAccess => {
112                HookPoint::Custom("workflow_shared_data".to_string())
113            }
114            WorkflowExecutionPhase::ErrorHandling => HookPoint::WorkflowError,
115            WorkflowExecutionPhase::ConditionEvaluation => {
116                HookPoint::Custom("condition_evaluation".to_string())
117            }
118            WorkflowExecutionPhase::BranchSelection => {
119                HookPoint::Custom("branch_selection".to_string())
120            }
121            WorkflowExecutionPhase::LoopIterationStart => {
122                HookPoint::Custom("loop_iteration_start".to_string())
123            }
124            WorkflowExecutionPhase::LoopIterationComplete => {
125                HookPoint::Custom("loop_iteration_complete".to_string())
126            }
127            WorkflowExecutionPhase::LoopTermination => {
128                HookPoint::Custom("loop_termination".to_string())
129            }
130            WorkflowExecutionPhase::ParallelFork => HookPoint::Custom("parallel_fork".to_string()),
131            WorkflowExecutionPhase::ParallelJoin => HookPoint::Custom("parallel_join".to_string()),
132            WorkflowExecutionPhase::ParallelSynchronization => {
133                HookPoint::Custom("parallel_synchronization".to_string())
134            }
135        };
136        let base_context = HookContext::new(hook_point, component_id);
137
138        Self {
139            base_context,
140            workflow_metadata,
141            workflow_state,
142            workflow_type,
143            step_context: None,
144            shared_data: HashMap::new(),
145            execution_phase,
146            pattern_context: HashMap::new(),
147        }
148    }
149
150    /// Set step context (for step boundary hooks)
151    pub fn with_step_context(mut self, step_context: StepContext) -> Self {
152        self.step_context = Some(step_context);
153        self
154    }
155
156    /// Set shared data
157    pub fn with_shared_data(mut self, shared_data: HashMap<String, JsonValue>) -> Self {
158        self.shared_data = shared_data;
159        self
160    }
161
162    /// Add pattern-specific context
163    pub fn with_pattern_context(mut self, key: String, value: JsonValue) -> Self {
164        self.pattern_context.insert(key, value);
165        self
166    }
167
168    /// Get hook point for this execution phase
169    pub fn get_hook_point(&self) -> HookPoint {
170        self.base_context.point.clone()
171    }
172}

174/// Enhanced workflow executor with hook integration
175#[derive(Clone)]
176pub struct WorkflowExecutor {
177    /// Hook executor for running hooks
178    hook_executor: Option<Arc<HookExecutor>>,
179    /// Hook registry for retrieving hooks
180    hook_registry: Option<Arc<HookRegistry>>,
181    /// Circuit breaker for performance protection
182    circuit_breaker: Option<Arc<CircuitBreaker>>,
183    /// Configuration
184    config: WorkflowLifecycleConfig,
185    /// Component ID for this workflow executor
186    #[allow(dead_code)]
187    component_id: HookComponentId,
188}
189
190impl WorkflowExecutor {
191    /// Create a new workflow executor with hook integration
192    pub fn new(
193        config: WorkflowLifecycleConfig,
194        hook_executor: Option<Arc<HookExecutor>>,
195        hook_registry: Option<Arc<HookRegistry>>,
196    ) -> Self {
197        let component_id =
198            HookComponentId::new(ComponentType::Workflow, "workflow_executor".to_string());
199
200        let circuit_breaker = if config.enable_circuit_breaker {
201            Some(Arc::new(CircuitBreaker::new("workflow_hooks".to_string())))
202        } else {
203            None
204        };
205
206        // If we have a hook registry but no executor, create one
207        let hook_executor = match (hook_executor, hook_registry.as_ref()) {
208            (Some(exec), _) => Some(exec),
209            (None, Some(_registry)) => {
210                let exec = Arc::new(HookExecutor::new());
211                // Note: HookExecutor should ideally have a method to set registry
212                // For now, we'll just use the executor as-is
213                Some(exec)
214            }
215            (None, None) => None,
216        };
217
218        Self {
219            hook_executor,
220            hook_registry,
221            circuit_breaker,
222            config,
223            component_id,
224        }
225    }
226
227    /// Execute hooks for a workflow phase
228    pub async fn execute_workflow_hooks(
229        &self,
230        workflow_context: WorkflowHookContext,
231    ) -> Result<()> {
232        if !self.config.enable_hooks {
233            return Ok(());
234        }
235
236        let hook_point = workflow_context.get_hook_point();
237        debug!(
238            "Executing workflow hooks for phase: {:?}",
239            workflow_context.execution_phase
240        );
241
242        // Check circuit breaker
243        if let Some(circuit_breaker) = &self.circuit_breaker {
244            if !circuit_breaker.can_execute() {
245                warn!("Circuit breaker open for workflow hooks, skipping execution");
246                return Ok(());
247            }
248        }
249
250        let start_time = Instant::now();
251
252        // Execute hooks if we have both executor and registry
253        if let (Some(hook_executor), Some(hook_registry)) =
254            (&self.hook_executor, &self.hook_registry)
255        {
256            // Convert workflow context to hook context
257            let mut hook_context = workflow_context.base_context.clone();
258
259            // Add workflow-specific metadata
260            hook_context.metadata.insert(
261                "workflow_type".to_string(),
262                workflow_context.workflow_type.clone(),
263            );
264            hook_context.metadata.insert(
265                "workflow_state".to_string(),
266                format!("{:?}", workflow_context.workflow_state),
267            );
268            hook_context.metadata.insert(
269                "execution_phase".to_string(),
270                format!("{:?}", workflow_context.execution_phase),
271            );
272
273            // Add step context if present
274            if let Some(step_ctx) = &workflow_context.step_context {
275                hook_context
276                    .metadata
277                    .insert("step_name".to_string(), step_ctx.name.clone());
278                hook_context
279                    .metadata
280                    .insert("step_index".to_string(), step_ctx.index.to_string());
281                hook_context
282                    .metadata
283                    .insert("step_type".to_string(), step_ctx.step_type.clone());
284            }
285
286            // Add pattern context
287            for (key, value) in &workflow_context.pattern_context {
288                hook_context
289                    .data
290                    .insert(format!("pattern_{}", key), value.clone());
291            }
292
293            // Get hooks from registry for this hook point
294            let hooks = hook_registry.get_hooks(&hook_point);
295
296            if !hooks.is_empty() {
297                // Execute hooks
298                let results = hook_executor.execute_hooks(&hooks, &mut hook_context).await;
299
300                match results {
301                    Ok(hook_results) => {
302                        // Check results for any that should block execution
303                        for result in hook_results {
304                            if let HookResult::Cancel(reason) = result {
305                                return Err(llmspell_core::LLMSpellError::Workflow {
306                                    message: format!(
307                                        "Hook cancelled workflow execution for phase {:?}: {}",
308                                        workflow_context.execution_phase, reason
309                                    ),
310                                    step: workflow_context
311                                        .step_context
312                                        .as_ref()
313                                        .map(|s| s.name.clone()),
314                                    source: None,
315                                });
316                            }
317                        }
318                    }
319                    Err(e) => {
320                        warn!(
321                            "Hook execution failed for phase {:?}: {}",
322                            workflow_context.execution_phase, e
323                        );
324                        // Continue execution - hooks should not break workflow functionality
325                    }
326                }
327            }
328
329            let duration = start_time.elapsed();
330            debug!(
331                "Workflow hooks executed for phase: {:?} in {:?}",
332                workflow_context.execution_phase, duration
333            );
334
335            // Record success with circuit breaker
336            if let Some(circuit_breaker) = &self.circuit_breaker {
337                circuit_breaker.record_success(duration);
338            }
339        }
340
341        // Audit logging if enabled
342        if self.config.enable_audit_logging {
343            self.log_audit_event(&workflow_context);
344        }
345
346        Ok(())
347    }
348
349    /// Execute hooks for step boundaries
350    pub async fn execute_step_hooks(
351        &self,
352        workflow_metadata: ComponentMetadata,
353        workflow_state: WorkflowState,
354        workflow_type: String,
355        step_context: StepContext,
356        _is_pre_execution: bool,
357    ) -> Result<()> {
358        let phase = WorkflowExecutionPhase::StepBoundary;
359
360        let component_id = HookComponentId::new(
361            ComponentType::Workflow,
362            format!("workflow_{}", workflow_metadata.name),
363        );
364
365        let mut hook_context = WorkflowHookContext::new(
366            component_id,
367            workflow_metadata,
368            workflow_state,
369            workflow_type,
370            phase,
371        );
372        hook_context = hook_context.with_step_context(step_context);
373
374        self.execute_workflow_hooks(hook_context).await
375    }
376
377    /// Execute hooks for state changes
378    pub async fn execute_state_change_hooks(
379        &self,
380        workflow_metadata: ComponentMetadata,
381        old_state: WorkflowState,
382        new_state: WorkflowState,
383        workflow_type: String,
384    ) -> Result<()> {
385        let component_id = HookComponentId::new(
386            ComponentType::Workflow,
387            format!("workflow_{}", workflow_metadata.name),
388        );
389
390        let mut hook_context = WorkflowHookContext::new(
391            component_id,
392            workflow_metadata,
393            new_state,
394            workflow_type,
395            WorkflowExecutionPhase::StateChange,
396        );
397
398        // Add old state to pattern context
399        hook_context = hook_context.with_pattern_context(
400            "old_state".to_string(),
401            JsonValue::String(format!("{:?}", old_state)),
402        );
403
404        self.execute_workflow_hooks(hook_context).await
405    }
406
407    /// Log audit event for workflow execution
408    fn log_audit_event(&self, context: &WorkflowHookContext) {
409        info!(
410            workflow = %context.workflow_metadata.name,
411            workflow_type = %context.workflow_type,
412            state = ?context.workflow_state,
413            phase = ?context.execution_phase,
414            "Workflow hook audit event"
415        );
416    }
417}
