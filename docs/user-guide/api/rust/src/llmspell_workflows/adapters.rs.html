12/// Adapter for converting between agent and workflow inputs
13pub struct WorkflowInputAdapter;
14
15impl WorkflowInputAdapter {
16    /// Convert AgentInput to WorkflowInput
17    ///
18    /// Extracts workflow-specific parameters from AgentInput:
19    /// - Main text becomes the primary input
20    /// - Parameters are mapped to context
21    /// - Special handling for timeout parameter
22    pub fn from_agent_input(input: AgentInput) -> WorkflowInput {
23        // Extract timeout if provided in parameters
24        let timeout = input
25            .parameters
26            .get("timeout_ms")
27            .and_then(|v| v.as_u64())
28            .map(Duration::from_millis)
29            .or_else(|| {
30                input
31                    .parameters
32                    .get("timeout_secs")
33                    .and_then(|v| v.as_u64())
34                    .map(Duration::from_secs)
35            });
36
37        // Convert agent parameters to workflow context
38        let mut context = HashMap::new();
39        for (key, value) in input.parameters {
40            // Skip special parameters that we handle separately
41            if key != "timeout_ms" && key != "timeout_secs" {
42                context.insert(key, value);
43            }
44        }
45
46        // Add execution context to workflow context if present
47        if let Some(exec_context) = &input.context {
48            context.insert(
49                "conversation_id".to_string(),
50                json!(exec_context.conversation_id),
51            );
52
53            // Store any additional context info
54            if let Some(session_id) = &exec_context.session_id {
55                context.insert("session_id".to_string(), json!(session_id));
56            }
57        }
58
59        // Create workflow input with text as primary input
60        WorkflowInput {
61            input: json!({
62                "text": input.text,
63                "original_type": "agent_input"
64            }),
65            context,
66            timeout,
67        }
68    }
69
70    /// Convert WorkflowInput back to AgentInput
71    ///
72    /// Useful for workflow composition where workflows call other agents
73    pub fn to_agent_input(workflow_input: WorkflowInput) -> AgentInput {
74        // Extract text from workflow input
75        let text = workflow_input
76            .input
77            .get("text")
78            .and_then(|v| v.as_str())
79            .map(|s| s.to_string())
80            .unwrap_or_else(|| {
81                // Fallback: serialize entire input as text
82                workflow_input.input.to_string()
83            });
84
85        // Convert context back to parameters
86        let mut parameters = HashMap::new();
87        for (key, value) in workflow_input.context {
88            parameters.insert(key, value);
89        }
90
91        // Add timeout back as parameter if present
92        if let Some(timeout) = workflow_input.timeout {
93            parameters.insert("timeout_ms".to_string(), json!(timeout.as_millis()));
94        }
95
96        // Extract conversation_id if it was stored and create execution context
97        let context = parameters
98            .remove("conversation_id")
99            .and_then(|v| v.as_str().map(String::from))
100            .map(ExecutionContext::with_conversation);
101
102        AgentInput {
103            text,
104            media: Vec::new(),
105            context,
106            parameters,
107            output_modalities: Vec::new(),
108        }
109    }
110}

112/// Adapter for converting between workflow and agent outputs
113pub struct WorkflowOutputAdapter;
114
115impl WorkflowOutputAdapter {
116    /// Convert WorkflowOutput to AgentOutput
117    ///
118    /// Preserves workflow execution metadata and results
119    pub fn to_agent_output(workflow_output: WorkflowOutput) -> AgentOutput {
120        // Create output text based on success/failure
121        let text = if workflow_output.success {
122            match &workflow_output.output {
123                Value::String(s) => s.clone(),
124                Value::Object(map) if map.contains_key("result") => map
125                    .get("result")
126                    .and_then(|v| v.as_str())
127                    .unwrap_or("Workflow completed successfully")
128                    .to_string(),
129                _ => format!("Workflow completed: {}", workflow_output.output),
130            }
131        } else {
132            format!(
133                "Workflow failed: {}",
134                workflow_output.error.as_deref().unwrap_or("Unknown error")
135            )
136        };
137
138        // Build metadata from workflow execution details
139        let mut metadata = OutputMetadata {
140            execution_time_ms: Some(workflow_output.duration.as_millis() as u64),
141            ..Default::default()
142        };
143
144        // Add workflow-specific metadata
145        metadata.extra.insert(
146            "workflow_success".to_string(),
147            json!(workflow_output.success),
148        );
149        metadata.extra.insert(
150            "steps_executed".to_string(),
151            json!(workflow_output.steps_executed),
152        );
153        metadata.extra.insert(
154            "steps_failed".to_string(),
155            json!(workflow_output.steps_failed),
156        );
157
158        // Add final context as metadata
159        for (key, value) in workflow_output.final_context {
160            metadata.extra.insert(format!("context_{}", key), value);
161        }
162
163        // Include raw workflow output for complete data preservation
164        metadata.extra.insert(
165            "workflow_output".to_string(),
166            workflow_output.output.clone(),
167        );
168
169        // Create agent output with text and metadata
170        let mut agent_output = AgentOutput::text(text);
171        agent_output.metadata = metadata;
172
173        // If workflow output contains structured tool calls, we could map them
174        // For now, we'll store the workflow output in metadata
175        if workflow_output.output.is_object() || workflow_output.output.is_array() {
176            agent_output.metadata.extra.insert(
177                "structured_output".to_string(),
178                workflow_output.output.clone(),
179            );
180        }
181
182        agent_output
183    }
184
185    /// Convert AgentOutput back to WorkflowOutput
186    ///
187    /// Useful when agents are used as workflow steps
188    pub fn from_agent_output(agent_output: AgentOutput, duration: Duration) -> WorkflowOutput {
189        // Extract success status from metadata if available
190        let success = agent_output
191            .metadata
192            .extra
193            .get("success")
194            .and_then(|v| v.as_bool())
195            .unwrap_or(true); // Default to success if not specified
196
197        // Extract error message if present
198        let error = if !success {
199            Some(agent_output.text.clone())
200        } else {
201            None
202        };
203
204        // Convert agent output to workflow output
205        // Check if there's structured output in metadata
206        let output = agent_output
207            .metadata
208            .extra
209            .get("structured_output")
210            .cloned()
211            .unwrap_or_else(|| json!(agent_output.text));
212
213        // Extract context from metadata
214        let mut final_context = HashMap::new();
215        for (key, value) in &agent_output.metadata.extra {
216            if key.starts_with("context_") {
217                let context_key = key.strip_prefix("context_").unwrap().to_string();
218                final_context.insert(context_key, value.clone());
219            }
220        }
221
222        WorkflowOutput {
223            output,
224            success,
225            duration,
226            steps_executed: 1, // Single agent execution
227            steps_failed: if success { 0 } else { 1 },
228            final_context,
229            error,
230        }
231    }
232}

234/// Convenience functions for adapter usage
235pub mod prelude {
236    use super::*;
237
238    /// Convert AgentInput to WorkflowInput
239    pub fn agent_to_workflow_input(input: AgentInput) -> WorkflowInput {
240        WorkflowInputAdapter::from_agent_input(input)
241    }
242
243    /// Convert WorkflowInput to AgentInput
244    pub fn workflow_to_agent_input(input: WorkflowInput) -> AgentInput {
245        WorkflowInputAdapter::to_agent_input(input)
246    }
247
248    /// Convert WorkflowOutput to AgentOutput
249    pub fn workflow_to_agent_output(output: WorkflowOutput) -> AgentOutput {
250        WorkflowOutputAdapter::to_agent_output(output)
251    }
252
253    /// Convert AgentOutput to WorkflowOutput
254    pub fn agent_to_workflow_output(output: AgentOutput, duration: Duration) -> WorkflowOutput {
255        WorkflowOutputAdapter::from_agent_output(output, duration)
256    }
257}

259#[cfg(test)]

263    #[test]
264    fn test_agent_to_workflow_input_conversion() {
265        let mut params = HashMap::new();
266        params.insert("timeout_ms".to_string(), json!(5000));
267        params.insert("max_retries".to_string(), json!(3));
268
269        let mut agent_input = AgentInput::text("Process this data");
270        agent_input.parameters = params;
271        agent_input.context = Some(ExecutionContext::with_conversation("conv-123".to_string()));
272
273        let workflow_input = WorkflowInputAdapter::from_agent_input(agent_input);
274
275        assert_eq!(
276            workflow_input.input.get("text").unwrap().as_str().unwrap(),
277            "Process this data"
278        );
279        assert_eq!(workflow_input.timeout, Some(Duration::from_millis(5000)));
280        assert_eq!(
281            workflow_input.context.get("max_retries").unwrap(),
282            &json!(3)
283        );
284        assert_eq!(
285            workflow_input.context.get("conversation_id").unwrap(),
286            &json!("conv-123")
287        );
288    }
289
290    #[test]
291    fn test_workflow_to_agent_output_conversion() {
292        let mut context = HashMap::new();
293        context.insert("processed_items".to_string(), json!(42));
294
295        let workflow_output = WorkflowOutput {
296            output: json!({"result": "Data processed successfully"}),
297            success: true,
298            duration: Duration::from_secs(2),
299            steps_executed: 5,
300            steps_failed: 0,
301            final_context: context,
302            error: None,
303        };
304
305        let agent_output = WorkflowOutputAdapter::to_agent_output(workflow_output);
306
307        assert_eq!(agent_output.text, "Data processed successfully");
308
309        let metadata = &agent_output.metadata;
310        assert_eq!(metadata.execution_time_ms, Some(2000));
311        assert_eq!(
312            metadata.extra.get("workflow_success").unwrap(),
313            &json!(true)
314        );
315        assert_eq!(metadata.extra.get("steps_executed").unwrap(), &json!(5));
316        assert_eq!(
317            metadata.extra.get("context_processed_items").unwrap(),
318            &json!(42)
319        );
320    }

322    #[test]

324        let original_input = AgentInput::text("Test input");

326        let converted_back = WorkflowInputAdapter::to_agent_input(workflow_input);
