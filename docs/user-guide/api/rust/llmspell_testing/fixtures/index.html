ABOUTME: Common test fixtures and data for consistent testing ABOUTME: Provides pre-configured test objects and sample data Test fixtures and common test data.

This module provides pre-configured test objects and sample data that can be used across different test suites for consistency.

ยงExamples

use llmspell_testing::fixtures::{
    sample_component_metadata,
    sample_agent_input,
    sample_workflow_steps,
};

// Use pre-configured test data
let metadata = sample_component_metadata();
assert_eq!(metadata.name, "test-component");

let input = sample_agent_input();
assert_eq!(input.text, "Test prompt");
