ABOUTME: Test helpers for tool creation and testing ABOUTME: Provides common utilities for tool tests including mock tools and test data Tool testing helpers.

This module provides common test utilities for testing tools including mock tool creation, test input generation, and tool execution helpers.

ยงExamples

use llmspell_testing::tool_helpers::{
    create_test_tool,
    create_test_tool_input,
    create_mock_tool,
};
use llmspell_core::{BaseAgent, ComponentMetadata};

// Create a test tool
let tool = create_test_tool(
    "test-tool",
    "Test Tool",
    vec![("param1", "string"), ("param2", "number")]
);

// Create test input
let input = create_test_tool_input(vec![
    ("operation", "test"),
    ("param1", "value"),
    ("param2", "42"),
]);

// Execute tool
let output = tool.execute(input, Default::default()).await.unwrap();
